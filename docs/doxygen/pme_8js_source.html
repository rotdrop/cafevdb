704/**
705 * Post the content of a pme-form via AJAX into a dialog
706 * widget. Useful for editing, viewing etc. because this avoids the
707 * need to reload the base table (when only viewing single
708 * data-sets).
709 *
710 * @param {jQuery} form The form to take the informatoin from, including the
711 * name of the PHP class which generates the response.
712 *
713 * @param {jQuery} element The input element which initiates the "form
714 * submit". In particular, we assume PME "view operation" if element
715 * carries a CSS class "pme-viewXXXXX" with XXXXX being anything.
716 *
717 * @param {string} containerSel TBD.
718 *
719 * @returns {boolean}
720 */
721const tableDialog = function(form, element, containerSel) {
722
723 let post = form.serialize();
724 let templateRenderer = form.find('input[name="templateRenderer"]');
725
726 if (templateRenderer.length === 0) {
727 // This just does not work.
728 return false;
729 }
730 templateRenderer = templateRenderer.val();
731
732 let viewOperation = false;
733
734 const initialName = element.attr('name');
735 const initialValue = element.val();
736 if (initialName) {
737 post += '&' + $.param({ [initialName]: initialValue });
738 }
739 if (element.hasClass(pmeToken('add'))) {
740 // start with all tabs open in when adding data
741 post += '&' + $.param({ [pmeSys('cur_tab')]: 'all' });
742 }
743
744 const cssClass = element.attr('class');
745 if (cssClass) {
746 viewOperation = cssClass.indexOf(pmeToken('view')) > -1;
747 }
748
749 let dialogCSSId = PHPMyEdit.dialogCSSId;
750 containerSel = pmeSelector(containerSel);
751 if (containerSel !== pmeDefaultSelector) {
752 if (containerSel.charAt(0) === '#') {
753 dialogCSSId = containerSel.substring(1) + '-' + dialogCSSId;
754 } else {
755 dialogCSSId = containerSel + '.' + dialogCSSId;
756 }
757 }
758
759 const tableOptions = {
760 ambientContainerSelector: containerSel,
761 dialogHolderCSSId: dialogCSSId,
762 templateRenderer,
763 initialViewOperation: viewOperation,
764 initialName,
765 initialValue,
766 reloadName: initialName,
767 reloadValue: initialValue,
768 modalDialog: true,
769 modified: false, // avoid reload of base table unless necessary
770 };
771 pmeTableDialogOpen(tableOptions, post);
772 return true;
773};
