232 public function callInternalRoute(
233 string $route,
234 string $method = self::POST,
235 array $routeParams = [],
236 array $requestData = [],
237 string $postType = self::JSON,
238 ):array {
239 if (!$this->session->isClosed()) {
240 if ($this->closeSession) {
241 $this->session->close();
242 } else {
243 throw new Exceptions\SessionStillOpenException(
244 $this->l->t('Cannot call internal route while the session is open.'),
245 session: $this->session
246 );
247 }
248 }
249
250 // $this->logInfo('CALL ROUTE ' . $route . ' ' . print_r($routeParams, true) . ' ' . print_r($requestData, true));
251
252 $headers = [];
253
254 $requestToken = \OCP\Util::callRegister();
255 $requestData['requesttoken'] = $requestToken;
256 $urlParameters = [
257 'requesttoken' => $requestToken,
258 'format' => 'json',
259 ];
260
261 if ($method == self::GET) {
262 $urlParameters = array_merge($urlParameters, $requestData);
263 $requestData = null;
264 }
265 if (!empty($requestData)) {
266 switch ($postType) {
267 case self::JSON:
268 if (is_array($requestData)) {
269 $requestData = \OC_JSON::encode($requestData);
270 $headers[] = 'Content-Type: application/json';
271 }
272 break;
273 case self::URL_ENCODED:
274 if (is_array($requestData)) {
275 $requestData = http_build_query($requestData, '', '&');
276 }
277 break;
278 default:
279 throw new InvalidArgumentException(
280 $this->l->t('Supported data formats are "%1$s" and "%2$s", specified was "%3$s".', [
281 self::JSON, self::URL_ENCODED, $postType
282 ]));
283 break;
284 }
285 }
286
287 if (str_starts_with($route, 'http')) {
288 unset($urlParameters['format']);
289 $appendSep = strrchr($url, '?') === false ? '?' : '&';
290 $url = $route . $appendSep . http_build_query($urlParameters, '', '&');
291 } else {
292 $url = $this->urlGenerator->linkToRouteAbsolute($route, array_merge($routeParams, $urlParameters));
293 }
294
295 $cookies = array();
296 foreach ($this->request->cookies as $name => $value) {
297 $cookies[] = "$name=" . urlencode($value);
298 }
299
300 $headers[] = 'OCS-APIRequest: true';
301 $headers[] = 'Accept: application/json';
302 $headers[] = 'requesttoken: ' . $requestToken;
303
304 $c = curl_init($url);
305 curl_setopt($c, CURLOPT_VERBOSE, 0);
306 curl_setopt($c, CURLOPT_HEADER, 0);
307 curl_setopt($c, CURLOPT_HTTPHEADER, $headers);
308 switch ($method) {
309 case self::GET:
310 curl_setopt($c, CURLOPT_HTTPGET, 1);
311 curl_setopt($c, CURLOPT_POST, 0);
312 break;
313 case self::POST:
314 curl_setopt($c, CURLOPT_POST, 1);
315 break;
316 case self::PUT:
317 curl_setopt($c, CURLOPT_POST, 0);
318 curl_setopt($c, CURLOPT_CUSTOMREQUEST, 'PUT');
319 break;
320 case self::DELETE:
321 curl_setopt($c, CURLOPT_POST, 0);
322 curl_setopt($c, CURLOPT_CUSTOMREQUEST, 'DELETE');
323 break;
324 }
325 if (!empty($requestData)) {
326 curl_setopt($c, CURLOPT_POSTFIELDS, $requestData);
327 $this->logDebug('CURL REQUEST DATA ' . $requestData);
328 }
329 if (count($cookies) > 0) {
330 curl_setopt($c, CURLOPT_COOKIE, join("; ", $cookies));
331 }
332 curl_setopt($c, CURLOPT_RETURNTRANSFER, 1);
333 // this is internal, so there is no point in verifying certs:
334 curl_setopt($c, CURLOPT_SSL_VERIFYHOST, 0);
335 curl_setopt($c, CURLOPT_SSL_VERIFYPEER, 0);
336
337 $result = curl_exec($c);
338 curl_close($c);
339
340 $responseData = json_decode($result, true);
341 if (!is_array($responseData)) {
342 throw new RuntimeException(
343 $this->l->t('Invalid response from API call: "%s"', print_r($result, true)));
344 }
345
346 $this->logDebug('RESPONSE DATA ' . print_r($responseData, true));
347
348 // Some apps still return HTTP_STATUS_OK and code errors and success in
349 // the old way instead of using HTTP-error-codes.
350 if (($responseData['status']??null) != 'success' && isset($responseData['data'])) {
351 throw new RuntimeException(
352 $this->l->t('Error response from call to internal route "%1$s": %2$s', [
353 $route, $responseData['data']['message']??print_r($responseData, true)
354 ]));
355 }
356
357 // parse OCS responses if format matches
358 if (isset($responseData['ocs'])) {
359 $meta = $responseData['ocs']['meta']??null;
360 $data = $responseData['ocs']['data']??null;
361 if ($meta === null || $data === null) {
362 throw new RuntimeException(
363 $this->l->t('Invalid OCS response from call to internal route "%1$s": %2$s', [
364 $route, $responseData,
365 ]));
366 }
367 if ($meta['statuscode'] !== 100) {
368 throw new RuntimeException(
369 $this->l->t('Error response from call to internal route "%1$s": %2$s -- %3$s', [
370 $route, $meta['status']??'unknwown status', $meta['message']??print_r($meta, true)
371 ]));
372 }
373 return $data;
374 }
375
376 if (isset($responseData['data'])) {
377 return $responseData['data'];
378 } else {
379 return $responseData;
380 }
381 }
