50const confirmedReceivablesUpdate = function(updateStrategy, requestHandler, single) {
51 const handlerWithProgress = function() {
52 ProgressStatus.create(-1, 0, { field: null, musician: '', receivable: '' })
53 .fail(Ajax.handleError)
54 .done(function(data) {
55 if (!Ajax.validateResponse(data, ['id'])) {
56 return;
57 }
58 const progressToken = data.id;
59 const progressWrapperTemplate = $('#progressWrapperTemplate');
60 const progressWrapperId = 'project-participant-fields-progress';
61 const progressWrapper = progressWrapperTemplate.octemplate({
62 wrapperId: progressWrapperId,
63 caption: '',
64 label: '',
65 });
66 const oldProgressWrapper = $('#' + progressWrapperId);
67 if (oldProgressWrapper.length === 0) {
68 $('body').append(progressWrapper);
69 } else {
70 oldProgressWrapper.replaceWith(progressWrapper);
71 }
72 progressWrapper.find('span.progressbar').progressbar({ value: 0, max: 100 });
73 let progressOpen = false;
74 progressWrapper.cafevDialog({
75 title: t(appName, 'Updating recurring receivables'),
76 width: 'auto',
77 height: 'auto',
78 modal: true,
79 closeOnEscape: false,
80 resizable: false,
81 dialogClass: 'progress-status progress',
82 buttons: [
83 {
84 text: t(appName, 'cancel'),
85 title: t(appName, 'Cancel the operation in progress.'),
86 class: 'cancel',
87 click() {
88 // if (ajaxRequest) {
89 // ajaxRequest.abort('cancelled');
90 // ajaxRequest = null;
91 // }
92 $(this).dialog('close');
93 },
94 },
95 ],
96 open() {
97 const dialogHolder = $(this);
98 DialogUtils.toBackButton(dialogHolder);
99 DialogUtils.customCloseButton(dialogHolder, function(event, container) {
100 dialogHolder.dialog('widget').find('.cancel.ui-button').trigger('click');
101 return false;
102 });
103 progressOpen = true;
104 ProgressStatus.poll(progressToken, {
105 update(id, current, target, data) {
106 if (data.field) {
107 try {
108 dialogHolder.dialog(
109 'option', 'title',
110 t(appName, 'Updating receivables for {field}, {receivable}, {musician}',
111 { field: data.field || '', receivable: data.receivable || '', musician: data.musician || '' })
112 );
113 } catch (e) {
114 // don't care
115 }
116 }
117 if (current >= 0 && target > 0) {
118 progressWrapper.find('.progressbar .label').text(
119 t(appName, '{current} of {target}', { current, target }));
120 }
121 progressWrapper.find('.progressbar').progressbar('option', 'value', current / target * 100.0);
122 return current < 0 || current !== target;
123 },
124 fail(xhr, status, errorThrown) { Ajax.handleError(xhr, status, errorThrown); },
125 interval: 500,
126 });
127 },
128 close() {
129 progressOpen = false;
130 ProgressStatus.poll.stop();
131 ProgressStatus.delete(progressToken);
132 progressWrapper.dialog('destroy');
133 progressWrapper.hide();
134 },
135 });
136 requestHandler(progressToken, function() {
137 if (progressOpen) {
138 try {
139 progressWrapper.dialog('close');
140 } catch (e) {
141 // don't care
142 }
143 }
144 });
145 });
146 };
147 const handler = single
148 ? function() { requestHandler(null, function() {}); }
149 : handlerWithProgress;
150 if (updateStrategy === 'replace') {
151 Dialogs.confirm(
152 t(appName, 'Update strategy "{updateStrategy}" replaces the value of existing receivables, please confirm that you want to continue.', { updateStrategy }),
153 t(appName, 'Overwrite Existing Records?'),
154 (answer) => (answer && handler()));
155 } else {
156 handler();
157 }
