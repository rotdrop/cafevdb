148 public static function validateAddress(string $address):bool
149 {
150 $phpMailer = new PHPMailer(exceptions: true);
151 $parser = new Mail_RFC822(null, null, null, false);
152 $parsedAddresses = $parser->parseAddressList($address);
153 $parseError = $parser->parseError();
154 if ($parseError !== false) {
155 return false;
156 }
157 if (count($parsedAddresses) !== 1) {
158 return false;
159 }
160 $emailRecord = reset($parsedAddresses);
161 $email = $emailRecord->mailbox.'@'.$emailRecord->host;
162 if ($emailRecord->host == 'localhost') {
163 return false;
164 } elseif (!$phpMailer->validateAddress($email)) {
165 return false;
