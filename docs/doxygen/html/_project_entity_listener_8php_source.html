103 private function registerPreCommitAction(Entities\Project $project):void
104 {
105 if (!empty($this->preCommitActions[$project->getId()])) {
106 return;
107 }
108 if (empty($this->eventsService)) {
109 $this->eventsService = $this->appContainer->get(EventsService::class);
110 }
111 $this->preCommitActions[$project->getId()] = new GenericUndoable(
112 function() use ($project) {
113 $oldRegistrationEvent = $this->eventsService->findProjectRegistrationEvent($project);
114 if (!empty($oldRegistrationEvent)) {
115 $oldRegistrationEvent = Util::cloneArray($oldRegistrationEvent);
116 }
118 $this->eventsService->ensureProjectRegistrationEvent($project);
119 return $oldRegistrationEvent;
120 },
121 function(?array $oldRegistrationEvent) {
123 $calDavService = $this->appContainer->get(CalDavService::class);
124 if (!empty($oldRegistrationEvent)) {
125 try {
126 $calDavService->restoreCalendarObject(object: $oldRegistrationEvent);
127 } catch (Throwable $t) {
128 $this->logException($t, 'Cannot restore ' . $oldRegistrationEvent['uri'] . '.');
129 try {
130 $calendarId = $oldRegistrationEvent['calendarid'];
131 $localUri = $oldRegistrationEvent['uri'];
132 $data = $oldRegistrationEvent['calendardata'];
133 $calDavService->createCalendarObject($calendarId, $localUri, $data);
134 } catch (Throwable $t) {
135 $this->logException($t, 'Cannot recreate ' . $oldRegistrationEvent['uri'] . '.');
136 }
137 }
138 } else {
139 $registrationEvent = $this->eventsService->findProjectRegistrationEvent($project);
140 if (!empty($registrationEvent)) {
141 try {
142 $calDavService->deleteCalendarObject(
143 $registrationEvent['calendarid'],
144 $registrationEvent['uri'],
145 );
146 } catch (Throwable $t) {
147 $this->logException($t, 'Cannot delete ' . $registrationEvent['uri'] . '.');
148 }
149 }
150 }
151 }
152 );
153 $this->entityManager->registerPreCommitAction($this->preCommitActions[$project->getId()]);
