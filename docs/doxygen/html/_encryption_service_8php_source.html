265 public function initUserKeyPair(bool $forceNewKeyPair = false):void
266 {
267 $exception = null;
268 try {
269 $this->asymKeyService->initEncryptionKeyPair($this->userId, $this->userPassword, $forceNewKeyPair);
270 } catch (Exceptions\EncryptionException $exception) {
271 // Gracefully accept a broken key-pair if the app-encryption key is empty.
272 try {
273 $userKey = $this->getUserEncryptionKey();
274 $exception = null;
275 if (empty($userKey)) {
276 // after all, this means that all values are unencrypted, so be graceful here
277 $this->asymKeyService->initEncryptionKeyPair($this->userId, $this->userPassword, forceNewKeyPair: true);
278 }
279 } catch (\Throwable $t) {
280 // give up
281 $this->logException($t, 'User\'s "' . $this->userId . '" asymmetric key pair is broken and encryption key appears to be non-empty.');
282 }
283 }
284 $this->userAsymmetricCryptor = $this->asymKeyService->getCryptor($this->userId);
285 if (!empty($exception)) {
286 $this->userAsymmetricCryptor
287 ->setPrivateKey(null)
288 ->setPublicKey(null);

390 public function initAppEncryptionKey():bool
391 {
392 if (!$this->bound()) {
393 throw new Exceptions\EncryptionKeyException($this->l->t('Cannot initialize global encryption key without bound user credentials.'));
394 }
395
396 $userDatabaseKey = $this->getUserEncryptionKey();
397 if (empty($userDatabaseKey)) {
398 // No key -> unencrypted
399 $this->logDebug("No Encryption Key, setting to empty string in order to disable encryption.");
400 $this->appCryptor->setEncryptionKey(''); // not null, just empty
401 } else {
402 $this->appCryptor->setEncryptionKey($userDatabaseKey);
403 }
404
405 // compare the user-key with the stored encryption key hash
406 $sysDatabaseKeyHash = $this->getConfigValue(self::APP_ENCRYPTION_KEY_HASH_KEY);
407 if (!$this->verifyHash($userDatabaseKey, $sysDatabaseKeyHash)) {
408 // Failed
409 $this->appCryptor->setEncryptionKey(null);
410 throw new Exceptions\EncryptionKeyException($this->l->t('Failed to validate user encryption key.'));
411 $this->logError('Unable to validate HASH for encryption key.');
412 } else {
413 $this->logDebug('Encryption keys validated'.(empty($userDatabaseKey) ? ' (no encryption)' : '').'.');
414 }
415
416 return true;
417 }

476 public function encryptionKeyValid(?string $encryptionKey = null):bool
477 {
478 if (!$this->bound()) {
479 throw new Exceptions\EncryptionKeyException($this->l->t('Cannot validate app encryption key without bound user credentials'));
480 }
481
482 if (empty($encryptionKey)) {
483 $encryptionKey = $this->appCryptor->getEncryptionKey();
484 }
485 if (empty($encryptionKey)) {
486 $this->logWarn('Provided encryption-key is empty, encryption is switched off.');
487 }
488
489 $sysDatabaseKeyHash = $this->getConfigValue(self::APP_ENCRYPTION_KEY_HASH_KEY);
490
491 if (empty($sysDatabaseKeyHash) !== empty($encryptionKey)) {
492 if (empty($sysDatabaseKeyHash)) {
493 $this->logError('Stored encryption key is empty while provided encryption key is not empty.');
494 } else {
495 $this->logError('Stored encryption key is not empty while provided encryption key is empty.');
496 }
497 return false;
498 }
499
500 $match = $this->verifyHash($encryptionKey, $sysDatabaseKeyHash);
501
502 if (!$match) {
503 $this->logError('Unable to validate HASH for encryption key.');
504 } else {
505 $this->logDebug('Validated HASH for encryption key.');
506 }
507
508 return $match;

592 public function getConfigValue(string $key, mixed $default = null, bool $ignoreLock = false)
593 {
594 if (!$ignoreLock && !empty($this->getAppValue(self::CONFIG_LOCK_KEY))) {
595 throw new Exceptions\ConfigLockedException('Configuration locked, not retrieving value for ' . $key);
596 }
597 $value = $this->getAppValue($key, $default);
598
599 if (!empty($value) && ($value !== $default) && array_search($key, self::NEVER_ENCRYPT) === false) {
600 // null is error or uninitialized, string '' means no encryption
601 if ($this->appCryptor->getEncryptionKey() === null) {
602 if (!empty($this->userId)) {
603 $message = $this->l->t('Decryption requested for user "%s", but not configured, empty encryption key.', $this->userId);
604 throw new Exceptions\EncryptionKeyException($message);
605 // $this->logError($message);
606 }
607 return false;
608 }
609 try {
610 $value = $this->appCryptor->decrypt($value);
611 } catch (\Throwable $t) {
612 throw new Exceptions\DecryptionFailedException($this->l->t('Unable to decrypt value "%s" for "%s"', [$value, $key]), $t->getCode(), $t);
613 }
614 //$this->logInfo("Decrypted value for $key: ".$value);
615 }
616
617 return $value;
618 }
