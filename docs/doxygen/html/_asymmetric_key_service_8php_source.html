153 public function initEncryptionKeyPair(?string $ownerId = null, ?string $keyPassphrase = null, bool $forceNewKeyPair = false)
154 {
155 if (empty($ownerId)) {
156 $ownerId = $this->getSessionUserId();
157 }
158
159 if (empty($keyPassphrase)) {
160 $keyPassphrase = $this->getLoginPassword($ownerId);
161 }
162
163 if (empty($ownerId) || empty($keyPassphrase)) {
164 throw new Exceptions\EncryptionKeyException($this->l->t('Cannot initialize SSL key-pair without user and password'));
165 }
166
167 if (!$forceNewKeyPair && !empty(self::$keyPairs[$ownerId])) {
168 return self::$keyPairs[$ownerId];
169 }
170
171 try {
172 $keyPair = $forceNewKeyPair ? null : $this->keyStorage->getKeyPair($ownerId, $keyPassphrase);
173 } catch (\Throwable $t) {
174 $this->logException($t, $this->l->t('Unable to retrieve old key-pair for "%s".', $ownerId));
175 $keyPair = [];
176 }
177 if (empty($keyPair[self::PRIVATE_ENCRYPTION_KEY_CONFIG]) || empty($keyPair[self::PUBLIC_ENCRYPTION_KEY_CONFIG])) {
178
179 $oldKeyPair = self::$keyPairs[$ownerId] ?? null;
180 if (empty($oldKeyPair) && $ownerId == $this->getSessionUserId()) {
181 $loginPassword = $this->getLoginPassword($ownerId);
182 if (!empty($loginPassword)) {
183 try {
184 $oldKeyPair = $this->keyStorage->getKeyPair($ownerId, $loginPassword);
185 } catch (\Throwable $t) {
186 $this->logException($t, 'Unable to fetch old encryption key pair for "' . $ownerId . '".');
187 }
188 }
189 }
190
191 // make sure the old key-pair cache is set
192 self::$keyPairs[$ownerId] = $oldKeyPair;
193
194 $this->eventDispatcher->dispatchTyped(new Events\BeforeEncryptionKeyPairChanged($ownerId, $oldKeyPair));
195
196 $this->keyStorage->backupKeyPair($ownerId); // make a backup-copy if old key exists
197 $keyPair = $this->keyStorage->generateKeyPair($ownerId, $keyPassphrase);
198
199 $this->eventDispatcher->dispatchTyped(new Events\AfterEncryptionKeyPairChanged($ownerId, $oldKeyPair, $keyPair));
200 }
201
202 self::$keyPairs[$ownerId] = $keyPair;
203
204 // ensure that the cached cryptor has the correct key
205 /* $cryptor = */$this->getCryptor($ownerId)
206 ->setPrivateKey($keyPair[self::PRIVATE_ENCRYPTION_KEY_CONFIG] ?? null)
207 ->setPublicKey($keyPair[self::PUBLIC_ENCRYPTION_KEY_CONFIG] ?? null);
208
209 return $keyPair;
210 }

316 public function setSharedPrivateValue(string $ownerId, string $key, mixed $value):void
317 {
318 $value = (string)$value;
319 $configKey = self::CONFIG_KEY_PREFIX . $key;
320 if (empty($value)) {
321 $this->cloudConfig->deleteUserValue($ownerId, $this->appName, $configKey);
322 return;
323 }
324 $cryptor = $this->getCryptor($ownerId);
325 if (!$cryptor->canEncrypt()) {
326 throw new Exceptions\CannotEncryptException($this->l->t('Cannot encrypt personal value "%1$s" for "%2$s".', [ $key, $ownerId ]));
327 }
328 $this->cloudConfig->setUserValue($ownerId, $this->appName, $configKey, $cryptor->encrypt($value));
329 }

344 public function getSharedPrivateValue(string $ownerId, string $key, mixed $default = null):?string
345 {

348 if (empty($value) || $value === $default) { // allow empty and default values
