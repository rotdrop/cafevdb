116 public function initEncryptionKeyPair(?string $ownerId = null, ?string $keyPassphrase = null, bool $forceNewKeyPair = false)
117 {
118 if (empty($ownerId)) {
119 $ownerId = $this->getSessionUserId();
120 }
121
122 if (empty($ownerId)) {
123 throw new Exceptions\EncryptionException($this->l->t('The owner-id was not specified and could not be determined automatically.'));
124 }
125
126 if (empty($keyPassphrase)) {
127 $keyPassphrase = $this->getLoginPassword($ownerId);
128 }
129
130 if (empty($ownerId) || empty($keyPassphrase)) {
131 throw new Exceptions\EncryptionKeyException($this->l->t('Cannot initialize SSL key-pair without user and password'));
132 }
133
134 if (!$forceNewKeyPair && !empty(self::$keyPairs[$ownerId])) {
135 return self::$keyPairs[$ownerId];
136 }
137
138 try {
139 $keyPair = $forceNewKeyPair ? null : $this->keyStorage->getKeyPair($ownerId, $keyPassphrase);
140 } catch (\Throwable $t) {
141 $this->logException($t, $this->l->t('Unable to retrieve old key-pair for "%s".', $ownerId));
142 $keyPair = [];
143 }
144 if (empty($keyPair[self::PRIVATE_ENCRYPTION_KEY_CONFIG]) || empty($keyPair[self::PUBLIC_ENCRYPTION_KEY_CONFIG])) {
145
146 $oldKeyPair = self::$keyPairs[$ownerId] ?? null;
147 if (empty($oldKeyPair) && $ownerId == $this->getSessionUserId()) {
148 $loginPassword = $this->getLoginPassword($ownerId);
149 if (!empty($loginPassword)) {
150 try {
151 $oldKeyPair = $this->keyStorage->getKeyPair($ownerId, $loginPassword);
152 } catch (\Throwable $t) {
153 $this->logException($t, 'Unable to fetch old encryption key pair for "' . $ownerId . '".');
154 }
155 }
156 }
157
158 // make sure the old key-pair cache is set
159 self::$keyPairs[$ownerId] = $oldKeyPair;
160
161 $this->eventDispatcher->dispatchTyped(new Events\BeforeEncryptionKeyPairChanged($ownerId, $oldKeyPair));
162
163 $this->keyStorage->backupKeyPair($ownerId); // make a backup-copy if old key exists
164 $keyPair = $this->keyStorage->generateKeyPair($ownerId, $keyPassphrase);
165
166 $this->eventDispatcher->dispatchTyped(new Events\AfterEncryptionKeyPairChanged($ownerId, $oldKeyPair, $keyPair));
167 }
168
169 self::$keyPairs[$ownerId] = $keyPair;
170
171 // ensure that the cached cryptor has the correct key
172 /* $cryptor = */$this->getCryptor($ownerId)
173 ->setPrivateKey($keyPair[self::PRIVATE_ENCRYPTION_KEY_CONFIG] ?? null)
174 ->setPublicKey($keyPair[self::PUBLIC_ENCRYPTION_KEY_CONFIG] ?? null);
175
176 return $keyPair;
177 }

283 public function setSharedPrivateValue(string $ownerId, string $key, mixed $value):void
284 {
285 $value = (string)$value;
286 $configKey = self::CONFIG_KEY_PREFIX . $key;
287 if (empty($value)) {
288 $this->cloudConfig->deleteUserValue($ownerId, $this->appName, $configKey);
289 return;
290 }
291 $cryptor = $this->getCryptor($ownerId);
292 if (!$cryptor->canEncrypt()) {
293 throw new Exceptions\CannotEncryptException($this->l->t('Cannot encrypt personal value "%1$s" for "%2$s".', [ $key, $ownerId ]));
294 }
295 $this->cloudConfig->setUserValue($ownerId, $this->appName, $configKey, $cryptor->encrypt($value));
296 }

311 public function getSharedPrivateValue(string $ownerId, string $key, mixed $default = null):?string
312 {
313 $configKey = self::CONFIG_KEY_PREFIX . $key;
314 $value = $this->cloudConfig->getUserValue($ownerId, $this->appName, $configKey, $default);
315 if (empty($value) || $value === $default) { // allow empty and default values
316 return $value === null ? null : (string)$value;
317 }
318 $cryptor = $this->getCryptor($ownerId);
319 if (!$cryptor->canDecrypt()) {
320 throw new Exceptions\CannotDecryptException($this->l->t('Cannot decrypt personal value "%1$s" for "%2$s".', [ $key, $ownerId ]));
321 }
