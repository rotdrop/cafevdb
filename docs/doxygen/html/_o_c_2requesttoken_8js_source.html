22import { emit } from '@nextcloud/event-bus'
23
24/**
25 * @private
26 * @param {Document} global the document to read the initial value from
27 * @param {Function} emit the function to invoke for every new token
28 * @returns {Object}
29 */
30export const manageToken = (global, emit) => {
31 let token = global.getElementsByTagName('head')[0].getAttribute('data-requesttoken')
32
33 return {
34 getToken: () => token,
35 setToken: newToken => {
36 token = newToken
37
38 emit('csrf-token-update', {
39 token,
40 })
41 },
42 }
43}
44
45const manageFromDocument = manageToken(document, emit)
46
47/**
48 * @returns {string}
49 */
50export const getToken = manageFromDocument.getToken
51
52/**
53 * @param {String} newToken new token
54 */
55export const setToken = manageFromDocument.setToken
