261 public function open(File $fileNode, ?int $sizeLimit = null, ?string $password = null):?ArchiveService
262 {
263 if (!$this->canOpen($fileNode)) {
264 throw new Exceptions\ArchiveCannotOpenException($this->t('Unable to open archive file %s (%s)', [
265 $fileNode->getPath(), self::getLocalPath($fileNode),
266 ]));
267 }
268 $this->archiver = ArchiveBackend::open(self::getLocalPath($fileNode), password: $password);
269 if (empty($this->archiver)) {
270 throw new Exceptions\ArchiveCannotOpenException($this->t('Unable to open archive file %s (%s)', [
271 $fileNode->getPath(), self::getLocalPath($fileNode),
272 ]));
273 }
274 if ($sizeLimit === null) {
275 $sizeLimit = $this->sizeLimit;
276 }
277 $this->fileNode = $fileNode;
278 $archiveInfo = $this->getArchiveInfo();
279 $archiveSize = $archiveInfo['originalSize'];
280 if ($sizeLimit !== null && $archiveSize > $sizeLimit) {
281 $this->archiver = null;
282 $this->fileNode = null;
283 throw new Exceptions\ArchiveTooLargeException(
284 $this->t('Uncompressed size of archive "%1$s" is too large: %2$s > %3$s', [
285 $fileNode->getInternalPath(), CloudUtil::humanFileSize($archiveSize), CloudUtil::humanFileSize($sizeLimit),
286 ]),
287 $sizeLimit,
288 $archiveInfo,
289 );
290 }
291
292 return $this;
293 }

334 public static function getArchiveFolderName(string $archiveFileName):?string
335 {
336 // double to account for "nested" archive types
337 $archiveFolderName = pathinfo($archiveFileName, PATHINFO_FILENAME);
338 $secondExtension = pathinfo($archiveFolderName, PATHINFO_EXTENSION);
339
340 // as a rule of thumb we only strip the second extension if it contains no
341 // spaces and is no longer that 4 characters.
342 if (strlen($secondExtension) <= 4 && str_replace(' ', '', $secondExtension) === $secondExtension) {
343 $archiveFolderName = pathinfo($archiveFolderName, PATHINFO_FILENAME);
344 }
345
346 return $archiveFolderName;
347 }
