136 private function yearFraction(\DateTimeInterface $insuranceStart, ?\DateTimeInterface $insuranceEnd, \DateTimeInterface $dueDate)
137 {
138 $timeZone = new DateTimeZone('UTC'); // $this->getDateTimeZone();
139 $startDate = self::convertToTimezoneDate(self::convertToDateTime($insuranceStart), $timeZone);
140 $dueDate = self::convertToTimezoneDate(self::convertToDateTime($dueDate), $timeZone);
141
142 $startDistance = $startDate->diff($dueDate);
143
144 // $dueDate is before $insuranceStart
145 if ($startDistance->invert) {
146 return 0.0;
147 }
148
149 // for our purpose everything > 0 days is a month, we only charge
150 // full months
151 $startDistance->d = 0;
152
153 $months = $startDistance->y > 0 ? 12.0 : $startDistance->m;
154
155 if (!empty($insuranceEnd)) {
156 // to get the diff right -- $insuranceEnd is the last day where the
157 // instrument was included by into the insurance, we have to add one
158 // day. E.g.: Start 01.07.YYYY, end 30.06.ZZZZ should yield one year and
159 // not 365 days.
160 $endDate = self::convertToTimezoneDate(self::convertToDateTime($insuranceEnd), $timeZone)->modify('+1 day');
161 $endDistance = $dueDate->diff($endDate);
162 if ($endDistance->invert) {
163 // due-date after end-date
164 if ($endDistance->y > 0) {
165 // ended longer than one year ago, so return 0
166 return 0.0;
167 }
168 $endDistance->d = 0; // just include fractional months
169 $months -= $endDistance->m;
170 }
171 }
172
173 $fraction = $months / 12.0;
174
175 return $fraction;
176 }

202 public function insuranceFeesYearly($musicianOrId, $date = null)
203 {
204 $timeZone = $this->getDateTimeZone();
205 if (empty($date)) {
206 $date = new DateTime();
207 }
208 $date = self::convertToTimezoneDate($date, $timeZone);
209 $yearUntil = $date->format('Y');
210
211 $result = [];
212
213 $insurances = $this->billableInsurances($musicianOrId);
215 foreach ($insurances as $insurance) {
216 $amount = $insurance->getInsuranceAmount();
218 $rate = $insurance->getInsuranceRate();
219 $annualFee = $amount * $rate->getRate();
220
221 $insuranceStart = self::convertToTimezoneDate($insurance->getStartOfInsurance(), $timeZone);
222 $insuranceEnd = $insurance->getDeleted();
223 if (!empty($insuranceEnd)) {
224 $insuranceEnd = self::convertToTimezoneDate($insuranceEnd, $timeZone);
225 }
226 $startYear = $insuranceStart->format('Y');
227
228 $lastDueDate = $this->dueDate($rate->getDueDate(), ($startYear - 1).'-06-01');
229 for ($year = $startYear; $year <= $yearUntil; ++$year) {
230 $dueDate = $this->dueDate($rate->getDueDate(), $year.'-06-01');
231 if ($lastDueDate > $insuranceEnd) {
232 break;
233 }
234 $yearFraction = $this->yearFraction($insuranceStart, $insuranceEnd, $dueDate);
235 if ($yearFraction != 0.0) {
236 $fee = $yearFraction * $annualFee * self::TAXES;
237 $result[$year] = ($result[$year] ?? 0.0) + $fee;
238 }
239 $lastDueDate = $dueDate;
240 }
241 }
242 return $result;
243 }

277 public function insuranceFee(mixed $musicianOrId, $date = null, ?array &$dueInterval = null):float
278 {
279 $timeZone = $this->getDateTimeZone();
280 if (empty($date)) {
281 $date = new DateTime();
282 }
283 $date = self::convertToTimezoneDate($date, $timeZone);
284
285 $payables = $this->billableInsurances($musicianOrId);
286
287 $fee = 0.0;
290 $minDueDate = $maxDueDate = null;
292 foreach ($payables as $insurance) {
293 $insuranceStart = self::convertToTimezoneDate($insurance->getStartOfInsurance(), $timeZone);
294 $insuranceEnd = $insurance->getDeleted();
295 if (!empty($insuranceEnd)) {
296 $insuranceEnd = self::convertToTimezoneDate($insuranceEnd, $timeZone);
297 }
298
300 $rate = $insurance->getInsuranceRate();
301
302 // end of insurance period
303 $dueDate = $this->dueDate($rate->getDueDate(), $date);
304 if (!empty($insuranceEnd) && $dueDate->modify('-1 year') > $insuranceEnd) {
305 continue;
306 }
307 $minDueDate = empty($minDueDate) ? $dueDate : min($dueDate, $minDueDate);
308 $maxDueDate = empty($maxDueDate) ? $dueDate : max($dueDate, $maxDueDate);
309
310 $amount = $insurance->getInsuranceAmount();
311 $annualFee = $amount * $rate->getRate();
312 $annualFee *= $this->yearFraction($insuranceStart, $insuranceEnd, $dueDate);
313
314 $fee += $annualFee * (1.0 + self::TAXES);
315 }
316 $dueInterval = [ 'min' => $minDueDate, 'max' => $maxDueDate ];
317 return round($fee, 2);
318 }
