70 array $criteria,
71 ?array $orderBy = null,
72 ?int $limit = null,
73 ?int $offset = null,
74 ): array {
75 $qb = $this->createQueryBuilder('table');
76 $andX = $qb->expr()->andX();
77 foreach ($criteria as $key => &$value) {
78 $value = str_replace('*', '%', $value);
79 if (strpos($value, '%') !== false) {
80 $andX->add($qb->expr()->like('table'.'.'.$key, ':'.$key));
81 } else {
82 $andX->add($qb->expr()->eq('table'.'.'.$key, ':'.$key));
83 }
84 }
85 foreach ($criteria as $key => $value) {
86 $qb->setParameter($key, $value);
87 }
88 $qb->where($andX);
89
90 self::addOrderBy($qb, $orderBy, $limit, $offset, 'table');
91
92 return $qb->getQuery()->execute();

319 protected function prepareFindBy(array $criteria, ?array $orderBy = null)
320 {
321 $orderBy = $orderBy?:[];
322
323 // filter out instances of criteria
324 $collectionCriteria = [];
325 foreach ($criteria as $key => $value) {
326 if ($value instanceof Collections\Criteria) {
327 $collectionCriteria[] = $value;
328 unset($criteria[$key]);
329 }
330 }
331
332 $whereCriteria = [];
333 $havingCriteria = [];
334 foreach ($criteria as $key => $value) {
335 // if key is numeric we assume $value is a pair [ EXPR => VALUE ]
336 if (is_numeric($key)) {
337 $key = array_key_first($value);
338 $value = $value[$key];
339 }
340 $groupFunction = null;
341 $fctPos = strpos($key, '@');
342 if ($fctPos !== false) {
343 $groupFunction = substr($key, $fctPos + 1);
344 $key = substr($key, 0, $fctPos);
345 }
346 $sqlFunction = null;
347 $fctPos = strpos($key, '#');
348 if ($fctPos !== false) {
349 $sqlFunction = substr($key, $fctPos +1);
350 $key = substr($key, 0, $fctPos);
351 }
352 $fieldType = null;
353 $typePos = strpos($key, ':');
354 if ($typePos !== false) {
355 $fieldType = substr($key, $typePos + 1);
356 $key = substr($key, 0, $typePos);
357 }
358
359 $criterion = [
360 'field' => $key,
361 'fieldType' => $fieldType,
362 'value' => $value,
363 'modifiers' => [],
364 'junctors' => [],
365 'comparator' => null,
366 'literal' => false,
367 'index' => count($whereCriteria) + count($havingCriteria),
368 'groupFunction' => $groupFunction,
369 'sqlFunction' => $sqlFunction,
370 ];
371
372 if (preg_match('/^[!=<>&|()]+/', $key, $matches)) {
373
374 $operators = $matches[0];
375 $key = substr($key, strlen($operators));
376
377 $criterion['field'] = $key;
378
379 while (!empty($operators)) {
380 // reduce multiple ! signs
381 while (strpos($operators, '!!') !== false) {
382 $operators = str_replace('!!', '', $operators);
383 }
384 // (&, (|, !(|, !(&
385 foreach (self::$junctors as $abbr => $junctor) {
386 $pos = strpos($operators, $abbr);
387 if ($pos === 0) {
388 $operators = substr($operators, strlen($abbr));
389 $criterion['junctors'] = array_merge($criterion['junctors'], $junctor);
390 }
391 if (empty($operators)) {
392 break 2;
393 }
394 }
395 // this is only !
396 foreach (self::$modifiers as $abbr => $modifier) {
397 $pos = strpos($operators, $abbr);
398 if ($pos === 0) {
399 $operators = substr($operators, strlen($abbr));
400 $criterion['modifiers'][] = $modifier;
401 }
402 if (empty($operators)) {
403 break 2;
404 }
405 }
406 foreach (self::$comparisons as $abbr => $comparator) {
407 $pos = strpos($operators, $abbr);
408 if ($pos === 0) {
409 if (!empty($criterion['comparator'])) {
410 throw new DatabaseException('Comparison for key "' . $key . '" already set to "' . $criterion['comparator'] . '".');
411 }
412 if ($abbr == '=' && $value === null) {
413 throw new DatabaseException('Comparison with null is undefined, only equality is allowed.');
414 }
415 if (is_array($value)) {
416 throw new DatabaseException('Array-valued comparisons are not allowed.');
417 }
418 $operators = substr($operators, strlen($abbr));
419 $criterion['comparator'] = $comparator;
420 }
421 if (empty($operators)) {
422 break 2;
423 }
424 }
425 }
426 } // !empty($operators)
427
428 if ($groupFunction === null) {
429 $whereCriteria[] = $criterion;
430 } else {
431 $havingCriteria[] = $criterion;
432 }
433 }
434
435 // walk through criteria and find associations ASSOCIATION.FIELD
436 $joinEntities = [];
437 foreach (array_merge($whereCriteria, $havingCriteria) as $criterion) {
438 $field = $criterion['field'];
439 $dotPos = strpos($field, '.');
440 if ($dotPos !== false) {
441 $joinEntities[substr($field, 0, $dotPos)] = true;
442 }
443 }
444 $indexBy = [];
445 foreach ($orderBy as $key => $ordering) {
446 $dotPos = strpos($key, '.');
447 if ($dotPos !== false) {
448 $tableAlias = substr($key, 0, $dotPos);
449 $field = $key;
450 $joinEntities[$tableAlias] = true;
451 } else {
452 $tableAlias = 'mainTable';
453 $field = $tableAlias.'.'.$key;
454 }
455 if (strtoupper($ordering) == 'INDEX') {
456 $indexBy[$tableAlias] = $field;
457 unset($orderBy[$key]);
458 }
459 }
460
461 // modify and also return
462 $queryParts = [
463 'criteria' => [
464 'where' => $whereCriteria,
465 'having' => $havingCriteria,
466 ],
467 'orderBy' => $orderBy,
468 'joinEntities' => $joinEntities,
469 'indexBy' => $indexBy,
470 'collectionCriteria' => $collectionCriteria,
471 ];
472
473 return $queryParts;
474 }

523 protected function generateFindByWhere(
524 ORM\QueryBuilder $qb,
525 array $queryParts,
526 ?int $limit = null,
527 ?int $offset = null,
528 ):ORM\QueryBuilder {
529
530 // unpack parameter array
531 // foreach ($queryParts as $key => $part) {
532 // ${$key} = $part;
533 // }
534 extract($queryParts);
535
536 foreach (['where', 'having'] as $conditionType) {
537
538 if (!empty($criteria[$conditionType])) {
539 $groupExpression = [
540 0 => [
541 'junctor' => 'andX',
542 'components' => [],
543 ]
544 ];
545 $groupLevel = 0;
546
547 foreach ($criteria[$conditionType] as &$criterion) {
548
549 $field = $criterion['field'];
550 $value = $criterion['value'];
551
552 // $this->log('FIELD ' . $conditionType . ': ' . $field);
553
554 foreach ($criterion['junctors'] as $junctor) {
555 if ($junctor !== ')') {
556 $expression = [
557 'junctor' => $junctor,
558 'components' => [],
559 ];
560 $groupExpression[++$groupLevel] = $expression;
561 } else {
562 $expression = array_pop($groupExpression);
563 $groupLevel--;
564 $junctor = $expression['junctor'];
565 $components = $expression['components'];
566 $compositeExpr = call_user_func_array([ $qb->expr(), $junctor ], $components);
567 $groupExpression[$groupLevel]['components'][] = $compositeExpr;
568 }
569 }
570 if (empty($field)) {
571 continue;
572 }
573 // $this->log('FIELD ' . $field);
574 $dotPos = strpos($field, '.');
575 if ($dotPos !== false) {
576 $tableAlias = substr($field, 0, $dotPos);
577 } else {
578 $tableAlias = 'mainTable';
579 $field = $tableAlias . '.' . $field;
580 }
581 $param = str_replace('.', '_', $field) . '_' . $criterion['index'];
582 if (!empty($criterion['groupFunction'])) {
583 $field = sprintf($criterion['groupFunction'], $field);
584 } elseif (!empty($criterion['sqlFunction'])) {
585 $field = sprintf($criterion['sqlFunction'], $field);
586 }
587 $comparator = $criterion['comparator'] ?? 'eq';
588 if ($value === null) {
589 $criterion['literal'] = true;
590 $expr = $qb->expr()->isNull($field);
591 } elseif (is_array($value)) {
592 // special case empty array:
593 // - in principle always FALSE (nothing is in an empty set)
594 // - FIELD == NULL ? NULL in any given set would be FALSE, we
595 // keep it that way, even if the set is empty
596 if (empty($value)) {
597 $criterion['literal'] = true;
598 // unfortunately, a literal 0 just cannot be modelled with the query builder
599 $expr = $qb->expr()->eq(1, 0);
600 } else {
601 // array values could contain wildcards
602 $value = str_replace('*', '%', $value);
603 if (!empty(array_filter($value, fn($x) => strpos($x, '%') !== false))) {
604 $value = implode('|', array_map(fn($x) => str_replace('%', '.*', preg_quote($x)), $value));
605 $value = '^' . $value . '$';
606 $expr = $qb->expr()->eq(new Expr\Func('REGEXP', [ $field, ':' . $param ]), 1);
607 $criterion['value'] = $value;
608 } else {
609 $expr = $qb->expr()->in($field, ':' . $param);
610 }
611 }
612 } elseif (is_string($value)) {
613 $value = str_replace('*', '%', $value);
614 if (strpos($value, '%') !== false) {
615 $expr = $qb->expr()->like($field, ':' . $param);
616 $criterion['value'] = $value;
617 } else {
618 $expr = $qb->expr()->$comparator($field, ':' . $param);
619 }
620 } else {
621 $expr = $qb->expr()->$comparator($field, ':' . $param);
622 }
623 foreach ($criterion['modifiers'] as $modifier) {
624 $expr = $qb->expr()->$modifier($expr);
625 }
626 $groupExpression[$groupLevel]['components'][] = $expr;
627 }
628
629 // closing parenthesis maybe omitted at end, can be arbitrary many
630 while ($groupLevel >= 0) {
631 $expression = array_pop($groupExpression);
632 $groupLevel--;
633 $junctor = $expression['junctor'];
634 $components = $expression['components'];
635 $compositeExpr = call_user_func_array([ $qb->expr(), $junctor ], $components);
636 if ($groupLevel >= 0) {
637 $groupExpression[$groupLevel]['components'][] = $compositeExpr;
638 } else {
639 $qb->{$conditionType}($compositeExpr);
640 }
641 }
642 unset($criterion); // break reference
643 }
644
645 foreach ($criteria[$conditionType] as $criterion) {
646 if ($criterion['literal']) {
647 continue;
648 }
649 $field = $criterion['field'];
650 if (empty($field)) {
651 continue;
652 }
653 // $this->log('PARAMETER ' . $field);
654 $dotPos = strpos($field, '.');
655 if ($dotPos !== false) {
656 $tableAlias = substr($field, 0, $dotPos);
657 $column = substr($field, $dotPos+1);
658 } else {
659 $tableAlias = 'mainTable';
660 $column = $field;
661 $field = $tableAlias . '.' . $column;
662 }
663 $param = str_replace('.', '_', $field) . '_' . $criterion['index'];
664
665 $value = $criterion['value'];
666 $fieldType = null;
667 if ($criterion['fieldType'] !== null) {
668 $fieldType = $criterion['fieldType'];
669 } elseif ($tableAlias == 'mainTable') {
670 if (!is_array($value)
671 && empty($criterion['groupFunction'])
672 && empty($criterion['sqlFunction'])) {
673 // try to deduce the type as this is NOT done by ORM here
674 $fieldType = $this->getClassMetadata()->getTypeOfField($column);
675 }
676 }
677 $qb->setParameter($param, $value, $fieldType);
678 }
679 }
680
681 self::addOrderBy($qb, $orderBy, $limit, $offset, 'mainTable');
682
683 foreach ($collectionCriteria as $selectableCriteria) {
684 $qb->addCriteria($selectableCriteria);
685 }
686
687 return $qb;
688 }
