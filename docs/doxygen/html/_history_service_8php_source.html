104 public function push(array $data):void
105 {
106 ksort($data);
107 $md5 = md5(serialize($data));
108 $historyData = $this->historyRecords[$this->historyPosition];
109 if ($historyData['md5'] != $md5) {
110 // add the new record if it appears to be new
111 array_splice($this->historyRecords, 0, $this->historyPosition);
112 array_unshift($this->historyRecords, [ 'md5' => $md5,
113 'data' => $data ]);
114 $this->historyPosition = 0;
115 $this->historySize = count($this->historyRecords);
116 while ($this->historySize > self::MAX_HISTORY_SIZE) {
117 array_pop($this->historyRecords);
118 --$this->historySize;
119 }
120 if ($this->debug) {
121 $this->printRecords();
122 }
