95 public function push(array $data):void
96 {
97 ksort($data);
98 $md5 = md5(serialize($data));
99 $historyData = $this->historyRecords[$this->historyPosition];
100 if ($historyData['md5'] != $md5) {
101 // add the new record if it appears to be new
102 array_splice($this->historyRecords, 0, $this->historyPosition);
103 array_unshift($this->historyRecords, [ 'md5' => $md5,
104 'data' => $data ]);
105 $this->historyPosition = 0;
106 $this->historySize = count($this->historyRecords);
107 while ($this->historySize > self::MAX_HISTORY_SIZE) {
108 array_pop($this->historyRecords);
109 --$this->historySize;
110 }
111 if ($this->debug) {
112 $this->printRecords();
113 }
114 }
