106 public function getDTEndFromVEvent(mixed $vevent)
107 {
108 if ($vevent->DTEND) {
109 $dtend = $vevent->DTEND;
110 } else {
111 $dtend = clone $vevent->DTSTART;
112 // clone creates a shallow copy, also clone DateTime
113 $dtend->setDateTime(clone $dtend->getDateTime());
114 if ($vevent->DURATION) {
115 $duration = strval($vevent->DURATION);
116 $invert = 0;
117 if ($duration[0] == '-') {
118 $duration = substr($duration, 1);
119 $invert = 1;
120 }
121 if ($duration[0] == '+') {
122 $duration = substr($duration, 1);
123 }
124 $interval = new DateInterval($duration);
125 $interval->invert = $invert;
126 $dtend->getDateTime()->add($interval);
127 }
128 }
129 return $dtend;
130 }

141 public function cleanByAccessClass(string $ownerId, mixed $vobject)
142 {
143 // Do not clean your own calendar
144 if ($ownerId === $this->userId()) {
145 return $vobject;
146 }
147
148 if (isset($vobject->VEVENT)) {
149 $velement = $vobject->VEVENT;
150 } elseif (isset($vobject->VJOURNAL)) {
151 $velement = $vobject->VJOURNAL;
152 } elseif (isset($vobject->VTODO)) {
153 $velement = $vobject->VTODO;
154 }
155
156 if (isset($velement->CLASS) && $velement->CLASS->getValue() == 'CONFIDENTIAL') {
157 foreach ($velement->children as &$property) {
158 switch ($property->name) {
159 case 'CREATED':
160 case 'DTSTART':
161 case 'RRULE':
162 case 'DURATION':
163 case 'DTEND':
164 case 'CLASS':
165 case 'UID':
166 break;
167 case 'SUMMARY':
168 $property->setValue($this->l->t('Busy'));
169 break;
170 default:
171 $velement->__unset($property->name);
172 unset($property);
173 break;
174 }
175 }
176 }
177 return $vobject;
178 }

360 public function validateRequest(mixed $request)
361 {
362 $errnum = 0;
363 $errarr = array('summary'=>'false', 'cal'=>'false', 'from'=>'false', 'fromtime'=>'false', 'to'=>'false', 'totime'=>'false', 'endbeforestart'=>'false');
364 if ($request['summary'] == '') {
365 $errarr['summary'] = 'true';
366 $errnum++;
367 }
368
369 $fromday = substr($request['from'], 0, 2);
370 $frommonth = substr($request['from'], 3, 2);
371 $fromyear = substr($request['from'], 6, 4);
372 if (!checkdate($frommonth, $fromday, $fromyear)) {
373 $errarr['from'] = 'true';
374 $errnum++;
375 }
376 $allday = isset($request['allday']);
377 if (!$allday && $this->checkTime(urldecode($request['fromtime']))) {
378 $errarr['fromtime'] = 'true';
379 $errnum++;
380 }
381
382 $today = substr($request['to'], 0, 2);
383 $tomonth = substr($request['to'], 3, 2);
384 $toyear = substr($request['to'], 6, 4);
385 if (!checkdate($tomonth, $today, $toyear)) {
386 $errarr['to'] = 'true';
387 $errnum++;
388 }
389 if ($request['repeat'] != 'doesnotrepeat') {
390 if (($request['interval'] !== strval(intval($request['interval']))) || intval($request['interval']) < 1) {
391 $errarr['interval'] = 'true';
392 $errnum++;
393 }
394 if (isset($request['repeat']) && !isset($this->getRepeatOptions()[$request['repeat']])) {
395 $errarr['repeat'] = 'true';
396 $errnum++;
397 }
398 if (isset($request['advanced_month_select']) && !isset($this->getMonthOptions()[$request['advanced_month_select']])) {
399 $errarr['advanced_month_select'] = 'true';
400 $errnum++;
401 }
402 if (isset($request['advanced_year_select']) && !isset($this->getYearOptions()[$request['advanced_year_select']])) {
403 $errarr['advanced_year_select'] = 'true';
404 $errnum++;
405 }
406 if (isset($request['weekofmonthoptions']) && !isset($this->getWeekofMonth()[$request['weekofmonthoptions']])) {
407 $errarr['weekofmonthoptions'] = 'true';
408 $errnum++;
409 }
410 if ($request['end'] != 'never') {
411 if (!isset($this->getEndOptions()[$request['end']])) {
412 $errarr['end'] = 'true';
413 $errnum++;
414 }
415 if ($request['end'] == 'count' && is_nan($request['byoccurrences'])) {
416 $errarr['byoccurrences'] = 'true';
417 $errnum++;
418 }
419 if ($request['end'] == 'date') {
420 list($bydate_day, $bydate_month, $bydate_year) = explode('-', $request['bydate']);
421 if (!checkdate($bydate_month, $bydate_day, $bydate_year)) {
422 $errarr['bydate'] = 'true';
423 $errnum++;
424 }
425 }
426 }
427 if (isset($request['weeklyoptions'])) {
428 foreach ($request['weeklyoptions'] as $option) {
429 if (!in_array($option, $this->getWeeklyOptions())) {
430 $errarr['weeklyoptions'] = 'true';
431 $errnum++;
432 }
433 }
434 }
435 if (isset($request['byyearday'])) {
436 foreach ($request['byyearday'] as $option) {
437 if (!isset($this->getByYearDayOptions()[$option])) {
438 $errarr['byyearday'] = 'true';
439 $errnum++;
440 }
441 }
442 }
443 if (isset($request['weekofmonthoptions'])) {
444 if (is_nan((double)$request['weekofmonthoptions'])) {
445 $errarr['weekofmonthoptions'] = 'true';
446 $errnum++;
447 }
448 }
449 if (isset($request['bymonth'])) {
450 foreach ($request['bymonth'] as $option) {
451 if (!in_array($option, $this->getByMonthOptions())) {
452 $errarr['bymonth'] = 'true';
453 $errnum++;
454 }
455 }
456 }
457 if (isset($request['byweekno'])) {
458 foreach ($request['byweekno'] as $option) {
459 if (!in_array($option, $this->getByWeekNoOptions())) {
460 $errarr['byweekno'] = 'true';
461 $errnum++;
462 }
463 }
464 }
465 if (isset($request['bymonthday'])) {
466 foreach ($request['bymonthday'] as $option) {
467 if (!isset($this->getByMonthDayOptions()[$option])) {
468 $errarr['bymonthday'] = 'true';
469 $errnum++;
470 }
471 }
472 }
473 }
474 if (!$allday && $this->checkTime(urldecode($request['totime']))) {
475 $errarr['totime'] = 'true';
476 $errnum++;
477 }
478 if ($today < $fromday && $frommonth == $tomonth && $fromyear == $toyear) {
479 $errarr['endbeforestart'] = 'true';
480 $errnum++;
481 }
482 if ($today == $fromday && $frommonth > $tomonth && $fromyear == $toyear) {
483 $errarr['endbeforestart'] = 'true';
484 $errnum++;
485 }
486 if ($today == $fromday && $frommonth == $tomonth && $fromyear > $toyear) {
487 $errarr['endbeforestart'] = 'true';
488 $errnum++;
489 }
490 if (!$allday && $fromday == $today && $frommonth == $tomonth && $fromyear == $toyear) {
491 list($tohours, $tominutes) = explode(':', $request['totime']);
492 list($fromhours, $fromminutes) = explode(':', $request['fromtime']);
493 if ($tohours < $fromhours) {
494 $errarr['endbeforestart'] = 'true';
495 $errnum++;
496 }
497 if ($tohours == $fromhours && $tominutes < $fromminutes) {
498 $errarr['endbeforestart'] = 'true';
499 $errnum++;
500 }
501 }
502 if ($errnum) {
503 return $errarr;
504 }
505 return false;
506 }

515 protected static function checkTime(string $time)
516 {
517 if (strpos($time, ':') === false) {
518 return true;
519 }
520 list($hours, $minutes) = explode(':', $time);
521 return empty($time)
522 || $hours < 0 || $hours > 24
523 || $minutes < 0 || $minutes > 60;
524 }

565 public function updateVCalendarFromRequest(mixed $request, mixed $vcalendar)
566 {
567 $accessclass = isset($request["accessclass"]) ? $request["accessclass"] : null;
568 $summary = $request["summary"];
569 $location = $request["location"];
570 $categories = explode(',', $request["categories"]);
571 $allday = isset($request["allday"]);
572 $from = $request["from"];
573 $to = $request["to"];
574 if (!$allday) {
575 $fromtime = $request['fromtime'];
576 $totime = $request['totime'];
577 }
578 $vevent = $vcalendar->VEVENT;
579 // $this->logInfo(get_class($vcalendar));
580 $description = $request["description"];
581 $repeat = $request["repeat"];
582 if ($repeat != 'doesnotrepeat') {
583 $rrule = '';
584 $interval = $request['interval'];
585 $end = $request['end'];
586 $byoccurrences = $request['byoccurrences'];
587 switch ($repeat) {
588 case 'daily':
589 $rrule .= 'FREQ=DAILY';
590 break;
591 case 'weekly':
592 $rrule .= 'FREQ=WEEKLY';
593 if (isset($request['weeklyoptions'])) {
594 $byday = '';
595 $daystrings = array_flip($this->getWeeklyOptions());
596 foreach ($request['weeklyoptions'] as $days) {
597 if ($byday == '') {
598 $byday .= $daystrings[$days];
599 } else {
600 $byday .= ',' .$daystrings[$days];
601 }
602 }
603 $rrule .= ';BYDAY=' . $byday;
604 }
605 break;
606 case 'weekday':
607 $rrule .= 'FREQ=WEEKLY';
608 $rrule .= ';BYDAY=MO,TU,WE,TH,FR';
609 break;
610 case 'biweekly':
611 $rrule .= 'FREQ=WEEKLY';
612 $interval = $interval * 2;
613 break;
614 case 'monthly':
615 $rrule .= 'FREQ=MONTHLY';
616 if ($request['advanced_month_select'] == 'monthday') {
617 break;
618 } elseif ($request['advanced_month_select'] == 'weekday') {
619 if ($request['weekofmonthoptions'] == 'auto') {
620 list($_day, $_month, $_year) = explode('-', $from);
621 $weekofmonth = floor($_day/7);
622 } else {
623 $weekofmonth = $request['weekofmonthoptions'];
624 }
625 $days = array_flip($this->getWeeklyOptions());
626 $byday = '';
627 foreach ($request['weeklyoptions'] as $day) {
628 if ($byday == '') {
629 $byday .= $weekofmonth . $days[$day];
630 } else {
631 $byday .= ',' . $weekofmonth . $days[$day];
632 }
633 }
634 if ($byday == '') {
635 $byday = 'MO,TU,WE,TH,FR,SA,SU';
636 }
637 $rrule .= ';BYDAY=' . $byday;
638 }
639 break;
640 case 'yearly':
641 $rrule .= 'FREQ=YEARLY';
642 if ($request['advanced_year_select'] == 'bydate') {
643 list($_day, $_month, $_year) = explode('-', $from);
644 $bymonth = date('n', mktime(0, 0, 0, $_month, $_day, $_year));
645 $bymonthday = date('j', mktime(0, 0, 0, $_month, $_day, $_year));
646 $rrule .= ';BYDAY=MO,TU,WE,TH,FR,SA,SU;BYMONTH=' . $bymonth . ';BYMONTHDAY=' . $bymonthday;
647 } elseif ($request['advanced_year_select'] == 'byyearday') {
648 list($_day, $_month, $_year) = explode('-', $from);
649 $byyearday = date('z', mktime(0, 0, 0, $_month, $_day, $_year)) + 1;
650 if (isset($request['byyearday'])) {
651 foreach ($request['byyearday'] as $yearday) {
652 $byyearday .= ',' . $yearday;
653 }
654 }
655 $rrule .= ';BYYEARDAY=' . $byyearday;
656 } elseif ($request['advanced_year_select'] == 'byweekno') {
657 list($_day, $_month, $_year) = explode('-', $from);
658 $rrule .= ';BYDAY=' . strtoupper(substr(date('l', mktime(0, 0, 0, $_month, $_day, $_year)), 0, 2));
659 $byweekno = '';
660 foreach ($request['byweekno'] as $weekno) {
661 if ($byweekno == '') {
662 $byweekno = $weekno;
663 } else {
664 $byweekno .= ',' . $weekno;
665 }
666 }
667 $rrule .= ';BYWEEKNO=' . $byweekno;
668 } elseif ($request['advanced_year_select'] == 'bydaymonth') {
669 if (isset($request['weeklyoptions'])) {
670 $days = array_flip($this->getWeeklyOptions());
671 $byday = '';
672 foreach ($request['weeklyoptions'] as $day) {
673 if ($byday == '') {
674 $byday .= $days[$day];
675 } else {
676 $byday .= ',' . $days[$day];
677 }
678 }
679 $rrule .= ';BYDAY=' . $byday;
680 }
681 if (isset($request['bymonth'])) {
682 $monthes = array_flip($this->getByMonthOptions());
683 $bymonth = '';
684 foreach ($request['bymonth'] as $month) {
685 if ($bymonth == '') {
686 $bymonth .= $monthes[$month];
687 } else {
688 $bymonth .= ',' . $monthes[$month];
689 }
690 }
691 $rrule .= ';BYMONTH=' . $bymonth;
692
693 }
694 if (isset($request['bymonthday'])) {
695 $bymonthday = '';
696 foreach ($request['bymonthday'] as $monthday) {
697 if ($bymonthday == '') {
698 $bymonthday .= $monthday;
699 } else {
700 $bymonthday .= ',' . $monthday;
701 }
702 }
703 $rrule .= ';BYMONTHDAY=' . $bymonthday;
704
705 }
706 }
707 break;
708 default:
709 break;
710 }
711 if ($interval != '') {
712 $rrule .= ';INTERVAL=' . $interval;
713 }
714 if ($end == 'count') {
715 $rrule .= ';COUNT=' . $byoccurrences;
716 }
717 if ($end == 'date') {
718 list($bydate_day, $bydate_month, $bydate_year) = explode('-', $request['bydate']);
719 $rrule .= ';UNTIL=' . $bydate_year . $bydate_month . $bydate_day;
720 }
721 $vevent->RRULE = $rrule;
722 $repeat = "true";
723 } else {
724 $repeat = "false";
725 }
726
727 $now = new DateTime('now');
728 $now->setTimeZone(new DateTimeZone('UTC'));
729 $lastModified = $vevent->__get('LAST-MODIFIED');
730 if (is_null($lastModified)) {
731 $lastModified = $vevent->add('LAST-MODIFIED');
732 }
733 $lastModified->setValue($now);
734 $vevent->DTSTAMP = $now;
735
736 $vevent->SUMMARY = $summary;
737
738 if ($allday) {
739 $start = new DateTime($from);
740 $end = new DateTime($to.' +1 day');
741
742 $vevent->DTSTART = $start;
743 $vevent->DTEND = $end;
744
745 $vevent->DTSTART['VALUE'] = 'DATE';
746 $vevent->DTEND['VALUE'] = 'DATE';
747 } else {
748 //$timezone = OC_Calendar_App::getTimezone();
749 //$timezone = new DateTimeZone($timezone);
750 $timezone = $this->getDateTimeZone();
751
752 $start = new DateTime($from.' '.$fromtime, $timezone);
753 $end = new DateTime($to.' '.$totime, $timezone);
754
755 $vevent->DTSTART = $start;
756 $vevent->DTEND = $end;
757 }
758
759 unset($vevent->DURATION);
760
761 if ($accessclass !== null) {
762 $vevent->CLASS = $accessclass;
763 }
764 $vevent->LOCATION = $location;
765 $vevent->DESCRIPTION = $description;
766 if (count($categories) > 0) {
767 $vevent->CATEGORIES = $categories;
768 } else {
769 unset($vevent->CATEGORIES);
770 }
771
772 /*if ($repeat == "true") {
773 $vevent->RRULE = $repeat;
774 }*/
775
776 return $vcalendar;
777 }
