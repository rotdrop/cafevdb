98 public function generateReceivables():Collection
99 {
100 $receivableOptions = $this->serviceFeeField->getDataOptions();
101
103 $managementOption = $this->serviceFeeField->getManagementOption();
104 if (empty($managementOption)) {
105 throw new RuntimeException(
106 $this->l->t(
107 'Unable to find management option for participant field "%s".',
108 $this->serviceFeeField->getName()
109 ));
110 }
111 $managementDate = Util::convertToDateTime($managementOption->getLimit());
112 if (empty($managementDate)) {
113 $startingDate = new DateTimeImmutable;
114 } else {
115 $startingDate = $managementDate;
116 }
117
119 $startingDate = $startingDate->setTimezone($this->timeZone);
120 $seconds = ($startingDate->getTimestamp() + $this->intervalSeconds - 1)
121 / $this->intervalSeconds * $this->intervalSeconds;
122 $startingDate = $startingDate->setTimestamp($seconds);
123 $managementOption->setLimit($startingDate->getTimestamp());
124 $endingSeconds = (new DateTimeImmutable)->getTimestamp();
125
126 for (; $seconds <= $endingSeconds; $seconds += $this->intervalSeconds) {
127 $timeStamp = $this->formatTimeStamp($seconds);
128 $labelText = $this->l->t($labelTemplate = 'Option %s', $timeStamp);
129 $tooltipTemplate = $this->toolTipsService['periodic-recurring-receivable-option']??'';
130 $tooltipText = $this->l->t($tooltipTemplate);
131
132 $existingReceivables = $receivableOptions->matching(self::criteriaWhere(['data' => (string)$seconds]));
133 if ($existingReceivables->isEmpty()) {
134 // add a new option
135 $receivable = (new Entities\ProjectParticipantFieldDataOption)
136 ->setField($this->serviceFeeField)
137 ->setKey(Uuid::create())
138 ->setLabel($labelText)
139 ->setToolTip($tooltipText)
140 ->setData($seconds) // may change in the future
141 ->setLimit(null); // may change in the future
142 $receivableOptions->set($receivable->getKey()->getBytes(), $receivable);
143 } else {
144 // update display things, but keep the essential data untouched
146 $receivable = $existingReceivables->first();
147 $receivable->setLabel($labelText)
148 ->setTooltip($tooltipText);
149 }
150 $this->translate($receivable, 'label', null, sprintf($labelTemplate, $timeStamp))
151 ->translate($receivable, 'tooltip', null, $tooltipTemplate);
152 }
153 return $this->serviceFeeField->getSelectableOptions();
154 }
