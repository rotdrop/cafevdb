122 public function executeActions(bool $gracefully = false):bool
123 {
124 if ($this->executing) {
125 return true;
126 }
127 $this->executing = true;
128 $this->runQueueExceptions = [];
129 $this->undoExceptions = [];
130 $this->undoStack = [];
131 while (!empty($this->actionQueue)) {
132 try {
133 $action = array_shift($this->actionQueue); // from front
134 $action->do();
135 array_unshift($this->undoStack, $action); // at front
136 } catch (\Throwable $t) {
137 $message = $this->l->t(
138 'Exception during execution of run-queue: %1$s. Number of successful actions: %2$d.', [
139 $t->getMessage(),
140 count($this->undoStack),
141 ]);
142 $this->runQueueExceptions[] = $t;
143 if ($gracefully) {
144 $this->logException($t, $message);
145 } else {
146 throw new UndoableRunQueueException($this, $message, 0, $t);
147 }
148 }
149 }
