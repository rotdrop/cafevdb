117 public function executeActions(bool $gracefully = false):bool

128 $action = array_shift($this->actionQueue); // from front
129 $action->do();
130 array_unshift($this->undoStack, $action); // at front

136 ]);
137 $this->runQueueExceptions[] = $t;
138 if ($gracefully) {
139 $this->logException($t, $message);
140 } else {
141 $this->executing = false;
142 throw new UndoableRunQueueException($this, $message, 0, $t);
