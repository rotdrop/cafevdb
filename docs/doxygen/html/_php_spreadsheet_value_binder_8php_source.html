80 public function bindValue(PhpSpreadsheet\Cell\Cell $cell, mixed $value = null)
81 {
82 // sanitize UTF-8 strings
83 if (is_string($value)) {
84 $value = PhpSpreadsheet\Shared\StringHelper::sanitizeUTF8($value);
85 }
86
87 // Find out data type
88 $dataType = parent::dataTypeForValue($value);
89
90 // Copied over with bug-fixes: currency values can be
91 // negative. And we want to support €, of course.
92 if ($dataType === PhpSpreadsheet\Cell\DataType::TYPE_STRING && !$value instanceof PhpSpreadsheet\RichText\RichText) {
93 // Check for currency
94 $monetary = $this->fuzzyInputService->parseCurrency($value, true);
95 if ($monetary !== false) {
96 switch ($monetary['currency']) {
97 case '€':
98 // Set value
99 $cell->setValueExplicit($monetary['amount'], PhpSpreadsheet\Cell\DataType::TYPE_NUMERIC);
100 // Set style
101 $format = PhpSpreadsheet\Style\NumberFormat::FORMAT_CURRENCY_EUR_SIMPLE;
102 //$format = '#.##0,00 [$€];[ROT]-#.##0,00 [$€]';
103 $format = '#,##0.00 [$€]';
104 $format = '#,##0.00 [$€];[RED]-#,##0.00 [$€]';
105 $cell->getWorksheet()->getStyle($cell->getCoordinate())
106 ->getNumberFormat()->setFormatCode($format);
107 return true;
108 case '$':
109 $cell->setValueExplicit($monetary['amount'], PhpSpreadsheet\Cell\DataType::TYPE_NUMERIC);
110 // Set style
111 $cell->getWorksheet()->getStyle($cell->getCoordinate())
112 ->getNumberFormat()->setFormatCode(PhpSpreadsheet\Style\NumberFormat::FORMAT_CURRENCY_USD_SIMPLE);
113 return true;
114 }
115 }
116
117 // Interpret some basic html
118 $hrefre = '#<a[^>]+href="([^"]+)"[^>]*>(.*?)</a>#i';
119 if (preg_match_all($hrefre, $value, $matches, PREG_SET_ORDER)) {
120 $absUrls = [];
121 $urlUuid = Uuid::create();
122 $absUrlCount = 0;
123 foreach ($matches as &$match) {
124 $url = $match[1];
125 if ($url[0] == '/' || !str_starts_with($url, 'mailto:')) {
126 $absUrl = $this->urlGenerator->getAbsoluteURL($url);
127 $match[1] = $absUrl;
128 $absUrlKey = $urlUuid . '_' . $absUrlCount;
129 $absUrls[$absUrlKey] = $absUrl;
130 $value = str_replace($url, $absUrlKey, $value);
131 $absUrlCount++;
132 }
133 if (str_starts_with($url, 'mailto:')) {
134 // strip potential query parameters
135 $queryPos = strrpos($url, '?');
136 if ($queryPos !== false) {
137 $match[1] = substr($url, 0, $queryPos);
138 str_replace($url, $match[1], $value);
139 }
140 }
141 }
142 $value = str_replace(array_keys($absUrls), array_values($absUrls), $value);
143 $cell->getHyperlink()->setUrl($matches[0][1]);
144 if (count($matches) == 1) {
145 $cell->setValueExplicit($matches[0][2], PhpSpreadsheet\Cell\DataType::TYPE_STRING);
146 } else {
147 $h2t = new Html2Text($value, [ 'width' => 0 ]);
148 $value = trim($h2t->get_text());
149 $cell->setValueExplicit($value, PhpSpreadsheet\Cell\DataType::TYPE_STRING);
150 }
151 return true;
152 }
153
154 // Handle remaining stuff by html2text
155 $h2t = new Html2Text($value, [ 'width' => 0 ]);
156 $value = trim($h2t->get_text());
157
158 // Well, 'ja' or 'nein' ... should also count for truth values, maybe
159 switch (strtoupper($value)) {
160 case strtoupper($this->l->t('yes')):
161 case strtoupper($this->l->t('true')):
162 $value = PhpSpreadsheet\Calculation\Calculation::getTRUE();
163 break;
164 case strtoupper($this->l->t('no')):
165 case strtoupper($this->l->t('false')):
166 $value = PhpSpreadsheet\Calculation\Calculation::getFALSE();
167 break;
168 }
169
170 // Test for booleans using locale-setting
171 if ($value == PhpSpreadsheet\Calculation\Calculation::getTRUE()) {
172 $cell->setValueExplicit(true, PhpSpreadsheet\Cell\DataType::TYPE_BOOL);
173 return true;
174 } elseif ($value == PhpSpreadsheet\Calculation\Calculation::getFALSE()) {
175 $cell->setValueExplicit(false, PhpSpreadsheet\Cell\DataType::TYPE_BOOL);
176 //$cell->setValueExplicit('=FALSE()', PhpSpreadsheet\Cell\DataType::TYPE_FORMULA);
177 return true;
178 }
179
180 // Check for number in scientific format
181 if (preg_match('/^'.PhpSpreadsheet\Calculation\Calculation::CALCULATION_REGEXP_NUMBER.'$/', $value)) {
182 $cell->setValueExplicit((float) $value, PhpSpreadsheet\Cell\DataType::TYPE_NUMERIC);
183 return true;
184 }
185
186 // Check for fraction
187 if (false) {
188 //Nope, phone-number look like fractions but are not
189 if (preg_match('/^([+-]?) *([0-9]*)\s?\/\s*([0-9]*)$/', $value, $matches)) {
190 // Convert value to number
191 $value = $matches[2] / $matches[3];
192 if ($matches[1] == '-') {
193 $value = 0 - $value;
194 }
195 $cell->setValueExplicit((float) $value, PhpSpreadsheet\Cell\DataType::TYPE_NUMERIC);
196 // Set style
197 $cell->getWorksheet()->getStyle($cell->getCoordinate())
198 ->getNumberFormat()->setFormatCode('??/??');
199 return true;
200 } elseif (preg_match('/^([+-]?)([0-9]*) +([0-9]*)\s?\/\s*([0-9]*)$/', $value, $matches)) {
201 // Convert value to number
202 $value = $matches[2] + ($matches[3] / $matches[4]);
203 if ($matches[1] == '-') {
204 $value = 0 - $value;
205 }
206 $cell->setValueExplicit((float) $value, PhpSpreadsheet\Cell\DataType::TYPE_NUMERIC);
207 // Set style
208 $cell->getWorksheet()->getStyle($cell->getCoordinate())
209 ->getNumberFormat()->setFormatCode('# ??/??');
210 return true;
211 }
212 }
213
214 // Check for percentage
215 if (preg_match('/^\-?[0-9]*\.?[0-9]*\s?\%$/', $value)) {
216 // Convert value to number
217 $value = (float) str_replace('%', '', $value) / 100;
218 $cell->setValueExplicit($value, PhpSpreadsheet\Cell\DataType::TYPE_NUMERIC);
219 // Set style
220 $cell->getWorksheet()->getStyle($cell->getCoordinate())
221 ->getNumberFormat()->setFormatCode(PhpSpreadsheet\Style\NumberFormat::FORMAT_PERCENTAGE_00);
222 return true;
223 }
224
225 // Check for time without seconds e.g. '9:45', '09:45'
226 if (preg_match('/^(\d|[0-1]\d|2[0-3]):[0-5]\d$/', $value)) {
227 // Convert value to number
228 list($hours, $minutes) = Util::explode(':', $value);
229 $days = $hours / 24 + $minutes / 1440;
230 $cell->setValueExplicit($days, PhpSpreadsheet\Cell\DataType::TYPE_NUMERIC);
231 // Set style
232 $cell->getWorksheet()->getStyle($cell->getCoordinate())
233 ->getNumberFormat()->setFormatCode(PhpSpreadsheet\Style\NumberFormat::FORMAT_DATE_TIME3);
234 return true;
235 }
236
237 // Check for time with seconds '9:45:59', '09:45:59'
238 if (preg_match('/^(\d|[0-1]\d|2[0-3]):[0-5]\d:[0-5]\d$/', $value)) {
239 // Convert value to number
240 list($hours, $minutes, $seconds) = Util::explode(':', $value);
241 $days = $hours / 24 + $minutes / 1440 + $seconds / 86400;
242 // Convert value to number
243 $cell->setValueExplicit($days, PhpSpreadsheet\Cell\DataType::TYPE_NUMERIC);
244 // Set style
245 $cell->getWorksheet()->getStyle($cell->getCoordinate())
246 ->getNumberFormat()->setFormatCode(PhpSpreadsheet\Style\NumberFormat::FORMAT_DATE_TIME4);
247 return true;
248 }
249
250 // Check for datetime, e.g. '2008-12-31', '2008-12-31 15:59', '2008-12-31 15:59:10'
251 $date = PhpSpreadsheet\Shared\Date::stringToExcel($value);
252 if ($date !== false) {
253 // Convert value to number
254 $cell->setValueExplicit($date, PhpSpreadsheet\Cell\DataType::TYPE_NUMERIC);
255 // Determine style. Either there is a time part or not. Look for ':'
256 if (strpos($value, ':') !== false) {
257 $formatCode = 'dd.mm.yyyy h:mm:ss';
258 } else {
259 $formatCode = 'dd.mm.yyyy';
260 }
261 $cell->getWorksheet()->getStyle($cell->getCoordinate())
262 ->getNumberFormat()->setFormatCode($formatCode);
263 return true;
264 }
265
266 // Check for newline character "\n"
267 if (strpos($value, "\n") !== false) {
268 $cell->setValueExplicit($value, PhpSpreadsheet\Cell\DataType::TYPE_STRING);
269 // Set style
270 $cell->getWorksheet()->getStyle($cell->getCoordinate())
271 ->getAlignment()->setWrapText(true);
272 $cell->getWorksheet()->getStyle($cell->getCoordinate())
273 ->getAlignment()->setHorizontal(PhpSpreadsheet\Style\Alignment::HORIZONTAL_JUSTIFY);
274 return true;
275 }
276 }
277
278 // Not bound yet? Use parent...
279 return parent::bindValue($cell, $value);
280 }
