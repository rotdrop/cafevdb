284 public function populateFontsFolder():array
285 {
286 $fontsFolder = $this->getFontsFolder();
287 $fontsFolderLocalPath = $this->getFontsFolderName();
288
289 $fonts = [];
290 foreach (self::FONT_FILE_NAMES as $family => $variants) {
291 $fonts[$family] = [ 'family' => $family ];
292 foreach ($variants as $variant => $fontFileName) {
293 $fontFilePath = $this->findOfficeFontFile($fontFileName);
294 if (empty($fontFilePath)) {
295 $this->logInfo('Unable to find ' . $fontFileName);
296 $fontAliasName = self::FONT_ALIAS_NAMES[$family][$variant] ?? null;
297 if (!empty($fontAliasName)) {
298 $this->logInfo('Substituting ' . $fontAliasName . ' for ' . $fontFileName);
299 $fontFilePath = $this->findOfficeFontFile($fontAliasName);
300 if (empty($fontFilePath)) {
301 $this->logInfo('Unable to find ' . $fontAliasName);
302 }
303 }
304 }
305 if (empty($fontFilePath)) {
306 $fonts[$family][$variant] = false;
307 continue;
308 }
309 $fonts[$family][$variant] = $fontFileName;
310 $this->logInfo('Found ' . $fontFilePath);
311 if (str_starts_with($fontFilePath, $fontsFolderLocalPath)) {
312 // already installed, just skip
313 continue;
314 }
315 $fontData = file_get_contents($fontFilePath);
316 if ($fontsFolder->nodeExists($fontFileName)) {
317 $fontsFolder->get($fontFileName)->putContent($fontData);
318 } else {
319 $fontsFolder->newFile($fontFileName, $fontData);
320 }
321 }
322 }
323
324 return $fonts;

335 public function findOfficeFontFile(string $fontFileName, bool $useLocate = false):?string
336 {
337 $this->getFontsFolder();
338 $fontsFolderLocalName = $this->getFontsFolderName();
339 $cacheRealPath = $fontsFolderLocalName . Constants::PATH_SEP . $fontFileName;
340 if (is_file($cacheRealPath)) {
341 return $cacheRealPath;
342 }
343 if (empty($this->systemFontDataCache)) {
344 $fontDataDir = realpath(self::SYSTEM_FONT_DATA_DIR);
345 if (!empty($fontDataDir) && file_exists($fontDataDir) && is_dir($fontDataDir)) {
346 $this->logInfo('SYSTEM FONT DIR ' . $fontDataDir);
347 $iterator = new RegexIterator(
348 new RecursiveIteratorIterator(
349 new RecursiveDirectoryIterator($fontDataDir, FilesystemIterator::SKIP_DOTS),
350 RecursiveIteratorIterator::LEAVES_ONLY
351 ),
352 '/^.*\\.ttf$/i',
353 RegexIterator::GET_MATCH);
354 foreach ($iterator as $key => $info) {
355 // $this->logInfo('INFO ' . print_r($info, true));
356 $pathName = $key;
357 $this->systemFontDataCache[basename($pathName)] = $pathName;
358 }
359 $info = null;
360 }
361 $this->logInfo('FONT CACHE ' . print_r($this->systemFontDataCache, true));
362 }
363 if (!empty($this->systemFontDataCache[$fontFileName])) {
364 return $this->systemFontDataCache[$fontFileName];
365 }
366
367 if ($useLocate) {
368 // try locate if not found in standard path
369 $outputLines = [];
370 $resultCode = -1;
371 try {
372 exec('locate ' . $fontFileName, $outputLines, $resultCode);
373 foreach ($outputLines as $fontPathCandidate) {
374 if (file_exists($fontPathCandidate)) {
375 $this->systemFontDataCache[$fontFileName] = $fontPathCandidate;
376 return $fontPathCandidate;
377 }
378 }
379 } catch (Throwable $t) {
380 $this->logException($t, 'Locating "' . $fontFileName . '" caught an exception.');
381 }
382 }
383
384 unset($this->systemFontDataCache[$fontFileName]);
385
386 return null;
387 }
