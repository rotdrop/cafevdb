271 public function populateFontsFolder():array
272 {
273 $fontsFolder = $this->getFontsFolder();
274 $fontsFolderLocalPath = $this->getFontsFolderName();
275
276 $fonts = [];
277 foreach (self::FONT_FILE_NAMES as $family => $variants) {
278 $fonts[$family] = [ 'family' => $family ];
279 foreach ($variants as $variant => $fontFileName) {
280 $fontFilePath = $this->findOfficeFontFile($fontFileName);
281 if (empty($fontFilePath)) {
282 $this->logInfo('Unable to find ' . $fontFileName);
283 $fontAliasName = self::FONT_ALIAS_NAMES[$family][$variant] ?? null;
284 if (!empty($fontAliasName)) {
285 $this->logInfo('Substituting ' . $fontAliasName . ' for ' . $fontFileName);
286 $fontFilePath = $this->findOfficeFontFile($fontAliasName);
287 if (empty($fontFilePath)) {
288 $this->logInfo('Unable to find ' . $fontAliasName);
289 }
290 }
291 }
292 if (empty($fontFilePath)) {
293 $fonts[$family][$variant] = false;
294 continue;
295 }
296 $fonts[$family][$variant] = $fontFileName;
297 $this->logInfo('Found ' . $fontFilePath);
298 if (str_starts_with($fontFilePath, $fontsFolderLocalPath)) {
299 // already installed, just skip
300 continue;
301 }
302 $fontData = file_get_contents($fontFilePath);
303 if ($fontsFolder->nodeExists($fontFileName)) {
304 $fontsFolder->get($fontFileName)->putContent($fontData);
305 } else {
306 $fontsFolder->newFile($fontFileName, $fontData);
307 }
308 }
309 }
310
311 return $fonts;

322 public function findOfficeFontFile(string $fontFileName, bool $useLocate = false):?string
323 {
324 $this->getFontsFolder();
325 $fontsFolderLocalName = $this->getFontsFolderName();
326 $cacheRealPath = $fontsFolderLocalName . Constants::PATH_SEP . $fontFileName;
327 if (is_file($cacheRealPath)) {
328 return $cacheRealPath;
329 }
330 if (empty($this->systemFontDataCache)) {
331 $fontDataDir = realpath(self::SYSTEM_FONT_DATA_DIR);
332 if (!empty($fontDataDir) && file_exists($fontDataDir) && is_dir($fontDataDir)) {
333 $this->logInfo('SYSTEM FONT DIR ' . $fontDataDir);
334 $iterator = new RegexIterator(
335 new RecursiveIteratorIterator(
336 new RecursiveDirectoryIterator($fontDataDir, FilesystemIterator::SKIP_DOTS),
337 RecursiveIteratorIterator::LEAVES_ONLY
338 ),
339 '/^.*\\.ttf$/i',
340 RegexIterator::GET_MATCH);
341 foreach ($iterator as $key => $info) {
342 // $this->logInfo('INFO ' . print_r($info, true));
343 $pathName = $key;
344 $this->systemFontDataCache[basename($pathName)] = $pathName;
345 }
346 $info = null;
347 }
348 $this->logInfo('FONT CACHE ' . print_r($this->systemFontDataCache, true));
349 }
350 if (!empty($this->systemFontDataCache[$fontFileName])) {
351 return $this->systemFontDataCache[$fontFileName];
352 }
353
354 if ($useLocate) {
355 // try locate if not found in standard path
356 $outputLines = [];
357 $resultCode = -1;
358 try {
359 exec('locate ' . $fontFileName, $outputLines, $resultCode);
360 foreach ($outputLines as $fontPathCandidate) {
361 if (file_exists($fontPathCandidate)) {
362 $this->systemFontDataCache[$fontFileName] = $fontPathCandidate;
363 return $fontPathCandidate;
364 }
365 }
366 } catch (Throwable $t) {
367 $this->logException($t, 'Locating "' . $fontFileName . '" caught an exception.');
368 }
369 }
370
371 unset($this->systemFontDataCache[$fontFileName]);
372
373 return null;
374 }
