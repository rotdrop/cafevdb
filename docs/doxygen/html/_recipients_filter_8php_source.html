259 private function execute():void
260 {
261 // Maybe should also check something else. If submitted is true,
262 // then we use the form data, otherwise the defaults.
263 $this->submitted = $this->cgiValue('formStatus', '') === 'submitted';
264
265 // "sane" default setttings
266 $this->emailRecs = $this->parameterService->getParam($this->emailKey, []);
267 $this->emailTable = $this->parameterService->getParam($this->mtabKey, '');
268
269 $this->reload = false;
270 $this->snapshot = false;
271
272 if ($this->submitted) {
273 $this->loadHistory(); // Fetch the filter-history from the session, if any.
274 $this->emailRecs = $this->cgiValue($this->emailKey, []);
275 $this->emailTable = $this->cgiValue($this->mtabKey);
276 if ($this->cgiValue('resetInstrumentsFilter', false) !== false) {
277 $this->submitted = false; // fall back to defaults for everything
278 $this->cgiData = [];
279 $this->reload = true;
280 } elseif ($this->cgiValue('undoInstrumentsFilter', false) !== false) {
281 $this->applyHistory(1); // the current state
282 $this->reload = true;
283 } elseif ($this->cgiValue('redoInstrumentsFilter', false) !== false) {
284 $this->applyHistory(-1);
285 $this->reload = true;
286 } elseif ($this->cgiValue('historySnapshot', false) !== false) {
287 // fast mode, only CGI data, no DB access. Only the
288 // $this->filterHistory() should be queried afterwards,
289 // everything else is undefined.
290 $this->snapshot = true;
291 $this->pushHistory(true);
292 $this->storeHistory();
293 return;
294 }
295 }
296
297 $this->remapEmailRecords();
298 $this->getMemberStatusNames();
299 $this->initMemberStatusFilter();
300 $this->determineUserBase();
301 $this->getInstrumentsFromDB();
302 $this->fetchInstrumentsFilter();
303 $this->getMusiciansFromDB();
304
305 if (!$this->submitted) {
306 // Do this at end in order to have any tweaks around
307 $this->setDefaultHistory();
308 } elseif (!$this->reload) {
309 // add the current selection to the history if it is different
310 // from the previous filter selection (i.e.: no-ops like
311 // hitten apply over and over again or multiple double-clicks
312 // will not alter the history.
313 $this->pushHistory();
314 }
315
316 $this->storeHistory();
317 }

470 private function pushHistory(bool $fastMode = false):void
471 {
472 $filter = [];
473 foreach (self::HISTORY_KEYS as $key) {
474 $filter[$key] = $this->cgiValue($key, []);
475 }
476
477 if (!$fastMode) {
478 // exclude musicians deselected by the filter from the set of
479 // selected recipients before recording the history
480 $filter['selectedRecipients'] =
481 array_intersect(
482 $filter['selectedRecipients'],
483 array_keys($this->eMailsDpy)
484 );
485
486 // tweak: sort the selected recipients by key
487 }
488
489 sort($filter['selectedRecipients']);
490 $md5 = md5(serialize($filter));
491
492 /* Avoid pushing duplicate history entries. If the new
493 * filter-record matches the current one, then simply discard
494 * the new filter. This is in order to avoid bloating the
495 * history records by repeated user submits of the same filter
496 * or duplicated double-clicks.
497 */
498 $historyFilter = $this->filterHistory[$this->historyPosition];
499 if ($historyFilter['md5'] != $md5) {
500 // Pushing a new record removes the history up to the current
501 // position, i.e. redos are not longer possible then. This
502 // seems to be common behaviour as "re-doing" is no longer
503 // well defined in this case.
504 array_splice($this->filterHistory, 0, $this->historyPosition);
505 array_unshift($this->filterHistory, [ 'md5' => $md5, 'data' => $filter ]);
506 $this->historyPosition = 0;
507 $this->historySize = count($this->filterHistory);
508 while ($this->historySize > self::MAX_HISTORY_SIZE) {
509 array_pop($this->filterHistory);
510 --$this->historySize;
511 }
512 }
513 }

522 private function applyHistory(int $offset):void
523 {
524 $newPosition = $this->historyPosition + $offset;
525
526 // Check for valid position.
527 if ($newPosition >= $this->historySize || $newPosition < 0) {
528 if ($this->shouldDebug(ConfigService::DEBUG_EMAILFORM)) {
529 throw new OutOfBoundsException(
530 $this->l->t(
531 'Invalid history position %d request, history size is %d',
532 [ $newPosition, $this->historySize ])
533 );
534 }
535 return;
536 }
537
538 // Move to the new history position.
539 $this->historyPosition = $newPosition;
540
541 $filter = $this->filterHistory[$newPosition]['data'];
542 foreach (self::HISTORY_KEYS as $key) {
543 $this->cgiData[$key] = $filter[$key];
544 }

599 private function fetchMusicians(array $criteria):void
600 {
601 // add the instruments filter
602 if (!empty($this->instrumentsFilter)) {
603 $criteria[] = [ 'instruments.instrument' => $this->instrumentsFilter ];
604 if ($this->projectId > 0 && ($this->userBase & self::MUSICIANS_EXCEPT_PROJECT) == 0) {
605 $criteria[] = [ 'projectInstruments.instrument' => $this->instrumentsFilter ];
606 $criteria[] = [ 'projectInstruments.project' => $this->projectId ];
607 }
608 }
609 if ($this->frozen && $this->projectId > 0) {
610 $criteria[] = [ 'id' => $this->emailRecs ];
611 }
612 $criteria[] = [ '!memberStatus' => $this->memberStatusBlackList() ];
613
614 $musicians = $this->musiciansRepository->findBy($criteria, [ 'id' => 'INDEX' ]);
615
616 // use the mailer-class we are using later anyway in order to
617 // validate email addresses syntactically now. Add broken
618 // addresses to the "brokenEMails" list.
619 $mailer = new PHPMailer(true);
620
622 foreach ($musicians as $rec => $musician) {
623
624 $displayName = $musician->getPublicName(true);
625 if (!empty($musician->getEmail())) {
626 // We allow comma separated multiple addresses
627 $musMail = explode(',', $musician->getEmail());
628 foreach ($musMail as $emailVal) {
629 if (!$mailer->validateAddress($emailVal)) {
630 $bad = htmlspecialchars($displayName.' <'.$emailVal.'>');
631 if (isset($this->brokenEMails[$rec])) {
632 $this->brokenEMails[$rec]['label'] .= ', '.$bad;
633 } else {
634 $this->brokenEMails[$rec] = [
635 'participant' => $this->projectId > 0 && $musician->isMemberOf($this->projectId),
636 'label' => $bad,
637 ];
638
639 }
640 } else {
641 $isParticipant = $this->projectId > 0 && $musician->isMemberOf($this->projectId);
642 $isNonParticipant = $this->projectId > 0 && !$isParticipant;
643 $userBase = 0;
644 if ($isParticipant) {
645 $userBase |= self::MUSICIANS_FROM_PROJECT;
646 } elseif ($isNonParticipant) {
647 $userBase |= self::MUSICIANS_EXCEPT_PROJECT;
648 }
649 $emailRecord = [
650 'email' => $emailVal,
651 'name' => $displayName,
652 'status' => $musician['memberStatus'],
653 'project' => $this->projectId ?? 0,
654 'participant' => $isParticipant,
655 'userBase' => $userBase,
656 'dbdata' => $musician,
657 ];
658 $this->eMails[$rec] = $emailRecord;
659 $this->eMailsDpy[$rec] = htmlspecialchars($displayName.' <'.$emailVal.'>');
660 }
661 }
662 } else {
663 $this->brokenEMails[$rec] = [
664 'participant' => $this->projectId > 0 && $musician->isMemberOf($this->projectId),
665 'label' => htmlspecialchars($displayName),
666 ];
667 }
668 }

676 private function getMusiciansFromDB():void
677 {
678 $this->brokenEMails = [];
679 $this->eMails = [];
680 $this->eMailsDpy = []; // display records
681
682 $userBase = $this->userBase & self::DATABASE_MUSICIANS;
683 $criteria = [];
684 if (!empty($this->project) && $userBase != self::ALL_MUSICIANS) {
685
686 // OK, perhaps one should switch to sub-queries ...
687 //
688 // The problem is here that we need HAVING clauses in order to filter by
689 // the project id. WHERE and HAVING are implicitly joined by an AND, so
690 // we cannot first restrict to confirmed participation in the current
691 // project by WHERE and later allow non-project-musicians by HAVING.
692
693 if (($userBase & self::MUSICIANS_EXCEPT_PROJECT) && ($userBase & self::MUSICIANS_FROM_PROJECT)) {
694 $criteria[] = [
695 "(|projectParticipation.project@GROUP_CONCAT(IF(IDENTITY(%s) = {$this->projectId}, 1, NULL))" => null
696 // "(|projectParticipation.project@GROUP_CONCAT(CASE WHEN IDENTITY(%s) = {$this->projectId} THEN 1 ELSE NULL END))" => null
697 ];
698 if ($userBase & self::MUSICIANS_FROM_PROJECT_CONFIRMED) {
699 $criteria[] = [
700 "projectParticipation.project@GROUP_CONCAT(IF(IDENTITY(%s) = {$this->projectId}, projectParticipation.registration, NULL))" => 1
701 ];
702 } else {
703 $criteria[] = [
704 "projectParticipation.project@GROUP_CONCAT(IF(IDENTITY(%s) = {$this->projectId}, NULLIF(projectParticipation.registration, 0), NULL))" => null
705 ];
706 }
707 $criteria[] = [ ')@' => true ];
708 } elseif ($userBase & self::MUSICIANS_EXCEPT_PROJECT) {
709 $criteria[] = [
710 "projectParticipation.project@GROUP_CONCAT(IF(IDENTITY(%s) = {$this->projectId}, 1, NULL))" => null
711 ];
712 } elseif ($userBase & self::MUSICIANS_FROM_PROJECT) {
713 $criteria[] = [ '(&projectParticipation.project' => $this->projectId ];
714 switch ($userBase & self::MUSICIANS_FROM_PROJECT) {
715 case self::MUSICIANS_FROM_PROJECT:
716 break; // just all participants
717 case self::MUSICIANS_FROM_PROJECT_CONFIRMED:
718 $criteria[] = [ 'projectParticipation.registration' => 1 ];
719 break;
720 case self::MUSICIANS_FROM_PROJECT_PRELIMINARY:
721 $criteria[] = [ '(|projectParticipation.registration' => 0 ];
722 $criteria[] = [ 'projectParticipation.registration' => null ];
723 $criteria[] = [ ')' => true ];
724 break;
725 }
726 $criteria[] = [ ')' => true ];
727 }
728 if (empty($criteria)) {
729 return;
730 }
731 }
732 $this->fetchMusicians($criteria);
733
734 // Finally sort the display array
735 asort($this->eMailsDpy);
736 }

1355 public function substituteProjectMailingList(array &$selectedRecipients):bool
1356 {
1357 if (empty($this->project)) {
1358 // this is not a project email
1359 $this->logInfo('NO PROJECT');
1360 return false;
1361 }
1362
1363 if (!($this->userBase & self::MUSICIANS_FROM_PROJECT_CONFIRMED)) {
1364 // intentionally not for the list which should contain all confirmed
1365 // participants + further people.
1366 $this->logInfo('NOT FOR THE LIST');
1367 return false;
1368 }
1369
1370 $defaultByStatus = $this->defaultByStatus();
1371 $memberFilter = $this->memberFilter ?? [];
1372 if (array_intersect($memberFilter, $defaultByStatus) !== $defaultByStatus) {
1373 // intentionally not for the list as less than the default status type
1374 // were addressed
1375 $this->logInfo('MEMBER FILTER INCOMPATIBLE');
1376 return false;
1377 }
1378
1379 if (!empty($this->instrumentsFilter)) {
1380 // intentionally not for the list, which is also archived
1381 $this->logInfo('INSTRUMENTS FILTER NOT EMPTY');
1382 return false;
1383 }
1384
1385 $numberOfPossibleRecipients = count($this->emailRecipientsChoices());
1386 $numberOfSelectedRecipients = count($selectedRecipients);
1387 if ($numberOfPossibleRecipients != $numberOfSelectedRecipients) {
1388 // although the recipient options intentionally include the list members
1389 // there is a particular choice of selected recipients, so this should
1390 // not go to the list.
1391 $this->logInfo('SPECIFIC CHOICE OF RECIPIENTS');
1392 return false;
1393 }
1394
1395 // keep this check last as it implies a REST call which is time-consuming.
1396 $listInfo = $this->getMailingListInfo();
1397 if (empty($listInfo)) {
1398 // well, there is not list ...
1399 return false;
1400 }
1401
1402 $listId = $listInfo['list_id'];
1404 $listsService = $this->di(MailingListsService::class);
1405 $listMembers = $listsService->findMembers($listId, flat: true, criteria: [
1406 MailingListsService::MEMBER_DELIVERY_STATUS => MailingListsService::DELIVERY_STATUS_ENABLED,
1407 MailingListsService::MEMBER_DELIVERY_MODE => MailingListsService::DELIVERY_MODE_REGULAR,
1408 ]);
1409 $bulkEmailFromAddress = $this->getConfigValue('emailfromaddress');
1410 Util::unsetValue($listMembers, $bulkEmailFromAddress);
1411 $recipientsByEmail = [];
1412 foreach ($selectedRecipients as $recipient) {
1413 $recipientsByEmail[$recipient['email']] = $recipient;
1414 }
1415 $remainingRecipients = array_diff_key($recipientsByEmail, array_flip($listMembers));
1416 if (!empty($remainingRecipients)) {
1417 $this->logInfo('NOT ON THE LIST ' . print_r(array_keys($remainingRecipients), true));
1418 }
1419
1420 $remainingListMembers = array_diff($listMembers, array_keys($recipientsByEmail));
1421 if (!empty($remainingListMembers)) {
1422 $this->logInfo('EXCESS MEMBERS ' . print_r($remainingListMembers, true));
1423 // It is ok if there are more members on the list, as intentionally this
1424 // should be mailing list traffic. return false;
1425 }
1426
1427 // throw away all recipients which are also reached by posting to the list
1428 // and add the list address as additional recipient.
1429 $selectedRecipients = array_merge(
1430 $this->selectedRecipients(self::PROJECT_MAILING_LIST),
1431 array_values($remainingRecipients),
1432 );
1433 return true;
1434 }
