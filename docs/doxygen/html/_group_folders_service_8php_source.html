162 public function searchFolders(string $regexp, string $topic = self::SEARCH_TOPIC_MOUNT):array
163 {
164 if ($this->sharedFolders === null) {
165 $this->fetchFolders();
166 }
167 if ($regexp[0] != $regexp[-1]) {
168 $regexp = '|^' . $regexp . '$|';
169 }
170 $this->logDebug('REGEXP ' . $regexp . ' / ' . $topic);
171 switch ($topic) {
172 case self::SEARCH_TOPIC_MOUNT:
173 return array_filter($this->sharedFolders, function($folderInfo) use ($regexp) {
174 return preg_match($regexp, $folderInfo['mount_point']);
175 });
176 case self::SEARCH_TOPIC_GROUP:
177 return array_filter($this->sharedFolders, function($folderInfo) use ($regexp) {
178 foreach (array_keys($folderInfo['groups']) as $group) {
179 if (preg_match($regexp, $group)) {
180 return true;
181 }
182 return false;
183 }
184 });
185 default:
186 throw new RuntimeException($this->l->t('Unknown search topic "%1$s"', $topic));
187 }
188 }

342 private function changeFolderManager(string $mountPoint, string $managerId, string $type, bool $canManage):void
343 {
344 // POST BASEURL/groupfolders/folders/8/acl
345 // [ acl => 0 / 1 ]
346 //
347 // POST BASEURL/groupfolders/folders/4/manageACL
348 // [ mappingType => MANAGER_TYPE, mappingId => MANAGER_ID, manageAcl=> 1 ]
349 //
350 $folderInfo = $this->getFolder($mountPoint);
351 if (empty($folderInfo)) {
352 throw new RuntimeException($this->l->t('Shared folder for mount-point "%1$s" does not exist, cannot modify manager "%2$s".', [ $mountPoint, $managerId ]));
353 }
354
355 $this->logDebug('FOLDER INFO ' . print_r($folderInfo, true));
356
357 // first check if anything needs to be done
358 $aclEnabled = !!($folderInfo['acl']??false);
359 $isManager = 0 < count(array_filter($folderInfo['manage']??[], function($manager) use ($managerId, $type) {
360 return $manager['type'] == $type && $manager['id'] == $managerId;
361 }));
362
363 if ($isManager == $canManage && ($canManage || $isManager == $canManage)) {
364 return;
365 }
366
367 $folderId = $folderInfo['id'];
368 if (!$aclEnabled) {
369 $route = implode('.', [ self::GROUP_FOLDERS_APP, 'Folder', 'setACL']);
370 $this->requestService->postToRoute($route, [ 'id' => $folderId ], [ 'acl' => 1 ]);
371 }
372
373 if ($isManager != $canManage) {
374 $route = implode('.', [ self::GROUP_FOLDERS_APP, 'Folder', 'setManageACL']);
375 $this->requestService->postToRoute($route, [ 'id' => $folderId ], [
376 'mappingType' => $type,
377 'mappingId' => $managerId,
378 'manageAcl' => (int)$canManage,
379 ]);
380 }
381 if (!$aclEnabled || $isManager != $canManage) {
382 $this->getFolderById($folderId);
383 }
384 }

461 public function changeMountPoint(string $mountPoint, string $targetMountPoint, bool $moveChildren = true):void
462 {
463 if ($mountPoint == $targetMountPoint) {
464 return;
465 }
466 $folderInfo = $this->getFolder($mountPoint);
467 if (empty($folderInfo)) {
468 throw new RuntimeException($this->l->t('Shared folder for mount-point "%1$s" does not exist, cannot change its mount-point to "%2$s".', [ $mountPoint, ]));
469 }
470 // POST BASE_URL/groupfolders/folders/FOLDER_ID/mountpoint
471 // DATA [ mountpoint => NEW_MOUNT_POINT ]
472
473 $route = implode('.', [ self::GROUP_FOLDERS_APP, 'Folder', 'renameFolder']);
474 $this->requestService->postToRoute($route, [
475 'id' => $folderInfo['id'],
476 ], [
477 'mountpoint' => $targetMountPoint,
478 ]);
479 $this->sharedFolders[$targetMountPoint] = $this->sharedFolders[$mountPoint];
480 unset($this->sharedFolders[$mountPoint]);
481 $this->sharedFolders[$targetMountPoint]['mount_point'] = $targetMountPoint;
482
483 if ($moveChildren) {
484 $children = $this->searchFolders('|^' . $mountPoint . '/.*$|');
485 foreach ($children as $childInfo) {
486 $oldChildMount = $childInfo['mount_point'];
487 $newChildMount = $targetMountPoint . substr($oldChildMount, strlen($mountPoint));
488 try {
489 $this->changeMountPoint($oldChildMount, $newChildMount, moveChildren: false);
490 } catch (\Throwable $t) {
491 $this->logException($t, 'Failed to move child mount from ' . $oldChildMount . ' to ' . $newChildMount);
492 }
493 }
494 }
