30/**
31 * Add a to-back-button to the titlebar of a jQuery-UI dialog. The
32 * purpose is to be able to move the top-dialog to be bottom-most,
33 * juse above a potential "modal" window layer.
34 *
35 * @param {jQuery} dialogHolder TBD.
36 */
37function dialogToBackButton(dialogHolder) {
38 const dialogWidget = dialogHolder.dialog('widget');
39 const toBackButtonTitle = t(
40 appName,
41 'If multiple dialogs are open, '
42 + 'then move this one to the lowest layer '
43 + 'and display it below the others. '
44 + 'Clicking anywhere on the dialog will bring to the front again.');
45 const toBackButton = $('<button class="toBackButton customDialogHeaderButton" title="' + toBackButtonTitle + '"></button>');
46 toBackButton.button({
47 label: '_',
48 icons: { primary: 'ui-icon-minusthick', secondary: null },
49 text: false,
50 });
51 dialogWidget.find('.ui-dialog-titlebar').append(toBackButton);
52 toBackButton.cafevTooltip({ placement: 'auto' });
53
54 toBackButton.off('click');
55 toBackButton.on('click', function() {
56 const overlay = $('.ui-widget-overlay.ui-front').last();
57 let overlayIndex = 100; // OwnCloud header resides at 50.
58 if (overlay.length > 0) {
59 overlayIndex = parseInt(overlay.css('z-index'));
60 }
61 // will be only few, so what
62 let needShuffle = false;
63 $('.ui-dialog.ui-widget').not('.cafevdb-modalizer').each(function(index) {
64 const thisIndex = parseInt($(this).css('z-index'));
65 if (thisIndex === overlayIndex + 1) {
66 needShuffle = true;
67 }
68 }).each(function(index) {
69 if (needShuffle) {
70 const thisIndex = parseInt($(this).css('z-index'));
71 $(this).css('z-index', thisIndex + 1);
72 }
73 });
74 dialogWidget.css('z-index', overlayIndex + 1);
75 return false;
76 });
77}

79/**
80 * Add a full-screen-button to the titlebar of a jQuery-UI dialog.
81 *
82 * @param {jQuery} dialogHolder TBD.
83 *
84 * @param {Function} onChangeCallback TBD.
85 */
86function dialogFullScreenButton(dialogHolder, onChangeCallback) {
87 const dialogWidget = dialogHolder.dialog('widget');
88 const parent = dialogWidget.parent();
89 const fullScreenClass = appName + '-full-screen';
90 const buttonTitle = t(
91 appName,
92 'Maximize this dialog to cover the entire browser window.');
93 const button = $('<button class="fullScreenButton customDialogHeaderButton" title="' + buttonTitle + '"></button>');
94 button.button({
95 label: '_',
96 icons: { primary: 'ui-icon-arrow-4-diag', secondary: null },
97 text: false,
98 });
99 dialogWidget.find('.ui-dialog-titlebar').append(button);
100 button.cafevTooltip({ placement: 'auto' });
101
102 button
103 .off('click')
104 .on('click', function() {
105 $.fn.cafevTooltip.remove(); // remove any left-over items
106 if (onChangeCallback !== undefined) {
107 onChangeCallback(dialogWidget.hasClass(fullScreenClass) ? 'fullscreen' : 'dialog', 'before');
108 }
109 if (dialogWidget.hasClass(fullScreenClass)) {
110 dialogWidget.removeClass(fullScreenClass);
111 if (!parent.is('body')) {
112 dialogWidget.detach().prependTo(parent);
113 }
114 } else {
115 dialogWidget.addClass(fullScreenClass);
116 if (!parent.is('body')) {
117 dialogWidget.detach().prependTo('body');
118 }
119 }
120 if (onChangeCallback !== undefined) {
121 onChangeCallback(dialogWidget.hasClass(fullScreenClass) ? 'fullscreen' : 'dialog', 'after');
122 }
123 });
124}

126/**
127 * jQuery UI just is not flexible enough. We want to be able to
128 * completely intercept the things the close button initiates. I
129 * just did not find any other way than to hide the close button
130 * completely and add another button with the same layout instead,
131 * but this time with complete control over the events triggered by
132 * this button.
133 *
134 * If callback is undefined, then simply call the close
135 * method. Otherwise it is called like callback(event, dialogHolder).
136 *
137 * @param {jQuery} dialogHolder TBD.
138 *
139 * @param {Function} callback TBD.
140 *
141 */
142function dialogCustomCloseButton(dialogHolder, callback) {
143 const dialogWidget = dialogHolder.dialog('widget');
144 const customCloseButtonTitle = t(
145 appName,
146 'Close the current dialog and return to the view '
147 + 'which was active before this dialog had been opened. '
148 + 'If the current view shows a `Back\' button, then intentionally '
149 + 'clicking the close-button (THIS button) should just be '
150 + 'equivalent to clicking the `Back\' button');
151 const customCloseButton = $('<button class="customCloseButton customDialogHeaderButton" title="' + customCloseButtonTitle + '"></button>');
152 customCloseButton.button({
153 label: 'x',
154 icons: { primary: 'ui-icon-closethick', secondary: null },
155 text: false,
156 });
157 dialogWidget.find('.ui-dialog-titlebar').append(customCloseButton);
158 customCloseButton.cafevTooltip({ placement: 'auto' });
159
160 customCloseButton
161 .off('click')
162 .on('click', function(event) {
163 if (typeof callback === 'function') {
164 callback(event, dialogHolder);
165 } else {
166 dialogHolder.dialog('close');
167 }
168 return false;
169 });
170}
