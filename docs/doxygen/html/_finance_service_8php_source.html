130 public function targetDeadline(
131 int $businessOffset,
132 ?int $calendarOffset = null,
133 ?\DateTimeInterface $fromDate = null,
134 ):DateTimeInterface {
136 if (empty($fromDate)) {
137 $fromDate = new CarbonImmutable('now', $this->getDateTimeZone());
138 } else {
139 $fromDate = (new CarbonImmutable)
140 ->setTimezone($fromDate->getTimezone())
141 ->setTimestamp($fromDate->getTimestamp());
142 }
143
144 if (!empty($calendarOffset) && $businessOffset*$calendarOffset < 0) {
145 throw new RuntimeException(
146 $this->l->t('The business-day and calendar-day offset must have the same sign (%d / %d)', [
147 $businessOffset, $calendarOffset
148 ]));
149 }
150
151 CarbonImmutable::setHolidaysRegion('target2');
152 $businessDeadline = $fromDate->addBusinessDays($businessOffset);
153 if (!empty($calendarOffset)) {
154 $calendarDeadline = $fromDate->addDays($calendarOffset);
155
156 $businessInterval = $fromDate->diff($businessDeadline, true);
157 $calendarInterval = $fromDate->diff($calendarDeadline, true);
158
159 if ($calendarInterval->days > $businessInterval->days) {
160 $businessDeadline = $calendarDeadline->nextBusinessDay();
161 }
162 }
163
164 return $businessDeadline;

734 public function generateSepaMandateReference(Entities\SepaDebitMandate $mandate):string
735 {
736 $project = $this->ensureProject($mandate->getProject());
737 if (empty($project)) {
738 throw new InvalidArgumentException($this->l->t('The given mandate does not contain a valid project-id.'));
739 }
740 $mandate->setProject($project);
741
742 $projectId = $project['id'];
743 $projectName = $this->sepaTranslit($project['name']);
744
745 $projectYear = substr($projectName, -4);
746 if (is_numeric($projectYear)) {
747 $projectName = substr($projectName, 0, -4);
748 } else {
749 $projectYear = null;
750 }
751
752 $musician = $this->ensureMusician($mandate->getMusician());
753 if (empty($musician)) {
754 throw new InvalidArgumentException($this->l->t('The given mandate does not contain a valid musician-id.'));
755 }
756 $mandate->setMusician($musician);
757
758 $sequence = $mandate->getSequence();
759
760 $musicianId = $musician['id'];
761 $firstName = $this->sepaTranslit($musician['firstName']);
762 $surName = $this->sepaTranslit($musician['surName']);
763
764 $format = empty($projectYear)
765 ? '%04d-%04d-%\'X1.1s%\'X1.1s-%-\'X19.19s%.0s+%02d'
766 : '%04d-%04d-%\'X1.1s%\'X1.1s-%-\'X15.15s%04d+%02d';
767
768 $ref = sprintf(
769 $format,
770 $projectId, $musicianId,
771 $firstName, $surName,
772 $projectName, $projectYear,
773 (int)$sequence);
774
775 $ref = strtoupper(Util::normalizeSpaces($ref, 'X'));
776
777 if (strlen($ref) != self::SEPA_MANDATE_LENGTH) {
778 throw new RuntimeException(
779 $this->l->t(
780 'SEPA mandate-reference "%s" is too long (%d > %d).',
781 [ $ref, strlen($ref), self::SEPA_MANDATE_LENGTH]));
782 }
783
784 return $ref;
785 }

850 public function storeSepaMandate(mixed $newMandate):?Entities\SepaDebitMandate
851 {

853 !isset($newMandate['mandateReference']) ||
854 !isset($newMandate['musicianId']) ||
855 !isset($newMandate['projectId'])) {
856 return null;
857 }
858
859 if (isset($newMandate['sequenceType'])) {
860 $sequenceType = $newMandate['sequenceType'];

876 if ($stamp != strtotime($value)) {

881 unset($newMandate[$date]);

899 switch ($key) {

901 $targetKey = 'musician';
902 break;
903 case 'projectId':
904 $targetKey = 'project';
905 break;
906 default:
907 $targetKey = $key;
908 break;
909 }

911 unset($newMandate[$key]);
912 $value = null;
913 }
914 $mandate[$targetKey] = $value; // @todo check date and time-stamps.
915 }
916 } else {
917 $mandate = Entities\SepaDebitMandate::create();
918 foreach ($newMandate as $key => $value) {
919 switch ($key) {
920 case 'musicianId':
921 $targetKey = 'musician';
922 break;
923 case 'projectId':
924 $targetKey = 'project';
925 break;
926 default:
927 $targetKey = $key;
928 break;
929 }
930 $mandate[$targetKey] = $value; // @todo check date and time-stamps.
931 }
932 $this->persist($mandate);
933 }
934 $this->flush($mandate); // persist
935
936 return $mandate;

1155 * 32-Bit-CPUs koennen mit PHP aber nur maximal 9 Stellen mit allen Ziffern genutzt werden.

1160 *
1161 * @return bool

1163 public function testIBAN(string $iban):bool
1164 {

1166 $iban1 = substr($iban, 4)
1167 . strval(ord($iban[0]) - 55)
1168 . strval(ord($iban[1]) - 55)
1169 . substr($iban, 2, 2);
1170
1171 for ($i = 0; $i < strlen($iban1); $i++) {
1172 if (ord($iban1[$i]) > 64 && ord($iban1[$i]) < 91) {
1173 $iban1 = substr($iban1, 0, $i) . strval(ord($iban1[$i]) - 55) . substr($iban1, $i+1);
1174 }
1175 }
1176 $rest = 0;
1177 for ($pos = 0; $pos < strlen($iban1); $pos += 7) {
1178 $part = strval($rest) . substr($iban1, $pos, 7);
1179 $rest = intval($part) % 97;
1180 }
1181 $checkSum = sprintf("%02d", 98 - $rest);
1182
1183 if (substr($iban, 2, 2)=='00') {
1184 return substr_replace($iban, $checkSum, 2, 2);

1186 return ($rest == 1) ? true : false;
