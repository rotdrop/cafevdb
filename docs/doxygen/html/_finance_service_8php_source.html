140 public function targetDeadline(
141 int $businessOffset,
142 ?int $calendarOffset = null,
143 ?\DateTimeInterface $fromDate = null,
144 ):DateTimeInterface {
146 if (empty($fromDate)) {
147 $fromDate = new CarbonImmutable('now', $this->getDateTimeZone());
148 } else {
149 $fromDate = (new CarbonImmutable)
150 ->setTimezone($fromDate->getTimezone())
151 ->setTimestamp($fromDate->getTimestamp());
152 }
153
154 if (!empty($calendarOffset) && $businessOffset*$calendarOffset < 0) {
155 throw new RuntimeException(
156 $this->l->t('The business-day and calendar-day offset must have the same sign (%d / %d)', [
157 $businessOffset, $calendarOffset
158 ]));
159 }
160
161 CarbonImmutable::setHolidaysRegion('target2');
162 $businessDeadline = $fromDate->addBusinessDays($businessOffset);
163 if (!empty($calendarOffset)) {
164 $calendarDeadline = $fromDate->addDays($calendarOffset);
165
166 $businessInterval = $fromDate->diff($businessDeadline, true);
167 $calendarInterval = $fromDate->diff($calendarDeadline, true);
168
169 if ($calendarInterval->days > $businessInterval->days) {
170 $businessDeadline = $calendarDeadline->nextBusinessDay();
171 }
172 }
173
174 return $businessDeadline;

744 public function generateSepaMandateReference(Entities\SepaDebitMandate $mandate):string
745 {
746 $project = $this->ensureProject($mandate->getProject());
747 if (empty($project)) {
748 throw new InvalidArgumentException($this->l->t('The given mandate does not contain a valid project-id.'));
749 }
750 $mandate->setProject($project);
751
752 $projectId = $project['id'];
753 $projectName = $this->sepaTranslit($project['name']);
754
755 $projectYear = substr($projectName, -4);
756 if (is_numeric($projectYear)) {
757 $projectName = substr($projectName, 0, -4);
758 } else {
759 $projectYear = null;
760 }
761
762 $musician = $this->ensureMusician($mandate->getMusician());
763 if (empty($musician)) {
764 throw new InvalidArgumentException($this->l->t('The given mandate does not contain a valid musician-id.'));
765 }
766 $mandate->setMusician($musician);
767
768 $sequence = $mandate->getSequence();
769
770 $musicianId = $musician['id'];
771 $firstName = $this->sepaTranslit($musician['firstName']);
772 $surName = $this->sepaTranslit($musician['surName']);
773
774 $format = empty($projectYear)
775 ? '%04d-%04d-%\'X1.1s%\'X1.1s-%-\'X19.19s%.0s+%02d'
776 : '%04d-%04d-%\'X1.1s%\'X1.1s-%-\'X15.15s%04d+%02d';
777
778 $ref = sprintf(
779 $format,
780 $projectId, $musicianId,
781 $firstName, $surName,
782 $projectName, $projectYear,
783 (int)$sequence);
784
785 $ref = strtoupper(Util::normalizeSpaces($ref, 'X'));
786
787 if (strlen($ref) != self::SEPA_MANDATE_LENGTH) {
788 throw new RuntimeException(
789 $this->l->t(
790 'SEPA mandate-reference "%s" is too long (%d > %d).',
791 [ $ref, strlen($ref), self::SEPA_MANDATE_LENGTH]));
792 }
793
794 return $ref;
795 }

860 public function storeSepaMandate(mixed $newMandate):?Entities\SepaDebitMandate
861 {
862 if (!is_array($newMandate) ||
863 !isset($newMandate['mandateReference']) ||
864 !isset($newMandate['musicianId']) ||
865 !isset($newMandate['projectId'])) {
866 return null;
867 }
868
869 if (isset($newMandate['sequenceType'])) {
870 $sequenceType = $newMandate['sequenceType'];
871 $newMandate['nonRecurring'] = $sequenceType == 'once';

875 $ref = $newMandate['mandateReference'];

877 $prj = $newMandate['projectId'];
878
879 // Convert to a date format understood by mySQL.
880 // @todo use \DateTime objects.
881 $dateFields = [ 'lastUsedDate', 'mandateDate', ];
882 foreach ($dateFields as $date) {
883 if (!empty($newMandate[$date])) {
884 $stamp = strtotime($newMandate[$date]);
885 $value = date('Y-m-d', $stamp);
886 if ($stamp != strtotime($value)) {
887 return null;

889 $newMandate[$date] = $value;
890 } else {
891 unset($newMandate[$date]);
892 }
893 }
894
895 // fetch the old mandate, but keep the old values encrypted
896 $mandate = $this->fetchSepaMandate($prj, $mus, false);

900 !isset($mandate['musician']) ||

902 $mandate['mandateReference'] != $ref ||
903 $mandate['musician']['id'] != $mus ||
904 $mandate['project']['id'] != $prj) {
905 return null;
906 }
907 // passed: issue an update query
908 foreach ($newMandate as $key => $value) {
909 switch ($key) {
910 case 'musicianId':
911 $targetKey = 'musician';
912 break;
913 case 'projectId':
914 $targetKey = 'project';
915 break;
916 default:
917 $targetKey = $key;
918 break;
919 }
920 if (empty($value) && $key != 'lastUsedDate') {
921 unset($newMandate[$key]);
922 $value = null;
923 }
924 $mandate[$targetKey] = $value; // @todo check date and time-stamps.
925 }
926 } else {
927 $mandate = Entities\SepaDebitMandate::create();
928 foreach ($newMandate as $key => $value) {
929 switch ($key) {
930 case 'musicianId':
931 $targetKey = 'musician';
932 break;
933 case 'projectId':
934 $targetKey = 'project';
935 break;
936 default:
937 $targetKey = $key;
938 break;
939 }
940 $mandate[$targetKey] = $value; // @todo check date and time-stamps.
941 }
942 $this->persist($mandate);
943 }
944 $this->flush($mandate); // persist
945
946 return $mandate;
947 }
