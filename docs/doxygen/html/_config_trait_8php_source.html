638 protected function getDocumentTemplatesPath(string $templateName = null, bool $dirName = false)
639 {
640 $pathComponents = [''];
641 $sharedFolder = $this->getSharedFolderPath();
642 if (empty($sharedFolder)) {
643 return null;
644 }
645 $pathComponents[] = $sharedFolder;
646 $templatesFolder = $this->getConfigValue(ConfigService::DOCUMENT_TEMPLATES_FOLDER);
647 if (empty($templatesFolder)) {
648 return null;
649 }
650 $pathComponents[] = $templatesFolder;
651 if (!empty($templateName)) {
652 $subFolder = ConfigService::DOCUMENT_TEMPLATES[$templateName]['folder']??null;
653 if (!empty($subFolder)) {
654 $subFolder = $this->getConfigValue($subFolder);
655 if (!empty($subFolder)) {
656 $pathComponents[] = $subFolder;
657 }
658 }
659 $templateFileName = $this->getConfigValue($templateName);
660 if (empty($templateFileName)) {
661 return null;
662 }
663 $pathComponents[] = $dirName ? '' : $templateFileName ;
664 }
665 return implode(Constants::PATH_SEP, $pathComponents);
