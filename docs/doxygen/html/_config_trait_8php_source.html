637 {
638 $pathComponents = [''];
639 $sharedFolder = $this->getSharedFolderPath();
640 if (empty($sharedFolder)) {
641 return null;
642 }
643 $pathComponents[] = $sharedFolder;
644 $templatesFolder = $this->getConfigValue(ConfigService::DOCUMENT_TEMPLATES_FOLDER);
645 if (empty($templatesFolder)) {
646 return null;
647 }
648 $pathComponents[] = $templatesFolder;
649 if (!empty($templateName)) {
650 $subFolder = ConfigService::DOCUMENT_TEMPLATES[$templateName]['folder']??null;
651 if (!empty($subFolder)) {
652 $subFolder = $this->getConfigValue($subFolder);

656 }

1058 $variants = array_unique([
1059 $name,

1064 strtolower($camelCase),
1065 strtoupper($camelCase),
1066 $words,

1070 array_map('strtolower', $variants),
