224 public function shortTitle()
225 {
226 if ($this->deleteOperation()) {
227 return $this->l->t('Remove all data of the displayed musician?');
228 } elseif ($this->copyOperation()) {
229 return $this->l->t('Copy the displayed musician?');
230 } elseif ($this->viewOperation()) {
231 return $this->l->t('Display of all stored personal data for the shown musician.');
232 } elseif ($this->changeOperation()) {
233 return $this->l->t('Edit the personal data of the displayed musician.');
234 } elseif ($this->addOperation()) {
235 return $this->l->t('Add a new musician to the data-base.');
236 } elseif (!$this->projectMode) {
237 return $this->l->t('Overview over all registered musicians');
238 } else {
239 return $this->l->t("Add musicians to the project `%s'", [ $this->projectName ]);
240 }
