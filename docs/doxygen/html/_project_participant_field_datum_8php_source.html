425 public function amountPayable():float
426 {
427 $value = 0.0;
428 switch ($this->field->getMultiplicity()) {
429 case Multiplicity::SINGLE():
430 case Multiplicity::MULTIPLE():
431 case Multiplicity::PARALLEL():
432 case Multiplicity::GROUPSOFPEOPLE():
433 $value = filter_var($this->dataOption->getData(), FILTER_VALIDATE_FLOAT);
434 if ($value === false) {
435 throw new RuntimeException('Stored value cannot be converted to float.');
436 }
437 break;
438 case Multiplicity::GROUPOFPEOPLE():
439 // value in management option of $field
440 $managementOption = $this->field->getManagementOption();
441 if (empty($managementOption)) {
442 throw new RuntimeException('Unable to access management option for obtaining the field value.');
443 }
444 $value = filter_var($managementOption->getData(), FILTER_VALIDATE_FLOAT);
445 if ($value === false) {
446 throw new RuntimeException('Stored value cannot be converted to float.');
447 }
448 break;
449 case Multiplicity::SIMPLE():
450 case Multiplicity::RECURRING():
451 if (!empty($this->optionValue)) {
452 $value = filter_var($this->optionValue, FILTER_VALIDATE_FLOAT);
453 if ($value === false) {
454 throw new RuntimeException('Stored value cannot be converted to float: ' . (string)$this->optionValue);
455 }
456 }
457 break;
458 default:
459 throw new RuntimeException('Unhandled multiplicity tag: '.(string)$this->field->getMultiplicity());
460 }
461 if ($this->field->getDataType() == DataType::LIABILITIES) {
462 $value = -$value;
463 }
464 return $value;
465 }
