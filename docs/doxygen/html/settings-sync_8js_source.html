26/**
27 * @param {string} settingsSection AJAX call goes to `apps/${appName}/settings/${settingsSection}`.
28 *
29 * @param {object} storageObject The object which receives the
30 * settings keys as properties. If not given defaults to this.
31 *
32 * @return {boolean} Success status, false
33 on error, true on success.
34 */
35async function fetchSettings(settingsSection, storageObject) {
36 if (storageObject === undefined) {
37 storageObject = this;
38 }
39 try {
40 const response = await axios.get(generateUrl('apps/' + appName + '/settings/' + settingsSection), {});
41 // Object.assign(storageObject, response.data);
42 for (const [key, value] of Object.entries(response.data)) {
43 if (Object.prototype.hasOwnProperty.call(storageObject, key)) {
44 // eslint-disable-next-line import/no-named-as-default-member
45 Vue.set(storageObject, key, value);
46 } else {
47 storageObject[key] = value;
48 }
49 }
50 return true;
51 } catch (e) {
52 console.info('ERROR', e);
53 let message = t(appName, 'reason unknown');
54 if (e.response && e.response.data) {
55 const responseData = e.response.data;
56 if (Array.isArray(responseData.messages)) {
57 message = responseData.messages.join(' ');
58 }
59 }
60 showError(t(appName, 'Unable to query the initial value of all settings: {message}', {
61 message,
62 }));
63 return false;
64 }
65}

108/**
109 * @param {string} settingsKey TDB.
110 *
111 * @param {string} settingsSection TDB.
112 *
113 * @param {Function} onSuccess Success callback, invoked with the
114 * response data and the arguments of this function.
115 *
116 * @return {boolean} TBD.
117 */
118async function saveSimpleSetting(settingsKey, settingsSection, onSuccess) {
119 const value = this[settingsKey];
120 try {
121 const response = await axios.post(generateUrl('apps/' + appName + '/settings/' + settingsSection + '/' + settingsKey), { value });
122 const responseData = response.data;
123 let displayValue;
124 if (responseData) {
125 if (responseData.newValue !== undefined) {
126 this[settingsKey] = responseData.newValue;
127 displayValue = this[settingsKey];
128 }
129 if (responseData.humanValue !== undefined) {
130 const humanKey = 'human' + settingsKey[0].toUpperCase() + settingsKey.substring(1);
131 this[humanKey] = responseData.humanValue;
132 displayValue = this[humanKey];
133 }
134 if (Array.isArray(displayValue)) {
135 displayValue = displayValue.toString();
136 }
137 }
138 if (displayValue === true) {
139 displayValue = t(appName, 'true');
140 }
141 if (displayValue && displayValue !== '') {
142 showInfo(t(appName, 'Successfully set "{settingsKey}" to {value}.', { settingsKey, value: displayValue }));
143 } else {
144 showInfo(t(appName, 'Setting "{settingsKey}" has been unset successfully.', { settingsKey }));
145 }
146 if (typeof onSuccess === 'function') {
147 onSuccess(responseData, value, settingsSection, settingsKey);
148 }
149 return true;
150 } catch (e) {
151 console.info('ERROR', e);
152 let message = t(appName, 'reason unknown');
153 if (e.response && e.response.data) {
154 const responseData = e.response.data;
155 if (Array.isArray(responseData.messages)) {
156 message = responseData.messages.join(' ');
157 }
158 }
159 if (value) {
160 showError(t(appName, 'Unable to set "{settingsKey}" to {value}: {message}.', {
161 settingsKey,
162 value: value === true ? t(appName, 'true') : '' + value,
163 message,
164 }), {
165 timeout: TOAST_PERMANENT_TIMEOUT,
166 });
167 } else {
168 showError(t(appName, 'Unable to unset "{settingsKey}": {message}', {
169 settingsKey,
170 value: this[settingsKey] || t(appName, 'false'),
171 message,
172 }), {
173 timeout: TOAST_PERMANENT_TIMEOUT,
174 });
175 }
176 return false;
177 }
178}

194async function saveConfirmedSetting(value, settingsSection, settingsKey, force, onSuccess) {
195 const self = this;
196 try {
197 const response = await axios.post(generateUrl('apps/' + appName + '/settings/' + settingsSection + '/' + settingsKey), { value, force });
198 const responseData = response.data;
199 if (responseData.status === 'unconfirmed') {
200 OC.dialogs.confirm(
201 responseData.feedback,
202 t(appName, 'Confirmation Required'),
203 function(answer) {
204 if (answer) {
205 self.saveConfirmedSetting(value, settingsSection, settingsKey, true);
206 } else {
207 showInfo(t(appName, 'Unconfirmed, reverting to old value.'));
208 self.getData();
209 }
210 },
211 true);
212 } else {
213 let displayValue;
214 if (responseData) {
215 if (responseData.newValue !== undefined) {
216 this[settingsKey] = responseData.newValue;
217 displayValue = this[settingsKey];
218 }
219 if (responseData.humanValue !== undefined) {
220 const humanKey = 'human' + settingsKey[0].toUpperCase() + settingsKey.substring(1);
221 this[humanKey] = responseData.humanValue;
222 displayValue = this[humanKey];
223 if (Array.isArray(displayValue)) {
224 displayValue = displayValue.toString();
225 }
226 }
227 }
228 if (displayValue && displayValue !== '') {
229 showSuccess(t(appName, 'Successfully set value for "{settingsKey}" to "{displayValue}"', { settingsKey, displayValue }));
230 } else {
231 showInfo(t(appName, 'Setting "{setting}" has been unset successfully.', { setting: settingsKey }));
232 }
233 if (typeof onSuccess === 'function') {
234 onSuccess(responseData, value, settingsSection, settingsKey, force);
235 }
236 }
237 return true;
238 } catch (e) {
239 console.info('ERROR', e);
240 let message = t(appName, 'reason unknown');
241 if (e.response && e.response.data) {
242 const responseData = e.response.data;
243 if (Array.isArray(responseData.messages)) {
244 message = responseData.messages.join(' ');
245 }
246 }
247 showError(t(appName, 'Could not set value for "{settingsKey}" to "{value}": {message}', {
248 settingsKey, value, message,
249 }), {
250 timeout: TOAST_PERMANENT_TIMEOUT,
251 });
252 self.getData();
253 return false;
254 }
