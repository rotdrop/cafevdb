96 public function log(mixed $level, string $message, array $context = [], int $shift = 0, bool $showTrace = false):void
97 {
98 $level = $this->mapLogLevels($level);
99 $trace = debug_backtrace();
100 $prefix = '';
101 $shift = min($shift, count($trace));
102
103 do {
104 $caller = $trace[$shift];
105 $file = $caller['file']??'unknown';
106 $line = $caller['line']??'unknown';
107 $caller = $trace[$shift+1]??'unknown';
108 $class = $caller['class']??'unknown';
109 $method = $caller['function'];
110
111 $prefix .= $file.':'.$line.': '.$class.'::'.$method.'(): ';
112 } while ($showTrace && --$shift > 0);
113
114 $this->logger()->log($level, $prefix.$message, $context);
