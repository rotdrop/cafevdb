17 }
18 function extractLast( term ) {
19 return split( term ).pop();
20 }
21 function showOptions() {
22 if(!self.element.autocomplete('widget').is(':visible') && self.element.val().trim() == '') {
23 self.element.autocomplete('search', '');
24 }
25 }
26 //console.log('_create: ' + this.options['id']);

29 });
30 this.element.bind('input', function( event ) {
31 showOptions();
32 });
33 this.element.bind('blur', function( event ) {
34 var tmp = self.element.val().trim();
35 if(tmp[tmp.length-1] == ',') {
36 self.element.val(tmp.substring(0, tmp.length-1));
37 } else {
38 self.element.val(tmp);
39 }
40 if(self.element.val().trim() != '') {
41 self.element.trigger('change'); // Changes wasn't saved when only using the dropdown.
42 }
43 });
44 this.element.bind( "keydown", function( event ) {

46 $( this ).data( "autocomplete" ).menu.active ) {
47 event.preventDefault();
48 }

50 .autocomplete({
51 minLength: 0,
52 source: function( request, response ) {
53 // delegate back to autocomplete, but extract the last term
54 response( $.ui.autocomplete.filter(
55 self.options.source, extractLast( request.term ) ) );
56 },
57 focus: function() {
58 // prevent value inserted on focus
59 return false;
60 },

62 var terms = split( this.value );

68 terms.push( "" );
69 this.value = terms.join( ", " );
70 return false;
71 }
