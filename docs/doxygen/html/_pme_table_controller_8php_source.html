159 private function load():Http\Response
160 {
161 $this->logInfo('Start');
162 try {
163 $templateRenderer = $this->parameterService->getParam('templateRenderer');
164 $template = $this->parameterService->getParam('template');
165 $dialogMode = !empty($this->parameterService->getParam('ambientContainerSelector'));
166 $reloadAction = false;
167 $reloadAction = $this->parameterService->getParam(
168 $this->pme->cgiSysName('_reloadfilter'),
169 $this->parameterService->getParam($this->pme->cgiSysName('_reloadlist'))
170 ) !== null;
171
172 if (empty($templateRenderer)) {
173 return self::grumble(['error' => $this->l->t('missing arguments'),
174 'message' => $this->l->t('No template-renderer submitted.'), ]);
175 }
176
178 $renderer = $this->appContainer->query($templateRenderer);
179 if (empty($renderer)) {
180 return self::response(
181 $this->l->t("Template-renderer `%s' cannot be found.", [$templateRenderer]),
182 Http::INTERNAL_SERVER_ERROR);
183 }
184 // $renderer->navigation(false); NOPE, navigation is needed, number of query records may change.
185
186 if ($dialogMode || $reloadAction) {
187 if (!$renderer->needPhpSession()) {
188 $this->logInfo('Closing session');
189 $this->session->close();
190 }
191 $historyAction = PageController::HISTORY_ACTION_LOAD;
192 } else {
193 $this->historyService->push($this->parameterService->getParams());
194 $historyAction = PageController::HISTORY_ACTION_PUSH;
195 }
196
197 $template = 'pme-table';
198 $templateParameters = [
199 'renderer' => $renderer,
200 'templateRenderer' => $templateRenderer,
201 'template' => $template,
202 'recordId' => $this->pme->getCGIRecordId(),
203 ];
204
205 $response = new PreRenderedTemplateResponse($this->appName, $template, $templateParameters, 'blank');
206
207 $response->addHeader('X-'.$this->appName.'-history-action', $historyAction);
208
209 if ($renderer->needPhpSession()) {
210 $response->preRender();
211 }
212
213 if (!$dialogMode && !$reloadAction) {
214 $this->historyService->store();
215 }
216
217 if (!$this->session->isClosed()) {
218 $this->logInfo('Closing session');
219 $this->session->close();
220 }
221
222 return $response;
223
224 } catch (\Throwable $t) {
225 $this->logException($t);
226 return self::grumble($this->exceptionChainData($t));
227 }
228 }

238 private function export():Http\Response
239 {
240 $exportFormat = $this->parameterService['exportFormat'];
241 if (empty($exportFormat)) {
242 return self::grumble($this->l->t('No export-format submitted'));
243 }
244
245 $template = $this->parameterService->getParam('template');
246 if (empty($template)) {
247 return self::grumble(['error' => $this->l->t('missing arguments'),
248 'message' => $this->l->t('No template submitted.'), ]);
249 }
250
252 $exporter = $this->appContainer->query('export'.':'.$template);
253 if (empty($exporter)) {
254 return self::response(
255 $this->l->t('Template-exporter for template "%s" cannot be found.', [$template]),
256 Http::STATUS_BAD_REQUEST);
257 }
258
259 $tmpFile = $this->tempManager->getTemporaryFile($this->appName());
260 register_shutdown_function(function() {
261 $this->tempManager->clean();
262 });
263
264 $fileMeta = $exporter->export($tmpFile, $exportFormat);
265
266 $data = file_get_contents($tmpFile);
267 unlink($tmpFile);
268
269 $fileName = implode('-', [
270 $this->formatTimeStamp($fileMeta['date']),
271 $this->appName(),
272 Util::normalizeSpaces($this->transliterate($fileMeta['name']), '-'),
273 ]) . '.' . $fileMeta['extension'];
274
275 return $this->dataDownloadResponse($data, $fileName, $fileMeta['mimeType']);
276 }
