135const cancel = function(event) {
136 event.preventDefault();
137 // $('#blogtextarea').tinymce().save();
138 if ($('#blogtextarea').val() === globalState.Blog.text) {
139 $('#blogedit').dialog('close').remove();

154const submit = function(event) {
155 event.preventDefault();
156 // $('#blogtextarea').tinymce().save();
157 let popupValue = 0;
158 if ($('#blogpopupset').prop('checked')) {
159 popupValue = 1;
160 } else if ($('#blogpopupclear').prop('checked')) {
161 popupValue = -1;
162 }
163 let clearReaderValue = 0;
164 if ($('#blogreaderclear').prop('checked')) {
165 clearReaderValue = 1;
166 } else {
167 clearReaderValue = 0;
168 }
169
170 console.info('GLOBAL', globalState.Blog);
171 const action = globalState.Blog.blogId > 0 ? 'modify' : 'create';
172 $.post(generateUrl('blog/action/' + action), {
173 blogId: globalState.Blog.blogId,
174 inReplyTo: globalState.Blog.inReplyTo,
175 content: $('#blogtextarea').val(),
176 priority: $('#blogpriority').val(),
177 popup: popupValue,
178 clearReader: clearReaderValue,
179 })
180 .fail(function(xhr, status, errorThrown) {
181 const message = Ajax.failMessage(xhr, status, errorThrown);
182 Dialogs.alert(message, t('cafevdb', 'Error'));
183 })
184 .done(function(data) {
185 $('#blogedit').dialog('close').remove();
186 updateThreads(data);
187 return true;
188 }, 'json');
189 return false;
190};
