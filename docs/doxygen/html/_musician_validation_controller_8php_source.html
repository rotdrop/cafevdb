418 // Compute the "severity" of the match, kind of hacky.
419
420 // email address or any of the two phone numbers and first name matches
421 // -> treat as exact match, 100 %
422
423 // first-name and city and street and street-number match
424 // -> 100 %
425
426 // names match
427 // -> 50 %
428
429 // any of the email address or phone numbers match
430 // -> 50 % (we have different participants which share their comms)
431
432 $duplicatesPropability = 0.0;
433
434 $commsMatch = (
435 (!empty($email) && $email == $musician->getEmail())
436 || (!empty($fixedLinePhone) && $fixedLinePhone == $musician->getFixedLinePhone())
437 || (!empty($mobilePhone) && $mobilePhone == $musician->getMobilePhone()));
438
439 $addressMatch = (
440 (!empty($street) && !empty($streetNumber) && !empty($city))
441 && $street == $musician->getStreet()
442 && $streetNumber == $musician->getStreetNumber()
443 && $city == $musician->getCity());
444
445 $firstNameMatch = !empty($firstName) && $firstName == $musician->getFirstName();
446
447 $surNameMatch = !empty($surName) && $surName == $musician->getSurName();
448
449 $namesMatch = $firstNameMatch && $surNameMatch;
450
451 if ($firstNameMatch) {
452 if ($commsMatch) {
453 $duplicatesPropability = 1.0; // treat as exact match
454 }
455 if ($addressMatch) {
456 $duplicatesPropability = 1.0; // treat as exact match
457 }
458 }
459
460 if ($duplicatesPropability < 1) {
461 if ($namesMatch) {
462 $duplicatesPropability = max($duplicatesPropability, 0.5);
463 }
464 if ($commsMatch) {
465 $duplicatesPropability = max($duplicatesPropability, 0.5);
466 }
467 }
468
469 if ($duplicatesPropability > 0) {
470 $reasons = [];
471 if ($namesMatch) {
472 $reasons[] = $this->l->t('full name');
473 } else {
474 $firstNameMatch && $reasons[] = $this->l->t('first name');
475 $surNameMatch && $reasons[] = $this->l->t('surname');
476 }
477 $commsMatch && $reasons[] = $this->l->t('communication');
478 $addressMatch && $reasons[] = $this->l->t('address');
479
480 $duplicates[$musicianId] = $this->flattenMusician($musician, only: []);
481 $duplicates[$musicianId]['duplicatesPropability'] = $duplicatesPropability;
482 $duplicates[$musicianId]['reasons'] = implode(', ', $reasons);
483 }
484 }
485
486 $message = [];
487 if (count($duplicates) > 0) {
488 $message[] = $this->l->t('Musician(s) with the same first and sur-name already exist: %s', $duplicateNames);
489 }
490
491 return self::dataResponse([
492 'message' => $message,
493 'duplicates' => $duplicates,
494 ]);
495 break;
496 default:
497 break;
498 }
499 return self::grumble($this->l->t('Unknown Request'));
