627 public function generateSubject(?Closure $transliterate = null):string
628 {
629 if (empty($transliterate)) {
630 $transliterate = fn(string $x) => $x;
631 }
632
633 $projectName = $this->project->getName();
634 $projectYear = $this->project->getYear();
635 if (substr($projectName, -4) == (string)$projectYear) {
636 $projectName = substr($projectName, 0, -4);
637 $yearSeparator = '';
638 } else {
639 $yearSeparator = '-';
640 }
641
642 $projectName = Util::dashesToCamelCase(
643 $transliterate(strtolower(Util::camelCaseToDashes($projectName, separator: ' '))),
644 capitalizeFirstCharacter: true,
645 dashes: ' ',
646 );
647
648 // collect the DatabaseStorageFolder's
649 $balanceDocuments = [];
650 if (!empty($this->balanceDocumentsFolder)) {
651 $folderName = $this->balanceDocumentsFolder->getName();
652 if (!empty($folderName)) {
653 $balanceDocuments[$folderName] = true;
654 }
655 }
657 foreach ($this->projectPayments as $projectPayment) {
658 $documentFolder = $projectPayment->getBalanceDocumentsFolder();
659 $folderName = !empty($documentFolder) ? $documentFolder->getName() : null;
660 if (!empty($folderName)) {
661 $balanceDocuments[$folderName] = true;
662 }
663 }
664
665 $yearSequences = [];
666 foreach (array_keys($balanceDocuments) as $folderName) {
667 $sequence = substr($folderName, -3);
668 $year = substr($folderName, -8, 4);
669 $yearSequences[$year] = $yearSequences[$year] ?? [];
670 $yearSequences[$year][] = $sequence;
671 }
672 ksort($yearSequences);
673
674 $balanceSequences = [];
675 if (count($yearSequences) == 1) {
676 $projectYear = array_keys($yearSequences)[0];
677 $projectName .= $yearSeparator . ($projectYear % 100);
678 $sequences = $yearSequences[$projectYear];
679 sort($sequences, SORT_NUMERIC);
680 $len = (int)log10(end($sequences)) + 1;
681 foreach ($sequences as $sequence) {
682 $balanceSequences[] = sprintf('%0' . $len . 'd', $sequence);
683 }
684 } else {
685 foreach ($yearSequences as $year => $sequences) {
686 sort($sequences, SORT_NUMERIC);
687 $len = (int)log10(end($sequences)) + 1;
688 foreach ($sequences as $sequence) {
689 $balanceSequences[] = ($year % 100) . '-' . sprintf('%0' . $len . 'd', $sequence);
690 }
691 }
692 }
693
694 $subjectPrefix = $projectName;
695 if (count($balanceSequences) >= 1) {
696 $subjectPrefix .= '-' . implode(trim(self::SUBJECT_PREFIX_SEPARATOR), $balanceSequences);
697 }
698
699 $subjects = $this->projectPayments
700 ->map(fn(ProjectPayment $payment) => $transliterate($payment->getSubject()))
701 ->toArray();
702 natsort($subjects);
703
704 // try to compactify the composite subject by attempting to group similar subjects
705 $oldPrefix = false;
706 $postfix = [];
707 $purpose = '';
708 foreach ($subjects as $subject) {
709 $parts = array_map(
710 fn($part) => Util::dashesToCamelCase(
711 preg_replace('/([^0-9]|^)[0-9]{2}([0-9]{2})([^0-9]|$)/', '\1\2\3', $part),
712 capitalizeFirstCharacter: true,
713 dashes: ' -_',
714 ),
715 Util::explode(trim(self::SUBJECT_OPTION_SEPARATOR), $subject, Util::TRIM|Util::OMIT_EMPTY_FIELDS|Util::ESCAPED),
716 );
717 $prefix = $parts[0];
718 if (count($parts) < 2 || $oldPrefix != $prefix) {
719 $purpose .= implode(self::SUBJECT_ITEM_SEPARATOR, $postfix);
720 if (strlen($purpose) > 0) {
721 $purpose .= self::SUBJECT_GROUP_SEPARATOR;
722 }
723 $purpose .= $prefix;
724 if (count($parts) >= 2) {
725 $purpose .= self::SUBJECT_OPTION_SEPARATOR;
726 $oldPrefix = $prefix;
727 } else {
728 $oldPrefix = false;
729 }
730 $postfix = [];
731 }
732 if (count($parts) >= 2) {
733 $postfix = array_merge($postfix, array_splice($parts, 1));
734 }
735 }
736 if (!empty($postfix)) {
737 $purpose .= implode(self::SUBJECT_ITEM_SEPARATOR, $postfix);
738 }
739 $purpose = $transliterate(Util::unescapeDelimiter($purpose, trim(self::SUBJECT_OPTION_SEPARATOR)));
740
741 $subject = $subjectPrefix . self::SUBJECT_PREFIX_SEPARATOR . $purpose;
742
743 if (strlen($subject) > self::SUBJECT_FIELD_LENGTH) {
744 $subject = substr(
745 Util::shortenCamelCaseString($subject, self::SUBJECT_FIELD_LENGTH, minLen: 4),
746 0,
747 self::SUBJECT_FIELD_LENGTH,
748 );
749 }
750
751 return $subject;
752 }
