398 public function connected():bool
399 {
400 $connection = $this->getConnection();
401 if (empty($connection)) {
402 return false;
403 }
404 $params = $connection->getParams();
405 $impossible = false;
406 foreach ([ 'host', 'user', 'password', 'dbname' ] as $key) {
407 if (empty($params[$key])) {
408 $impossible = true;
409 }
410 }
411 if ($impossible) {
412 $this->logError('Unable to access database, connection parameters are unset');
413 return false;
414 }
415 try {
416 if (!$connection->ping()) {
417 if (!$connection->connect()) {
418 $this->logError('db cannot connect');
419 return false;
420 }
421 }
422 } catch (\Throwable $t) {
423 $this->logException($t, 'Caught execption trying to ping database server ' . $params['user'] . '@' . $params['host'] . ':' . $params['dbname']);
424 return false;
425 }
426 return true;

861 public function persist(mixed $entity):mixed
862 {
863 $meta = $this->getClassMetadata(get_class($entity));
864 if ($meta->containsForeignIdentifier) {
865 $columnValues = $meta->getIdentifierColumnValues($entity);
866
867 foreach ($meta->associationMappings as $property => $association) {
868
869 if (!$meta->isSingleValuedAssociation($property)) {
870 // only to-one makes sense
871 continue;
872 }
873
874 if (array_search($property, $meta->identifier) !== false) {
875 // all primary keys must already be there
876 continue;
877 }
878
879 if (!empty($meta->getFieldValue($entity, $property))) {
880 // don't override values already present
881 continue;
882 }
883
884 try {
885 $targetEntity = $association['targetEntity'];
886 $targetMeta = $this->getClassMetadata($targetEntity);
887
888 // the actual keys may need remapping through join columns
889 $targetColumnValues = $columnValues;
890 foreach ($association['joinColumns'] as $joinColumn) {
891 $sourceColumn = $joinColumn['name'];
892 $targetColumn = $joinColumn['referencedColumnName'];
893 if (isset($targetColumnValues[$sourceColumn])) {
894 $value = $targetColumnValues[$sourceColumn];
895 unset($targetColumnValues[$sourceColumn]);
896 $targetColumnValues[$targetColumn] = $value;
897 }
898 }
899
