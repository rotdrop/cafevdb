414 public function connected():bool
415 {
416 $connection = $this->getConnection();
417 if (empty($connection)) {
418 return false;
419 }
420 $params = $connection->getParams();
421 $impossible = false;
422 foreach ([ 'host', 'user', 'password', 'dbname' ] as $key) {
423 if (empty($params[$key])) {
424 $impossible = true;
425 }
426 }
427 if ($impossible) {
428 $this->logError('Unable to access database, connection parameters are unset');
429 return false;
430 }
431 try {
432 if (!$connection->ping()) {
433 if (!$connection->connect()) {
434 $this->logError('db cannot connect');
435 return false;
436 }
437 }
438 } catch (\Throwable $t) {
439 $this->logException($t, 'Caught execption trying to ping database server ' . $params['user'] . '@' . $params['host'] . ':' . $params['dbname']);
440 return false;
441 }
442 return true;
443 }

878 public function persist(mixed $entity):mixed
879 {
880 $meta = $this->getClassMetadata(get_class($entity));
881 if ($meta->containsForeignIdentifier) {
882 $columnValues = $meta->getIdentifierColumnValues($entity);
883
884 foreach ($meta->associationMappings as $property => $association) {
885
886 if (!$meta->isSingleValuedAssociation($property)) {
887 // only to-one makes sense
888 continue;
889 }
890
891 if (array_search($property, $meta->identifier) !== false) {
892 // all primary keys must already be there
893 continue;
894 }
895
896 if (!empty($meta->getFieldValue($entity, $property))) {
897 // don't override values already present
898 continue;
899 }
900
901 try {
902 $targetEntity = $association['targetEntity'];
903 $targetMeta = $this->getClassMetadata($targetEntity);
904
905 // the actual keys may need remapping through join columns
906 $targetColumnValues = $columnValues;
907 foreach ($association['joinColumns'] as $joinColumn) {
908 $sourceColumn = $joinColumn['name'];
909 $targetColumn = $joinColumn['referencedColumnName'];
910 if (isset($targetColumnValues[$sourceColumn])) {
911 $value = $targetColumnValues[$sourceColumn];
912 unset($targetColumnValues[$sourceColumn]);
913 $targetColumnValues[$targetColumn] = $value;
914 }
