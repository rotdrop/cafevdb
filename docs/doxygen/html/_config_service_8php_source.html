935 public function encryptConfigValues(array $override = []):void
936 {
937 $this->encryptionCache = array_merge($this->encryptionCache, $override);
938 $appKeys = $this->getAppKeys();
939 $cacheKeys = array_keys($this->encryptionCache);
940 foreach (array_diff($appKeys, $cacheKeys) as $uncached) {
941 if (preg_match('/::[0-9]+$/', $uncached)) {
942 // skip backup keys
943 continue;
944 }
945 $this->logWarn("Found un-cached configuration key $uncached");
946 $this->getConfigValue($uncached, ignoreLock: true);
947 }
948 foreach (array_diff($cacheKeys, $appKeys) as $unstored) {
949 $this->logWarn("Found un-persisted configuration key $unstored");
950 }
951 $cacheKeys = array_keys($this->encryptionCache);
952 //$this->logInfo('keys: '.print_r($cacheKeys, true));
953 foreach ($cacheKeys as $key) {
954 $this->setConfigValue($key, $this->encryptionCache[$key], ignoreLock: true);
955 }

1005 $this->setUser($oldUser);
