911 public function encryptConfigValues(array $override = []):void
912 {
913 $this->encryptionCache = array_merge($this->encryptionCache, $override);
914 $appKeys = $this->getAppKeys();
915 $cacheKeys = array_keys($this->encryptionCache);
916 foreach (array_diff($appKeys, $cacheKeys) as $uncached) {
917 if (preg_match('/::[0-9]+$/', $uncached)) {
918 // skip backup keys
919 continue;
920 }
921 $this->logWarn("Found un-cached configuration key $uncached");
922 $this->getConfigValue($uncached, ignoreLock: true);
923 }
924 foreach (array_diff($cacheKeys, $appKeys) as $unstored) {
925 $this->logWarn("Found un-persisted configuration key $unstored");
926 }
927 $cacheKeys = array_keys($this->encryptionCache);
928 //$this->logInfo('keys: '.print_r($cacheKeys, true));
929 foreach ($cacheKeys as $key) {
930 $this->setConfigValue($key, $this->encryptionCache[$key], ignoreLock: true);
931 }
932 }

969 public function sudo(string $uid, callable $callback)
970 {
971 $oldUser = $this->setUserId($uid);
972 if (empty($oldUser)) {
973 return false;
974 }
975 try {
976 $result = $callback($uid);
977 } catch (Throwable $t) {
978 $this->setUser($oldUser);

980 }
981 $this->setUser($oldUser);

1086 public function localeCountryNames(?string $displayLocale = null):array

1097 foreach ($locales as $locale) {

1099 if ($country) {
1100 $countryCodes[$country] = locale_get_display_region($locale, $displayLanguage);
