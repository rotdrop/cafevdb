909 public function encryptConfigValues(array $override = []):void
910 {
911 $this->encryptionCache = array_merge($this->encryptionCache, $override);
912 $appKeys = $this->getAppKeys();
913 $cacheKeys = array_keys($this->encryptionCache);
914 foreach (array_diff($appKeys, $cacheKeys) as $uncached) {
915 if (preg_match('/::[0-9]+$/', $uncached)) {
916 // skip backup keys
917 continue;
918 }
919 $this->logWarn("Found un-cached configuration key $uncached");
920 $this->getConfigValue($uncached, ignoreLock: true);
921 }
922 foreach (array_diff($cacheKeys, $appKeys) as $unstored) {
923 $this->logWarn("Found un-persisted configuration key $unstored");
924 }
925 $cacheKeys = array_keys($this->encryptionCache);
926 //$this->logInfo('keys: '.print_r($cacheKeys, true));
927 foreach ($cacheKeys as $key) {
928 $this->setConfigValue($key, $this->encryptionCache[$key], ignoreLock: true);
929 }
