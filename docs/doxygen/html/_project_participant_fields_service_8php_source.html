367 public function participantFieldSurcharge(?string $key, ?string $value, Entities\ProjectParticipantField $participantField):float
368 {
369 switch ($participantField->getMultiplicity()) {
370 case Multiplicity::SIMPLE():
371 return (float)$value;
372 case Multiplicity::GROUPOFPEOPLE():
373 if (empty($key)) {
374 break;
375 }
376 return (float)$participantField->getManagementOption()->getData();
377 case Multiplicity::SINGLE():
378 if (empty($key)) {
379 break;
380 }
382 $dataOption = $participantField->getDataOptions()->first();
383 // Non empty value means "yes".
384 if ((string)$dataOption['key'] != $key) {
385 $this->logWarn('Stored value "'.$key.'" unequal to stored key "'.$dataOption['key'].'"');
386 }
387 return (float)$dataOption['data'];
388 case Multiplicity::GROUPSOFPEOPLE():
389 case Multiplicity::MULTIPLE():
390 if (empty($key)) {
391 break;
392 }
393 foreach ($participantField->getDataOptions() as $dataOption) {
394 if ((string)$dataOption['key'] == $key) {
395 return (float)$dataOption['data'];
396 }
397 }
398 $this->logError('No data item for multiple choice key "'.$key.'"');
399 return 0.0;
400 case Multiplicity::PARALLEL():
401 if (empty($key)) {
402 break;
403 }
404 $keys = Util::explode(',', $key);
405 $found = false;
406 $amount = 0.0;
407 foreach ($participantField->getDataOptions() as $dataOption) {
408 if (array_search((string)$dataOption['key'], $keys) !== false) {
409 $amount += (float)$dataOption['data'];
410 $found = true;
411 }
412 }
413 if (!$found) {
414 $this->logError('No data item for parallel choice key "'.$key.'"');
415 }
416 return $amount;
417 case Multiplicity::RECURRING():
418 if (empty($key) || empty($value)) {
419 break;
420 }
421 // $keys = Util::explode(',', $key);
422 $values = Util::explodeIndexed($value);
423
424 $amount = 0.0;
425 foreach ($values as $key => $value) {
426 $amount += $value;
427 }
428 return $amount;
429 }
430 return 0.0;

730 public function explodeDataOptions(mixed $values, bool $addProto = true, bool $trimInactive = false):array
731 {
732 $options = empty($values) ? [] : (is_array($values) ? $values : json_decode($values, true));
733 if (is_string($options)) {
734 // perhaps double encoded, try ...
735 $options = json_decode($options, true);
736 }
737 if (is_string($values) && $values != '[]' && !empty($values) && empty($options)) {
738 $options = [
739 array_merge(
740 $this->dataOptionPrototype(),
741 [ 'key' => Uuid::create(), 'label' => $values, ]),
742 ];
743 }
744 if (isset($options[-1])) {
745 throw new Exception(
746 $this->l->t(
747 'Option index -1 should not be present here, options: %s',
748 print_r($options, true)));
749 }
750 $options = array_values($options);
751 $protoType = $this->dataOptionPrototype();
752 $protoKeys = array_keys($protoType);
753 $result = [];
754 foreach ($options as $option) {
755 $keys = array_keys($option);
756 if ($keys !== $protoKeys) {
757 throw new InvalidArgumentException(
758 $this->l->t(
759 'Prototype keys "%s" and options keys "%s" differ',
760 [ implode(',', $protoKeys), implode(',', $keys) ])
761 );
762 }
763 if ($trimInactive && !empty($option['deleted'])) {
764 continue;
765 }
766 $result[(string)$option['key']] = $option;
767 }
768 if ($addProto) {
769 $result[$protoType['key']] = $protoType;
770 }
771 return $result;
772 }
