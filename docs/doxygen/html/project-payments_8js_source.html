57/**
58 * Generate a popup in order to add a new split-transaction, i.e. a
59 * ProjectPayment entity which is always subordinate to a
60 * CompositePayment entity.
61 *
62 * @param {string} containerSel The ambient element of the container
63 * (i.e. the base page, or the div holding the dialog this one was
64 * initiated from.
65 *
66 * @param {object} post Arguments object:
67 * { projectName: 'NAME', projectId: XX }
68 */
69const projectPaymentPopup = function(containerSel, post) {
70 // Prepare the data-array for PHPMyEdit.tableDialogOpen(). The
71 // instrumentation numbers are somewhat nasty and require too
72 // many options.
73
74 const template = 'project-payments';
75 const tableOptions = {
76 ambientContainerSelector: containerSel,
77 dialogHolderCSSId: template + '-dialog',
78 template,
79 templateRenderer: Page.templateRenderer(template),
80 Table: 'CompositePayments',
81 projectId: post.projectId,
82 projectName: post.projectName,
83 // Now special options for the dialog popup
84 initialViewOperation: false,
85 initialName: pmeSys('operation'),
86 initialValue: 'Change',
87 reloadName: pmeSys('operation'),
88 reloadValue: 'Change',
89 // reloadMode: 'discard',
90 [pmeSys('operation')]: 'Change',
91 modalDialog: false,
92 modified: false,
93 };
94 PHPMyEdit.tableDialogOpen($.extend({}, tableOptions, post));
95};

97const backgroundDecryption = function(container) {
98 const $container = PHPMyEdit.container(container);
99 rejectDecryptionPromise();
100 console.time('DECRYPTION PROMISE');
101 decryptionPromise.done((maxJobs) => {
102 console.timeEnd('DECRYPTION PROMISE');
103 console.info('MAX DECRYPTION JOBS HANDLED', maxJobs);
104 });
105 lazyDecrypt($container);
106};

108const ready = function(selector, pmeParameters, resizeCB) {
109
110 const $container = $(selector);
111
112 if (pmeParameters.reason !== 'dialogClose') {
113 backgroundDecryption($container);
114 }
115
116 if (pmeParameters.reason === 'dialogOpen') {
117
118 // AJAX download support
119 $container
120 .on('click', 'a.download-link.ajax-download', function(event) {
121 const $this = $(this);
122 fileDownload($this.attr('href'));
123 return false;
124 });
125
126 $container
127 .on('contextmenu', 'table.pme-main tr.composite-payment.first td', function(event) {
128 if (event.ctrlKey) {
129 return; // let the user see the normal context menu
130 }
131 const $row = $(this).closest('tr.composite-payment.first');
132 event.stopImmediatePropagation();
133 $row.toggleClass('following-hidden');
134 $row.find('input.expanded-marker').val($row.hasClass('following-hidden') ? 0 : 1);
135 return false;
136 });
137
138 $container
139 .on('change', 'select.instrumentation-id', function(event) {
140 const $this = $(this);
141 const musicianId = $this.val();
142 const $receivables = $container.find('select.receivable');
143 const $receivableOptions = $receivables.find('option');
144 if (musicianId !== '') {
145 const $musicianOption = SelectUtils.optionByValue($this, musicianId);
146 const allowedOptionKeys = $musicianOption.data('data').split(',');
147 $receivableOptions.each(function(index) {
148 const $option = $(this);
149 if ($option.val() !== '') {
150 $option.prop('disabled', allowedOptionKeys.indexOf($option.val()) < 0);
151 }
152 });
153 } else {
154 $receivableOptions.prop('disabled', false);
155 }
156 SelectUtils.refreshWidget($receivables);
157 });
158
159 $container
160 .on('change', 'select.receivable', function(event) {
161 const $this = $(this);
162 const receivableKey = $this.val();
163 const $musicians = $container.find('select.instrumentation-id');
164 const $musiciansOptions = $musicians.find('option');
165 if (receivableKey !== '') {
166 $musiciansOptions.each(function(index) {
167 const $option = $(this);
168 if ($option.val() !== '') {
169 $option.prop('disabled', $option.data('data').split(',').indexOf(receivableKey) < 0);
170 }
171 });
172 } else {
173 $musiciansOptions.prop('disabled', false);
174 }
175 SelectUtils.refreshWidget($musicians);
176 });
177
178 $container
179 .on('change', 'select.payment-id', function(event) {
180 const $this = $(this);
181
182 // deselect action option
183 const $actionOption = SelectUtils.optionByValue($this, -1);
184 if ($actionOption.is(':selected')) {
185 $actionOption.prop('selected', false);
186 SelectUtils.refreshWidget($this);
187
188 let musicianIds = [];
189 let projectIds = [];
190 $this.find('option').each(function() {
191 const $option = $(this);
192 if ($option.is($actionOption)) {
193 return;
194 }
195 const optionData = $option.data('data');
196 musicianIds.push(optionData.musicianId);
197 projectIds.push(optionData.projectId);
198 });
199 musicianIds = [...new Set(musicianIds)];
200 projectIds = [...new Set(projectIds)];
201
202 if (projectIds.length > 1) {
203 Dialogs.alert(
204 t(appName, 'Too many Projects'),
205 t(appName, 'Currently merging composite-payments for different projects ({projects}) is not supported, sorry.',
206 { projects: projectIds.join(', ') }),
207 false, true);
208 }
209 if (musicianIds.length > 1) {
210 Dialogs.alert(
211 t(appName, 'Too many Musicians'),
212 t(appName, 'Internal error: splits of composite-payments cannot belong to different musicians ({musicians}).',
213 { musicians: musicianIds.join(', ') }),
214 false, true);
215 }
216 const projectId = projectIds[0];
217 const musicianId = musicianIds[0];
218
219 const recordId = $actionOption.data('data').recordId;
220 const compositePaymentId = recordId.id;
221
222 projectPaymentPopup(selector, {
223 projectId,
224 musicianId,
225 [pmeData('id')]: compositePaymentId,
226 [pmeData('Musicians:id')]: musicianId,
227 [pmeSys('rec')]: recordId,
228 [pmeSys('groupby_rec')]: { id: compositePaymentId, ['ProjectPayments' + pageRenderer.masterFieldSuffix]: 0 },
229 });
230
231 return false;
232 }
233 return false;
234 });
235
236 // upload supporting document(s)
237 const musicianId = findByName($container, pmeData('Musicians:id')).val();
238 $container
239 .find('tr.supporting-document td.pme-value .file-upload-row')
240 .each(function() {
241 initFileUploadRow.call(
242 this,
243 -1, // projectId
244 musicianId,
245 resizeCB, {
246 upload: 'finance/payments/documents/upload',
247 delete: 'finance/payments/documents/delete',
248 });
249 });
250
251 $container
252 .on('change', 'select.project-balance-documents', function(event) {
253 const $this = $(this);
254 const $option = SelectUtils.selectedOptions($this);
255 const $cell = $this.closest(pmeValueSelector);
256 const $filesAppAnchor = $cell.find('.open-parent');
257 const filesAppLink = $filesAppAnchor.data('parent-link');
258 if ($option.length > 0) {
259 $filesAppAnchor.attr('href', filesAppLink + '/' + $option.data('data'));
260 } else {
261 $filesAppAnchor.attr('href', filesAppLink);
262 }
263 return false;
264 });
265
266 } // reason === 'dialogOpen'
267
268 const tableOptions = pmeParameters.tableOptions || {};
269 if (tableOptions.ambientContainerSelector) {
270
271 const $pmeForm = (pmeParameters.reason === 'dialogClose')
272 ? $(pmeParameters.htmlResponse).find(pmeFormSelector)
273 : $container.find(pmeFormSelector);
274
275 const paymentsRowTagName = pmeData('ProjectPayments:row_tag');
276 const rowTag = findByName($pmeForm, paymentsRowTagName).val();
277
278 // Make sure the current payment is present in the "master"
279 // form. Otherwise a form submit would delete the current payment
280 // again.
281 if (rowTag && !isCompositeRow(rowTag)) {
282 const $ambientContainer = $(tableOptions.ambientContainerSelector);
283 const $ambientForm = $ambientContainer.find(pmeFormSelector);
284 const paymentsIdName = pmeData('ProjectPayments:id[]');
285 const $paymentOption = findByName($ambientContainer, paymentsIdName).find('option[value="' + rowTag + '"]');
286 if ($paymentOption.length === 0) {
287 const amountName = pmeData('ProjectPayments:amount');
288 const subjectName = pmeData('ProjectPayments:subject');
289 $ambientForm.append('<input type="hidden" name="' + paymentsIdName + '" value="' + rowTag + '"/>');
290 const amountInput = findByName($ambientForm, amountName);
291 amountInput.val(amountInput.val() + ',' + rowTag + ':' + findByName($pmeForm, amountName).val());
292 const subjectInput = findByName($ambientForm, subjectName);
293 subjectInput.val(subjectInput.val() + ',' + rowTag + ':' + findByName($pmeForm, subjectName).val());
294 }
295 }
296 }
297
298 resizeCB();
299};

301const documentReady = function() {
302
303 CAFEVDB.addReadyCallback(function() {
304
305 const container = PHPMyEdit.container();
306
307 if (!container.hasClass('project-payments')) {
308 return;
309 }
310
311 ready(container, { reason: 'dialogOpen' }, function() {});
312 });

314};
