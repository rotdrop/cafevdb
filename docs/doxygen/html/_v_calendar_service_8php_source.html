764 private function updateVTodoFromRequest(VCalendar $vCalendar, array $request):VCalendar
765 {
766 if (!empty($this->validateVTodoRequest($request))) {
767 return null;
768 }
769
770 $vTodo = $vCalendar->VTODO;
771 $timezone = $this->getDateTimeZone();
772
773 $vTodo->{'LAST-MODIFIED'} = new DateTimeImmutable('now', new DateTimeZone('UTC'));
774 $vTodo->DTSTAMP = new DateTimeImmutable('now', new DateTimeZone('UTC'));
775
776 $vTodo->SUMMARY = $request['summary'];
777
778 if (!empty($request['due'])) {
779 $due = new DateTimeImmutable($request['due'], $timezone);
780 if ($request['allday']) {
781 $due = $due->setTime(0, 0, 0);
782 $vTodo->DUE = $due;
783 $vTodo->DUE['VALUE'] = 'DATE';
784 } else {
785 $vTodo->DUE = $due;
786 }
787 }
788 if (!empty($request['start'])) {
789 $dtStart = new DateTimeImmutable($request['start'], $timezone);
790 if ($request['allday']) {
791 $dtStart = $dtStart->setTime(0, 0, 0);
792 $vTodo->DTSTART = $dtStart;
793 $vTodo->DTSTART['VALUE'] = 'DATE';
794 } else {
795 $vTodo->DTSTART = $dtStart;
796 }
797 }
798
799 $optionalKeys = [
800 'priority' => 'PRIORITY',
801 'description' => 'DESCRIPTION',
802 'location' => 'LOCATION',
803 'status' => 'STATUS',
804 ];
805 foreach ($optionalKeys as $rqKey => $vKey) {
806 if (!empty($request[$rqKey])) {
807 $vTodo->{$vKey} = $request[$rqKey];
808 } else {
809 unset($vTodo->{$vKey});
810 }
811 }
812
813 $categories = [];
814 if (!empty($request['categories'])) {
815 $categories = explode(',', $request['categories']);
816 }
817 if (count($categories)) {
818 $vTodo->CATEGORIES = $categories;
819 } else {
820 unset($vTodo->{'CATEGORIES'});
821 }
822
823 $this->addVAlarmsFromRequest($vCalendar, $vCalendar->VTODO, $request);
824 $this->addParticipants($vCalendar, $vCalendar->VTODO, $request);
825 $this->addRelations($vCalendar, $vCalendar->VTODO, $request);
826
827 return $vCalendar;
828 }

844 private function createVAlarmFromSeconds(
845 VCalendar $vCalendar,
846 int $seconds,
847 ?string $related = null,
848 ?string $description = null,
849 string $action = self::ALARM_ACTION_DISPLAY,
850 ):?VAlarm {
851 if (empty($seconds)) {
852 return null;
853 }
854 /*
855BEGIN:VALARM
856DESCRIPTION:
857ACTION:DISPLAY
858TRIGGER;VALUE=DURATION:-P1D
859X-KDE-KCALCORE-ENABLED:TRUE
860END:VALARM
861 */
862 $inverted = $seconds < 0;
863 if ($inverted) {
864 $seconds = -$seconds;
865 $related = $related ?? self::VALARM_FROM_END;
866 } else {
867 $related = $related ?? self::VALARM_FROM_START;
868 }
869 $vAlarm = $vCalendar->createComponent('VALARM');
870 $vAlarm->DESCRIPTION = !empty($description) ? $description : $this->l->t('Default Event Notification');
871 $vAlarm->ACTION = $action;
872 if ($inverted) {
873 $dinterval = (new DateTimeImmutable)->add(new DateInterval('PT' . ($seconds + 1) . 'S'));
874 } else {
875 $dinterval = (new DateTimeImmutable)->sub(new DateInterval('PT' . $seconds . 'S'));
876 }
877 $interval = $dinterval->diff(new DateTimeImmutable);
878 $alarmValue = sprintf(
879 '%sP%s%s%s%s',
880 $interval->format('%r'),
881 $interval->format('%d') > 0 ? $interval->format('%dD') : null,
882 ($interval->format('%h') > 0 || $interval->format('%i') > 0) ? 'T' : null,
883 $interval->format('%h') > 0 ? $interval->format('%hH') : null,
884 $interval->format('%i') > 0 ? $interval->format('%iM') : null);
885 $vAlarm->add('TRIGGER', $alarmValue, [ 'VALUE' => 'DURATION', 'RELATED' => $related ]);
886
887 // $this->logInfo('TRY CREATE ALARM ' . $vAlarm->TRIGGER . ' SECONDS ' . ($inverted ? '-' : '') . $seconds);
888
889 return $vAlarm;
890 }
