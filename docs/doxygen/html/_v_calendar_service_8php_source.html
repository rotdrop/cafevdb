250 public function addRelations(VCalendar $vRoot, VComponent $vComponent, mixed $objectData):void
251 {
252 if (empty($objectData['related'])) {
253 return;
254 }
255 $related = $objectData['related'];
256 if (empty(array_intersect(array_keys($related), ['PARENT', 'CHILD', 'SIBLING']))) {
257 $related = [ 'PARENT' => $related ];
258 }
259 $ownUid = $vComponent->UID;
260 unset($vComponent->{'RELATED-TO'});
261 foreach ($related as $relType => $uids) {
262 foreach ($uids as $uid) {
263 if ($uid == $ownUid) {
264 continue; // gracefully ignore
265 }
266 $vComponent->add(
267 'RELATED-TO',
268 $uid, [
269 'RELTYPE' => $relType,
270 ],
271 );

567 public static function getDTEnd(VCalendar $vCalendar):ICalendar\DateTime
568 {
569 $vObject = self::getVObject($vCalendar);
570 if ($vObject->DTEND) {
571 return $vObject->DTEND;
572 }
573 $dtEnd = clone $vObject->DTSTART;
574 // clone creates a shallow copy, also clone DateTime
575 $dtEnd->setDateTime(clone $dtEnd->getDateTime());
576 if ($vObject->DURATION) {
577 $duration = strval($vObject->DURATION);
578 $invert = 0;
579 if ($duration[0] == '-') {
580 $duration = substr($duration, 1);
581 $invert = 1;
582 }
583 if ($duration[0] == '+') {
584 $duration = substr($duration, 1);
585 }
586 $interval = new DateInterval($duration);
587 $interval->invert = $invert;
588 $dtEnd->getDateTime()->add($interval);
589 }
590 return $dtEnd;

681 private function updateVTodoFromRequest(VCalendar $vCalendar, array $request):VCalendar
682 {
683 if (!empty($this->validateVTodoRequest($request))) {
684 return null;
685 }
686
687 $vTodo = $vCalendar->VTODO;
688 $timezone = $this->getDateTimeZone();
689
690 $vTodo->{'LAST-MODIFIED'} = new DateTimeImmutable('now', new DateTimeZone('UTC'));
691 $vTodo->DTSTAMP = new DateTimeImmutable('now', new DateTimeZone('UTC'));
692
693 $vTodo->SUMMARY = $request['summary'];
694
695 if (!empty($request['due'])) {
696 $due = new DateTimeImmutable($request['due'], $timezone);
697 if ($request['allday']) {
698 $due = $due->setTime(0, 0, 0);
699 $vTodo->DUE = $due;
700 $vTodo->DUE['VALUE'] = 'DATE';
701 } else {
702 $vTodo->DUE = $due;
703 }
704 }
705 if (!empty($request['start'])) {
706 $dtStart = new DateTimeImmutable($request['start'], $timezone);
707 if ($request['allday']) {
708 $dtStart = $dtStart->setTime(0, 0, 0);
709 $vTodo->DTSTART = $dtStart;
710 $vTodo->DTSTART['VALUE'] = 'DATE';
711 } else {
712 $vTodo->DTSTART = $dtStart;
713 }
714 }
715
716 $optionalKeys = [
717 'priority' => 'PRIORITY',
718 'description' => 'DESCRIPTION',
719 'location' => 'LOCATION',
720 'status' => 'STATUS',
721 ];
722 foreach ($optionalKeys as $rqKey => $vKey) {
723 if (!empty($request[$rqKey])) {
724 $vTodo->{$vKey} = $request[$rqKey];
725 } else {
726 unset($vTodo->{$vKey});
727 }
728 }
729
730 $categories = [];
731 if (!empty($request['categories'])) {
732 $categories = explode(',', $request['categories']);
733 }
734 if (count($categories)) {
735 $vTodo->CATEGORIES = $categories;
736 } else {
737 unset($vTodo->{'CATEGORIES'});
738 }
739
740 $this->addVAlarmsFromRequest($vCalendar, $vCalendar->VTODO, $request);
741 $this->addParticipants($vCalendar, $vCalendar->VTODO, $request);
742 $this->addRelations($vCalendar, $vCalendar->VTODO, $request);
743
744 return $vCalendar;
745 }

761 private function createVAlarmFromSeconds(
762 VCalendar $vCalendar,
763 int $seconds,
764 ?string $related = null,
765 ?string $description = null,
766 string $action = self::ALARM_ACTION_DISPLAY,
767 ):?VAlarm {
768 if (empty($seconds)) {
769 return null;
770 }
771 /*
772BEGIN:VALARM
773DESCRIPTION:
774ACTION:DISPLAY
775TRIGGER;VALUE=DURATION:-P1D
776X-KDE-KCALCORE-ENABLED:TRUE
777END:VALARM
778 */
779 $inverted = $seconds < 0;
780 if ($inverted) {
781 $seconds = -$seconds;
782 $related = $related ?? self::VALARM_FROM_END;
783 } else {
784 $related = $related ?? self::VALARM_FROM_START;
785 }
786 $vAlarm = $vCalendar->createComponent('VALARM');
787 $vAlarm->DESCRIPTION = !empty($description) ? $description : $this->l->t('Default Event Notification');
788 $vAlarm->ACTION = $action;
789 if ($inverted) {
790 $dinterval = (new DateTimeImmutable)->add(new DateInterval('PT' . ($seconds + 1) . 'S'));
791 } else {
792 $dinterval = (new DateTimeImmutable)->sub(new DateInterval('PT' . $seconds . 'S'));
793 }
794 $interval = $dinterval->diff(new DateTimeImmutable);
795 $alarmValue = sprintf(
796 '%sP%s%s%s%s',
797 $interval->format('%r'),
798 $interval->format('%d') > 0 ? $interval->format('%dD') : null,
799 ($interval->format('%h') > 0 || $interval->format('%i') > 0) ? 'T' : null,
800 $interval->format('%h') > 0 ? $interval->format('%hH') : null,
801 $interval->format('%i') > 0 ? $interval->format('%iM') : null);
802 $vAlarm->add('TRIGGER', $alarmValue, [ 'VALUE' => 'DURATION', 'RELATED' => $related ]);
803
804 // $this->logInfo('TRY CREATE ALARM ' . $vAlarm->TRIGGER . ' SECONDS ' . ($inverted ? '-' : '') . $seconds);
805
806 return $vAlarm;
807 }
