161 private function createVEventFromRequest(mixed $objectData):?VCalendar
162 {
163 if (!isset($objectData['repeat'])) {
164 $objectData['repeat'] = 'doesnotrepeat';
165 }
167 $start = $objectData['start'] ?? null;
168 if ($start instanceof DateTimeInterface) {
169 $timezone = $this->getDateTimeZone($start->getTimestamp());
170 $start = $start->setTimezone($timezone);
171 $objectData['from'] = $objectData['start']->format('d-m-Y');
172 if (!isset($objectData['allday'])) {
173 $objectData['fromtime'] = $start->format('H:i:s');
174 }
175 }

181 $objectData['to'] = $objectData['end']->format('d-m-Y');
182 if (!isset($objectData['allday'])) {
183 $objectData['totime'] = $end->format('H:i:s');

785 private function updateVTodoFromRequest(VCalendar $vCalendar, array $request):VCalendar
786 {
787 if (!empty($this->validateVTodoRequest($request))) {
788 return null;
789 }
790
791 $vTodo = $vCalendar->VTODO;
792 $timezone = $this->getDateTimeZone();
793
794 $vTodo->{'LAST-MODIFIED'} = new DateTimeImmutable('now', new DateTimeZone('UTC'));
795 $vTodo->DTSTAMP = new DateTimeImmutable('now', new DateTimeZone('UTC'));
796
797 $vTodo->SUMMARY = $request['summary'];
798
799 if (!empty($request['due'])) {
800 $due = new DateTimeImmutable($request['due'], $timezone);
801 if ($request['allday']) {
802 $due = $due->setTime(0, 0, 0);
803 $vTodo->DUE = $due;
804 $vTodo->DUE['VALUE'] = 'DATE';
805 } else {
806 $vTodo->DUE = $due;
807 }
808 }
809 if (!empty($request['start'])) {
810 $dtStart = new DateTimeImmutable($request['start'], $timezone);
811 if ($request['allday']) {
812 $dtStart = $dtStart->setTime(0, 0, 0);
813 $vTodo->DTSTART = $dtStart;
814 $vTodo->DTSTART['VALUE'] = 'DATE';
815 } else {
816 $vTodo->DTSTART = $dtStart;
817 }
818 }
819
820 $optionalKeys = [
821 'priority' => 'PRIORITY',
822 'description' => 'DESCRIPTION',
823 'location' => 'LOCATION',
824 'status' => 'STATUS',
825 ];
826 foreach ($optionalKeys as $rqKey => $vKey) {
827 if (!empty($request[$rqKey])) {
828 $vTodo->{$vKey} = $request[$rqKey];
829 } else {
830 unset($vTodo->{$vKey});
831 }
832 }
833
834 $categories = [];
835 if (!empty($request['categories'])) {
836 $categories = explode(',', $request['categories']);
837 }
838 if (count($categories)) {
839 $vTodo->CATEGORIES = $categories;
840 } else {
841 unset($vTodo->{'CATEGORIES'});
842 }
843
844 $this->addVAlarmsFromRequest($vCalendar, $vCalendar->VTODO, $request);
845 $this->addParticipants($vCalendar, $vCalendar->VTODO, $request);
846 $this->addRelations($vCalendar, $vCalendar->VTODO, $request);

865 private function createVAlarmFromSeconds(
866 VCalendar $vCalendar,
867 int $seconds,
868 ?string $related = null,
869 ?string $description = null,
870 string $action = self::ALARM_ACTION_DISPLAY,
871 ):?VAlarm {
872 if (empty($seconds)) {
873 return null;
874 }
875 /*
876BEGIN:VALARM
877DESCRIPTION:
878ACTION:DISPLAY
879TRIGGER;VALUE=DURATION:-P1D
880X-KDE-KCALCORE-ENABLED:TRUE
881END:VALARM
882 */
883 $inverted = $seconds < 0;
884 if ($inverted) {
885 $seconds = -$seconds;
886 $related = $related ?? self::VALARM_FROM_END;
887 } else {
888 $related = $related ?? self::VALARM_FROM_START;
889 }
890 $vAlarm = $vCalendar->createComponent('VALARM');
891 $vAlarm->DESCRIPTION = !empty($description) ? $description : $this->l->t('Default Event Notification');
892 $vAlarm->ACTION = $action;
893 if ($inverted) {
894 $dinterval = (new DateTimeImmutable)->add(new DateInterval('PT' . ($seconds + 1) . 'S'));
895 } else {
896 $dinterval = (new DateTimeImmutable)->sub(new DateInterval('PT' . $seconds . 'S'));
897 }
898 $interval = $dinterval->diff(new DateTimeImmutable);
899 $alarmValue = sprintf(
900 '%sP%s%s%s%s',
901 $interval->format('%r'),
902 $interval->format('%d') > 0 ? $interval->format('%dD') : null,
903 ($interval->format('%h') > 0 || $interval->format('%i') > 0) ? 'T' : null,
904 $interval->format('%h') > 0 ? $interval->format('%hH') : null,
905 $interval->format('%i') > 0 ? $interval->format('%iM') : null);
906 $vAlarm->add('TRIGGER', $alarmValue, [ 'VALUE' => 'DURATION', 'RELATED' => $related ]);
907
908 // $this->logInfo('TRY CREATE ALARM ' . $vAlarm->TRIGGER . ' SECONDS ' . ($inverted ? '-' : '') . $seconds);
909
910 return $vAlarm;
911 }
