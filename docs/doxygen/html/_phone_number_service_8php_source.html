170 public function normalize(string $number):string
171 {
172
173 // convert html entities back to their character expressions
174 $number = html_entity_decode($number);
175
176 // convert non-breaking unicode space to normal space
177 $number = str_replace("\xc2\xa0", "\x20", $number);
178
179 // remove space
180 $number = preg_replace('/\s+/', '', $number);
181
182 // remove brackets
183 $number = preg_replace('/^\‍((\+\d\d)\‍)/', '$1', $number);
184
185 // remove intermediate (0) stuff
186 $number = preg_replace('/^([^(]*\d+[^(]*)\‍(0\‍)/', '$1', $number);
187
188 // replace leading dounle 00 by + as this seems to be better
189 // understood by the libphonenumber backend.
190 $number = preg_replace('/^00/', '+', $number);
191
192 if ($number !== '') {
193 // add local area code
194 if (!preg_match('!^[0+]!', $number) && $this->getDefaultPrefix()) {
195 $number = '0' . $this->getDefaultPrefix() . $number;
196 }
197 }
198
199 return $number;
200 }
