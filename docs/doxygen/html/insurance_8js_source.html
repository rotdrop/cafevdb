86const updateInsuranceFee = function(elements) {
87
88 const $scopeSelect = elements.$scopeSelect;
89 const $brokerSelect = elements.$brokerSelect;
90 const $insuranceAmount = elements.$insuranceAmount;
91 const $insuranceRate = elements.$insuranceRate;
92 const $insuranceFee = elements.$insuranceFee;
93
94 const scope = SelectUtils.selected($scopeSelect);
95 if (!scope) {
96 return false;
97 }
98 const $broker = SelectUtils.selectedOptions($brokerSelect);
99 if ($broker.length === 0) {
100 return false;
101 }
102 const rates = $broker.data('data');
103 const rateMeta = rates.find(rate => rate.geographicalScope === scope);
104 if (!rateMeta) {
105 return false;
106 }
107 const rate = rateMeta.rate;
108
109 $insuranceRate.find('.pme-input').val(rate);
110 const $rateDisplay = $insuranceRate.find('.insurance-rate-display');
111 $rateDisplay.data('value', rate);
112 $rateDisplay.html((rate * 100.0).toLocaleString(lang) + ' %');
113
114 $insuranceFee.html(
115 new Intl.NumberFormat(
116 lang, {
117 style: 'currency',
118 currency: $insuranceFee.data('currencyCode'),
119 })
120 .format(
121 $insuranceAmount.val() * rate * (1.0 + $insuranceFee.data('taxRate'))
122 )
123 );
124};

126const enableScopeOptions = function($scopeSelect, $broker) {
127 if ($broker.length === 0 || $broker.val() === '') {
128 return;
129 }

131 if (rates === undefined) {
132 console.error('RATES UNDEFINED', $broker.data());
133 return;
134 }
135 $scopeSelect.find('option').each(function(idx) {
136 const $option = $(this);
137 $option.prop('disabled', rates.find(rate => rate.geographicalScope === $option.val()) === undefined);
138 });
139 $scopeSelect.trigger('change'); // update insurance fees
140 SelectUtils.refreshWidget($scopeSelect);

143const pmeFormInit = function(containerSel) {
144 containerSel = PHPMyEdit.selector(containerSel);
145 const container = PHPMyEdit.container(containerSel);
146 const form = container.find('form[class^="pme-form"]');
147 const submitSel = 'input.pme-save,input.pme-apply,input.pme-more';
148 const submits = form.find(submitSel);
149
150 if (submits.length > 0) {
151
152 const rateDialog = container.find('select.broker').length > 0;
153 const brokerDialog = container.find('input.broker').length > 0;
154
155 let textInputs;
156 let key;
157
158 if (brokerDialog) {
159 textInputs = {
160 broker: container.find('input.broker'),
161 brokerName: container.find('input.brokername'),
162 brokerAddress: container.find('textarea.brokeraddress'),
163 };
164 } else if (rateDialog) {
165 textInputs = {
166 rate: container.find('input.rate'),
167 policy: container.find('input.policy'),
168 };
169 } else {
170 // need to disable all of these on blur in order to avoid
171 // focus ping-pong
172 textInputs = {
173 insuredItem: container.find('input.insured-item'),
174 manufacturer: container.find('input.manufacturer'),
175 constructionYear: container.find('input.construction-year'),
176 amount: container.find('input.amount'),

180 const oldValues = {};
181 for (key in textInputs) {
182 oldValues[key] = textInputs[key].val();
183 }

185 const validate = function(control, button, lockCallback) {
186
187 if (lockCallback === undefined) {
188 lockCallback = function(lock) {};
189 }
190
191 const validateLock = function() {
192 lockCallback(true);
193 };
194
195 const validateUnlock = function() {
196 lockCallback(false);
197 };
198
199 let post = form.serialize();
200 post += '&control=' + control;
201
202 // until end of validation
203 validateLock(true);
204
205 Notification.hide(function() {
206 $.post(generateUrl('insurance/validate/' + control), post)
207 .fail(function(xhr, status, errorThrown) {
208 Ajax.handleError(xhr, status, errorThrown, function() {
209 for (const key in textInputs) {
210 textInputs[key].val(oldValues[key]);
211 }
212 validateUnlock();
213 });
214 })
215 .done(function(data) {
216 if (!Ajax.validateResponse(
217 data,
218 Object.keys(textInputs),
219 validateUnlock)) {
220 for (const key in textInputs) {
221 textInputs[key].val(oldValues[key]);
222 }
223 return;
224 }
225
226 Notification.messages(data.message);
227
228 if (typeof textInputs[control] !== 'undefined') {
229 textInputs[control].val(data[control]);
230 }
231 if (control === 'submit') {
232 for (key in textInputs) {
233 textInputs[key].val(data[key]);
234 }
235 if (typeof button !== 'undefined') {
236 $(form).off('click', submitSel);
237 button.trigger('click');
238 } else {
239 form.submit();
240 }
241 }
242 for (key in textInputs) {
243 oldValues[key] = textInputs[key].val();
244 }
245
246 validateUnlock();
247 });
248 });
249 }; // validate end
