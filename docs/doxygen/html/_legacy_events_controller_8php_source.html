130 public function serviceSwitch(string $topic, string $subTopic):Http\Response
131 {
132 switch ($topic) {
133 case 'forms':
134 switch ($subTopic) {
135 case self::SUBTOPIC_NEW:
136 return $this->newEventForm();
137 case self::SUBTOPIC_EDIT:
138 $this->session->close();
139 return $this->editEventForm($subTopic);
140 case self::SUBTOPIC_CLONE:
141 return $this->editEventForm($subTopic);
142 default:
143 break;
144 }
145 break;
146 case 'actions':
147 $this->session->close();
148 switch ($subTopic) {
149 case self::SUBTOPIC_NEW:
150 return $this->newEvent();
151 case self::SUBTOPIC_EDIT:
152 return $this->editEvent();
153 case self::SUBTOPIC_DELETE:
154 return $this->deleteEvent();
155 case self::SUBTOPIC_EXPORT:
156 return $this->exportEvent();

158 break;
159 }
160 break;
161 default:
162 $this->session->close();
163 break;
164 }
165 return self::grumble($this->l->t("unknown service requested: `%s/%s'.", [$topic, $subTopic]));
