165 private function registerPreCommitAction(Entity $entity, bool $remove):void
166 {
167 $entityId = self::entityId($entity);
168 if (!empty($this->preCommitActions[$entityId])) {
169 return;
170 }
172 $cloudConfig = $this->appContainer->get(IConfig::class);
173
174 $appName = $this->appContainer->get('appName');
175 $orchestraGroupId = $cloudConfig->getAppValue($appName, ConfigService::USER_GROUP_KEY);
176
177 if (empty($orchestraGroupId)) {
178 return;
179 }
180
182 $encryptionService = $this->appContainer->get(EncryptionService::class);
183 $executiveBoardProjectId = $encryptionService->getConfigValue(ConfigService::EXECUTIVE_BOARD_PROJECT_ID_KEY, -1);
184 if ($entity->getProject->getId() != $executiveBoardProjectId) {
185 // nothing to do
186 return;
187 }
188
190 $groupManager = $this->appContainer->get(IGroupManager::class);
191
192 $managementGroupId = $orchestraGroupId . AuthorizationService::MANAGEMENT_GROUP_SUFFIX;
194 $managementGroup = $groupManager->get($managementGroupId);
196 $orchestraGroup = $groupManager->get($orchestraGroupId);
197
199 $userManager = $this->appContainer->get(IUserManager::class);
200 $userId = $entity->getMusician()->getUserIdSlug();
201 $user = $userManager->get($userId);
202
203 $closureArguments = compact([
204 'userId',
205 'user',
206 'managementGroup',
207 'orchestraGroup',
208 'groupManager',
209 'userManager',
210 ]);
211
212 $this->preCommitActions[$entityId] = new GenericUndoable(
213 function() use ($closureArguments) {
214 extract($closureArguments);
215 $previousStatus = [
216 'orchestraGroup' => $orchestraGroup->inGroup($user),
217 'managementGroup' => $managementGroup->inGroup($user),
218 ];
219 if ($remove) {
220 $managementGroup->removeUser($user);
221 } else {
222 $managementGroup->addUser($user);
223 $orchestraGroup->addUser($user);
224 }
225 return $previousStatus;
226 },
227 function(array $previousStatus) use ($closureArguments) {
228 extract($closureArguments);
229 foreach ($previousStatus as $groupKey => $status) {
231 $group = $closureArguments[$groupKey];
232 if ($status && !$group->inGroup($user)) {
233 $group->addUser($user);
234 }
235 if (!$status && $group->inGroup($user)) {
236 $group->removeUser($user);
237 }
238 }
239 },
240 );
241 }
