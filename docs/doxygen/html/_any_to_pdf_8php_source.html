100 public function convertData(string $data, ?string $mimeType = null):string
101 {
102 if (empty($mimeType) || $mimeType == 'application/octet-stream') {
103 $mimeType = $this->mimeTypeDetector->detectString($data);
104 }
105
106 $converters = self::CONVERTERS[$mimeType] ?? self::CONVERTERS['default'];
107
108 foreach ($converters as $converter) {
109 if (!is_array($converter)) {
110 $converter = [ $converter ];
111 }
112
113 $convertedData = null;
114 foreach ($converter as $tryConverter) {
115 try {
116 $method = $tryConverter . 'Convert';
117 $convertedData = $this->$method($data);
118 break;
119 } catch (Throwable $t) {
120 $this->logException($t, 'Ignoring failed converter ' . $tryConverter);
121 }
122 }
123 if (empty($convertedData)) {
124 throw new RuntimeException($this->l->t('Converter "%1$s" has failed trying to convert mime-type "%2$s"', [ print_r($converter, true), $mimeType ]));
125 }
126 $data = $convertedData;
127 $convertedData = null;
128 }
129
130 return $data;
131 }

148 protected function unoconvConvert(string $data):string
149 {
150 $converterName = 'unoconv';
151 $converter = $this->findExecutable($converterName);
152 $retry = false;
153 do {
154 $process = new Process([
155 $converter,
156 '-f', 'pdf',
157 '--stdin', '--stdout',
158 '-e', 'ExportNotes=False'
159 ]);
160 $process->setInput($data);
161 try {
162 $process->run();
163 $retry = false;
164 } catch (Throwable $t) {
165 $this->logException($t);
166 $this->logError('RETRY');
167 $retry = true;
168 }
169 } while ($retry);
170
171 return $process->getOutput();

230 protected function tiff2pdfConvert(string $data):string
231 {
232 $converterName = 'tiff2pdf';
233 $converter = $this->findExecutable($converterName);
234 $inputFile = $this->tempManager->getTemporaryFile();
235 $outputFile = $this->tempManager->getTemporaryFile();
236 file_put_contents($inputFile, $data);
237
238 // As of mow tiff2pdf cannot write to stdout.
239 $process = new Process([
240 $converter,
241 '-p', $this->paperSize,
242 '-o', $outputFile,
243 $inputFile,
244 ]);
245 $process->run();
246 $data = file_get_contents($outputFile);
247
248 unlink($inputFile);
249 unlink($outputFile);
250 return $data;

263 protected function findExecutable(string $program)
264 {
265 if (empty($this->executables[$program])) {
266 $executable = $this->executableFinder->find($program);
267 if (empty($executable)) {
268 $this->executables[$program] = [
269 'exception' => throw new Exceptions\EnduserNotificationException($this->l->t('Please install the "%s" program on the server.', $program)),
270 'path' => null,
271 ];
272 } else {
273 $this->executables[$program] = [
274 'exception' => null,
275 'path' => $executable,
276 ];
277 }
278 }
279 if (empty($this->executables[$program]['path'])) {
280 throw $this->executables[$program]['exception'];
281 }
282 return $this->executables[$program]['path'];
283 }
