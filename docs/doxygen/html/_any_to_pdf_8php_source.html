117 public function convertData(string $data, ?string $mimeType = null):string
118 {
119 if (empty($mimeType) || $mimeType == 'application/octet-stream') {
120 $mimeType = $this->mimeTypeDetector->detectString($data);
121 }
122
123 $converters = self::CONVERTERS[$mimeType] ?? self::CONVERTERS['default'];
124
125 foreach ($converters as $converter) {
126 if (!is_array($converter)) {
127 $converter = [ $converter ];
128 }
129
130 $convertedData = null;
131 foreach ($converter as $tryConverter) {
132 try {
133 $method = $tryConverter . 'Convert';
134 $convertedData = $this->$method($data);
135 break;
136 } catch (Throwable $t) {
137 $this->logException($t, 'Ignoring failed converter ' . $tryConverter);
138 }
139 }
140 if (empty($convertedData)) {
141 throw new RuntimeException($this->l->t('Converter "%1$s" has failed trying to convert mime-type "%2$s"', [ print_r($converter, true), $mimeType ]));
142 }
143 $data = $convertedData;
144 $convertedData = null;
145 }
146
147 return $data;
148 }

165 protected function unoconvConvert(string $data):string
166 {
167 $converterName = 'unoconv';
168 $converter = $this->findExecutable($converterName);
169 $retry = false;
170 do {
171 $process = new Process([
172 $converter,
173 '-f', 'pdf',
174 '--stdin', '--stdout',
175 '-e', 'ExportNotes=False'
176 ]);
177 $process->setInput($data);
178 try {
179 $process->run();
180 $retry = false;
181 } catch (Throwable $t) {
182 $this->logException($t);
183 $this->logError('RETRY');
184 $retry = true;
185 }
186 } while ($retry);
187
188 return $process->getOutput();

251 $inputFile = $this->tempManager->getTemporaryFile();
252 $outputFile = $this->tempManager->getTemporaryFile();

261 ]);
262 $process->run();
263 $data = file_get_contents($outputFile);
264
265 unlink($inputFile);
266 unlink($outputFile);

280 protected function findExecutable(string $program)
281 {
282 if (empty($this->executables[$program])) {
283 $executable = $this->executableFinder->find($program);
284 if (empty($executable)) {
285 $this->executables[$program] = [
286 'exception' => throw new Exceptions\EnduserNotificationException($this->l->t('Please install the "%s" program on the server.', $program)),
287 'path' => null,
288 ];
289 } else {
290 $this->executables[$program] = [
291 'exception' => null,
292 'path' => $executable,
293 ];
294 }
295 }
296 if (empty($this->executables[$program]['path'])) {
297 throw $this->executables[$program]['exception'];
298 }
299 return $this->executables[$program]['path'];
300 }
