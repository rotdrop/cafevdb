84 public function validate(string $topic):DataResponse
85 {
86 $projectValues = $this->parameterService->getPrefixParams($this->pme->cgiDataName());
87 switch ($topic) {
88 case 'name':
89 $required = [
90 'year' => $this->l->t("project-year"),
91 'name' => $this->l->t("project-name"),
92 'type' => $this->l->t("project-kind"),
93 ];
94 foreach ($required as $key => $subject) {
95 if (empty($projectValues[$key])) {
96 return self::grumble($this->l->t("The %s must not be empty.", [$subject]));
97 }
98 }
99 $control = $this->parameterService->getParam('control', 'name');
100 $record = $this->pme->getCGIRecordId();
101 $projectId = $record['id']??null;
102 $projectName = $projectValues['name'];
103 $projectYear = $projectValues['year'];
104 $attachYear = !empty($projectValues['type']) && $projectValues['type'] == ProjectType::TEMPORARY;
105
106 $infoMessage = "";
107 switch ($control) {
108 case "submit":
109 case "name":
110 // No whitespace, s.v.p., and CamelCase
111 $origName = $projectName;
112
114 $projectService = $this->di(ProjectService::class);
115 $projectName = $projectService->sanitizeName($projectName);
116 if ($origName != $projectName) {
117 $infoMessage .= $this->l->t(
118 'The project name has been simplified from "%s" to "%s".',
119 [ $origName, $projectName ]
120 );
121 }
122 $matches = [];
123 // Get the year from the name, if set
124 if (preg_match('/^(.*\D)?(\d{4})$/', $projectName, $matches) == 1) {
125 $projectName = $matches[1];
126 if ($control != "submit" && $attachYear) {
127 // the year-control wins when submitting the form
128 $projectYear = $matches[2];
129 }
130 if ($projectName == "") {
131 return self::grumble($this->l->t("The project-name must not only consist of the year-number."));
132 }
133 } elseif ($projectName == "") {
134 return self::grumble($this->l->t("No project-name given."));
135 }
136 if (mb_strlen($projectName) > Renderer::NAME_LENGTH_MAX) {
137 return self::grumble($this->l->t(
138 "The project-name is too long, ".
139 "please use something less than %d characters ".
140 "(excluding the attached year). Thanks",
141 [ Renderer::NAME_LENGTH_MAX ]
142 ));
143 }
144 // fallthrough
145 case "year":
146 if ($projectYear == "") {
147 return self::grumble($this->l->t("No project-year given."));
148 }
149 if (preg_match('/^\d{4}$/', $projectYear) !== 1) {
150 return self::grumble($this->l->t("The project-year has to consist of four digits, e.g. ``1984''."));
151 }
152
153 // Strip the year from the name and replace with the given year
154 $origName = $origName??$projectName;
155 $strippedName = preg_replace('/\d+$/', '', $projectName);
156 if ($attachYear) {
157 $projectName = $strippedName . $projectYear;
158 if ($projectName != $origName) {
159 $infoMessage .= $this->l->t("The year %s has been appended to the project-slug %s.", [ $projectYear, $strippedName ]);
160 }
161 }
162 // Project name may be empty at this point. Why not
163 break;
164 default:
165 return self::grumble($this->l->t('Unknown Request'));
166 }
167
168 $repository = $this->getDatabaseRepository(Entities\Project::class);
169 $projects = $repository->shortDescription();
170 foreach ($projects['projects'] as $id => $nameYear) {
171 if ($id != $projectId && $nameYear['name'] == $projectName && $nameYear['year'] == $projectYear) {
172 return self::grumble($this->l->t(
173 'A project with the name "%1$s" already exists in the year %2$s with the id %3$d (new: %4$d). Please choose a different name or year.',
174 [ $projectName, $projectYear, $id, $projectId ]
175 ));
176 }
177 }
178 return self::dataResponse(['projectYear' => $projectYear,
179 'projectName' => $projectName,
180 'message' => $infoMessage ]);
181 default:
182 break;
183 }
184 return self::grumble($this->l->t('Unknown Request'));
185 }

196 public function changeInstrumentation(string $instruments, string $voices):DataResponse
197 {
198 $instrumentsKey = str_replace('[]', '', $instruments);
199 $instruments = array_filter($this->parameterService[$instrumentsKey]??[]);
200
201 $voicesKey = str_replace('[]', '', $voices);
202 $voices = array_filter($this->parameterService[$voicesKey]??[]);
203
204 $instrumentInfo =
205 $this->getDatabaseRepository(Entities\Instrument::class)->describeALL();
206
207 // convert voices array to instrumentId => [ voices ]
208 $voicesByInstrument = Util::explodeIndexedMulti(implode(',', $voices));
209
210 $voicesSelectArray = [];
211 foreach ($instruments as $instrument) {
212 $highestConfigured = empty($voicesByInstrument[$instrument]) ? 0 : max($voicesByInstrument[$instrument]);
213 $highest = max(Renderer::NUM_VOICES_MIN, $highestConfigured + Renderer::NUM_VOICES_EXTRA);
214 for ($i = 1; $i <= $highest+1; ++$i) {
215 $voiceIndicator = ($i <= $highest) ? $i : '?';
216 $value = $instrument . Renderer::JOIN_KEY_SEP . $voiceIndicator;
217 $voiceOption = [
218 'value' => $value,
219 'name' => $instrumentInfo['byId'][$instrument] . ' ' . $voiceIndicator,
220 'group' => $instrumentInfo['idGroups'][$instrument],
221 ];
222 $voicesSelectArray[] = $voiceOption;
223 // make sure there are no holes, all voices up to the maximum are
224 // selected
225 if ($i <= $highestConfigured) {
226 $voices[] = $value;
227 }
228 }
229 }
230
231 $voicesSelectOptions = PageNavigation::selectOptions($voicesSelectArray, $voices);
232
233 return self::dataResponse([ 'voices' => $voicesSelectOptions ]);
234 }

295 if (!$force) {

303 'message' => $this->l->t('Successfully deleted "%s".', $listId),
304 'list_id' => $listId,

306 'status' => 'unset',
307 'l10nStatus' => $this->l->t('unset'),
308 ]);

310 case self::LIST_OPERATION_SUBSCRIBE:
312 $listsService = $this->di(MailingListsService::class);
314 $projectService = $this->di(ProjectService::class);
316 $project = $projectService->findById($projectId);
317 $listId = $project->getMailingListId();
318
319 $failures = [];
320 $newCount = 0;
321 $keptCount = 0;
322 $noEmailCount = 0;
323 $notConfirmedCount = 0;
325 foreach ($project->getParticipants() as $participant) {
326 if (!$participant->getRegistration()) {
327 ++$notConfirmedCount;
328 continue; // only subscribe confirmed participants
329 }
330 $email = $participant->getMusician()->getEmail();
331 if (empty($email)) {
332 ++$noEmailCount;
333 continue; // after all, it is a mailing list ...
334 }
335 $displayName = $participant->getPublicName(firstNameFirst: true);
336 try {
337 $result = $projectService->ensureMailingListSubscription($participant);
338 $newCount += (int)$result;
339 $keptCount += (int)!$result;
340 } catch (\Throwable $t) {
341 $failures[] = [
342 'email' => $email,
343 'displayName' => $displayName,
344 'message' => $t->getMessage(),
345 ];
346 }
347 }
348
349 $messages = [
350 $this->l->t('%1$d new subscriptions, %2$d existing subscriptions have been kept.', [ $newCount, $keptCount ]),
351 ];
352 if ($notConfirmedCount > 0) {
353 $messages[] = $this->l->n(
354 '%n person has not been subscribed because of unconfirmed participation.',
355 '%n persons have not been subscribed because their participation is not confirmed.',
356 $notConfirmedCount
357 );
358 }
359 if ($noEmailCount > 0) {
360 $messages[] = $this->l->t('%d persons have not been subscribed because we do not have their email-address.', $noEmailCount);
361 }
362 if (count($failures) > 0) {
363 $messages[] = $this->l->t('%d subscription requests have failed.', [ count($failures) ]);
364 }
365 foreach ($failures as $failure) {
366 $messages[] = $this->l->t('Error for "%1$s <%2$s>": %3$s', [
367 $failure['displayName'], $failure['email'], $failure['message'],
368 ]);
369 }
370 return self::dataResponse([
371 'message' => $messages,
372 'listId' => $listId,
373 'status' => 'unchanged',
374 ]);
375 }
376 return self::grumble($this->l->t('Unknown Request: "%s".', $operation));

416 public function get(int $projectId, string $topic = '', string $subTopic = ''):DataResponse
417 {
419 $projectService = $this->di(ProjectService::class);
421 $project = $projectService->findById($projectId);
422 if (empty($project)) {
423 return self::grumble($this->l->t('Unable to find project with id "%d".', $projectId));
424 }
425 switch ($topic) {
426 case '':
427 return self::dataResponse($this->flattenProject($project));
428 case self::GET_PARTICIPANT_FIELDS:
429 $multiplicity = $this->parameterService->getParam('multiplicity');
430 $type = $this->parameterService->getParam('type');
431 $data = [];
433 foreach ($project->getParticipantFields() as $field) {
434 $fieldType = $field->getDataType();
435 $fieldMultiplicity = $field->getMultiplicity();
436 if (!empty($type) && $fieldType != $type) {
437 continue;
438 }
439 if (!empty($multiplicity) && $fieldMultiplicity != $multiplicity) {
440 continue;
441 }
442 $data[] = [
443 'id' => $field->getId(),
444 'name' => $field->getName(),
445 'untranslatedName' => $field->getUntranslatedName(),

449 ];

466 usort($data, fn($a, $b) => strcmp($a['publicName'], $b['publicName']));
467 return self::dataResponse($data);

469 switch ($subTopic) {
470 case ProjectService::FOLDER_TYPE_DOWNLOADS:
471 return self::dataResponse([
472 $projectService->ensureDownloadsShare($project, noCreate: true),

476 return self::grumble($this->l->t('Unknown share type "%s".', $subTopic));

478 break;
479 case self::GET_PROJECT_FOLDER:
480 switch ($subTopic) {
481 case '':
482 case 'all':
483 return self::dataResponse($projectService->getProjectFolder($project));
484 case ProjectService::FOLDER_TYPE_PROJECT:
485 case ProjectService::FOLDER_TYPE_BALANCE:
486 case ProjectService::FOLDER_TYPE_PARTICIPANTS:
487 case ProjectService::FOLDER_TYPE_POSTERS:
488 case ProjectService::FOLDER_TYPE_DOWNLOADS:

497 }
