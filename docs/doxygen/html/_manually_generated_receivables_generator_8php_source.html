149 $added = 0;
150 $removed = 0;

152 $fieldDataOptions = $this->serviceFeeField->getSelectableOptions();
153
154 // First step: subscribe the participant to all labelled options. This
155 // must be, otherwise the UI will be a nightmare.
157 foreach ($fieldDataOptions as $fieldOption) {
158 if (!empty($fieldOption->getLabel())) {
159 $added += $this->subscribeParticipant($participant, $fieldOption);
160 }
161 }
162
163 $participantFieldsData = $participant->getParticipantFieldsData();
165 $emptyFieldData = $participantFieldsData->matching(self::criteriaWhere([
166 'field' => $this->serviceFeeField,
167 '&(|optionValue' => null,
168 'optionValue' => '',
169 ')' => null,
170 ]));
171
172 // keep options with non-empty label.
173 $emptyFieldData = $emptyFieldData->filter(
174 fn(Entities\ProjectParticipantFieldDatum $datum) => empty($datum->getDataOption()->getLabel()));
175
176 if (!empty($receivable)) {
177 // dedicated receivables update ... just update this one
178
180 $fieldData = $receivable->getMusicianFieldData($participant->getMusician());
181 foreach ($fieldData as $fieldDatum) {
182 if (empty($fieldDatum->getOptionValue()) && $emptyFieldData->count() > 1) {
183 $this->remove($fieldDatum);
184 $this->remove($fieldDatum);
185 $participantFieldsData->removeElement($fieldDatum);
186 $participant->getMusician()->getProjectParticipantFieldsData()->removeElement($fieldDatum);
187 $fieldDatum->getDataOption()->getFieldData()->removeElement($fieldDatum);
188 $emptyFieldData->removeElement($fieldDatum);
189 ++$removed;
190 }
191 }
192

195 'removed' => $removed,
196 'changed' => 0, // never
197 'skipped' => 0,
198 'notices' => [],
199 ];
200 }
201
202 // then remove all but one empty option
203 while ($emptyFieldData->count() > 1) {
205 $fieldDatum = $emptyFieldData->first();
206 $this->remove($fieldDatum);
207 $this->remove($fieldDatum);
208 $participantFieldsData->removeElement($fieldDatum);

210 $fieldDatum->getDataOption()->getFieldData()->removeElement($fieldDatum);

213 }
214
215 $this->logInfo('Empty field data count: ' . $emptyFieldData->count());
216
217 if ($emptyFieldData->isEmpty()) {
218
219 $this->logInfo('No empty option data, generate one empty item');
220

222
223 // if no empty option is there, add one
224 //

245 $this->logInfo('Option ' . (string)$fieldOption->getKey() . ' already used by musician ' . $participant->getMusician()->getPublicName());
246 }
247 }
248

260 $this->persist($emptyFieldOption);
261 $this->flush();

264 // use it, add empty data to musician
265
266 // use it, add empty data to musician
