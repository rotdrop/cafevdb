110 public static function slugifyFileName(string $base, string $userIdSlug, bool $ignoreExtension = false):string
111 {
112 if ($ignoreExtension) {
113 $fileName = $base;
114 $extension = '';
115 } else {
116 $pathInfo = pathinfo($base);
117 $fileName = $pathInfo['filename'];
118 $extension = empty($pathInfo['extension']) ? '' : '.' . $pathInfo['extension'];
119 }
120 $fileName = str_replace('.', '-', $fileName);
121 return $fileName . self::getSlugPostfix($userIdSlug) . $extension;
122 }

146 public static function unSlugifyFileName(string $path, string $userIdSlug, bool $keepExtension = true):string
147 {
148 $pathInfo = pathinfo($path);
149 $fileName = $pathInfo['filename'];
150 $postfix = self::getSlugPostfix($userIdSlug);
151 $extension = empty($pathInfo['extension']) || !$keepExtension ? '' : '.' . $pathInfo['extension'];
152 if (str_ends_with($fileName, $postfix)) {
153 return substr($fileName, 0, -strlen($postfix)) . $extension;
154 }
155 return $fileName . $extension;
156 }

195 protected function impersonateMusician(Entities\Musician $musician):void
196 {
197 $financialArtifacts = ProjectParticipantFieldsService::participantMonetaryObligations($musician);
198 if ($financialArtifacts['sum'] !== $financialArtifacts['received']) {
199 throw new Exceptions\EnduserNotificationException(
200 $this->l->t(
201 'Musician "%1$s" cannot be removed because there are unbalanced financial obligations totals/payed/remaining = %$2d/%3d/%3d.'
202 . ' A negative remaining amount means the musician still needs to be refund by the orchestra.', [
203 $musician->getPublicName(),
204 $financialArtifacts['sum'],
205 $financialArtifacts['received'],
206 $financialArtifacts['sum'] - $financialArtifacts['received'],
207 ]
208 )
209 );
210 }
211 if ($musician->getPayments()->count() > 0) {
212 throw new Exceptions\EndUserNotificationException(
213 $this->l->t('Musician "%s" cannot be removed because there are financial records which depend on this musician.', $musician->getPublicName()));
214 }
215
217 $participantFieldsService = $this->di(ProjectParticipantFieldsService::class);
218
220 foreach ($musician->getProjectParticipantFieldsData() as $participantDatum) {
221 // cleanup as required, in particular the file-system
222 $removeEntity = false;
224 $participantField = $participantDatum->getField();
225 switch ($participantField->getDataType()) {
226 case FieldDataType::CLOUD_FILE:
228 $cloudFile = $participantFieldsService->getEffectiveFieldDatum($participantDatum);
229 $cloudFile->delete();
230 $removeEntity = true;
231 break;
232 case FieldDataType::CLOUD_FOLDER:
234 $cloudFolder = $participantFieldsService->getEffectiveFieldDatum($participantDatum);
235 $cloudFolder->delete();
236 $removeEntity = true;
237 break;
238 case FieldDataType::DB_FILE:
240 $dbFile = $participantFieldsService->getEffectiveFieldDatum($participantDatum);
241 $this->remove($dbFile, true);
242 $removeEntity = true;
243 break;
244 case FieldDataType::RECEIVABLES:
245 case FieldDataType::LIABILITIES:
246 $dbFile = $participantDatum->getSupportingDocument();
247 $this->remove($dbFile, true);
248 $participantDatum->setSupportingDocument(null);
249 $this->persist($participantDatum); // ??? needed ???
250 break;
251 default:
252 break;
253 }
254 switch ($participantField->getMultiplicity()) {
255 case FieldMultiplicity::SIMPLE:
256 $removeEntity = true;
257 break;
258 case FieldMultiplicity::RECURRING:
259 $removeEntity = true;
260 break;
261 default:
262 break;
263 }
264 if ($removeEntity) {
265 $musician->getProjectParticipantFieldsData()->removeElement($participantDatum);
266 $participantDatum->getDataOption()->getFieldData()->removeElement($participantDatum);
267 $participantDatum->getField()->getFieldData()->removeElement($participantDatum);
268 $this->remove($participantDatum, true);

319 public function deleteMusician(Entities\Musician $musician):void
320 {
321 // Unsubscribe the musician from the mailing-list
322 $list = $this->getConfigValue('announcementsMailingList');
323 try {
324 $this->listsService->unsubscribe($list, $musician->getEmail());
325 } catch (\Throwable $t) {
326 // for now we ignore any mailing list related errors in order not to
327 // annoy the persons who are desperately trying to add persons to
328 // the data-base. We should, however, find a notification channel
329 // for end-user messages.
330 $this->logException(
331 $t,
332 'Failed to unsubscribe' . $musician->getPublicName(firstNameFirst: true)
333 . ' from the mailing-list ' . $list);
334 }
335
336 if (empty($musician->getDeleted())) {
337 // $this->remove($musician, true); // this should be soft-delete
338 // for now skip the soft-deleteable cascade and just set the deleted-data manually
339 $musician->setDeleted('now');
340 }
341 if ($musician->getPayments()->count() == 0) {
342 if ($musician->getProjectParticipation()->isEmpty()) {
343 $this->logInfo($musician->getPublicName() . ' is unused, issuing hard-delete');
344 foreach ($musician->getInstruments() as $instrument) {
345 $this->remove($instrument, true);
346 $this->remove($instrument, true);
347 }
348 $this->remove($musician, true); // this should be hard-delete
349 } else {
350 $this->logInfo($musician->getPublicName() . ' is used, trying to impersonate');
351 // Perhaps: remove all personal data and keep a dummy record as
352 // project-participant, until finally all projects have been deleted and
353 // cleaned up.
354 $this->impersonateMusician($musician);
355 }
356 } else {
357 $this->logInfo($musician->getPublicName() . ' has recorded payments, just marking the person as deleted.');
358 $this->flush();
359 }
