116 public static function slugifyFileName(string $base, string $userIdSlug, bool $ignoreExtension = false):string
117 {
118 if ($ignoreExtension) {
119 $fileName = $base;
120 $extension = '';
121 } else {
122 $pathInfo = pathinfo($base);
123 $fileName = $pathInfo['filename'];
124 $extension = empty($pathInfo['extension']) ? '' : '.' . $pathInfo['extension'];
125 }
126 $fileName = str_replace('.', '-', $fileName);
127 return $fileName . self::getSlugPostfix($userIdSlug) . $extension;
128 }

152 public static function unSlugifyFileName(string $path, string $userIdSlug, bool $keepExtension = true):string
153 {
154 $pathInfo = pathinfo($path);
155 $fileName = $pathInfo['filename'];
156 $postfix = self::getSlugPostfix($userIdSlug);
157 $extension = empty($pathInfo['extension']) || !$keepExtension ? '' : '.' . $pathInfo['extension'];
158 if (str_ends_with($fileName, $postfix)) {
159 return substr($fileName, 0, -strlen($postfix)) . $extension;
160 }
161 return $fileName . $extension;
162 }

201 protected function impersonateMusician(Entities\Musician $musician):void
202 {
203 $financialArtifacts = ProjectParticipantFieldsService::participantMonetaryObligations($musician);
204 if ($financialArtifacts['sum'] !== $financialArtifacts['received']) {
205 throw new Exceptions\EnduserNotificationException(
206 $this->l->t(
207 'Musician "%1$s" cannot be removed because there are unbalanced financial obligations totals/payed/remaining = %$2d/%3d/%3d.'
208 . ' A negative remaining amount means the musician still needs to be refund by the orchestra.', [
209 $musician->getPublicName(),
210 $financialArtifacts['sum'],
211 $financialArtifacts['received'],
212 $financialArtifacts['sum'] - $financialArtifacts['received'],
213 ]
214 )
215 );
216 }
217 if ($musician->getPayments()->count() > 0) {
218 throw new Exceptions\EndUserNotificationException(
219 $this->l->t('Musician "%s" cannot be removed because there are financial records which depend on this musician.', $musician->getPublicName()));
220 }
221
223 $participantFieldsService = $this->di(ProjectParticipantFieldsService::class);
224
226 foreach ($musician->getProjectParticipantFieldsData() as $participantDatum) {
227 // cleanup as required, in particular the file-system
228 $removeEntity = false;
230 $participantField = $participantDatum->getField();
231 switch ($participantField->getDataType()) {
232 case FieldDataType::CLOUD_FILE:
234 $cloudFile = $participantFieldsService->getEffectiveFieldDatum($participantDatum);
235 $cloudFile->delete();
236 $removeEntity = true;
237 break;
238 case FieldDataType::CLOUD_FOLDER:
240 $cloudFolder = $participantFieldsService->getEffectiveFieldDatum($participantDatum);
241 $cloudFolder->delete();
242 $removeEntity = true;
243 break;
244 case FieldDataType::DB_FILE:
246 $dbFile = $participantFieldsService->getEffectiveFieldDatum($participantDatum);
247 $this->remove($dbFile, true);
248 $removeEntity = true;
249 break;
250 case FieldDataType::RECEIVABLES:
251 case FieldDataType::LIABILITIES:
252 $dbFile = $participantDatum->getSupportingDocument();
253 $this->remove($dbFile, true);
254 $participantDatum->setSupportingDocument(null);
255 $this->persist($participantDatum); // ??? needed ???
256 break;
257 default:
258 break;
259 }
260 switch ($participantField->getMultiplicity()) {
261 case FieldMultiplicity::SIMPLE:
262 $removeEntity = true;
263 break;
264 case FieldMultiplicity::RECURRING:
265 $removeEntity = true;
266 break;
267 default:
268 break;
269 }
270 if ($removeEntity) {
271 $musician->getProjectParticipantFieldsData()->removeElement($participantDatum);
272 $participantDatum->getDataOption()->getFieldData()->removeElement($participantDatum);
273 $participantDatum->getField()->getFieldData()->removeElement($participantDatum);

325 public function deleteMusician(Entities\Musician $musician):void
326 {
327 // Unsubscribe the musician from the mailing-list
328 $list = $this->getConfigValue('announcementsMailingList');
329 try {
330 $this->listsService->unsubscribe($list, $musician->getEmail());
331 } catch (\Throwable $t) {
332 // for now we ignore any mailing list related errors in order not to
333 // annoy the persons who are desperately trying to add persons to
334 // the data-base. We should, however, find a notification channel
335 // for end-user messages.
336 $this->logException(
337 $t,
338 'Failed to unsubscribe' . $musician->getPublicName(firstNameFirst: true)
339 . ' from the mailing-list ' . $list);
340 }
341
342 if (empty($musician->getDeleted())) {
343 // $this->remove($musician, true); // this should be soft-delete
344 // for now skip the soft-deleteable cascade and just set the deleted-data manually
345 $musician->setDeleted('now');
346 }
347 if ($musician->getPayments()->count() == 0) {
348 if ($musician->getProjectParticipation()->isEmpty()) {
349 $this->logInfo($musician->getPublicName() . ' is unused, issuing hard-delete');
350 foreach ($musician->getInstruments() as $instrument) {
351 $this->remove($instrument, true);
352 $this->remove($instrument, true);
353 }
354 $this->remove($musician, true); // this should be hard-delete
355 } else {
356 $this->logInfo($musician->getPublicName() . ' is used, trying to impersonate');
357 // Perhaps: remove all personal data and keep a dummy record as
358 // project-participant, until finally all projects have been deleted and
359 // cleaned up.
360 $this->impersonateMusician($musician);
361 }
362 } else {
363 $this->logInfo($musician->getPublicName() . ' has recorded payments, just marking the person as deleted.');
364 $this->flush();
365 }
