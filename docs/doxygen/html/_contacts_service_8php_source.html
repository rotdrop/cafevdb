133 public function emailContacts():array
134 {
136 $musicianCardBackend = $this->di(MusicianCardBackend::class);
137 $musiciansKey = $musicianCardBackend->getURI();
138 $result = [];
139 $addressBooks = $this->contactsManager->getUserAddressBooks();

142 if ($addressBook->getKey() == $musiciansKey) {
143 continue;
144 }
145 $bookName = $addressBook->getDisplayName();
146 $contacts = $addressBook->search('', [ 'FN', 'EMAIL' ], [] /* options */);
147 foreach ($contacts as $contact) {
148 // $this->logInfo('CONTACT: ' . print_r($contact, true));
149 $uid = $contact['UID'];
150 $fullName = $contact['FN']??'';
151 $emails = $contact['EMAIL']??null;
152 if (empty($emails)) {
153 continue;
154 }
155 if (!is_array($emails)) {
156 $emails = [ $emails ];
157 }
158 $theseContacts = [];
159 foreach ($emails as $email) {
160 if (empty($email)) {
161 continue;
162 }
163 $theseContacts[] = [
164 'uid' => $uid,
165 'email' => $email,
166 'name' => $fullName,
167 'addressBook' => $bookName,
168 ];
169 }
170 usort($theseContacts, function($a, $b) {
171 $aName = $a['name'] != '' ? $a['name'] : $a['email'];
172 $bName = $b['name'] != '' ? $b['name'] : $b['email'];
173 return strcmp($aName, $bName);
174 });
175 $result = array_merge($result, $theseContacts);
176 }
177 }
178 return $result;

219 public function flattenVCard(string $cardUri, VCard $vCard, bool $withTypes = true):array
220 {
221 $result = [
222 'URI' => $cardUri,
223 ];
224 foreach ($vCard->children() as $property) {

226 if ($property->getValueType() === 'BINARY' && $this->getTypeFromProperty($property)) {
227 $uri = 'data:image/'.strtolower($property['TYPE']).';base64,'.$property->getRawMimeDirValue();
228 $result[$property->name] = 'VALUE=uri:' . $uri;
229 } elseif ($property->getValueType() === 'URI') {
230 $result[$property->name] = 'VALUE=uri:' . $property->getValue();
231 } else {
232 $result[$property->name] = $property->getValue();
233 }
234 } elseif (in_array($property->name, ['URL', 'GEO', 'CLOUD', 'ADR', 'EMAIL', 'IMPP', 'TEL', 'X-SOCIALPROFILE', 'RELATED', 'LANG', 'X-ADDRESSBOOKSERVER-MEMBER'])) {
235 if (!isset($result[$property->name])) {
236 $result[$property->name] = [];
237 }
238
239 $type = $this->getTypeFromProperty($property);
240 if ($withTypes) {
241 $result[$property->name][] = [
242 'type' => $type,
243 'value' => $property->getValue()
244 ];
245 } else {
246 $result[$property->name][] = $property->getValue();
247 }
248 } else {
249 $result[$property->name] = $property->getValue();
250 }
251 }
252 return $result;
