162 public function usage(mixed $identifier, bool $brief = false):?array

164 $selects = [

173 'MAX(p.dateOfReceipt) AS dateOfLastReceipt',
174 'MAX(t.created) AS dateOfLastCreatedTransaction',
175 'MAX(t.submitDate) AS dateOfLastSubmittedDebitNote',
176 // 'IF(p.dateOfReceipt = MAX(t.dueDate), p.debitMessageId, NULL) AS debitNoteLastNotified',
177 'CASE WHEN p.dateOfReceipt = MAX(t.dueDate) THEN p.notificationMessageId ELSE \'\' END AS debitNoteLastNotified',
178 ]);
179 }
180
181 // the what part ...
182 $qb = $this->createQueryBuilder('m')
183 ->select(implode(',', $selects))
184 ->leftJoin('m.payments', 'p')
185 ->leftJoin('p.sepaTransaction', 't')
186 ->groupBy('m.mandateReference');
187
188 // the where part ...
189 if (is_string($identifier)) { // assume it is the mandate-reference
190 $qb->where('m.mandateReference = :reference')
191 ->setParameter('reference', $identifier);
192 } elseif (is_array($identifier) || ($identifier instanceof Entities\SepaDebitMandate)) {
193 $qb->where('m.project = :project')
194 ->andWhere('m.musician = :musician')
195 ->andWhere('m.sequence = :sequence')
196 ->setParameter('project', $identifier['project'])
197 ->setParameter('musician', $identifier['musician'])
198 ->setParameter('sequence', $identifier['sequence']);
199 } else {
200 throw new Exception('Mandate identifier is '.(empty($identifier) ? 'empty' : 'unsupported'));
201 }
