304 public function updateUserSqlViews(?string $dataBaseName):void
305 {
306 $cloudDbUser = $this->checkAndGetCloudDbUser();
307 $currentStatement = null;
308 try {
309 foreach (self::USER_SQL_VIEWS as $name => $sql) {
310 $viewName = $this->viewName($dataBaseName, self::USER_SQL_PREFIX, $name);
311 $statements = [
312 sprintf($sql, $viewName, $this->appName),
313 sprintf(self::GRANT_SELECT, $viewName, $cloudDbUser),
314 ];
315 if ($name === 'User') {
316 // allow changing the password from the cloud
317 $statements[] = sprintf(self::GRANT_FIELD_UPDATE, $viewName, $cloudDbUser, 'password');
318 // allow deactivation of users from the cloud
319 $statements[] = sprintf(self::GRANT_FIELD_UPDATE, $viewName, $cloudDbUser, 'inactive');
320 }
321 foreach ($statements as $sql) {
322 $currentStatement = $sql;
323 $this->logDebug('SQL ' . $currentStatement);
324 $this->connection->prepare($sql)->execute();
325 }
326 }
327 } catch (\Throwable $t) {
328 throw new Exceptions\DatabaseCloudConnectorViewException(
329 $this->l->t('Unable to create or update the user-sql cloud-connector views: %s.', $currentStatement),
330 $t->getCode(),
331 $t
332 );
333 }

478 public function configureCloudUserBackend(bool $erase = false):array
479 {
481 $requestService = $this->appContainer->get(RequestService::class);
482
483 $configKeys = $this->cloudConfig->getAppKeys($this->appName);
484 $prefix = self::CLOUD_USER_BACKEND . ':';
485 $prefixLen = strlen($prefix);
486 $cloudUserBackendKeys = array_map(function($key) use ($prefixLen) {
487 return substr($key, $prefixLen);
488 }, array_filter($configKeys, function($key) use ($prefix) {
489 return str_starts_with($key, $prefix);
490 }));
491
492 $this->logDebug('USER SQL KEYS ' . print_r($cloudUserBackendKeys, true));
493
494 $cloudUserBackendParams = [];
495 foreach ($cloudUserBackendKeys as $cloudUserBackendKey) {
496 $cloudUserBackendValue = $erase ? '' : $this->cloudConfig->getAppValue($this->appName, $prefix . $cloudUserBackendKey);
497 if (preg_match('/%system:(\w+)%/', $cloudUserBackendValue, $matches)) {
498 $cloudUserBackendValue = $this->cloudConfig->getSystemValue($matches[1]);
499 }
500 // $this->cloudConfig->setAppValue(self::CLOUD_USER_BACKEND, $cloudUserBackendKey, $cloudUserBackendValue);
501 $cloudUserBackendParams[str_replace('.', '-', $cloudUserBackendKey)] = $cloudUserBackendValue;
502 }
503 $this->logInfo('USER SQL POST PARAMS ' . print_r($cloudUserBackendParams, true));
504
505 // try also to clear the cache after and before changing the configuration
506 $this->clearUserBackendCache();
507
508 $messages = [];
509
511 $requestService = $this->appContainer->get(RequestService::class);
512
513 $route = implode('.', [
514 self::CLOUD_USER_BACKEND,
515 'settings',
516 'saveProperties',
517 ]);
518 $result = $requestService->postToRoute($route, requestData: $cloudUserBackendParams, postType: RequestService::URL_ENCODED);
519 $messages[] = $result['message'] ?? $this->l->t('"%s" configuration may have succeeded.', self::CLOUD_USER_BACKEND);
520
521 // try also to clear the cache after and before changing the configuration
522 $this->clearUserBackendCache();
523
524 return $messages;
525 }
