124 if ($flush) {

137 $document = $yearFolder->addDocument($file)

140 $this->persist($document);
141 $yearFolder
142 ->setCreated(min($file->getCreated(), $yearFolder->getCreated()))
143 ->setUpdated(max($file->getUpdated(), $yearFolder->getUpdated()));
144
145 if ($flush) {
146 $this->flush();
147 $this->entityManager->commit();
148 }
149
150 } catch (Throwable $t) {

152 $this->entityManager->rollback();
