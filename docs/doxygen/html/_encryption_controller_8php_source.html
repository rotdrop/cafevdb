324 private function recryptForUser(string $userId):?string
325 {
326 $this->entityManager = $this->appContainer->get(EntityManager::class);
328 $musician = $this->getDatabaseRepository(Entities\Musician::class)->findByUserId($userId);
329 if (!empty($musician)) {
330 $accessToken = \random_bytes(Entities\MusicianRowAccessToken::HASH_LENGTH / 8);
331 $this->entityManager->beginTransaction();
332 try {
333 $this->keyService->setSharedPrivateValue($userId, self::ROW_ACCESS_TOKEN_KEY, $accessToken);
334 $tokenEntity = $musician->getRowAccessToken();
335 if (empty($tokenEntity)) {
336 $tokenEntity = new Entities\MusicianRowAccessToken($musician, $accessToken);
337 $this->persist($tokenEntity);
338 } else {
339 $tokenEntity->setAccessToken($accessToken);
340 }
341 $this->flush();
342 $this->entityManager->commit();
343 } catch (\Throwable $t) {
344 $this->entityManager->rollback();
345 $this->keyService->setSharedPrivateValue($userId, self::ROW_ACCESS_TOKEN_KEY, null);
346 $this->logException($t, 'Unable to set row access-token for user ' . $userId);
347 throw new OCS\OCSBadRequestException($this->l->t('Unable to set row access-token for user "%s".', $userId), $t);
348 }
349
350 // next we should try to recrypt all encrypted entities of the user ...
351 $encryptedEntities = [];
352 $encryptedEntities = array_merge($encryptedEntities, $musician->getSepaBankAccounts()->toArray());
354 foreach ($musician->getEncryptedFiles() as $encryptedFile) {
355 $encryptedEntities[] = $encryptedFile->getFileData();
356 }
357 try {
358 $this->entityManager->recryptEntityList($encryptedEntities);
359 } catch (\Throwable $t) {
360 $this->logException($t, 'Unable to recrypt encrypted data for user ' . $userId);
361 throw new OCS\OCSBadRequestException($this->l->t('Unable to recrypt encrypted data for user "%s".', $userId), $t);
362 }
363 }
364
365 $appEncryptionKey = null;
366
368 $authorizationService = $this->appContainer->get(AuthorizationService::class);
369 if ($authorizationService->authorized($userId)) {
370 // set encryption key for this user
372 $encryptionService = $this->appContainer->get(EncryptionService::class);
373 if (!$encryptionService->encryptionKeyValid()) {
374 throw new OCS\OCSBadRequestException($this->l->t('Encryption key is invalid'));
375 }
376 try {
377 $appEncryptionKey = $encryptionService->getAppEncryptionKey();
378 $encryptionService->setUserEncryptionKey($appEncryptionKey, $userId);
379 } catch (Exceptions\EncryptionException $e) {
380 throw new OCS\OCSBadRequestException($this->l->t('Unable to set app-encryption-key for user "%s".', $userId), $e);
381 }
382 }
383
384 return $appEncryptionKey;
385 }
