311 private function recryptForUser(string $userId):?string
312 {
313 $this->entityManager = $this->appContainer->get(EntityManager::class);
315 $musician = $this->getDatabaseRepository(Entities\Musician::class)->findByUserId($userId);
316 if (!empty($musician)) {
317 $accessToken = \random_bytes(Entities\MusicianRowAccessToken::HASH_LENGTH / 8);
318 $this->entityManager->beginTransaction();
319 try {
320 $this->keyService->setSharedPrivateValue($userId, self::ROW_ACCESS_TOKEN_KEY, $accessToken);
321 $tokenEntity = $musician->getRowAccessToken();
322 if (empty($tokenEntity)) {
323 $tokenEntity = new Entities\MusicianRowAccessToken($musician, $accessToken);
324 $this->persist($tokenEntity);
325 } else {
326 $tokenEntity->setAccessToken($accessToken)
327 ->setUserId($musician->getUserIdSlug());
328 }
329 $this->flush();
330 $this->entityManager->commit();
331 } catch (\Throwable $t) {
332 $this->entityManager->rollback();
333 $this->keyService->setSharedPrivateValue($userId, self::ROW_ACCESS_TOKEN_KEY, null);
334 $this->logException($t, 'Unable to set row access-token for user ' . $userId);
335 throw new OCS\OCSBadRequestException($this->l->t('Unable to set row access-token for user "%s".', $userId), $t);
336 }
337
338 // next we should try to recrypt all encrypted entities of the user ...
339 $encryptedEntities = [];
340 $encryptedEntities = array_merge($encryptedEntities, $musician->getSepaBankAccounts()->toArray());
342 foreach ($musician->getEncryptedFiles() as $encryptedFile) {
343 $encryptedEntities[] = $encryptedFile->getFileData();
344 }
345 try {
346 $this->entityManager->recryptEntityList($encryptedEntities);
347 } catch (\Throwable $t) {
348 $this->logException($t, 'Unable to recrypt encrypted data for user ' . $userId);
349 throw new OCS\OCSBadRequestException($this->l->t('Unable to recrypt encrypted data for user "%s".', $userId), $t);
350 }
351 }
352
353 $appEncryptionKey = null;
354
356 $authorizationService = $this->appContainer->get(AuthorizationService::class);
357 if ($authorizationService->getUserPermissions($userId) != AuthorizationService::PERMISSION_NONE) {
358 // set encryption key for this user
360 $encryptionService = $this->appContainer->get(EncryptionService::class);
361 if (!$encryptionService->encryptionKeyValid()) {
362 throw new OCS\OCSBadRequestException($this->l->t('Encryption key is invalid'));
363 }
364 try {
365 $appEncryptionKey = $encryptionService->getAppEncryptionKey();
366 $encryptionService->setUserEncryptionKey($appEncryptionKey, $userId);
367 } catch (Exceptions\EncryptionException $e) {
368 throw new OCS\OCSBadRequestException($this->l->t('Unable to set app-encryption-key for user "%s".', $userId), $e);
369 }
370 }
371
372 return $appEncryptionKey;
373 }
