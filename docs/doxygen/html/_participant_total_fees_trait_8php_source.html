249 public function totalFeesPreFilterTrigger(PHPMyEdit $pme, string $op, string $step):bool
250 {
251 $summaryName = $this->joinTableFieldName(PMETableViewBase::PROJECT_PAYMENTS_TABLE, 'amount');
252 $outstandingName = $this->joinTableFieldName(PMETableViewBase::PROJECT_PARTICIPANT_FIELDS_DATA_TABLE, 'total_amount_outstanding');

254 if (!isset($pme->fdn[$summaryName])) {

258
259 $summaryIndex = $pme->fdn[$summaryName];
260 $outstandingIndex = $pme->fdn[$outstandingName];
261
262 $searchField = array_search($summaryIndex, $pme->sfn);
263 if ($searchField === false) {
264 $searchField = array_search(-$summaryIndex, $pme->sfn);

266

271 : (string)$outstandingIndex;
272 }
273
274 return true;
