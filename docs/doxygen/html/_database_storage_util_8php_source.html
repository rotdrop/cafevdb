141 public function getCollectionArchive(array $items, ?string $folderName = null)
142 {
143 $folderName = empty($folderName) ? '' : $folderName . self::PATH_SEP;
144
145 $dataStream = fopen("php://memory", 'w');
146 $zipStream = new ZipStream(outputName: $folderName, outputStream: $dataStream);
147
148 foreach ($items as $item) {
149 if (empty($item)) {
150 continue;
151 }
152 $file = $this->get($item);
153 $zipStream->addFile($folderName . $file->getFileName(), $file->getFileData()->getData());
154 }
155
156 $zipStream->finish();
157 rewind($dataStream);
158 $data = stream_get_contents($dataStream);
159 fclose($dataStream);
160
161 return $data;
162 }
