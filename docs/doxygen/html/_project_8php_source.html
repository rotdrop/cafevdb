182 {
183 if ($this->id) {
184 return;
185 }
186 $this->id = null;
187 $oldInstrumentationNumbers = $this->instrumentationNumbers;
188 $oldParticipantFields = $this->participantFields;
189 $this->__construct();
190
191 // clone all instrumentation numbers
192 foreach ($oldInstrumentationNumbers as $oldInstrumentationNumber) {
194 $instrumentationNumber = clone $oldInstrumentationNumber;
195 $instrumentationNumber->setProject($this);
196 $this->instrumentationNumbers->add($instrumentationNumber);
197 }
198
199 // clone all participant fields
200 foreach ($oldParticipantFields as $oldParticipantField) {
202 $participantField = clone $oldParticipantField;
203 $participantField->setProject($this);
204 $this->participantFields->add($participantField);
205 }
206
207 // anything else stays empty
