206 public function __clone()
207 {
208 if ($this->id) {
209 return;
210 }
211 $this->id = null;
212 $oldInstrumentationNumbers = $this->instrumentationNumbers;
213 $oldParticipantFields = $this->participantFields;
214 $this->__construct();
215
216 // clone all instrumentation numbers
217 foreach ($oldInstrumentationNumbers as $oldInstrumentationNumber) {
219 $instrumentationNumber = clone $oldInstrumentationNumber;
220 $instrumentationNumber->setProject($this);
221 $this->instrumentationNumbers->add($instrumentationNumber);
222 }
223
224 // clone all participant fields
225 foreach ($oldParticipantFields as $oldParticipantField) {
227 $participantField = clone $oldParticipantField;
228 $participantField->setProject($this);
229 $this->participantFields->add($participantField);
230 }
231
232 // anything else stays empty
233 }
