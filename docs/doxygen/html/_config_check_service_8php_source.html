441 public function checkShareOwner(string $shareOwnerId, ?string $shareOwnerPassword = null):bool
442 {
443 $shareGroupId = $this->getAppValue(ConfigService::USER_GROUP_KEY, false);
444 if (empty($shareGroupId)) {
445 return false; // need at least this group!
446 }
447
448 $created = false;
449 $shareOwner = null;
450
451 // Create the user if necessary
452 if (!$this->userManager()->userExists($shareOwnerId)) {
453 $this->logError("User does not exist");
454 if (empty($shareOwnerPassword)) {
455 $shareOwnerPassword = $this->generateRandomBytes(30);
456 }
457
458 if (!$this->isSubAdminOfGroup()) {
459 $this->logError("Permission denied: ".$this->userId()." is not a group admin of ".$shareGroupId.".");
460 return false;
461 }
462 $userBackend = $this->user()->getBackend();
463
464 $shareOwner = $this->userManager()->createUserFromBackend($shareOwnerId, $shareOwnerPassword, $userBackend);
465 if (!empty($shareOwner)) {
466 $this->logInfo("User created");
467 // trigger address book creation and things
468 $this->eventDispatcher->dispatch(\OCP\IUser::class . '::firstLogin', new GenericEvent($shareOwner));
469 $created = true;
470 } else {
471 $this->logError("User could not be created");
472 return false;
473 }
474 } else {
475 $shareOwner = $this->userManager()->get($shareOwnerId);
476 }
477
478 if (empty($shareOwner)) {
479 $this->logError("Share-owner " . $shareOwnerId . " could not be found or created.");
480 return false;
481 }
482
483 // Sutff the user in its appropriate group
484 if (!$this->inGroup($shareOwnerId, $shareGroupId)) {
485 $this->logError("not in group");
486 $shareGroup = $this->group($shareGroupId);
487 if (empty($shareGroup)) {
488 $this->logError("Could not get group " . $shareGroupId . ".");
489 } else {
490 $shareGroup->addUser($shareOwner);
491 }
492 // check again, addUser() has no return value
493 if (!$this->inGroup($shareOwnerId, $shareGroupId)) {
494 $this->logError("Could not add " . $shareOwnerId . " to group " . $shareGroupId . ".");
495 if ($created) {
496 $this->logError("Deleting just created user " . $shareOwnerId . ".");
497 $shareOwner->delete();
498 }
499 return false;
500 }
501 $this->logDebug("added to group");
502 }
503
504 return $this->shareOwnerExists($shareOwnerId);
