245 private function generatePurpose(string $subject):array
246 {
247 if (strlen($subject) > FinanceService::SEPA_PURPOSE_LENGTH) {
248 $subject = Util::removeSpaces($subject);
249 }
250 if (strlen($subject) > FinanceService::SEPA_PURPOSE_LENGTH) {
251 $subject = Util::shortenCamelCaseString($subject, FinanceService::SEPA_PURPOSE_LENGTH, 4);
252 }
253
254 $purpose = [];
255 for ($i = 0; $i < FinanceService::SEPA_PURPOSE_LENGTH; $i += self::PURPOSE_LINE_LENGTH) {
256 $purpose[] = '"' . substr($subject, $i, self::PURPOSE_LINE_LENGTH) . '"';
257 }
258 return $purpose;
259 }
