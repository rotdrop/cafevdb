328 public function getSelectableOptions(bool $includeDeleted = false):Collection
329 {
330 // this unfortunately just does not work.
331 // return $this->dataOptions->matching(DBUtil::criteriaWhere([ '!key' => Uuid::NIL, 'deleted' => null, ]));
332 $filtered = $this->dataOptions->filter(function($option) use ($includeDeleted) {
334 return ($includeDeleted || empty($option->getDeleted()))
335 && (string)$option->getKey() != ProjectParticipantFieldDataOption::GENERATOR_KEY;
336 });
337 $iterator = $filtered->getIterator();
338 $iterator->uasort(function(ProjectParticipantFieldDataOption $a, ProjectParticipantFieldDataOption $b) {
339 $cmp = strcmp($a->getLabel(), $b->getLabel());
340 return $cmp !== 0 ? $cmp : strcmp($a->getKey(), $b->getKey());
341 });
342 return new ArrayCollection(iterator_to_array($iterator));

896 $this->preUpdatePosted[$field] = $changeSet[0];

904 $this->preUpdatePosted[$field] = $changeSet[0];
