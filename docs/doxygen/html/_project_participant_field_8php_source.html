877 public function preUpdate(Event\PreUpdateEventArgs $event)
878 {
879 $field = 'name';
880 $changeSet = $this->getTranslationChangeSet($event, $field);
881 if ($changeSet) {
883 $entityManager = EntityManager::getDecorator($event->getEntityManager());
884 $entityManager->dispatchEvent(new Events\PreRenameProjectParticipantField($this, $changeSet[0], $changeSet[1]));
885 $this->preUpdatePosted[$field] = $changeSet[0];
886 }
887 $field = 'tooltip';
888 $changeSet = $this->getTranslationChangeSet($event, $field);
889 if ($changeSet) {
891 $entityManager = EntityManager::getDecorator($event->getEntityManager());
892 $entityManager->dispatchEvent(new Events\PreChangeProjectParticipantFieldTooltip($this, $changeSet[0], $changeSet[1]));
893 $this->preUpdatePosted[$field] = $changeSet[0];
894 }
895 $field = 'dataType';
896 $changeSet = $this->getTranslationChangeSet($event, $field);
897 if ($changeSet) {
899 $entityManager = EntityManager::getDecorator($event->getEntityManager());
900 $entityManager->dispatchEvent(new Events\PreChangeProjectParticipantFieldType($this, $changeSet[0], $changeSet[1]));
901 $this->preUpdatePosted[$field] = $changeSet[0];
902 }
903 }

924 $field = 'dataType';
