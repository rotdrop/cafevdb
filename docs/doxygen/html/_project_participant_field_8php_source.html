338 public function getSelectableOptions(bool $includeDeleted = false):Collection
339 {
340 // this unfortunately just does not work.
341 // return $this->dataOptions->matching(DBUtil::criteriaWhere([ '!key' => Uuid::NIL, 'deleted' => null, ]));
342 $filtered = $this->dataOptions->filter(function($option) use ($includeDeleted) {
344 return ($includeDeleted || empty($option->getDeleted()))
345 && (string)$option->getKey() != ProjectParticipantFieldDataOption::GENERATOR_KEY;
346 });
347 $iterator = $filtered->getIterator();
348 $iterator->uasort(function(ProjectParticipantFieldDataOption $a, ProjectParticipantFieldDataOption $b) {
349 $cmp = strcmp($a->getLabel(), $b->getLabel());
350 return $cmp !== 0 ? $cmp : strcmp($a->getKey(), $b->getKey());
351 });
352 return new ArrayCollection(iterator_to_array($iterator));
353 }
