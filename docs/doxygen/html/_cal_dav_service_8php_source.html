656 public function findCalendarObjectByCategories(
657 array $categories,
658 ?string $userId = null,
659 array $calendarUris = [],
660 ?string $type = null,
661 ):array {
662 $principalUri = 'principals/users/' . ($userId ?? $this->userId());
663 $query = $this->calendarManager->newQuery($principalUri);
664 foreach ($calendarUris as $calendarUri) {
665 $query->addSearchCalendar($calendarUri);
666 }
667 $categories = array_values($categories);
668 if (!empty($categories)) {
669 $firstCategory = $categories[0];
670 $query->addSearchProperty(ICalendarQuery::SEARCH_PROPERTY_CATEGORIES);
671 $query->setSearchPattern($firstCategory);
672 }
673 if (!empty($type)) {
674 $query->addType($type);
675 }
676 $calendarObjects = $this->calendarManager->searchForPrincipal($query);
677 $result = [];
678 foreach ($calendarObjects as $objectInfo) {
679 if (!empty($categories)) {
680 $match = false;
681 foreach ($objectInfo['objects'] as $calendarObject) {
682 if (isset($calendarObject['CATEGORIES'])) {
683 $objectCategories = explode(',', $calendarObject['CATEGORIES'][0][0]);
684 if (empty(array_diff($categories, $objectCategories))) {
685 $match = true;
686 break;
687 }
688 }
689 }
690 if (!$match) {
691 continue;
692 }
693 }
694 $result[] = $this->getCalendarObject($objectInfo['calendar-key'], $objectInfo['uri']);
695 }
696 return $result;
697 }
