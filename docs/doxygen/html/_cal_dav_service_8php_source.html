543 private function getObjectUri(int $calendarId, string $objectIdentifier):?string
544 {
545 if (str_ends_with($objectIdentifier, self::URI_SUFFIX)) {
546 $localUri = $objectIdentifier;
547 } else {
548 $uid = $objectIdentifier;
549 $principalUri = $this->calendarPrincipalUri($calendarId);
550 if (empty($principalUri)) {
551 return null;
552 }
553 $uri = $this->calDavBackend->getCalendarObjectByUID($principalUri, $uid);
554 if (empty($uri)) {
555 return null;
556 }
557 $localUri = basename($uri);
558 }
559 return $localUri;
