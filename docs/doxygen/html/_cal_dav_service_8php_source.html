509 private function getObjectUri(int $calendarId, string $objectIdentifier):?string
510 {
511 if (str_ends_with($objectIdentifier, self::URI_SUFFIX)) {
512 $localUri = $objectIdentifier;
513 } else {
514 $uid = $objectIdentifier;
515 $principalUri = $this->calendarPrincipalUri($calendarId);
516 if (empty($principalUri)) {
517 return null;
518 }
519 $uri = $this->calDavBackend->getCalendarObjectByUID($principalUri, $uid);
520 if (empty($uri)) {
521 return null;
522 }
523 $localUri = basename($uri);
524 }
525 return $localUri;
