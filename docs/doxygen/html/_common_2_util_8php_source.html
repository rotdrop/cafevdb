109 {

147 public static function normalizeSpaces(?string $name, string $singleSpace = ' ', bool $stripLinebreaks = false):?string
148 {
149 if ($name === null) {
150 return null;
151 }
152 $name = str_replace("\xc2\xa0", "\x20", $name);
153 $name = trim($name);
154 $name = str_replace("\r\n", "\n", $name);
155 if ($stripLinebreaks) {
156 $name = str_replace("\n", $singleSpace, $name);
157 }
158 $name = preg_replace('/\h+/u', $singleSpace, $name);
159 $name = preg_replace('/\h+([\n.,;:?!])/u', '$1', $name);
160
161 return $name;

263 public static function explode(string $delim, ?string $string, int $flags = self::OMIT_EMPTY_FIELDS|self::ESCAPED, string $escape = '\\'):array
264 {
265 if (empty($flags)) {
266 return explode($delim, $string);

268 $pregFlags = ($flags & self::OMIT_EMPTY_FIELDS) ? PREG_SPLIT_NO_EMPTY : 0;
269 $trimExpr = ($flags & self::TRIM) ? '\s*' : '';
270 if (($flags & self::ESCAPED) && !empty($escape)) {
271 return
272 self::unescapeDelimiter(
273 preg_split('/'.$trimExpr.preg_quote($escape, '/').'.'.'(*SKIP)(*FAIL)|'.preg_quote($delim, '/').$trimExpr.'/s', $string, -1, $pregFlags),
274 $delim,
275 $escape);
276 } else {
277 return preg_split('/'.$trimExpr.preg_quote($delim, '/').$trimExpr.'/', $string, -1, $pregFlags);

490 public static function humanFileSize(int $bytes, ?string $locale = null, bool $binary = true, int $digits = 2):string
491 {
492 $prefix = [ '', 'K', 'M', 'G', 'T', 'P', 'E', 'Z' ];
493 $locale = $locale ?? 'en_US_POSIX';
494 $fmt = new NumberFormatter($locale, \NumberFormatter::DECIMAL);
495 $multiplier = $binary ? 1024.0 : 1000.0;
496 $bytes = (float)$bytes;
497 $exp = 0;
498 $prefixCount = count($prefix);
499 while ($exp < $prefixCount && $bytes > $multiplier) {
500 ++$exp;
501 $bytes /= $multiplier;
502 }
503 $postfix = $prefix[$exp];
504 if ($binary && $postfix !== '') {
505 $postfix .= 'i';
506 }
507 $postfix .= 'B';
508 return $fmt->format(round($bytes, $digits)) . ' ' . $postfix;
509 }

637 public static function hexDump(string $string, ?array $options = null)
638 {
639 if (!is_scalar($string)) {
640 throw new InvalidArgumentException('$string argument must be a string');
641 }
642 if (!is_array($options)) {
643 $options = array();
644 }
645 $lineSep = isset($options['line_sep']) ? $options['line_sep'] : "\n";
646 $bytesPerLine = $options['bytes_per_line'] ? $options['bytes_per_line'] : 16;
647 $padChar = isset($options['pad_char']) ? $options['pad_char'] : '.'; // padding for non-readable characters
648
649 $textLines = str_split($string, $bytesPerLine);
650 $hexLines = str_split(bin2hex($string), $bytesPerLine * 2);
651
652 $offset = 0;
653 $output = array();
654 $bytesPerLineDiv2 = (int)($bytesPerLine / 2);
655 foreach ($hexLines as $i => $hexLine) {
656 $textLine = $textLines[$i];
657 $output [] =
658 sprintf('%08X', $offset) . ' ' .
659 str_pad(
660 strlen($textLine) > $bytesPerLineDiv2
661 ?
662 implode(' ', str_split(substr($hexLine, 0, $bytesPerLine), 2)) . ' ' .
663 implode(' ', str_split(substr($hexLine, $bytesPerLine), 2))
664 :
665 implode(' ', str_split($hexLine, 2)), $bytesPerLine * 3) .
666 ' |' . preg_replace('/[^\x20-\x7E]/', $padChar, $textLine) . '|';
667 $offset += $bytesPerLine;
668 }
669 $output[] = sprintf('%08X', strlen($string));
670 return $options['want_array'] ? $output : join($lineSep, $output) . $lineSep;
671 }

709 public static function dateTime($arg1 = "now", $arg2 = null, $arg3 = null):DateTimeImmutable
710 {
711 if ($arg1 instanceof \DateTimeImmutable) {
712 if ($arg2 !== null || $arg3 !== null) {
713 throw \InvalidArgumentException('Excess arguments, expected 1, got 3.');
714 }
715 return $arg1;
716 }
717 if ($arg1 instanceof \DateTime) {
718 if ($arg2 !== null || $arg3 !== null) {
719 throw \InvalidArgumentException('Excess arguments, expected 1, got 3.');
720 }
721 return \DateTimeImmutable::createFromMutable($arg1);
722 }
723 if ($arg1 instanceof \DateTimeInterface) {
724 if ($arg2 !== null || $arg3 !== null) {
725 throw \InvalidArgumentException('Excess arguments, expected 1, got 3.');
726 }
727 return \DateTimeImmutable::createFromInterface($arg1);
728 }
729 if (is_string($arg1) && is_string($arg2)) {
730 return \DateTimeImmutable::createFromFormat($arg1, $arg2, $arg3);
731 } elseif ($arg2 === null && $arg3 === null) {
732 $timeStamp = filter_var($arg1, FILTER_VALIDATE_INT, [ 'min' => 0 ]);
733 if ($timeStamp !== false) {
734 return (new DateTimeImmutable())->setTimestamp($timeStamp);
735 } elseif (is_string($arg1)) {
736 return new DateTimeImmutable($arg1);
737 }
738 } elseif ($arg3 === null) {
739 return new DateTimeImmutable($arg1, $arg2);
740 }
741 throw new InvalidArgumentException('Unsupported arguments');
742 }
