1184 public function beforeInsertTrigger(

1187 string $step,
1188 ?array $oldVals,
1189 array &$changed,
1190 array &$newVals,
1191 ):bool {
1192 $this->debugPrintValues($oldVals, $changed, $newVals, null, 'before');
1193
1194 if (empty($newVals['name'])) {
1195 return false;
1196 }
1197 $newVals['name'] = $this->projectService->sanitizeName($newVals['name']);
1198 if (empty($newVals['name'])) {
1199 return false;
1200 }
1201
1202 // sanitize instrumentation numbers
1203 $instrumentsColumn = $this->joinTableFieldName(self::PROJECT_INSTRUMENTATION_NUMBERS_TABLE, 'instrument_id');
1204 $voicesColumn = $this->joinTableFieldName(self::PROJECT_INSTRUMENTATION_NUMBERS_TABLE, 'voice');
1205
1206 // Add zeros to the voices data as the "0" voice is needed for
1207 // convenience in either case, otherwise adding musicians to the
1208 // ProjectParticipants table fails.
1209 $instruments = Util::explode(',', $newVals[$instrumentsColumn]??'');
1210 foreach (Util::explode(',', $newVals[$instrumentsColumn]??'') as $instrument) {
1211 $newVals[$voicesColumn] = $instrument . self::JOIN_KEY_SEP . '0' . ','
1212 . ($newVals[$voicesColumn]??'');
1213 }
1214 $voiceItems = Util::explode(',', $newVals[$voicesColumn]);
1215 foreach ($voiceItems as $key => $voiceItem) {
1216 list($instrument, $voice) = explode(self::JOIN_KEY_SEP, $voiceItem);
1217 if (array_search($instrument, $instruments) === false) {
1218 $this->debug('REMOVE VOICE ' . $voice . ' FOR INSTRUMENT ' . $instrument);
1219 unset($voiceItems[$key]);
1220 }
1221 }
1222 sort($voiceItems, SORT_NATURAL);
1223 $newVals[$voicesColumn] = implode(',', array_unique($voiceItems));
1224
1225 foreach ([$instrumentsColumn, $voicesColumn] as $column) {
1226 Util::unsetValue($changed, $column);
1227 if (!empty($newVals[$column])) {
1228 $changed[] = $column;
1229 }
1230 }
1231
1232 if ($newVals['type'] == ProjectType::TEMPLATE) {
1233 // do not create mailing lists for templates
1234 $newVals['mailing_list_id'] = 'keep-empty';
1235 }
1236
1237 // unset 'copy_participants'
1238 Util::unsetValue($changed, 'copy_participants');
1239
1240 $this->debugPrintValues($oldVals, $changed, $newVals, null, 'after');
1241
1242 return true;

1306 ${$dataArray}[$voicesColumn] = implode(',', array_unique($items));
