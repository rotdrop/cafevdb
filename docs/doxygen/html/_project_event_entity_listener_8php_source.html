111 private function registerPreCommitAction(Entities\ProjectEvent $projectEvent):void
112 {
113 if (!empty($this->preCommitActions[$projectEvent->getId()])) {
114 return;
115 }
116 if (empty($this->eventsService)) {
117 $this->eventsService = $this->appContainer->get(EventsService::class);
118 }
119 $recordAbsenceCategory = $this->eventsService->getRecordAbsenceCategory();
120 $calendarId = $projectEvent->getCalendarId();
121 $eventUri = $projectEvent->getEventUri();
122 $recurrenceId = $projectEvent->getRecurrenceId();
123 $this->preCommitActions[$projectEvent->getId()] = new GenericUndoable(
124 function() use ($projectEvent, $recordAbsenceCategory, $calendarId, $eventUri, $recurrenceId) {
125 if (empty($projectEvent->getAbsenceField())) {
126 $additions = [];
127 $removals[] = $recordAbsenceCategory;
128 } else {
129 $additions[] = $recordAbsenceCategory;
130 $removals = [];
131 }
132 try {
133 $changed = $this->eventsService->changeCategories(
134 $calendarId,
135 $eventUri,
136 $recurrenceId,
137 $additions,
138 $removals,
139 );
140 } catch (Exceptions\CalendarEntryNotFoundException $e) {
141 // this can happen if the project events table got out of sync
142 $changed = false;
143 }
144 return $changed;
145 },
146 function(bool $changed) use ($projectEvent, $recordAbsenceCategory, $calendarId, $eventUri, $recurrenceId) {
147 if (!$changed) {
148 return;
149 }
150 // just exchange additions and removals
151 if (!empty($projectEvent->getAbsenceField())) {
152 $additions = [];
153 $removals[] = $recordAbsenceCategory;
154 } else {
155 $additions[] = $recordAbsenceCategory;
156 $removals = [];
157 }
158 $this->eventsService->changeCategories(
159 $calendarId,
160 $eventUri,
161 $recurrenceId,
162 $additions,
163 $removals,
164 );
165 },
166 );
167 $action = $this->preCommitActions[$projectEvent->getId()];
168 if ($this->entityManager->isTransactionActive()) {
169 // preferable
170 $this->entityManager->registerPreCommitAction($action);
171 } else {
172 $this->entityManager->registerPreFlushAction($action);
