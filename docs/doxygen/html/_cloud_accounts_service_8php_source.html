138 public function addUserToBackend(IUser $user, string|UserInterface $newBackend)
139 {
140 if (is_string($newBackend)) {
141 $newBackend = array_filter(
142 $this->userManager->getBackends(),
143 fn(IUserBackend $backend) => $backend->getBackendName() == $newBackend,
144 );
145 if (count($newBackend) != 1) {
146 return false;
147 }
148 $newBackend = reset($newBackend);
149 }
150
151 $oldBackend = $user->getBackend();
152
153 if (!($oldBackend instanceof IUserBackend) || !($newBackend instanceof IUserBackend)) {
154 $this->logError('Only support moving between sane user-backends.');
155 return false;
156 }
157
158 $oldBackendName = $oldBackend->getBackendName();
159 $newBackendName = $newBackend->getBackendName();
160
161 if ($oldBackendName == $newBackendName) {
162 return true;
163 }
164
165 $email = $user->getEMailAddress() ?? '';
166
167 if ($email === '') {
168 // This also means that the setup is somehow totally broken ...
169 $this->logError('User "' . $userId . '" does not have an email.');
170 return false;
171 }
172
173 $userId = $user->getUID();
174 $displayName = $user->getDisplayName();
175 $quota = $user->getQuota();
176
177 // try to transfer the avatar
179 $avatar = $this->avatarManager->getAvatar($userId);
180 if ($avatar->isCustomAvatar()) {
181 $avatarImage = $avatar->get(-1);
182 }
183
184 $newUser = $this->doAddUserToBackend(
185 userId: $userId,
186 UserInterface: $newBackend,
187 displayName: $displayName,
188 email: $email,
189 quota: $quota,
190 avatarImage: $avatarImage ?? null,
191 );
192
193 if (empty($newUser)) {
194 $this->logger->error('Failed to transfer "' . $userId . '" to the user-backend "' . $newBackend->getBackendName());
195 return false;
196 }
197
198 // If this has succeeded, perhaps the user should be deleted in the old backend, if possible
199 try {
201 $oldBackend->deleteUser($user->getUID());
202 $this->logInfo('Deleted user "' . $userId . '" in old backend "' . $oldBackendName . '"');
203 } catch (Throwable $t) {
204 $this->logException($t, 'Cannot delete user "' . $userId . '" in old backend "' . $oldBackendName . '"');
205 }
206
207 return true;
208 }

225 protected function doAddUserToBackend(
226 string $userId,
227 UserInterface $backend,
228 string $displayName,
229 string $email,
230 string $quota,
231 ?IAvatar $avatarImage,
232 ):?IUser {
233 $backendName = $backend->getBackendName();
234 if ($backend->userExists($userId)) {
235 $this->logWarn('User "' . $userId . '" already exists in backend "' . $backendName . '"');
236 if (!($this->userManager instanceof \OC\User\Manager) || !method_exists($this->userManager, 'getUserObject')) {
237 $this->logError('The IUserManager implementation seems to have changed, cannot retrieve the user object from the backend.');
238 return null;
239 }
241 $userManager = $this->userManager;
242 return $userManager->getUserObject($userId, $backend);
243 }
244
245 $passwordEvent = new GenerateSecurePasswordEvent();
246 $this->eventDispatcher->dispatchTyped($passwordEvent);
247 $password = $passwordEvent->getPassword();
248
249 if ($password === null) {
250 // Fallback: ensure to pass password_policy in any case
251 $password = $this->secureRandom->generate(10)
252 . $this->secureRandom->generate(1, ISecureRandom::CHAR_UPPER)
253 . $this->secureRandom->generate(1, ISecureRandom::CHAR_LOWER)
254 . $this->secureRandom->generate(1, ISecureRandom::CHAR_DIGITS)
255 . $this->secureRandom->generate(1, ISecureRandom::CHAR_SYMBOLS);
256 }
257 $generatePasswordResetToken = true;
258
259 try {
260 $newUser = $this->userManager->createUserFromBackend($userId, $password, $backend);
261
262 $this->logger->info('Successful addUser call with userid: ' . $userId, ['app' => 'ocs_api']);
263
264 $newUser->setDisplayName($displayName);
265 $newUser->setQuota($quota);
266
267 if (!empty($avatarImage)) {
268 try {
269 $avatar = $this->avatarManager->getAvatar($userId);
270 $avatar->set($avatarImage);
271 } catch (Throwable $t) {
272 $this->logException($t, 'Unable to restore the avatar image after moving "' . $userId . '" to the new backend "' . $backendName . '".');
273 }
274 }
275
276 $newUser->setEMailAddress($email);
277 if ($this->cloudConfig->getAppValue('core', 'newUser.sendEmail', 'yes') === 'yes') {
278 try {
279 $emailTemplate = $this->newUserMailHelper->generateTemplate($newUser, $generatePasswordResetToken);
280 $this->newUserMailHelper->sendMail($newUser, $emailTemplate);
281 } catch (Throwable $e) {
282 // Mail could be failing hard or just be plain not configured
283 // Logging error as it is the hardest of the two
284 $this->logger->error(
285 "Unable to send the invitation mail to $email",
286 [
287 'app' => 'ocs_api',
288 'exception' => $e,
289 ]
290 );
291 }
292 }
293 return $newUser;
294 } catch (HintException $e) {
295 $this->logger->warning(
296 'Failed addUser attempt with hint exception.',
297 [
298 'app' => 'ocs_api',
299 'exception' => $e,
300 ]
301 );
302 return null;
303 } catch (InvalidArgumentException $e) {
304 $this->logger->error(
305 'Failed addUser attempt with invalid argument exception.',
306 [
307 'app' => 'ocs_api',
308 'exception' => $e,
309 ]
310 );
311 return null;
312 } catch (Throwable $e) {
313 $this->logger->error(
314 'Failed addUser attempt with exception.',
315 [
316 'app' => $this->appName,
317 'exception' => $e
318 ]
319 );
320 return null;
321 }

364 public function promoteGroupMembership(IUser $user, IGroup $group, array $requiredBackends = [])
365 {
366 if (empty($requiredBackends)) {
367 $requiredBackends = [ 'Database' ];
368 $orchestraUserAndGroupBackend = $this->cloudConfig->getAppValue(
369 $this->appName,
370 ConfigService::USER_AND_GROUP_BACKEND_KEY,

373 if ($orchestraUserAndGroupBackend !== null) {
374 $requiredBackends[] = $orchestraUserAndGroupBackend;

386 $backendName = $backend->getBackendName();
387 if (!in_array($backendName, $requiredBackends)) {

393 $backend->addToGroup($userId, $groupId);

397 $this->logException($t, 'Backend ' . get_class($backend) . ' cannot add user ' . $user->getUID());
398 }

423 public function ensureGroupBackends(IGroup $group, array $requiredBackends = []):array
424 {
425 if (empty($requiredBackends)) {
426 $requiredBackends = [ 'Database' ];
427 $orchestraUserAndGroupBackend = $this->cloudConfig->getAppValue(
428 $this->appName,
429 ConfigService::USER_AND_GROUP_BACKEND_KEY,
430 null,
431 );
432 if ($orchestraUserAndGroupBackend !== null) {
433 $requiredBackends[] = $orchestraUserAndGroupBackend;
434 }
435 }
436
437 $status = array_fill_keys($requiredBackends, true); // set to \false on failure below
438 $gid = $group->getGID();
439 $displayName = $group->getDisplayName();
440 $groupBackendNames = $group->getBackendNames();
441 $missingBackends = array_diff($requiredBackends, $groupBackendNames);
442 foreach ($this->groupManager->getBackends() as $backend) {
443 if (!($backend instanceof INamedBackend)) {
444 continue;
445 }
446 $backendName = $backend->getBackendName();
447 if (!in_array($backendName, $missingBackends)) {
448 continue;
449 }
450 if (!($backend instanceof ICreateGroupBackend)) {
451 $this->logError('Group ' . $gid . ' is missing in backend "' . $backendName . '", but the backend is not able to create groups.');
452 $status[$backendName] = false;
453 continue;
454 }
455 if (!$backend->createGroup($gid)) {
456 $this->logError('Group ' . $gid . ' is missing in backend "' . $backendName . '", but the backend failed to create it.');
457 $status[$backendName] = false;
458 continue;
459 }
460 if (!$backend->groupExists($gid)) {
461 $this->logError('Backend "' . $backendName . '" did not error out creating the group "' . $gid . '" but the group does not exist in the backend.');
462 $status[$backendName] = false;
463 continue;
464 }
465 if (!($backend instanceof ISetDisplayNameBackend)) {
466 continue;
467 }
468 if (($backend instanceof IGetDisplayNameBackend) && $backend->getDisplayName($gid) == $displayName) {
469 continue;
470 }
471 $backend->setDisplayName($gid, $displayName);
472 // ignore the result, not so important
473 }
474 if (($this->groupManager instanceof \OC\Group\Manager) && method_exists($this->groupManager, 'clearCaches')) {
475 $this->groupManager->clearCaches();
476 }
477 return $status;
478 }
