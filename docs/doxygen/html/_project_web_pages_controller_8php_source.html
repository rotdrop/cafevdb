68 public function serviceSwitch(
69 string $topic,
70 ?int $projectId = null,
71 ?int $articleId = null,
72 array $articleData = [],
73 ):Http\Response {
74 if ($topic != 'ping' && $projectId <= 0) {
75 return self::grumble($this->l->t("Invalid or unset project-id: `%s'", [ $projectId ]));
76 }
77
78 if (count($articleData) > 0 &&
79 $articleId >= 0 &&
80 $articleData['articleId'] != $articleId) {
81 return self::grumble(
82 $this->l->t(
83 "Submitted article id %d ".
84 "does not match the id %d stored in the article-data.",
85 [ $articleId, $articleData['articleId'] ]));
86 }
87
88 if ($topic != 'add' && $topic != 'ping') {
89 // require both, articleId and articleData
90 if ($articleId < 0) {
91 return self::grumble($this->l->t("Invalid or unset article-id: `%s'", [ $articleId ]));
92 }
93 if (count($articleData) == 0) {
94 return self::grumble(
95 $this->l->t(
96 "Cannot perform action `%s' with article with id %d, project with id %d: ".
97 "missing article-data.",
98 [ $topic, $articleId, $projectId ]));
99 }
100 }
101
102 switch ($topic) {
103 case 'ping':
104 if ($this->projectService->pingWebPages() === false) {
105 return self::grumble($this->l->t('Unable to ping project web-pages CMS'));
106 } else {
107 return self::response('OK');
108 }
109 break;
110 case 'add':
111 try {
112 // This simply means: create a new page for the project.
113 $article = $this->projectService->createProjectWebPage($projectId);
114 } catch (\Throwable $t) {
115 return self::grumble($this->exceptionChainData($t));
116 }
117 $message = $this->l->t(
118 "Created a new public web page with name %s and id %d ".
119 "for the project with id %d.",
120 [ $article['articleName'], $article['articleId'], $projectId ]);
121 // If there is no rehearsal page attached to the project, then attach one
122 $rehearsalsCat = $this->getConfigValue('redaxoRehearsals');
123 $rehearsal = null;
124 try {
125 $articles = $this->projectService->fetchProjectWebPages($projectId);
126 foreach ($articles as $article) {
127 if ($article['categoryId'] == $rehearsalsCat) {
128 $rehearsal = $article;
129 break;
130 }
131 }
132 } catch (\Throwable $t) {
133 // ignore
134 }
135 if ($rehearsal === null) {
136 // create one, but ignore anypotential error
137 try {
138 $this->projectService->createProjectWebPage($projectId, 'rehearsals');
139 $message .= ' '.$this->l->t("Created additionally a new rehearsal web page.");
140 } catch (\Throwable $t) {
141 $message .= ' '.$this->l->t("Failed to create additionally a new rehearsal web-page.");
142 }
143 }
144 return self::response($message);
145 case 'link':
146 try {
147 $this->projectService->attachProjectWebPage($projectId, $articleData);
148 } catch (\Throwable $t) {
149 return self::grumble($this->exceptionChainData($t));
150 }
151 return self::response(
152 $this->l->t(
153 "Linked the existing public web-article %s (id %d) to the project with id %d",
154 [ $articleData['articleName'], $articleId, $projectId ]));
155 case 'unlink':
156 try {
157 $this->projectService->detachProjectWebPage($projectId, $articleId);
158 } catch (\Throwable $t) {
159 return self::grumble($this->exceptionChainData($t));
160 }
161 return self::response(
162 $this->l->t(
163 "Detached the public web-article %s (id %d) from the project with id %d",
164 [ $articleData['articleName'], $articleId, $projectId ]));
165 case 'delete':
166 try {
167 $this->projectService->deleteProjectWebPage($projectId, $articleData);
168 } catch (\Throwable $t) {
169 return self::grumble($this->exceptionChainData($t));
170 }
171 return self::response(
172 $this->l->t(
173 "Removed public web page %s (id %d) from the project with id %d.",
174 [ $articleData['articleName'], $articleId, $projectId ]));
175 break;
176 }
177
178 return self::grumble($this->l->t('Unknown Request'));
179 }
