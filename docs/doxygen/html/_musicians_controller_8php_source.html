132 public function search(

141
142 switch ($scope) {
143 case self::SCOPE_ADDRESSBOOK:
144 case self::SCOPE_CLOUD_USERS:
145 return self::grumble($this->l->t('Picking users from the cloud or the address-books is not yet supported, sorry.'));
146 case self::SCOPE_EXECUTIVE_BOARD:
147 $projectId = $this->getExecutiveBoardProjectId();
148 break;
149 case self::SCOPE_CLUB_MEMBERS:
150 $projectId = $this->getClubMembersProjectId();
151 break;
152 case self::SCOPE_MUSICIANS:
153 // just go
154 break;
155 }
156
157 if ($projectName !== null && $projectId === null) {
158 // our findLikeTrait cannot iterate to projectParticipation.project.name
159 $project = $this->getDatabaseRepository(Entities\Project::class)->findOneBy([ 'name' => $projectName ]);
160 $project = $project->getId();
161 } else {
162 $project = $projectId;
163 }
164
165 if (empty($pattern)) {
166 $criteria = [];
167 } else {
168 $pattern = str_replace('*', '%', $pattern);
169
170 if (strpos($pattern, '%') === false) {
171 if ($pattern[0] != '^') {
172 $pattern = '%' . $pattern;
173 }
174 if (substr($pattern, -1) != '$') {
175 $pattern = $pattern . '%';
176 }
177 }
178
179 $criteria = [
180 '(|surName' => $pattern,
181 'firstName' => $pattern,
182 'displayName' => $pattern,
183 'nickName' => $pattern,
184 'userIdSlug' => $pattern,
185 ];
186 $criteria[] = [ ')' => true ];
187 }
188
189 if ($project !== null) {
190 $criteria[] = [ 'projectParticipation.project' => $project ];
191 }
192
193 $musicians = $this->musiciansRepository->findBy($criteria, [
194 'surName' => 'ASC',
195 'firstName' => 'ASC'
196 ], $limit, $offset);
197
198 if ($limit !== null && count($ids) > 0) {
199 $criteria = [ [ 'id' => $ids ] ];
200 if ($project !== null) {
201 $criteria[] = [ 'projectParticipation.project' => $project ];
202 }
203 $byIdMusicians = $this->musiciansRepository->findBy($criteria);
204 $musicians = array_merge($musicians, $byIdMusicians);
205 }
206
207 $musiciansData = [];
209 foreach ($musicians as $musician) {
210 if (Uuid::asUuid($musician->getSurName()) !== null) {
211 continue; // skip dummy musicians
212 }
213 $musiciansData[] = $this->getFlatMusician($musician, only: []);
214 }
