396 public function sql_fetch(&$stmt, $type = 'a')
397 {
398 if (!$this->resultValid($stmt)) {
399 return false;
400 }
401 $this->dbh->getWrappedConnection()->setAttribute(PDO::ATTR_STRINGIFY_FETCHES, true);
402 $type = $type === 'n' ? FetchMode::NUMERIC : FetchMode::ASSOCIATIVE;
403 $result = $stmt->fetch($type);
404 // Work around bug https://jira.mariadb.org/browse/MDEV-27323
405 if ($result !== false
406 && $type == FetchMode::ASSOCIATIVE
407 && $this->queryHash == $this->generatedQueryHash
408 && !empty($this->columnAliases)) {
409 $missingColumns = array_diff($this->columnAliases, array_keys($result));
410 if (!empty($missingColumns)) {
411 // We assume that this is caused by an optimization bug and that
412 // either the corresponding qfNN_idx column or the corresponding
413 // non-..._idx column is there and contains the correct data
414 $this->logError('Potential MariaDB missing-columns bug (https://jira.mariadb.org/browse/MDEV-27323): ' . print_r($missingColumns, true));
415 $postfix = '_idx';
416 foreach ($missingColumns as $column) {
417 $dataColumn = strrpos($column, $postfix) === false ? $column . $postfix : substr($column, -strlen($postfix));
418 if (empty($result[$dataColumn])) {
419 $this->logError('MariaDB bug https://jira.mariadb.org/browse/MDEV-27323): unable to reconstruct data for column ' . $column);
420 }
421 $result[$column] = $result[$dataColumn];
422 }
423 }
424 }
425 $this->dbh->getWrappedConnection()->setAttribute(PDO::ATTR_STRINGIFY_FETCHES, false);
426
427 return $result;
428 }

464 public function myquery($query, $line = 0, $debug = false)
465 {
466 $this->queryHash = md5($query);
467 if ($debug || $this->debug) {
468 $this->logInfo("DEBUG QUERY: ".$query, [], 2);
469 }
470 $logEntry = [
471 'query' => $query,
472 ];
473 $startTime = hrtime(true); // [ns]
474 try {
475 $stmt = $this->dbh->executeQuery($query);
476 $endTime = hrtime(true);
477 $this->affectedRows = $stmt->rowCount();
478 $this->errorCode = $stmt->errorCode();
479 $this->errorInfo = $stmt->errorInfo();
480 $logEntry['affectedRows'] = $this->affectedRows;
481 $logEntry['errorCode'] = $this->errorCode;
482 $logEntry['errorInfo'] = $this->errorInfo;
483 $logEntry['duration'] = ($endTime - $startTime) / 1e6; // [ms]
484 $this->queryLog[] = $logEntry;
485 } catch (DBALException $t) {
486 $endTime = hrtime(true);
487 $this->logException($t);
488 throw $t;
489 $this->exception = $t;
490 $this->errorCode = $t->getCode();
491 $this->errorInfo = $t->getMessage();
492 $logEntry['affectedRows'] = $this->affectedRows;
493 $logEntry['errorCode'] = $this->errorCode;
494 $logEntry['errorInfo'] = $this->errorInfo;
495 $logEntry['duration'] = ($endTime - $startTime) / 1e6;
496 $this->queryLog[] = $logEntry;
497 return false;
498 }
499 return $stmt;
500 }

645 protected function makeTimeStampFromDatabase($databaseValue)

649 if ($timeStamp % (24 * 60 * 60) != 0 || preg_match("/\d{1,2}\:\d{1,2}/", $databaseValue)) {

660 $this->logDebug('ORIG / MOD ' . print_r($dateTime, true) . ' / ' . print_r($modDateTime, true));

664 $this->logError('Timezone adjustment failed for data-base input ' . $databaseValue);
665 }

675 public function makeUserTimeString($k, $row)
676 {
677 $value = '';
678 $data = $row["qf$k"."_timestamp"];
679 switch ($data) {
680 case '':
681 case 0:
682 case '0000-00-00':
683 // Invalid date and time
684 break;
685 default:
686 if (!is_numeric($data)) {
687 // Convert whatever is contained in the timestamp field to
688 // seconds since the epoch.
689 $data = $this->makeTimeStampFromDatabase($data);
690 }
691 $timeStamp = intval($data);
692 $dateFormat = $timeFormat = $this->fdd[$k]['datetimeformat'] ?? false;
693 $dateFormat = $this->fdd[$k]['dateformat'] ?? false;
694 $timeFormat = $this->fdd[$k]['timeformat'] ?? false;
695
696 if (!empty($dateFormat) && !empty($timeFormat)) {
697 if ($dateFormat === true && $timeFormat === true) {
698 return $this->dateTimeFormatter->formatDateTime($timeStamp);
699 } elseif ($timeFormat === true) {
700 return $this->dateTimeFormatter->formatDateTime($timeStamp, $dateFormat);
701 } else {
702 return $this->dateTimeFormatter->formatDateTime($timeStamp, $dateFormat, $timeFormat);
703 }
704 } elseif (!empty($dateFormat)) {
705 return $dateFormat === true
706 ? $this->dateTimeFormatter->formatDate($timeStamp)
707 : $this->dateTimeFormatter->formatDate($timeStamp, $dateFormat);
708 } elseif (!empty($timeFormat)) {
709 return $timeFormat === true
710 ? $this->timeTimeFormatter->formatTime($timeStamp)
711 : $this->timeTimeFormatter->formatTime($timeStamp, $timeFormat);
712 } elseif (!empty($this->fdd[$k]['datemask'])) {
713 return date($this->fdd[$k]['datemask'], $timeStamp);
714 } elseif (!empty($this->fdd[$k]['strftimemask'])) {
715 return strftime($this->fdd[$k]['strftimemask'], $timeStamp);
716 }
717 }
718 return $value;
719 }
