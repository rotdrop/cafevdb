411 public function sql_fetch(&$stmt, $type = 'a')
412 {
413 if (!$this->resultValid($stmt)) {
414 return false;
415 }
416 $this->dbh->getWrappedConnection()->setAttribute(PDO::ATTR_STRINGIFY_FETCHES, true);
417 $type = $type === 'n' ? FetchMode::NUMERIC : FetchMode::ASSOCIATIVE;
418 $result = $stmt->fetch($type);
419 // Work around bug https://jira.mariadb.org/browse/MDEV-27323
420 if ($result !== false
421 && $type == FetchMode::ASSOCIATIVE
422 && $this->queryHash == $this->generatedQueryHash
423 && !empty($this->columnAliases)) {
424 $missingColumns = array_diff($this->columnAliases, array_keys($result));
425 if (!empty($missingColumns)) {
426 // We assume that this is caused by an optimization bug and that
427 // either the corresponding qfNN_idx column or the corresponding
428 // non-..._idx column is there and contains the correct data
429 $this->logError('Potential MariaDB missing-columns bug (https://jira.mariadb.org/browse/MDEV-27323): ' . print_r($missingColumns, true));
430 $postfix = '_idx';
431 foreach ($missingColumns as $column) {
432 $dataColumn = strrpos($column, $postfix) === false ? $column . $postfix : substr($column, -strlen($postfix));
433 if (empty($result[$dataColumn])) {
434 $this->logError('MariaDB bug https://jira.mariadb.org/browse/MDEV-27323): unable to reconstruct data for column ' . $column);
435 }
436 $result[$column] = $result[$dataColumn];
437 }
438 }
439 }
440 $this->dbh->getWrappedConnection()->setAttribute(PDO::ATTR_STRINGIFY_FETCHES, false);
441
442 return $result;
443 }

479 public function myquery($query, $line = 0, $debug = false)
480 {
481 $this->queryHash = md5($query);
482 if ($debug || $this->debug) {
483 $this->logInfo("DEBUG QUERY: ".$query, [], 2);
484 }
485 $logEntry = [
486 'query' => $query,
487 ];
488 $startTime = hrtime(true); // [ns]
489 try {
490 $stmt = $this->dbh->executeQuery($query);
491 $endTime = hrtime(true);
492 $this->affectedRows = $stmt->rowCount();
493 $this->errorCode = $stmt->errorCode();
494 $this->errorInfo = $stmt->errorInfo();
495 $logEntry['affectedRows'] = $this->affectedRows;
496 $logEntry['errorCode'] = $this->errorCode;
497 $logEntry['errorInfo'] = $this->errorInfo;
498 $logEntry['duration'] = ($endTime - $startTime) / 1e6; // [ms]
499 $this->queryLog[] = $logEntry;
500 } catch (DBALException $t) {
501 $endTime = hrtime(true);
502 $this->logException($t);
503 throw $t;
504 $this->exception = $t;
505 $this->errorCode = $t->getCode();
506 $this->errorInfo = $t->getMessage();
507 $logEntry['affectedRows'] = $this->affectedRows;
508 $logEntry['errorCode'] = $this->errorCode;
509 $logEntry['errorInfo'] = $this->errorInfo;
510 $logEntry['duration'] = ($endTime - $startTime) / 1e6;
511 $this->queryLog[] = $logEntry;
512 return false;
513 }
514 return $stmt;
515 }

660 protected function makeTimeStampFromDatabase($databaseValue)
661 {
662 $timeStamp = parent::makeTimeStampFromDatabase($databaseValue);
663
664 if ($timeStamp % (24 * 60 * 60) != 0 || preg_match("/\d{1,2}\:\d{1,2}/", $databaseValue)) {
665 // assume UTC with time-stamp

668

670 $dateTime = (new DateTime)->setTimezone($timeZone)->setTimestamp($timeStamp);

673 $modDateTime = (new DateTime)->setTimezone($timeZone)->setTimestamp($modTimeStamp);
674
675 $this->logDebug('ORIG / MOD ' . print_r($dateTime, true) . ' / ' . print_r($modDateTime, true));

679 $this->logError('Timezone adjustment failed for data-base input ' . $databaseValue);
680 }

690 public function makeUserTimeString($k, $row)
691 {
692 $value = '';
693 $data = $row["qf$k"."_timestamp"];
694 switch ($data) {
695 case '':
696 case 0:
697 case '0000-00-00':
698 // Invalid date and time
699 break;
700 default:
701 if (!is_numeric($data)) {
702 // Convert whatever is contained in the timestamp field to
703 // seconds since the epoch.
704 $data = $this->makeTimeStampFromDatabase($data);
705 }
706 $timeStamp = intval($data);
707 $dateFormat = $timeFormat = $this->fdd[$k]['datetimeformat'] ?? false;
708 $dateFormat = $this->fdd[$k]['dateformat'] ?? false;
709 $timeFormat = $this->fdd[$k]['timeformat'] ?? false;
710
711 if (!empty($dateFormat) && !empty($timeFormat)) {
712 if ($dateFormat === true && $timeFormat === true) {
713 return $this->dateTimeFormatter->formatDateTime($timeStamp);
714 } elseif ($timeFormat === true) {
715 return $this->dateTimeFormatter->formatDateTime($timeStamp, $dateFormat);
716 } else {
717 return $this->dateTimeFormatter->formatDateTime($timeStamp, $dateFormat, $timeFormat);
718 }
719 } elseif (!empty($dateFormat)) {
720 return $dateFormat === true
721 ? $this->dateTimeFormatter->formatDate($timeStamp)
722 : $this->dateTimeFormatter->formatDate($timeStamp, $dateFormat);
723 } elseif (!empty($timeFormat)) {
724 return $timeFormat === true
725 ? $this->timeTimeFormatter->formatTime($timeStamp)
726 : $this->timeTimeFormatter->formatTime($timeStamp, $timeFormat);
727 } elseif (!empty($this->fdd[$k]['datemask'])) {
728 return date($this->fdd[$k]['datemask'], $timeStamp);
729 } elseif (!empty($this->fdd[$k]['strftimemask'])) {
730 return strftime($this->fdd[$k]['strftimemask'], $timeStamp);
731 }
732 }
733 return $value;
734 }
