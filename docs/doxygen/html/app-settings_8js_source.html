24import $ from './jquery.js';

29/**
30 * A variant of the old fashioned appsettings with a callback
31 * instead of script loading
32 *
33 * @param {string} route TBD.
34 *
35 * @param {object} callbacks TBD.
36 */
37const appSettings = function(route, callbacks) {
38 const defaultCallbacks = {
39 done() {},
40 fail() {},
41 always() {},
42 };
43 callbacks = $.extend({}, defaultCallbacks, callbacks);
44 const $popup = $('#appsettings_popup');
45 if ($popup.is(':visible')) {
46 $popup.addClass('hidden').html('');
47 // $popup.hide().html('');
48 } else {
49 // const arrowclass = $popup.hasClass('topright') ? 'up' : 'left';
50 $.get(generateUrl(route))
51 .done(function(data) {
52 $popup
53 .html(data)
54 .ready(function() {
55 // assume the first element is a container div
56 if ($popup.find('.popup-title').length > 0) {
57 $popup.find('.popup-title').append('<a class="close"></a>');
58 // $popup.find(">:first-child").prepend('<a class="close"></a>').show();
59 } else {
60 $popup.find('>:first-child').prepend('<div class="popup-title"><h2>' + t('core', 'Settings') + '</h2><a class="close"></a></div>');
61 }
62 $popup.find('.close').bind('click', function() {
63 $popup.addClass('hidden').html('');
64 });
65 callbacks.done.apply($popup.get(0), arguments);
66 $popup.find('>:first-child').removeClass('hidden');
67 $popup.removeClass('hidden');
68 });
69 })
70 .fail(function() {
71 console.log(arguments);
72 callbacks.fail.apply($popup.get(0), arguments);
73 })
74 .always(function() {
75 callbacks.always.apply($popup.get(0), arguments);
76 });
77 }
78};

80const documentReady = function() {
81
82 const appNav = $('#app-navigation');
83
84 appNav.on('click keydown', '#app-settings-header', function(event) {
85 if ($('#app-settings').hasClass('open')) {
86 $('#app-settings').switchClass('open', '');
87 } else {
88 $('#app-settings').switchClass('', 'open');
89 }
90 $('#app-settings-header').cafevTooltip('hide');
91 unfocus('#app-settings-header');
92 return false;
93 });
94
95 appNav.on('click', '#app-settings-further-settings', function(event) {
96 const $self = $(this);
97 $self.addClass('loading');
98 appSettings(
99 'settings/personal/form', {
100 done() {
101 const $popup = $(this);
102 $popup.trigger(appName + ':content-update'); // perhaps remove this
103 },
104 always() {
105 $self.removeClass('loading');
106 },
107 });
108
109 return false;
110 });
111
112};

114export default documentReady;
