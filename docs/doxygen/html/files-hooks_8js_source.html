60const acceptableMimeType = function(mimeType) {
61 return supportedMimeTypes.indexOf(mimeType) >= 0;

64const validTemplatePath = function(path) {
65 return path.startsWith(initialState.sharing.files.folders.templates);

68const getProjectNameFromProjectBalancesFolders = function(dirInfo) {
69 let dirName = dirInfo.path;
70 if (!dirName.startsWith(initialState.sharing.files.folders.projectBalances)) {
71 return null;
72 }
73 dirName = dirName.substring(initialState.sharing.files.folders.projectBalances.length);
74 dirName = dirName.replace(/^\/?(\d{4}|)\/?/, '');
75 const slashPos = dirName.indexOf('/');
76 const projectName = slashPos >= 0 ? dirName.substring(0, dirName.indexOf('/')) : dirName;
77 return projectName;
78};
79
80const getProjectYearFromProjectName = function(projectName) {
81 if (!projectName) {
82 return null;
83 }
84 const yearMatch = projectName.match(/\d{4}$/);
85 if (Array.isArray(yearMatch) && yearMatch.length === 1) {
86 return yearMatch[0];
87 }
88 return null;
89};
90
91const isProjectBalanceSupportingDocumentsTopFolder = function(dirInfo, projectName) {
92 projectName = projectName || getProjectNameFromProjectBalancesFolders(dirInfo);
93 if (!projectName) {
94 return false;
95 }
96 const dirName = dirInfo.path;
97 const baseName = dirInfo.name;
98 return dirName.startsWith(projectBalancesFolder)
99 && baseName === supportingDocumentsFolder;
100};
101
102const isProjectBalanceSupportingDocumentsFolder = function(dirInfo, projectName, projectYear) {
103 projectName = projectName || getProjectNameFromProjectBalancesFolders(dirInfo);
104 if (!projectName) {
105 return false;
106 }
107 projectYear = projectYear || getProjectYearFromProjectName(projectName);
108 const dirName = dirInfo.path;
109 const baseName = dirInfo.name;
110 if (projectYear) {
111 return isProjectBalanceSupportingDocumentsTopFolder(dirInfo, projectName);
112 } else {
113 const result = dirName.startsWith(projectBalancesFolder + '/' + projectName + '/' + supportingDocumentsFolder)
114 && baseName.match(/\d{4}$/);
115 return result;
116 }
117};

119const enableTemplateActions = function(fileInfo) {
120
121 if (fileInfo && fileInfo.isDirectory()) {
122 return false;
123 }
124
125 if (!acceptableMimeType(fileInfo.mimetype)) {
126 return false;
127 }
128
129 if (!validTemplatePath(fileInfo.path)) {
130 return false;
131 }
132
133 window.OCA.CAFEVDB.fileInfo = fileInfo;
134
135 return true; // TODO depend on subdir etc.
136};

258 async supportingDocumentHandler(name) {
259 const nameRegExp = new RegExp('^(?:' + this.folderPrefix + '-?)?' + '(\\d{3}|XXX)$');
260 const sequenceMatch = name.match(nameRegExp);
261 if (!sequenceMatch) {
262 showError(t(appName, 'The name of the new document in support must match the format "{projectName}-XXX" where "XXX" is a placeholder for 3 decimal digits or a literal "XXX" in which case the next available sequence number is chosen automatically.', this));
263 }
264 let sequence = sequenceMatch[1];
265 if (sequence === 'XXX') {
266 const sequences = [];
267 for (const file of this.fileList.files) {
268 if (file.mimetype !== 'httpd/unix-directory') {
269 continue;
270 }
271 if (!file.name.match(nameRegExp)) {
272 continue;
273 }
274 sequences.push(+file.name.substr(file.name.length - 3));
275 }
276 sequences.sort((a, b) => a - b);
277 let previous = 0;
278 for (const current of sequences) {
279 if (current - previous !== 1) {
280 break;
281 }
282 previous = current;
283 }
284 sequence = previous + 1;
285 }
286 sequence = String(sequence).padStart(3, '0');
287 showInfo(t(appName, 'Document sequence determined as {sequence}.', { sequence }));
288 const dirName = this.folderPrefix + '-' + sequence;
289 await this.fileList.createDirectory(dirName);
