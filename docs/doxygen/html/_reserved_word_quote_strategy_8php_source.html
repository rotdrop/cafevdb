43 private function getQuotedName(AbstractPlatform $platform, $name, $force = false)
44 {
45 $keywords = $platform->getReservedKeywordsList();
46 $parts = explode('.', $name);
47 foreach ($parts as $k => $v) {
48 if ($force || $keywords->isKeyword($v)) {
49 $parts[$k] = $platform->quoteIdentifier($v);
50 }
51 }
52
53 return implode('.', $parts);
54 }

153 public function getIdentifierColumnNames(ClassMetadata $class, AbstractPlatform $platform)
154 {
155 $quotedColumnNames = array();
156
157 foreach ($class->identifier as $fieldName) {
158 if (isset($class->fieldMappings[$fieldName])) {
159 $quotedColumnNames[] = $this->getColumnName($fieldName, $class, $platform);
160
161 continue;
162 }
163
164 // Association defined as Id field
165 $self = $this;
166 $joinColumns = $class->associationMappings[$fieldName]['joinColumns'];
167 $assocQuotedColumnNames = array_map(
168 function ($joinColumn) use ($platform, $self)
169 {
170 return $self->getQuotedName(
171 $platform,
172 $joinColumn['name'],
173 isset($joinColumn['quoted'])
174 );
175 },
176 $joinColumns
177 );
178
179 $quotedColumnNames = array_merge($quotedColumnNames, $assocQuotedColumnNames);
180 }
181
182 return $quotedColumnNames;
183 }

188 public function getColumnAlias($columnName, $counter, AbstractPlatform $platform, ClassMetadata $class = null)
189 {
190 // 1 ) Concatenate column name and counter
191 // 2 ) Trim the column alias to the maximum identifier length of the platform.
192 // If the alias is to long, characters are cut off from the beginning.
193 // 3 ) Strip non alphanumeric characters
194 // 4 ) Prefix with "_" if the result its numeric
195 $columnName = $columnName . '_' . $counter;
196 $columnName = substr($columnName, -$platform->getMaxIdentifierLength());
197 $columnName = preg_replace('/[^A-Za-z0-9_]/', '', $columnName);
198 $columnName = is_numeric($columnName) ? '_' . $columnName : $columnName;
199
200 return $platform->getSQLResultCasing($columnName);
201 }
