471 public function generateProjectPayments(Entities\ProjectParticipant $participant, array $receivableOptions, ?\DateTimeInterface $transactionDueDate = null):Entities\CompositePayment
472 {
473 $payments = new ArrayCollection();
474 $totalAmount = 0.0;
475 $project = $participant->getProject();
476 $musician = $participant->getMusician();
477
479 $compositePayment = (new Entities\CompositePayment)
480 ->setMusician($musician)
481 ->setProjectPayments($payments)
482 ->setAmount($totalAmount)
483 ->setSubject('');
484
485 if (empty($receivableOptions)) {
486 return $compositePayment;
487 }
488
490 foreach ($receivableOptions as $receivableOption) {
491 if ($project != $receivableOption->getField()->getProject()) {
492 throw new RuntimeException(
493 $this->l->t('Refusing to generate payments for mismatching projects, current is "%s", but the receivable belongs to project "%s".', [
494 $project->getName(),
495 $receivableOption->getField()->getProject()->getName(),
496 ]));
497 }
498 $receivableDueDate = $receivableOption->getField()->getDueDate();
499 $depositDueDate = $receivableOption->getField()->getDepositDueDate();
501 foreach ($receivableOption->getMusicianFieldData($musician) as $receivable) {
502 $paidAmount = $receivable->amountPaid();
503 $payableAmount = (float)$receivable->amountPayable();
504 $depositAmount = (float)$receivable->depositAmount();
505 if ((float)$payableAmount * (float)$depositAmount < 0) {
506 throw new RuntimeException($this->l->t('Payable amount "%f" and deposit amount "%f" should have the compatible signs.', [ $payableAmount, $depositAmount ]));
507 }
508 if (!empty($transactionDueDate) && !empty($receivableDueDate)) {
509 if ($payableAmount > 0) {
510 // debit note
511 if ($receivableDueDate <= $transactionDueDate) {
512 // past due-date, just keep billing the entire amount
513 } elseif ($depositDueDate <= $transactionDueDate) {
514 // past deposit-due date, charge the deposit
515 $payableAmount = $depositAmount;
516 } else {
517 // too early, just don't charge anything
518 $payableAmount = 0.0;
519 $this->logInfo('NOT YET DUE; DOING NOTHING');
520 }
521 } else {
522 // bank transfer
523 if ($transactionDueDate <= $depositDueDate) {
524 // early payment, only up to deposit
525 $payableAmount = $depositAmount;
526 } else {
527 // just transfer anything, i.e. keep the amount as is
528 }
529 }
530 }
531 $debitAmount = round($payableAmount - $paidAmount, 2);
532 if ($debitAmount == 0.0) {
533 // No need to debit empty amounts
534 // @todo Perhaps empty amounts should also be recorded.
535 continue;
536 }
538 $payment = (new Entities\ProjectPayment)
539 ->setProject($project)
540 ->setMusician($musician)
541 ->setReceivable($receivable)
542 ->setReceivableOption($receivableOption)
543 ->setAmount($debitAmount)
544 ->setCompositePayment($compositePayment)
545 ->setSubject($receivable->paymentReference());
546
547 $payments->add($payment);
548 $totalAmount += $debitAmount;
549 }
550 }
551
552 $compositePayment
553 ->setMusician($participant->getMusician())
554 ->setProject($participant->getProject())
555 ->setAmount($totalAmount)
556 ->updateSubject(fn($x) => $this->financeService->sepaTranslit($x));
557
558 return $compositePayment;
559 }
