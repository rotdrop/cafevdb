97 public function serviceSwitch(
98 string $operation,
99 string $list,
100 string $email,
101 ?string $displayName = null,
102 ?string $role = null,
103 ):DataResponse {
104 if ($list == 'announcements') {
105 $list = $this->getConfigValue('announcementsMailingList');
106 }
107 try {
108 switch ($operation) {
109 case self::OPERATION_INVITE:
110 $this->logInfo('INVITE ' . $list . ' / ' . $email);
111 $this->listsService->invite($list, email: $email, displayName: $displayName);
112 break;
113 case self::OPERATION_SUBSCRIBE:
114 $this->logInfo('SUBSCRIBE ' . $list . ' / ' . $email);
115 $this->listsService->subscribe($list, email: $email, displayName: $displayName, role: $role);
116 break;
117 case self::OPERATION_UNSUBSCRIBE:
118 $this->logInfo('UNSUBSCRIBE ' . $list . ' / ' . $email);
119 $this->listsService->unsubscribe($list, $email);
120 break;
121 case self::OPERATION_ACCEPT:
122 $this->logInfo('ACCEPT SUBSCRIPTION ' . $list . ' / ' . $email);
123 $this->listsService->handleSubscriptionRequest($list, $email, MailingListsService::MODERATION_ACTION_ACCEPT);
124 break;
125 case self::OPERATION_REJECT:
126 $this->logInfo('REJECT SUBSCRIPTION ' . $list . ' / ' . $email);
127 $this->listsService->handleSubscriptionRequest($list, $email, MailingListsService::MODERATION_ACTION_REJECT, 'test reason');
128 break;
129 case self::OPERATION_RELOAD:
130 break;
131 default:
132 return self::grumble($this->l->t('Unknown mailing list operation "%s"', $operation));
133 }
134 $status = $this->listsService->getSubscriptionStatus($list, $email);
135 } catch (HttpClientConnectException $e) {
136 throw new Exceptions\EnduserNotificationException($this->l->t('The mailing-list service is unreachable'), 0, $e);
137 }
138
139 return self::dataResponse([
140 'status' => $status,
141 ]);
142 }
