224 public function shortTitle()
225 {
226 if ($this->deleteOperation()) {
227 return $this->l->t('Remove this Bank-Account?');
228 } elseif ($this->viewOperation()) {
229 if ($this->projectId > 0 && $this->projectName != '') {
230 return $this->l->t('Bank-Account for %s', array($this->projectName));
231 } else {
232 return $this->l->t('Bank-Account');
233 }
234 } elseif ($this->changeOperation()) {
235 return $this->l->t('Change this Bank-Account');
236 }
237 if ($this->projectId > 0 && $this->projectName != '') {
238 return $this->l->t('Overview over all SEPA Bank Accounts for %s', [ $this->projectName ]);
239 } else {
240 return $this->l->t('Overview over all SEPA Bank Accounts');
241 }
