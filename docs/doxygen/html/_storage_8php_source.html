150 return $this->files[$dirName];

153 $dirComponents = Util::explode(self::PATH_SEPARATOR, $dirName);
154
155 $filterState = $this->disableFilter(EntityManager::SOFT_DELETEABLE_FILTER);
156
158 $folderDirEntry = $this->rootFolder;
159 if (!$rootIsMandatory && empty($dirName) && empty($folderDirEntry)) {
160 $this->files[$dirName] = [ '.' => $folderDirEntry ?? new DirectoryNode('.', new DateTimeImmutable('@1')) ];
161 if (empty($folderDirEntry)) {

174 $filterState && $this->enableFilter(EntityManager::SOFT_DELETEABLE_FILTER);

182 $this->files[$dirName] = [ '.' => $folderDirEntry ];

580 public function writeStream(string $path, $stream, int $size = null): int
581 {
582 // $this->logInfo('WRITE STREAM ' . $path);
583
584 if (!$this->touch($path)) {
585 return false;
586 }
588 $file = $this->fileFromFileName($path);
589 if (empty($file)) {
590 return false;
591 }
592 if ($size === null) {
593 $stream = CountWrapper::wrap($stream, function ($writtenSize) use (&$size) {
594 $size = $writtenSize;
595 });
596 }
597
598 $fileData = stream_get_contents($stream);
599 fclose($stream);
600
601 $this->entityManager->beginTransaction();
602 try {
603 $file->getFileData()->setData($fileData);
605 $mimeTypeDetector = $this->di(IMimeTypeDetector::class);
606 $file->setMimeType($mimeTypeDetector->detectString($fileData));
607 $file->setSize(strlen($fileData));
608
609 $this->flush();
610
611 $this->entityManager->commit();
612 } catch (Throwable $t) {
613 $this->logException($t, 'writeStream() failed');
614 if ($this->entityManager->isOwnTransactionActive()) {
615 $this->entityManager->rollback();
616 }
617 return -1;
618 }
619
620 return $size;
621 }
