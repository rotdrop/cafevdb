169 private function fillInternal(File $templateFile, string $templateFileName, array $templateData, array $blocks, bool $asPdf)
170 {
171 ob_start();
172
173 $fillData = $this->fillData($templateData);
174
175 $this->backend->ResetVarRef(false);
176 $this->backend->VarRef = $fillData;
177
178 unset($fillData['now']); // the cache is just for this request, so "now" from the first call is good enough
179 unset($fillData['date']); // the cache is just for this request, so "now" from the first call is good enough
180 $templateDataHash = md5(json_encode($fillData));
181
182 $this->backend->LoadTemplate($templateFile->fopen('r'), OPENTBS_ALREADY_UTF8);
183
184 // OpenTBS does not support DateTimeInterface or time-zones, so
185 // convert everything to a time-stamp and add the timezone-offset
186 // to get correct dates and times.
187 array_walk_recursive($this->backend->VarRef, function(&$value, $key) {
188 if ($value instanceof \DateTimeInterface) {
189 $stamp = $value->getTimestamp();
190 $stamp += $value->getOffset();
191 $value = $stamp;
192 }
193 });
194
195 // do a serialize - unserialize
196 $fileHash = $templateFile->getEtag();
197
198 if (empty($this->cache[$fileHash][$templateDataHash])) {
199
200 $this->backend->VarRef = json_decode(json_encode($this->backend->VarRef), true);
201
202 // Do an opportunistic block-merge for every key with is an array
203 foreach ($this->backend->VarRef as $key => $value) {
204 if (is_array($value)) {
205 // try to do a block-merge primarily to have shorter names in
206 // the tempalte. If the array is not a numeric array, then
207 // wrap it into a single element numeric array in order to
208 // please TBS.
209 $keys = array_keys($value);
210 if ($keys != array_filter($keys, 'is_int')) {
211 $value = [ $value ];
212 }
213 $this->backend->MergeBlock($key, $value);
214 }
215 }
216
217 // Do a block-merge for every explicitly defined block
218 foreach ($blocks as $key => $reference) {
219 $indices = explode('.', $reference);
220 $value = $this->backend->VarRef;
221 while (!empty($indices) && !empty($value)) {
222 $index = array_shift($indices);
223 $value = $value[$index]??null;
224 }
225 if (empty($value)) {
226 throw new RuntimeException($this->l->t('Data for block "%s" using the path "%s" could not be found in the substitution data.', [ $key, $reference ]));
227 }
228 $keys = array_keys($value);
229 if ($keys != array_filter($keys, 'is_int')) {
230 $value = [ $value ];
231 }
232 $this->backend->MergeBlock($key, $value);
233 }
234
235 $this->backend->show(OPENTBS_STRING);
236
237 $output = ob_get_contents();
238 ob_end_clean();
239
240 if (!empty($output)) {
241 throw new Exceptions\Exception($output);
242 }
243
244 $fileData = $this->backend->Source;
245 $mimeType = $templateFile->getMimeType();
246
247 $this->cache[$fileHash][$templateDataHash] = [
248 'fileData' => $fileData,
249 'mimeType' => $mimeType,
250 ];
251 } else {
252 list('fileData' => $fileData, 'mimeType' => $mimeType) = $this->cache[$fileHash][$templateDataHash];
253 }
254
255 $fileName = basename($templateFileName);
256
257 if ($asPdf) {
258
259 if (empty($this->cache[$fileHash][$templateDataHash]['pdfData'])) {
260 $fileData = $this->anyToPdf->convertData($fileData);
261 $this->cache[$fileHash][$templateDataHash]['pdfData'] = $fileData;
262 } else {
263 $fileData = $this->cache[$fileHash][$templateDataHash]['pdfData'];
264 }
265
266 $mimeType = 'application/pdf';
267 $pathInfo = pathinfo($fileName);
268 $fileName = $pathInfo['filename'] . '.pdf';
269 }
270
271 return [
272 $fileData,
273 $mimeType,
274 $fileName,
275 ];
276 }
