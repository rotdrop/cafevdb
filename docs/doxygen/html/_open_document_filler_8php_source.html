152 private function fillInternal(File $templateFile, string $templateFileName, array $templateData, array $blocks, bool $asPdf)
153 {
154 ob_start();
155
156 $fillData = $this->fillData($templateData);
157
158 $this->backend->ResetVarRef(false);
159 $this->backend->VarRef = $fillData;
160
161 unset($fillData['now']); // the cache is just for this request, so "now" from the first call is good enough
162 unset($fillData['date']); // the cache is just for this request, so "now" from the first call is good enough
163 $templateDataHash = md5(json_encode($fillData));
164
165 $this->backend->LoadTemplate($templateFile->fopen('r'), OPENTBS_ALREADY_UTF8);
166
167 // OpenTBS does not support DateTimeInterface or time-zones, so
168 // convert everything to a time-stamp and add the timezone-offset
169 // to get correct dates and times.
170 array_walk_recursive($this->backend->VarRef, function(&$value, $key) {
171 if ($value instanceof \DateTimeInterface) {
172 $stamp = $value->getTimestamp();
173 $stamp += $value->getOffset();
174 $value = $stamp;
175 }
176 });
177
178 // do a serialize - unserialize
179 $fileHash = $templateFile->getEtag();
180
181 if (empty($this->cache[$fileHash][$templateDataHash])) {
182
183 $this->backend->VarRef = json_decode(json_encode($this->backend->VarRef), true);
184
185 // Do an opportunistic block-merge for every key with is an array
186 foreach ($this->backend->VarRef as $key => $value) {
187 if (is_array($value)) {
188 // try to do a block-merge primarily to have shorter names in
189 // the tempalte. If the array is not a numeric array, then
190 // wrap it into a single element numeric array in order to
191 // please TBS.
192 $keys = array_keys($value);
193 if ($keys != array_filter($keys, 'is_int')) {
194 $value = [ $value ];
195 }
196 $this->backend->MergeBlock($key, $value);
197 }
198 }
199
200 // Do a block-merge for every explicitly defined block
201 foreach ($blocks as $key => $reference) {
202 $indices = explode('.', $reference);
203 $value = $this->backend->VarRef;
204 while (!empty($indices) && !empty($value)) {
205 $index = array_shift($indices);
206 $value = $value[$index]??null;
207 }
208 if (empty($value)) {
209 throw new RuntimeException($this->l->t('Data for block "%s" using the path "%s" could not be found in the substitution data.', [ $key, $reference ]));
210 }
211 $keys = array_keys($value);
212 if ($keys != array_filter($keys, 'is_int')) {
213 $value = [ $value ];
214 }
215 $this->backend->MergeBlock($key, $value);
216 }
217
218 $this->backend->show(OPENTBS_STRING);
219
220 $output = ob_get_contents();
221 ob_end_clean();
222
223 if (!empty($output)) {
224 throw new Exceptions\Exception($output);
225 }
226
227 $fileData = $this->backend->Source;
228 $mimeType = $templateFile->getMimeType();
229
230 $this->cache[$fileHash][$templateDataHash] = [
231 'fileData' => $fileData,
232 'mimeType' => $mimeType,
233 ];
234 } else {
235 list('fileData' => $fileData, 'mimeType' => $mimeType) = $this->cache[$fileHash][$templateDataHash];
236 }
237
238 $fileName = basename($templateFileName);
239
240 if ($asPdf) {
241
242 if (empty($this->cache[$fileHash][$templateDataHash]['pdfData'])) {
243 $fileData = $this->anyToPdf->convertData($fileData);
244 $this->cache[$fileHash][$templateDataHash]['pdfData'] = $fileData;
245 } else {
246 $fileData = $this->cache[$fileHash][$templateDataHash]['pdfData'];
247 }
248
249 $mimeType = 'application/pdf';
250 $pathInfo = pathinfo($fileName);
251 $fileName = $pathInfo['filename'] . '.pdf';
252 }
253

256 $mimeType,
257 $fileName,
