169 private function preProcessKey(string $key):string
170 {
171 foreach (self::SUBKEY_PREFIXES as $prefix) {
172 if (str_starts_with($key, $prefix . '-')) {
173 $key = str_replace(['--', '+-'], ['-minus-', '-plus-'], $key);
174 $lastChar = substr($key, -1);
175 switch ($lastChar) {
176 case '-':
177 $key = substr($key, 0, -1) . '-minus';
178 break;
179 case '+':
180 $key = substr($key, 0, -1) . '-plus';
181 break;
182 }
183 return str_replace('-', self::SUB_KEY_SEP, $key);
184 }
185 }
186 return $key;
187 }

198 public function fetch(string $key, bool $escape = true):?string
199 {
200 if ($key == $this->lastKey && $escape) {
201 return $this->lastToolTip;
202 }
203 $this->lastKey = $key;
204
205 $this->makeToolTips();
206 $toolTipsData = $this->toolTipsData;
207
208 $key = $this->preprocessKey($key);
209 $keys = explode(self::SUB_KEY_SEP, $key);
210 while (\count($keys) > 0) {
211 $key = array_shift($keys);
212 $toolTipsData = $toolTipsData[$key] ?? ($toolTipsData['default'] ?? null);
213 }
214 $tip = $toolTipsData['default'] ?? $toolTipsData;
215
216 if (!is_scalar($tip)) {
217 $tip = null;
218 }
219
220 if (empty($tip)) {
221 $this->failedKeys[] = $this->lastKey;
222 if ($this->debug) {
223 $tip = $this->l->t('Unknown Tooltip for key "%s" requested.', $this->lastKey);
224 }
225 } else {
226 $this->failedKeys = [];
227 if ($this->debug) {
228 $tip .= ' (' . $this->l->t('ToolTip-Key "%s"', $this->lastKey) . ')';
229 }
230 }
231
232 $tip = preg_replace('/(^\s*[\n])+/m', '<p class="tooltip-paragraph">', $tip);
233
234 // idea: allow markdown?
235
236 $this->lastToolTip = empty($tip) ? null : ($escape ? htmlspecialchars($tip) : $tip);
237
238 return $this->lastToolTip;
239 }
