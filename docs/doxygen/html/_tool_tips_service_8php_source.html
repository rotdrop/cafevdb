160 private function preProcessKey(string $key):string
161 {
162 foreach (self::SUBKEY_PREFIXES as $prefix) {
163 if (str_starts_with($key, $prefix . '-')) {
164 $key = str_replace(['--', '+-'], ['-minus-', '-plus-'], $key);
165 $lastChar = substr($key, -1);
166 switch ($lastChar) {
167 case '-':
168 $key = substr($key, 0, -1) . '-minus';
169 break;
170 case '+':
171 $key = substr($key, 0, -1) . '-plus';
172 break;
173 }
174 return str_replace('-', self::SUB_KEY_SEP, $key);
175 }
176 }
177 return $key;
178 }

189 public function fetch(string $key, bool $escape = true):?string
190 {
191 if ($key == $this->lastKey && $escape) {
192 return $this->lastToolTip;
193 }
194 $this->lastKey = $key;
195
196 $toolTipsData = ToolTipsDataService::get();
197
198 $key = $this->preprocessKey($key);
199
200 $keys = explode(self::SUB_KEY_SEP, $key);
201 while (\count($keys) > 0) {
202 $key = array_shift($keys);
203 $toolTipsData = $toolTipsData[$key] ?? ($toolTipsData['default'] ?? null);
204 }
205 $tip = $toolTipsData['default'] ?? $toolTipsData;
206
207 if (!isset($tip['text'])) {
208 $tip = null;
209 }
210
211 if (empty($tip)) {
212 $this->failedKeys[] = $this->lastKey;
213 if ($this->debug) {
214 $tip = $this->l->t('Unknown Tooltip for key "%s" requested.', $this->lastKey);
215 }
216 } else {
217 $this->failedKeys = [];
218 $parameters = array_map(
219 fn(mixed $value) => $value instanceof Closure ? $value($this->l, $this->appContainer) : $value,
220 $tip['parameters'] ?? [],
221 );
222 $tip = $this->l->t($tip['text'], $parameters);
223 if ($this->debug) {
224 $tip .= ' (' . $this->l->t('ToolTip-Key "%s"', $this->lastKey) . ')';
225 }
226 }
227
228 $tip = preg_replace('/(^\s*[\n])+/m', self::PARAGRAPH . "\n", $tip);
229
230 // idea: allow markdown?
231
232 $this->lastToolTip = empty($tip) ? null : ($escape ? htmlspecialchars($tip) : $tip);
233
234 return $this->lastToolTip;
235 }
