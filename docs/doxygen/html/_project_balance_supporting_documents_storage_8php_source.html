175 private function isContainerDirectory(string $path):?int
176 {
177 switch ($this->project->getType()) {
178 case ProjectType::TEMPLATE:
179 break;
180 case ProjectType::PERMANENT:
181 // Folder have a year prefix and the supporting document has the year in its name
182 if (preg_match('|^/?' . '\d{4}' . '/' . $this->project->getName() . '-' . '\d{4}' . '-' . '(\d{3})/?$|', $path, $matches)) {
183 return $matches[1];
184 }
185 break;
186 case ProjectType::TEMPORARY:
187 // project-name contains already the year
188 if (preg_match('|^/?' . $this->project->getName() . '-' . '(\d{3})/?$|', $path, $matches)) {
189 return $matches[1];
190 }
191 break;
192 }
193 return null;
194 }

464 public function unlink($path)
465 {
466 if ($this->is_dir($path)) {
467 return false;
468 }
470 $dirEntry = $this->fileFromFileName($path);
471 if (empty($dirEntry)) {
472 throw new Exceptions\DatabaseStorageException(
473 $this->l->t('Unable to find database entity for path "%s".', $path)
474 );
475 }
476 if ($dirEntry instanceof Entities\DatabaseStorageFolder) {
477 throw new Exceptions\DatabaseStorageException(
478 $this->l->t('Path "%s" is a directory.', $path)
479 );
480 }
481
482 $parent = $dirEntry->getParent();
483 if (empty($parent)) {
484 throw new Exceptions\DatabaseStorageException(
485 $this->l->t('Unable to find document container for path "%s".', $path)
486 );
487 }
488 $file = $dirEntry->getFile();
489 if (empty($file)) {
490 throw new Exceptions\DatabaseStorageException(
491 $this->l->t('The directory entry "%s" is not linked to a file.', $path)
492 );
493 }
494
495 $this->entityManager->beginTransaction();
496 try {
497 $dirEntry->setParent(null);
498 $dirEntry->setFile(null);
499 $this->entityManager->remove($dirEntry);
500
501 // break the link to the supporting documents of payments
502 // @todo Maybe better use a find-by instead of bloating the entities with too many associations.
503
505 $repository = $this->entityManager->getRepository(Entities\CompositePayment::class);
506 $compositePayments = $repository->findBy([
507 'balanceDocumentsFolder' => $parent,
508 'supportingDocument' => $file,
509 ]);
511 foreach ($compositePayments as $compositePayment) {
512 $compositePayment->setBalanceDocumentsFolder(null);
513 }
514
516 $repository = $this->entityManager->getRepository(Entities\ProjectPayment::class);
517 $projectPayments = $repository->findBy([
518 'balanceDocumentsFolder' => $parent,
519 ]);
521 foreach ($projectPayments as $projectPayment) {
522 $supportingDocument = $projectPayment->getReceivable() ? $projectPayment->getReceivable()->getSupportingDocument() : null;
523 if ($supportingDocument == $file) {
524 $projectPayment->setBalanceDocumentsFolder(null);
525 }
526 }
527
528 $this->flush();
529
530 $this->entityManager->commit();
531
532 $this->unsetFileNameCache($path);
533 } catch (\Throwable $t) {
534 $this->logException($t);
535 if ($this->entityManager->isTransactionActive()) {
536 $this->entityManager->rollback();
537 }
538 return false;
