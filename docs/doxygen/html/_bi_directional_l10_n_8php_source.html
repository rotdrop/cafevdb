161 protected static function parseCSV(string $fileName, string $targetLang, string $keyLang = 'en'):?array
162 {
163 $file = fopen($fileName, 'r');
164 if ($file === false) {
165 return null;
166 }
167 $header = fgetcsv($file);
168 if (empty($header)) {
169 return null;
170 }
171
172 $targetColumn = array_search($targetLang, $header);
173 $keyColumn = array_search($keyLang, $header);
174
175 $targetLookup = [];
176 $keyLookup = [];
177 for ($row = fgetcsv($file); !empty($row); $row = fgetcsv($file)) {
178 if (empty($row[$targetColumn]) || empty($row[$keyColumn])) {
179 continue;
180 }
181 $keyPhrases = explode(';', $row[$keyColumn]);
182 $targetPhrases = explode(';', $row[$targetColumn]);
183 foreach ($keyPhrases as $keyPhrase) {
184 $keyPhrase = trim($keyPhrase);
185 foreach ($targetPhrases as $targetPhrase) {
186 $targetPhrase = trim($targetPhrase);
187 if (empty($targetLookup[$keyPhrase])) {
188 $targetLookup[$keyPhrase] = $targetPhrase;
189 }
190 if (empty($keyLookup[$targetPhrase])) {
191 $keyLookup[$targetPhrase] = $keyPhrase;
192 }
193 }
194 }
195 }
196 return [
197 self::FORWARD => $targetLookup,
198 self::REVERSE => $keyLookup,
199 ];
200 }
