104 public function move(
105 string $stashedFile,
106 string $destinationPath,
107 string $uploadMode = self::UPLOAD_MODE_COPY,
108 ?string $originalFileName = null,
109 string $storage = self::MOVE_DEST_CLOUD,
110 bool $encrypted = false,
111 int $ownerId = 0,
112 string $uploadFolder = AppStorage::UPLOAD_FOLDER
113 ):Response {
114 if ($uploadMode == self::UPLOAD_MODE_MOVE) {
115 if (empty($originalFileName)) {
116 return self::grumble($this->l->t(
117 'Original file path is not given, cannot move files.'
118 ));
119 }
121 $originalFile = $this->userStorage->getFile($originalFileName);
122 if (empty($originalFile)) {
123 return self::grumble($this->l->t(
124 'The original file "%s" cannot be found, cannot move files.',
125 $originalFileName
126 ));
127 }
128 if (!($originalFile->getPermissions() & CloudConstants::PERMISSION_DELETE)) {
129 return self::grumble($this->l->t(
130 'Original file "%s" cannot be deleted, moving it is therefore not possible.',
131 $originalFileName
132 ));
133 }
134 $originalFile->delete();
135 }
136
137 $appFile = $this->appStorage->getFile($uploadFolder, $stashedFile);
138 switch ($storage) {
139 case self::MOVE_DEST_CLOUD:
140 if ($uploadMode == self::UPLOAD_MODE_LINK) {
141 return self::grumble($this->l->t(
142 'Linking files is only support when the destination storage is backed by the data-base.'
143 ));
144 }
145
146 $this->userStorage->putContent($destinationPath, $appFile->getContent());
147 $downloadLink = $this->userStorage->getDownloadLink($destinationPath);
148 $appFile->delete();
149
150 return self::dataResponse([
151 'message' => $this->l->t('Moved "%s" to "%s".', [ $stashedFile, $destinationPath ]),
152 'fileName' => basename($destinationPath),
153 'downloadLink' => $downloadLink,
154 ]);
155 case self::MOVE_DEST_DB:
156 // here $destinationPath is the file-name in the data-base
157 if (empty($this->entityManager)) {
158 $this->entityManager = $this->di(EntityManager::class);
159 }
160
161 $dbFileClass = $encrypted ? Entities\EncryptedFile::class : Entities\File::class;
162 if ($uploadMode == self::UPLOAD_MODE_LINK) {
163 // this is somewhat academic as here is no dedicate storage
164 // location. However, this is how linking in principle works: just
165 // increase the link-count.
166 $dbFile = $this->entityManager->find(Entities\File::class, $originalFileName);
167 if (empty($dbFile)) {
168 return self::grumble($this->l->t('Link source cannot be found.'));
169 }
170 if ($encrypted && !($dbFile instanceof Entities\EncryptedFile)) {
171 return self::grumble($this->l->t(
172 'Encryption requested, but link-source "%s" is unencrypted',
173 $dbFile->getName()
174 ));
175 }
176 } else {
178 $dbFile = new $dbFileClass(
179 fileName: $destinationPath,
180 data: $appFile->getContent(),
181 mimeType: $appFile->getMimeType()
182 );
183 $dbFile->setFileName($originalFileName);
184 }
185 if ($encrypted && $ownerId > 0) {
186 $owner = $this->getDatabaseRepository(Entities\Musician::class)->find($ownerId);
187 $dbFile->addOwner($owner);
188 }
189
190 $this->entityManager->beginTransaction();
191 try {
192 $this->persist($dbFile);
193 $this->flush();
194 $this->entityManager->commit();
195 } catch (\Throwable $t) {
196 $this->logException($t);
197 $this->entityManager->rollback();
198 return self::grumble($this->exceptionChainData($t));
199 }
200
201 // ok, all fine
202 $appFile->delete();
203
204 $downloadLink = $this->urlGenerator()->linkToRoute(
205 $this->appName().'.downloads.get', [
206 'section' => 'database',
207 'object' => $dbFile->getId(),
208 ])
209 . '?requesttoken=' . urlencode(\OCP\Util::callRegister())
210 . '&fileName=' . urlencode(basename($destinationPath));
211
212 return self::dataResponse([
213 'message' => $this->l->t('Moved "%1$s" to db-storage with name "%2$s", id %d.', [ $stashedFile, $destinationPath, $dbFile->getId() ]),
214 'fileName' => basename($destinationPath),
215 'fileId' => $dbFile->getId(),
216 'downloadLink' => $downloadLink,
217 ]);
218 }
219 return self::grumble($this->l->t('Unknown request'));
220 }

254 public function stash(
255 array $cloudPaths = [],
256 string $uploadMode = self::UPLOAD_MODE_COPY,
257 string $uploadFolder = AppStorage::UPLOAD_FOLDER,
258 ):Response {
259 $uploadMaxFileSize = \OCP\Util::computerFileSize(ini_get('upload_max_filesize'));
260 $postMaxSize = \OCP\Util::computerFileSize(ini_get('post_max_size'));
261 $maxUploadFileSize = min($uploadMaxFileSize, $postMaxSize);
262 $maxHumanFileSize = \OCP\Util::humanFileSize($maxUploadFileSize);
263
264 $uploads = [];
265 if (!empty($cloudPaths)) {
266 foreach ($cloudPaths as $path) {
267 $files = [];
268
270 $cloudFile = $this->userStorage->get($path);
271
273 $dbFile = $this->userStorage->getDatabaseFile($path);
274 if (!empty($dbFile) && !($dbFile instanceof Entities\DatabaseStorageFile)) {
275 return self::grumble($this->l->t('File "%s" referes to a database storage folder, we expected a plain file.', $path));
276 }
277
278 if (empty($cloudFile)) {
279 return self::grumble($this->l->t('File "%s" could not be found in cloud storage.', $path));
280 }
281 if ($cloudFile->getType() != FileInfo::TYPE_FILE) {
282 return self::grumble($this->l->t('File "%s" is not a plain file, this is not yet implemented.'));
283 }
284
285 $fileName = $cloudFile->getName();
286 $fileEntityId = !empty($dbFile) ? $dbFile->getFile()->getId() : null;
287
288 switch ($uploadMode) {
289 case self::UPLOAD_MODE_TEST:
290 $uploadModes = [ self::UPLOAD_MODE_COPY, ];
291 if ($cloudFile->getPermissions() & CloudConstants::PERMISSION_DELETE) {
292 $uploadModes[] = self::UPLOAD_MODE_MOVE;
293 }
294 if (!empty($dbFile)) {
295 $uploadModes[] = self::UPLOAD_MODE_LINK;
296 }
297 $uploads[] = [
298 'original_name' => $path,
299 'upload_mode' => $uploadModes,
300 ];
301 continue 2;
302 case self::UPLOAD_MODE_LINK:
303 if (empty($dbFile)) {
304 return self::grumble($this->l->t(
305 'File "%s" is not backed by database-storage and thus cannot be linked.',
306 $cloudFile->getName()
307 ));
308 }
309 $originalName = $fileEntityId;
310 $tmpName = null;
311 break;
312 case self::UPLOAD_MODE_MOVE:
313 if (!($cloudFile->getPermissions() & CloudConstants::PERMISSION_DELETE)) {
314 return self::grumble($this->l->t(
315 'File "%s" cannot be deleted, moving it is therefor not possible.',
316 $fileName
317 ));
318 }
319 // the actual deletion should be post-poned until the stashed file
320 // has been moved into place.
321 // no break
322 case self::UPLOAD_MODE_COPY:
323 try {
324 $uploadFile = $this->appStorage->newTemporaryFile($uploadFolder);
325 $uploadFile->putContent($cloudFile->getContent());
326 } catch (\Throwable $t) {
327 return self::grumble($this->l->t('Could not copy cloud file "%s" to upload storage.', $fileName));
328 }
329 $originalName = $uploadMode == self::UPLOAD_MODE_MOVE ? $path : $fileName;
330 $tmpName = $uploadFile->getName();
331 break;
332 }
333
334 // We emulate an uploaded file here:
335 $fileRecord = [
336 'name' => $fileName,
337 'error' => 0,
338 'tmp_name' => $tmpName,
339 'type' => $cloudFile->getMimetype(),
340 'size' => $cloudFile->getSize(),
341 'upload_max_file_size' => $maxUploadFileSize,
342 'max_human_file_size' => $maxHumanFileSize,
343 'upload_mode' => $uploadMode,
344 'original_name' => $originalName,
345 ];
346 $uploads[] = $fileRecord;
347 }
348 } else {
349 if ($uploadMode != self::UPLOAD_MODE_COPY) {
350 return self::grumble($this->l->t(
351 'For client-uploads the only supported upload-mode is "copy", "%s" is not possible.',
352 $uploadMode
353 ));
354 }
355
356 $files = $this->request->files[self::UPLOAD_KEY];
357 if (empty($files)) {
358 // may be caused by PHP restrictions which are not caught by
359 // error handlers.
360 $contentLength = $this->request->server['CONTENT_LENGTH'];
361 $limit = \OCP\Util::uploadLimit();
362 if ($contentLength > $limit) {
363 return self::grumble(
364 $this->l->t('Upload size %s exceeds limit %s, contact your server administrator.', [
365 \OCP\Util::humanFileSize($contentLength),
366 \OCP\Util::humanFileSize($limit),
367 ]));
368 }
369 $error = error_get_last();
370 if (!empty($error)) {
371 return self::grumble(
372 $this->l->t('No file was uploaded, error message was "%s".', $error['message']));
373 }
374 return self::grumble($this->l->t('No file was uploaded. Unknown error'));
375 }
376
377 $files = Util::transposeArray($files);
378
379 $totalSize = 0;
380 foreach ($files as $file) {
381
382 $totalSize += $file['size'];
383
384 if ($maxUploadFileSize >= 0 and $totalSize > $maxUploadFileSize) {
385 return self::grumble([
386 'message' => $this->l->t('Not enough storage available'),
387 'upload_max_file_size' => $maxUploadFileSize,
388 'max_human_file_size' => $maxHumanFileSize,
389 ]);
390 }
391
392
393 $file['str_error'] = Util::fileUploadError($file['error'], $this->l);
394 if ($file['error'] != UPLOAD_ERR_OK) {
395 continue;
396 }
397
398 $file['upload_max_file_size'] = $maxUploadFileSize;
399 $file['max_human_file_size'] = $maxHumanFileSize;
400 $file['original_name'] = $file['name']; // clone
401 $file['upload_mode'] = self::UPLOAD_MODE_COPY;
402
403 try {
404 $uploadFile = $this->appStorage->newTemporaryFile($uploadFolder);
405 $this->appStorage->moveFileSystemFile($file['tmp_name'], $uploadFile);
406 $file['name'] = $uploadFile->getName();
407 $file['tmp_name'] = $file['name'];
408 } catch (\Throwable $t) {
409 $file['error'] = 99;
410 $file['str_error'] = $this->l->t('Couldn\'t save temporary file for: %s', $file['name']);
411 continue;
412 }
413 $uploads[] = $file;
414 }
415 }
416 return self::dataResponse($uploads);
417 }
