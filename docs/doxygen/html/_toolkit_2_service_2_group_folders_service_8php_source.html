163 public function searchFolders(string $regexp, string $topic = self::SEARCH_TOPIC_MOUNT):array
164 {
165 if ($this->sharedFolders === null) {
166 $this->fetchFolders();
167 }
168 if ($regexp[0] != $regexp[-1]) {
169 $regexp = '|^' . $regexp . '$|';
170 }
171 $this->logDebug('REGEXP ' . $regexp . ' / ' . $topic);
172 switch ($topic) {
173 case self::SEARCH_TOPIC_MOUNT:
174 return array_filter($this->sharedFolders, function($folderInfo) use ($regexp) {
175 return preg_match($regexp, $folderInfo['mount_point']);
176 });
177 case self::SEARCH_TOPIC_GROUP:
178 return array_filter($this->sharedFolders, function($folderInfo) use ($regexp) {
179 foreach (array_keys($folderInfo['groups']) as $group) {
180 if (preg_match($regexp, $group)) {
181 return true;
182 }
183 return false;
184 }
185 });
186 default:
187 throw new RuntimeException($this->l->t('Unknown search topic "%1$s"', $topic));
188 }
189 }

337 private function changeFolderManager(string $mountPoint, string $managerId, string $type, bool $canManage):void
338 {
339 // POST BASEURL/groupfolders/folders/8/acl
340 // [ acl => 0 / 1 ]
341 //
342 // POST BASEURL/groupfolders/folders/4/manageACL
343 // [ mappingType => MANAGER_TYPE, mappingId => MANAGER_ID, manageAcl=> 1 ]
344 //
345 $folderInfo = $this->getFolder($mountPoint);
346 if (empty($folderInfo)) {
347 throw new RuntimeException($this->l->t('Shared folder for mount-point "%1$s" does not exist, cannot modify manager "%2$s".', [ $mountPoint, $managerId ]));
348 }
349
350 $this->logDebug('FOLDER INFO ' . print_r($folderInfo, true));
351
352 // first check if anything needs to be done
353 $aclEnabled = !!($folderInfo['acl'] ?? false);
354 $isManager = 0 < count(array_filter($folderInfo['manage'] ?? [], function($manager) use ($managerId, $type) {
355 return $manager['type'] == $type && $manager['id'] == $managerId;
356 }));
357
358 if ($isManager == $canManage && ($canManage || $isManager == $canManage)) {
359 return;
360 }
361
362 $folderId = $folderInfo['id'];
363 if (!$aclEnabled) {
364 $this->folderManager->setFolderACL($folderId, true);
365 }
366
367 if ($isManager != $canManage) {
368 $this->folderManager->setManageACL($folderId, $type, $managerId, $canManage);
369 }
370 if (!$aclEnabled || $isManager != $canManage) {
371 $this->getFolderById($folderId);
372 }
373 }

444 public function changeMountPoint(string $mountPoint, string $targetMountPoint, bool $moveChildren = true):void
445 {
446 if ($mountPoint == $targetMountPoint) {
447 return;
448 }
449 $folderInfo = $this->getFolder($mountPoint);
450 if (empty($folderInfo)) {
451 throw new RuntimeException($this->l->t('Shared folder for mount-point "%1$s" does not exist, cannot change its mount-point to "%2$s".', [ $mountPoint, ]));
452 }
453 // POST BASE_URL/groupfolders/folders/FOLDER_ID/mountpoint
454 // DATA [ mountpoint => NEW_MOUNT_POINT ]
455
456 $this->folderManager->renameFolder($folderInfo['id'], $targetMountPoint);
457 $this->sharedFolders[$targetMountPoint] = $this->sharedFolders[$mountPoint];
458 unset($this->sharedFolders[$mountPoint]);
459 $this->sharedFolders[$targetMountPoint]['mount_point'] = $targetMountPoint;
460
461 if ($moveChildren) {
462 $children = $this->searchFolders('|^' . $mountPoint . '/.*$|');
463 foreach ($children as $childInfo) {
464 $oldChildMount = $childInfo['mount_point'];
465 $newChildMount = $targetMountPoint . substr($oldChildMount, strlen($mountPoint));
466 try {
467 $this->changeMountPoint($oldChildMount, $newChildMount, moveChildren: false);
468 } catch (\Throwable $t) {
469 $this->logException($t, 'Failed to move child mount from ' . $oldChildMount . ' to ' . $newChildMount);
470 }
471 }
472 }
