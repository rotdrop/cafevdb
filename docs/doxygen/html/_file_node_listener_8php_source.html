165 private function matchDocumentTemplates(string $path, ConfigService $configService)
166 {
167 foreach (ConfigService::DOCUMENT_TEMPLATES as $templateKey => $templateInfo) {
168 $subFolder = $templateInfo['folder']??'';
169 $subFolderName = empty($subFolder) ? '' : $configService->getConfigValue($subFolder);
170 if (!empty($subFolderName)) {
171 $subFolderName .= '/';
172 }
173 $templateFile = $subFolderName . $configService->getConfigValue($templateKey);
174 if ($path === $templateFile) {
175 return [ $templateKey, $subFolderName ];
176 }
177 }
178 $this->logInfo('NOT A SPECIAL FILE ' . $path);
