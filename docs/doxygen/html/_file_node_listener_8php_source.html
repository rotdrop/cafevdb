162 private function matchDocumentTemplates(string $path, ConfigService $configService)
163 {
164 foreach (ConfigService::DOCUMENT_TEMPLATES as $templateKey => $templateInfo) {
165 $subFolder = $templateInfo['folder']??'';
166 $subFolderName = empty($subFolder) ? '' : $configService->getConfigValue($subFolder);
167 if (!empty($subFolderName)) {
168 $subFolderName .= '/';
169 }
170 $templateFile = $subFolderName . $configService->getConfigValue($templateKey);
171 if ($path === $templateFile) {
172 return [ $templateKey, $subFolderName ];
173 }
174 }
175 $this->logInfo('NOT A SPECIAL FILE ' . $path);
176 return [ null, null ];
177 }
