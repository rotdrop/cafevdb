835 {
836 foreach ($newValues as $key => &$value) {
837 if (!is_scalar($value)) { // don't trim arrays
838 continue;
839 }
840
841 // Convert unicode space to ordinary space and trim
842 $value = Util::normalizeSpaces($value);
843
844 $fdn = $pme->fdn[$key];
845 if ($pme->col_has_multiple($fdn)) {
846 $value = preg_replace('/\s*,\s*/', self::VALUES_SEP, $value);
847 }
848
849 // @todo what is this
850 if ($op !== 'insert' && ($pme->skipped($fdn) || $pme->readonly($fdn))) {
851 continue;
852 }
853
854 $chgIdx = array_search($key, $changed);
855 if ($chgIdx === false && $oldValues[$key] != $value) {
856 $changed[] = $key;
857 }
858 if ($op == 'insert' && (string)$value === '' && $chgIdx !== false) {
859 unset($changed[$chgIdx]);
860 $changed = array_values($changed);
861 unset($newValues[$key]);
862 }
863 }
864
