177 public function callInternalRoute(
178 string $route,
179 string $method = self::POST,
180 array $routeParams = [],
181 array $requestData = [],
182 string $postType = self::JSON,
183 ):array {
184 // $this->logInfo('CALL ROUTE ' . $route . ' ' . print_r($routeParams, true) . ' ' . print_r($requestData, true));
185
186 $headers = [];
187
188 $requestToken = \OCP\Util::callRegister();
189 $requestData['requesttoken'] = $requestToken;
190 $urlParameters = [
191 'requesttoken' => $requestToken,
192 'format' => 'json',
193 ];
194
195 if ($method == self::GET) {
196 $urlParameters = array_merge($urlParameters, $requestData);
197 $requestData = null;
198 }
199 if (!empty($requestData)) {
200 switch ($postType) {
201 case self::JSON:
202 if (is_array($requestData)) {
203 $requestData = \OC_JSON::encode($requestData);
204 $headers[] = 'Content-Type: application/json';
205 }
206 break;
207 case self::URL_ENCODED:
208 if (is_array($requestData)) {
209 $requestData = http_build_query($requestData, '', '&');
210 }
211 break;
212 default:
213 throw new InvalidArgumentException(
214 $this->l->t('Supported data formats are "%1$s" and "%2$s", specified was "%3$s".', [
215 self::JSON, self::URL_ENCODED, $postType
216 ]));
217 break;
218 }
219 }
220
221 if (str_starts_with($route, 'http')) {
222 unset($urlParameters['format']);
223 $appendSep = strrchr($url, '?') === false ? '?' : '&';
224 $url = $route . $appendSep . http_build_query($urlParameters, '', '&');
225 } else {
226 $url = $this->urlGenerator->linkToRouteAbsolute($route, array_merge($routeParams, $urlParameters));
227 }
228
229 $cookies = array();
230 foreach ($this->request->cookies as $name => $value) {
231 $cookies[] = "$name=" . urlencode($value);
232 }
233
234 $headers[] = 'OCS-APIRequest: true';
235 $headers[] = 'Accept: application/json';
236 $headers[] = 'requesttoken: ' . $requestToken;
237
238 $c = curl_init($url);
239 curl_setopt($c, CURLOPT_VERBOSE, 0);
240 curl_setopt($c, CURLOPT_HEADER, 0);
241 curl_setopt($c, CURLOPT_HTTPHEADER, $headers);
242 switch ($method) {
243 case self::GET:
244 curl_setopt($c, CURLOPT_HTTPGET, 1);
245 curl_setopt($c, CURLOPT_POST, 0);
246 break;
247 case self::POST:
248 curl_setopt($c, CURLOPT_POST, 1);
249 break;
250 case self::PUT:
251 curl_setopt($c, CURLOPT_POST, 0);
252 curl_setopt($c, CURLOPT_CUSTOMREQUEST, 'PUT');
253 break;
254 case self::DELETE:
255 curl_setopt($c, CURLOPT_POST, 0);
256 curl_setopt($c, CURLOPT_CUSTOMREQUEST, 'DELETE');
257 break;
258 }
259 if (!empty($requestData)) {
260 curl_setopt($c, CURLOPT_POSTFIELDS, $requestData);
261 $this->logDebug('CURL REQUEST DATA ' . $requestData);
262 }
263 if (count($cookies) > 0) {
264 curl_setopt($c, CURLOPT_COOKIE, join("; ", $cookies));
265 }
266 curl_setopt($c, CURLOPT_RETURNTRANSFER, 1);
267 // this is internal, so there is no point in verifying certs:
268 curl_setopt($c, CURLOPT_SSL_VERIFYHOST, 0);
269 curl_setopt($c, CURLOPT_SSL_VERIFYPEER, 0);
270
271 $result = curl_exec($c);
272 curl_close($c);
273
274 $responseData = json_decode($result, true);
275 if (!is_array($responseData)) {
276 throw new RuntimeException(
277 $this->l->t('Invalid response from API call: "%s"', print_r($result, true)));
278 }
279
280 $this->logDebug('RESPONSE DATA ' . print_r($responseData, true));
281
282 // Some apps still return HTTP_STATUS_OK and code errors and success in
283 // the old way instead of using HTTP-error-codes.
284 if (($responseData['status']??null) != 'success' && isset($responseData['data'])) {
285 throw new RuntimeException(
286 $this->l->t('Error response from call to internal route "%1$s": %2$s', [
287 $route, $responseData['data']['message']??print_r($responseData, true)
288 ]));
289 }
290
291 // parse OCS responses if format matches
292 if (isset($responseData['ocs'])) {
293 $meta = $responseData['ocs']['meta']??null;
294 $data = $responseData['ocs']['data']??null;
295 if ($meta === null || $data === null) {
296 throw new RuntimeException(
297 $this->l->t('Invalid OCS response from call to internal route "%1$s": %2$s', [
298 $route, $responseData,
299 ]));
300 }
301 if ($meta['statuscode'] !== 100) {
302 throw new RuntimeException(
303 $this->l->t('Error response from call to internal route "%1$s": %2$s -- %3$s', [
304 $route, $meta['status']??'unknwown status', $meta['message']??print_r($meta, true)
305 ]));
306 }
307 return $data;
308 }
309
310 if (isset($responseData['data'])) {
311 return $responseData['data'];
312 } else {
313 return $responseData;
314 }
315 }
