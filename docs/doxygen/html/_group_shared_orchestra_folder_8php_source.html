122 public function execute():bool
123 {
124 try {
125 // roadmap:
126 //
127 // - create a new group-shared folder FOLDERNAME-new
128 //
129 // - copy all the data from the old shared folder to the new one, recursively
130 //
131 // - unshare the old folder, removing all shares. This will also
132 // invalidate member download shares, but so what. Or could this be
133 // avoided by tweaking the database?
134 //
135 // - rename the new group share to FOLDERNAME
136
137 $shareOwner = $this->encryptionService->getConfigValue(ConfigService::SHAREOWNER_KEY);
138 $sharedFolder = $this->encryptionService->getConfigValue(ConfigService::SHARED_FOLDER, false);
139 if (empty($sharedFolder)) {
140 throw new Exceptions\EnduserNotificationException(
141 $this->l->t('The shared orchester file-system folder is not configured, hence this migration cannot be performed.')
142 );
143 }
144
145 // Step 0: check if we are already done
146 $folderInfo = $this->groupFoldersService->getFolder($sharedFolder);
147
148 if (empty($folderInfo)) {
149 // Step 1, create the folder
150 if (!$this->configCheckService->checkGroupFolder($sharedFolder, self::MIGRATION_SUFFIX)) {
151 throw new Exceptions\EnduserNotificationException(
152 $this->l->t('Unable to create the intermediate shared folder "%s". Please contact the system administrator.', $sharedFolder . self::MIGRATION_SUFFIX)
153 );
154 }
155 $groupFolder = $sharedFolder . self::MIGRATION_SUFFIX;
156 } else {
157 $groupFolder = $sharedFolder;
158 $sharedFolder = $groupFolder . self::KEEP_OLD_SUFFIX;
159 }
160
161 // now copy everything ...
162 $this->userId = $shareOwner;
164 $userFolder = $this->getUserFolder();
165 $this->logInfo('USER FOLDER ' . $userFolder->getPath());
166
167 $sourceFolder = $userFolder->get($sharedFolder);
168 $this->logInfo('SOURCE: ' . $sourceFolder->getPath() . ' ' . $sourceFolder->getParent()->getPath());
169
170 $targetFolder = $userFolder->get($groupFolder);
171 $this->logInfo('TARGET: ' . $targetFolder->getPath() . ' ' . $targetFolder->getParent()->getPath());
172
173 $sourceStorage = $sourceFolder->getStorage();
174 $targetStorage = $targetFolder->getStorage();
175
176 $targetInternalPath = $targetFolder->getInternalPath();
177 $sourceInternalPath = $sourceFolder->getInternalPath();
178
179 $this->logInfo('INTERNAL PATHS ' . $targetInternalPath . ' vs ' . $sourceInternalPath);
180
181 $locked = false;
182 try {
183 $targetStorage->acquireLock($targetInternalPath, ILockingProvider::LOCK_EXCLUSIVE, $this->lockingProvider);
184 $locked = true;
185 $targetStorage->copyFromStorage($sourceStorage, $sourceInternalPath, $targetInternalPath);
186 $targetStorage->releaseLock($targetInternalPath, ILockingProvider::LOCK_EXCLUSIVE, $this->lockingProvider);
187 $locked = false;
188 $targetStorage->getScanner()->scan($targetInternalPath);
189 } catch (Throwable $t) {
190 $this->logException($t);
191 if ($locked) {
192 try {
193 $targetStorage->releaseLock($targetInternalPath, ILockingProvider::LOCK_EXCLUSIVE, $this->lockingProvider);
194 } catch (Throwable $t) {
195 $this->logException($t, 'Unable to unlock ' . $targetInternalPath);
196 }
197 }
198 try {
199 // try to cleanup if possible
200 $targetStorage->getScanner()->scan($targetInternalPath);
201 // $targetFolder->delete();
202 } catch (FileNotFoundException $e) {
203 // really ignore this one: nothing to be cleaned up
204 } catch (Throwable $t) {
205 $this->logException($t, 'Unable to cleanup target path.');
206 // otherwise ignore
207 }
208 throw Exceptions\EnduserNotificationException($this->l->t('Copying failed.'), 0, $t);
209 }
210
211 $shares = [];
212 foreach (self::SHARE_TYPES as $shareType) {
213 $shares = array_merge($shares, $this->shareManager->getSharesBy($shareOwner, $shareType, reshares: true, limit: -1));
214 }
215
216 $sourcePath = $sourceFolder->getPath();
217 $targetPath = $targetFolder->getPath();
218
220 foreach ($shares as $share) {
221 try {
222 $sharedNode = $share->getNode();
223 $shareType = $share->getShareType();
224 $shareSource = $sharedNode->getPath();
225 $sharedWith = $share->getSharedWith();
226 $this->logInfo(
227 'Share:'
228 . ' SRC: ' . $shareSource
229 . ' TGT: ' . $share->getTarget()
230 . ' TYPE ' . $shareType
231 . ' TOKEN ' . $share->getToken()
232 . ' WITH ' . $sharedWith
233 );
234 if ($shareType == IShare::TYPE_LINK || $shareType == IShare::TYPE_USER) {
235 // Update individual and shared-by links shares s.t. they point to
236 // the new location.
237 //
238 // typical source: /cameratashareholder/files/camerata
239 // typical target: /cameratashareholder/files/camerata-migration
240 // typical link share path:
241 // /cameratashareholder/files/camerata/projects/2019/Auvergne2019/downloads
242 $newShareSource = str_replace($sourcePath, '', $shareSource);
243 $newSharedNode = $targetFolder->get($newShareSource);
244 $share->setNode($newSharedNode);
245 $this->shareManager->updateShare($share);
246 } else {
247 // Delete all other shares.
248 $this->logInfo('Deleting share ' . $shareSource . ' -> ' . $sharedWith);
249 $this->shareManager->deleteShare($share);
250 }
251 } catch (NotFoundException $e) {
252 $this->logInfo('Cleaning share of non-existing node');
253 $this->shareManager->deleteShare($share);
254 }
255 }
256
257 if (!str_ends_with($sourcePath, self::KEEP_OLD_SUFFIX)) {
258 // then move the old folder out of the way
259 $sourceFolder->move($sourcePath . self::KEEP_OLD_SUFFIX);
260 }
261
262 if (str_ends_with($targetPath, self::MIGRATION_SUFFIX)) {
263 $this->groupFoldersService->changeMountPoint($groupFolder, $sharedFolder, moveChildren: true);
264 }
265
266 $orchestraGroup = $this->encryptionService->getConfigValue(ConfigService::USER_GROUP_KEY);
267 // group-share the old folder in order to give the orchestra group
268 // access for a while
269 $this->sharingService->groupShareNode(
270 $sourceFolder,
271 $orchestraGroup,
272 $sharedFolder . self::SHARED_KEEP_OLD_SUFFIX,
273 \OCP\Constants::PERMISSION_READ,
274 sharedByOwner: true,
275 );
276
277 // Remember that we do no longer need the folder from the share-owner
278 $this->encryptionService->setConfigValue(ConfigService::SHAREOWNER_FOLDER_SERVICE_KEY, false);
279
280 } catch (Throwable $t) {
281 throw new Exceptions\MigrationException('Migration has failed', 0, $t);
282 $this->logException($t);
283 return false;
284 }
285
286 return true;
287 }
