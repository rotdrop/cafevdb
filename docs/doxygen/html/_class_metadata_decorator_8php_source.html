251 private function updateInverseSide(
252 object $entity,
253 array $association,
254 object $targetEntity,
255 ClassMetadataInterface $targetMeta,
256 ):void {
257 if (!empty($association['mappedBy'])) {
258 $this->debug('WE ARE THE INVERSE SIDE ' . print_r($association, true));
259 }
260 // try to maintain connectivity if it does not cause additional direct database access
261 if (!empty($association['inversedBy'])
262 && (!($targetEntity instanceof Proxy) || $targetEntity->__isInitialized())) {
263 $inversedBy = $association['inversedBy'];
264 $associationType = $association['type'];
265 switch ($associationType) {
266 case ClassMetadata::ONE_TO_ONE:
267 $targetMeta->setFieldValue($targetEntity, $inversedBy, $entity);
268 break;
269 case ClassMetadata::MANY_TO_ONE:
270 $inversedByValue = $targetMeta->getFieldValue($targetEntity, $inversedBy);
271 if (!($inversedByValue instanceof Collection)
272 || ($inversedByValue instanceof PersistentCollection) && !$inversedByValue->isInitialized()) {
273 // skip if the collection would have to be fetches or if
274 // it is not there already (the latter should not happen).
275 break;
276 }
277 $targetAssociation = $targetMeta->associationMappings[$inversedBy];
278 $indexBy = $targetAssociation['indexBy'] ?? null;
279 // $orderBy = $targetAssociation['orderBy']; complicated
280 if (!empty($indexBy)) {
281 $indexByValue = (string)$this->metaData->getFieldValue($entity, $indexBy);
282 $inversedByValue->set($indexByValue, $entity);
283 } elseif (!$inversedByValue->contains($entity)) {
284 $inversedByValue->add($entity);
285 }
286 break;
287 }
288 }

308 public function getIdentifierColumnValues(mixed $entity)
309 {
310 $columnValues = [];
311 foreach ($this->metaData->getIdentifierValues($entity) as $field => $value) {
312 if (isset($this->metaData->associationMappings[$field])) {
313 $association = $this->metaData->associationMappings[$field];
314 $targetEntity = $association['targetEntity'];
316 $targetMeta = $this->entityManager->getClassMetadata($targetEntity);
317 if (count($association['joinColumns']) != 1) {
318 throw new Exception($this->l->t('Foreign keys as principle keys cannot be composite'));
319 }
320 $joinInfo = $association['joinColumns'][0];
321 $columnName = $joinInfo['name'];
322 $targetColumn = $joinInfo['referencedColumnName'];
323 $targetField = $targetMeta->fieldNames[$targetColumn];
324 if ($value instanceof $targetEntity) {
325 $columnValues[$columnName] = $targetMeta->getFieldValue($value, $targetField);
326 } else {
327 // avoid references with empty identifiers
328 if (empty($value)) {
329 $value = null;
330 $reference = null;
331 } else {
332 // replace the value by a reference
333 $reference = $this->entityManager->getReference($targetEntity, [ $targetField => $value ]);
334
335 // try to maintain connectivity if it does not cause additional direct database access
336 $this->updateInverseSide($entity, $association, $reference, $targetMeta);
337 }
338 $this->doSetFieldValue($this->metaData, $entity, $field, $reference);
339 // assume this is the column value, not the entity of the foreign key
340 $columnValues[$columnName] = $value;
341 }
342 } else {
343 $columnName = $this->metaData->fieldMappings[$field]['columnName'];
344 $columnValues[$columnName] = $value;
345 }
346 }
347 return $columnValues;

375 public function extractKeyValues(array $columnValues, bool $ignoreMissing = false):array
376 {
377 $entityId = [];
378 foreach ($this->metaData->identifier as $field) {
379 $dbalType = null;
380 if (isset($this->metaData->associationMappings[$field])) {
381 if (count($this->metaData->associationMappings[$field]['joinColumns']) != 1) {
382 throw new Exception($this->l->t('Foreign keys as principle keys cannot be composite'));
383 }
384 $columnName = $this->metaData->associationMappings[$field]['joinColumns'][0]['name'];
385 } else {
386 $columnName = $this->metaData->fieldMappings[$field]['columnName'];
387 if (!isset($columnValues[$columnName])) {
388 // possibly an attempt to extract from non-existing field.
389 if ($ignoreMissing || $this->metaData->usesIdGenerator()) {
390 continue;
391 }
392 throw new Exception(
393 $this->l->t('Missing value and no generator for identifier field: %s::%s', [ $this->getName(), $field ]));
394 }
395 $dbalType = Type::getType($this->metaData->fieldMappings[$field]['type']);
396 }
397 $value = $columnValues[$columnName];
398 if (!empty($dbalType) && is_string($value)) {
399 $value = $dbalType->convertToPHPValue($value, $this->entityManager->getPlatform());
400 }
401 $entityId[$field] = $value;
402 }
403 return $entityId;

439 private function doSetFieldValue(mixed $meta, mixed $entity, string $field, mixed $value):void
440 {
441 // try first the setter/getter of the entity
442 $method = 'set'.ucfirst($field);
443
444 // try to convert string to correct type if possible
445 try {
446 $dbalTypeName = $meta->getTypeOfField($field);
447 if (!empty($dbalTypeName)) {
448 $dbalType = Type::getType($dbalTypeName);
449 if (!empty($dbalType) && is_string($value)) {
450 $value = $dbalType->convertToPHPValue($value, $this->entityManager->getPlatform());
451 }
452 }
453 } catch (ConversionException $e) {
454 if (empty($value) && is_callable([ $entity, $method ])) {
455 $this->logException($e, 'Type conversion for field ' . $field . ' failed, continuing anyway.');
456 } else {
457 throw new DatabaseException($this->l->t('Type conversion for field "%s" failed.', $field), 0, $e);
458 }
459 }
460 if (is_callable([ $entity, $method ])) {
461 $entity->$method($value);
462 } else {
463 $this->logWarn('Probably missing method "'.$method.'" on an entity '.get_class($entity));
464 $meta->setFieldValue($entity, $field, $value);
465 }

477 public function setColumnValue(mixed $entity, string $column, mixed $value):void
478 {
479 $this->debug('Set column value for ' . $column . ' to ' . $value);
480
481 $meta = $this->metaData;
482 $numSetters = 0;
483 if (isset($meta->fieldNames[$column])) {
484 // simple case, just set the entity property
485 $this->doSetFieldValue($meta, $entity, $meta->fieldNames[$column], $value);
486 ++$numSetters;
487 }
488 $columnAssociations = $this->columnAssociations();
489 foreach (($columnAssociations[$column]??[]) as $field) {
490 // If the column is a join-column then try to create references to the
491 // respective entity. Multi-column associations are supported if the
492 // column is the only one which is not known.
493 if (empty($value)) {
494 // an empty join-column value must void the association as the
495 // join-columns are the identifier of the referenced entity and hence
496 // cannot be null unless the referenced entity is null
497 $this->debug('DO CLEAR FIELD VALUE FOR ' . $field);
498 $this->doSetFieldValue($meta, $entity, $field, null);
499 continue;
500 }
501 $associationMapping = $meta->associationMappings[$field];
502 $targetEntity = $associationMapping['targetEntity'];
503 $targetMeta = $this->entityManager->getClassMetadata($targetEntity);
504 $referencedColumnId = [];
505 foreach ($associationMapping['joinColumns'] as $joinInfo) {
506 $joinColumnName = $joinInfo['name'];
507 $targetColumn = $joinInfo['referencedColumnName'];
508 if ($joinColumnName == $column) {
509 $fieldValue = $meta->getFieldValue($entity, $field);
510 if (!empty($fieldValue) &&
511 $value == $targetMeta->getColumnValue($fieldValue, $targetColumn)) {
512 // do not destroy already installed associated entities
513 continue 2;
514 }
515 $targetValue = $value;
516 } else {
517 $targetValue = $this->getColumnValue($entity, $joinColumnName);
518 }
519 if (empty($targetValue)) {
520 continue 2;
521 }
522 $referencedColumnId[$targetColumn] = $targetValue;
523 }
524 $referencedId = $targetMeta->extractKeyValues($referencedColumnId);
525 $fieldValue = $this->entityManager->getReference($targetEntity, $referencedId);
526 $this->doSetFieldValue($meta, $entity, $field, $fieldValue);
527 $this->updateInverseSide($entity, $associationMapping, $fieldValue, $targetMeta);
528 ++$numSetters;
529 }
530 if ($numSetters == 0) {
531 // throw new \RuntimeException($this->l->t('Unable to feed a single field with the value of the column "' . $column . '".'));
532 $this->debug('Remember value for column ' . $column . ': ' . $value);
533 $this->temporaryColumnStorage[$column] = $value;
534 } else {
535 $this->debug('Clear value for column ' . $column);
536 unset($this->temporaryColumnStorage[$column]);
537 }
538 }

549 public function getColumnValue(mixed $entity, string $column)
550 {
551 $meta = $this->metaData;
552 // simple case ...
553 if (isset($meta->fieldNames[$column])) {
554 return $meta->getFieldValue($entity, $meta->fieldNames[$column]);
555 }
556 // look into the association mappings
557 $columnAssociations = $this->columnAssociations();
558 foreach (($columnAssociations[$column]??[]) as $targetColumn => $field) {
559 $fieldValue = $meta->getFieldValue($entity, $field);
560 if (empty($fieldValue)) {
561 continue;
562 }
563 $associationMapping = $meta->associationMappings[$field];
564 $targetEntity = $associationMapping['targetEntity'];
565 if ($fieldValue instanceof $targetEntity) {
566 $targetMeta = $this->entityManager->getClassMetadata($targetEntity);
567 return $targetMeta->getColumnValue($fieldValue, $targetColumn);
568 } else {
569 // assume the field-value is the just the column value of the single join-column
570 if (count($associationMapping['joinColumns']) != 1) {
571 throw new RuntimeException($this->l->t('Association field must eiter be an entity or the value of the single join column.') . print_r($associationMapping['joinColumns'], true));
572 }
573 return $fieldValue;
574 }
575 }
576 if (!empty($this->temporaryColumnStorage[$column])) {
577 $this->debug('USE TEMPORARY COLUMN STORAGE ' . $column . ' => ' . $this->temporaryColumnStorage[$column]);
578 return $this->temporaryColumnStorage[$column];
579 }
580 return null;
581 }
