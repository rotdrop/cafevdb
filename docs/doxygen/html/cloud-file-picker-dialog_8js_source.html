24import $ from './jquery.js';
25import { appName } from './app-info.js';
26import * as Ajax from './ajax.js';
27import * as Dialogs from './dialogs.js';
28import * as Notification from './notification.js';
29import generateUrl from './generate-url.js';
30import { parse as pathParse } from 'path';
31import escapeHtml from 'escape-html';
32
33const defaultOptions = {
34 setup() {},
35 cleanup() {},
36 handlePickedFiles(files, paths, cleanup) {
37 cleanup();
38 },
39 filePickerCaption: t(appName, 'Select a file from the cloud'),
40 stashUrl: 'upload/stash',
41 multiple: false,
42 mimeTypeFilter: undefined,
43 modal: undefined,
44 initialCloudFolder: '',
45};
46
47const cloudFilePickerDialog = function(options) {
48
49 options = {
50 ...defaultOptions,
51 ...options,
52 };
53
54 Dialogs.filePicker(
55 options.filePickerCaption,
56 function(paths) {
57 options.setup();
58 if (!paths) {
59 Dialogs.alert(t(appName, 'Empty response from file selection!'), t(appName, 'Error'));
60 options.cleanup();
61 return;
62 }
63 if (!Array.isArray(paths)) {
64 paths = [paths];
65 }
66 $.post(generateUrl(options.stashUrl), {
67 cloudPaths: paths,
68 uploadMode: 'test',
69 })
70 .fail(function(xhr, status, errorThrown) {
71 Ajax.handleError(xhr, status, errorThrown, options.cleanup);
72 })
73 .done(function(data) {
74
75 const performUpload = function(uploadMode) {
76 $.post(generateUrl(options.stashUrl), {
77 cloudPaths: paths,
78 uploadMode,
79 })
80 .fail(function(xhr, status, errorThrown) {
81 Ajax.handleError(xhr, status, errorThrown, options.cleanup);
82 })
83 .done(function(files) {
84 if (!Array.isArray(files) || (!options.multiple && files.length !== 1)) {
85 Dialogs.alert(
86 t(appName, 'Unable to copy selected file(s) {file}.', { file: paths.join(', ') }),
87 t(appName, 'Error'),
88 options.cleanup
89 );
90 return;
91 }
92 options.handlePickedFiles(files, paths, options.cleanup);
93 });
94 };
95
96 const uploadFiles = [];
97 const allUploadModes = ['copy', 'move', 'link'];
98 let uploadModes = allUploadModes;
99 for (const uploadInfo of data) {
100 uploadModes = uploadModes.filter(value => uploadInfo.upload_mode.includes(value));
101 uploadFiles.push(pathParse(uploadInfo.original_name));
102 }
103 const templateParameters = {
104 operations: uploadModes.join(' '),
105 files: uploadFiles.map(
106 (info) => `<span class="file-node tooltip-auto tooltip-wide"
107 title="${escapeHtml(info.dir + '/' + info.base)}"
108>
109 <span class="dirname">${escapeHtml(info.dir)}/</span>
110 <span class="basename">${escapeHtml(info.base)}</span>
111</span>`).join(''),
112 widgetCssClass: 'cloud-file-system-operations',
113 widgetRadioName: 'cloudFileSystemOperations',
114 };
115 for (const mode of allUploadModes) {
116 templateParameters[mode + 'Selected'] = '';
117 templateParameters[mode + 'CssClass'] = mode + '-control';
118 if (uploadModes.includes(mode)) {
119 templateParameters[mode + 'Disabled'] = '';
120 templateParameters[mode + 'CssClass'] += ' enabled';
121 } else {
122 templateParameters[mode + 'Disabled'] = 'disabled';
123 templateParameters[mode + 'CssClass'] += ' disabled';
124 }
125 }
126 templateParameters.copySelected = 'checked';
127
128 const $fileSystemOps = $('#cloudFileSystemOperations').octemplate(
129 templateParameters,
130 { escapeFunction: (x) => x }
131 );
132
133 let uploadMode = 'copy';
134 $('body')
135 .off('change', 'input.cloud-file-system-operations-input')
136 .on('change', 'input.cloud-file-system-operations-input', function(event) {
137 uploadMode = $(this).val();
138 console.info('UPLOAD MODE', uploadMode);
139 });
140 $('body')
141 .on('open', '#oc-dialog-0-content', function(event) {
142 console.info('DIALOG OPENED', event);
143 });
144
145 Dialogs.confirm(
146 $fileSystemOps.html(),
147 t(appName, 'Select File System Operation'), {
148 callback(answer) {
149 console.info('UPLOAD MODE', uploadMode);
150 if (answer) {
151 performUpload(uploadMode);
152 } else {
153 options.cleanup();
154 Notification.messages(t(appName, 'Operation has been cancelled.'));
155 }
156 },
157 buttons: {
158 type: OC.dialogs.YES_NO_BUTTONS,
159 confirm: t(appName, 'Apply'),
160 cancel: t(appName, 'Cancel'),
161 },
162 modal: true,
163 allowHtml: true,
164 }
165 )
166 .then(function() {
167 $('.oc-dialog .oc-dialog-content .cloud-file-system-operations-wrapper .tooltip-auto').cafevTooltip();
168 });
169
170 });
171 },
172 options.multiple, // multiselect
173 options.mimeTypeFilter, // mimetypeFilter
174 options.modal, // modal
175 undefined, // type
176 options.initialCloudFolder,
177 );
178
179};
180
181export default cloudFilePickerDialog;
