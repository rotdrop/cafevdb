170 protected function applyMigration(string $version, string $className):void
171 {
173 $instance = $this->appContainer->get($className);
174
175 $this->entityManager->close();
176 $this->clearDatabaseRepository();
177 $this->entityManager->reopen();
178
179 $exception = null;
180 try {
181 $result = $instance->execute();
182 } catch (Throwable $exception) {
183 $result = false;
184 }
185 if ($result !== true) {
186 throw new RuntimeException($this->l->t("Migration %s has failed to execute.", $className), 0, $exception);
187 }
188
189 $this->entityManager->close();
190 $this->entityManager->reopen();
191 $this->clearDatabaseRepository();
192
193 $migrationClassName = self::getBaseClassName($className);
194
195 foreach (['run', 'retry'] as $state) {
196 try {
197 $migrationRecord = $this->getDatabaseRepository(Entities\Migration::class)->find($version);
198 if (empty($migrationRecord)) {
200 $migrationRecord = (new Entities\Migration)
201 ->setVersion($version)
202 ->setMigrationClassName($migrationClassName);
203 $this->persist($migrationRecord);
204 } else {
205 $migrationRecord
206 ->incrementRunCount()
207 ->setMigrationClassName($migrationClassName);
208 }
209 $this->flush();
210 break;
211 } catch (Throwable $t) {
212 $this->logInfo('Try to sanitize the migrations table');
213 $this->sanitizeMigrationsTable();
214 }
215 }

262 protected function findLatestVersion(bool $autoFix = true):?string
263 {
264 if (!$this->entityManager->connected()) {
265 return null;
266 }
268 try {
269 $latestMigration = $this->getDatabaseRepository(Entities\Migration::class)->findOneBy([], [ 'version' => 'DESC' ]);
270 } catch (DBALException\TableNotFoundException $tnfe) {
271 // Ok, there is no migrations table, handle this inside the initial migration
272 $this->logInfo('NO MIGRATIONS TABLE');
273 if ($autoFix) {
274 $this->createMigrationsTable();
275 return $this->findLatestVersion(autoFix: false);
276 }
277 } catch (InvalidFieldNameException $ifne) {
278 $this->logInfo('MIGRATIONS TABLE SEEMS BROKEN');
279 if ($autoFix) {
280 $this->logInfo('ATTEMPTING TO FIX IT');
281 $this->sanitizeMigrationsTable();
282 return $this->findLatestVersion(autoFix: false);
283 }
284 } catch (Throwable $t) {
285 $this->logException($t);
286 }
287 if (empty($latestMigration)) {
288 $this->logInfo('NO MIGRATIONS HAVE BEEN APPLIED YET.');
289 return null;
290 }
291 $this->logInfo('LATEST ' . $latestMigration->getVersion());
292 return $latestMigration->getVersion();

318 protected function findMigrations(string $directory):array
319 {
320 $directory = realpath($directory);
321 if ($directory === false || !file_exists($directory) || !is_dir($directory)) {
322 return [];
323 }
324
325 $iterator = new RegexIterator(
326 new RecursiveIteratorIterator(
327 new RecursiveDirectoryIterator($directory, FilesystemIterator::SKIP_DOTS),
328 RecursiveIteratorIterator::LEAVES_ONLY
329 ),
330 '#^.+\\/Version\d+\\.php$#i',
331 \RegexIterator::GET_MATCH);
332
333 $files = array_keys(iterator_to_array($iterator));
334 uasort($files, function ($a, $b) {
335 preg_match('/^Version(\d+)\\.php$/', basename($a), $matchA);
336 preg_match('/^Version(\d+)\\.php$/', basename($b), $matchB);
337 if (!empty($matchA) && !empty($matchB)) {
338 if ($matchA[1] !== $matchB[1]) {
339 return ($matchA[1] < $matchB[1]) ? -1 : 1;
340 }
341 return ($matchA[2] < $matchB[2]) ? -1 : 1;
342 }
343 return (basename($a) < basename($b)) ? -1 : 1;
344 });
345
346 $migrations = [];
347
348 foreach ($files as $file) {
349 $className = basename($file, '.php');
350 $version = (string) substr($className, 7);
351 $migrations[$version] = sprintf('%s\\%s', self::MIGRATIONS_NAMESPACE, $className);
352 }
353
354 ksort($migrations);
355
356 return $migrations;
357 }
