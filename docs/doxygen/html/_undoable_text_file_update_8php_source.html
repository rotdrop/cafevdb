115 public function do():void
116 {
117 if (is_callable($this->name)) {
118 $this->name = call_user_func($this->name);
119 }
120 $this->name = self::normalizePath($this->name);
121
122 $this->renamedName = null;
123 // - $folderName does not exist
124 // -> just create the new folder, delete on undo
125 // - $folderName exists and is a folder
126 // -> just keep, do not delete on undo
127 // - $folderName exists and is a file
128 // -> delete, undelete on undo, or just rename
129 $components = explode(UserStorage::PATH_SEP, $this->name);
130 if (empty($components)) {
131 throw new Exception('Cannot create the root-node.');
132 }
133
134 if (!$this->mkdir && empty($this->userStorage->get(dirname($this->name)))) {
135 $this->nothingToUndo = true;
136 return;
137 }
138
139 // Make sure the dirname($name) exists. This will also NOT be undone.
140 $prefix = array_slice($components, 0, count($components) - 1);
141 $this->userStorage->ensureFolderChain($prefix);
142
143 $content = trim($this->content);
144 $replacableContent = trim($this->replacableContent);
145
147 $existingNode = $this->userStorage->get($this->name);
148 if (!empty($existingNode)) {
149 if ($existingNode->getType() != FileInfo::TYPE_FILE) {
150 // garbled, rename the beast
151 $this->renamedName = $this->renamedName($this->name);
152 $this->userStorage->rename($this->name, $this->renamedName);
153 } else {
154 // Plain file. In order not to generate excessivly many README.md
155 // files we just add to the content of the old README.md
156 $oldContent = $this->oldContent = $existingNode->getContent();
157
158 $separatorPos = strpos($oldContent, Constants::OLD_CONTENT_SEPARATOR);
159 $separatorPos === false && $separatorPos = null;
160 $oldContentHead = trim(substr($oldContent, 0, $separatorPos));
161
162 if ($content == $oldContentHead) {
163 $content = null;
164 $this->nothingToUndo = true;
165 return;
166 } elseif (!empty($replacableContent) && $oldContentHead == $replacableContent) {
167 if ($separatorPos !== null) {
168 $content .= substr($oldContent, $separatorPos);
169 }
170 } elseif (!empty($oldContent)) {
171 // annotate the old content
172 if (empty($content)) {
173 $content = '';
174 } else {
175 $content .= Constants::OLD_CONTENT_SEPARATOR;
176 }
177 $now = new DateTime;
178 $content .= $this->l->t('The old ``%1$s`` content on %2$s at %3$s was:', [
179 basename($this->name),
180 $this->dateTimeFormatter->formatDate($now),
181 $this->dateTimeFormatter->formatTime($now),
182 ]);
183 $content .= "\n\n" . $this->oldContent;
184 }
185 }
186 }
187
188 if (!empty($content)) {
189 try {
190 $this->userStorage->putContent($this->name, $content);
191 } catch (\Throwable $t) {
192 if (!$this->gracefully) {
193 throw $t;
194 }
195 $this->logException($t, 'Unable to store contents of readme-file');
196 }
197 } else {
198 $this->userStorage->delete($this->name);
199 }
200 }

212 }
213 } else {
214 $this->userStorage->putContent($this->name, $this->oldContent);
