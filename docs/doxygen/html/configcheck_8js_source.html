35/**
36 * jQuery ready-callback used elsewhere.
37 */
38function documentReady() {
39
40 const $container = $('#app-content');
41
42 $container.on('click', '#configrecheck', function(event) {
43 console.info('Hello recheck');
44 Page.loadPage({ template: 'maintenance/configcheck' });
45 return false;
46 });
47
48 const handleMigrations = function() {
49 if ($container.find('.config-check').length <= 0) {
50 return;
51 }
52
53 // check for pending migrations and handle them
54 $.get(generateUrl('maintenance/migrations/unapplied'))
55 .fail(Ajax.handleError)
56 .done(function(data) {
57 if (data.migrations.length <= 0) {
58 return;
59 }
60 let migrationList = '<dl class="migrations-list">';
61 for (const [version, description] of Object.entries(data.migrations)) {
62 migrationList += `<dt class="migration-version">${version}</dt><dd class="migration-description">${description}</dd>`;
63 }
64 migrationList += '</dl>';
65 Dialogs.info(
66 t(appName, 'Data-migrations need to be performed before proceeding to the orchestra app.')
67 + '<p>'
68 + migrationList
69 + '<p>'
70 + t(appName, 'Click the "ok" button to start the migrations.'),
71 t(appName, 'Data Migration'),
72 function(confirmation) {
73 if (confirmation !== true) {
74 return;
75 }
76 $.post(generateUrl('maintenance/migrations/apply/all'))
77 .fail(Ajax.handleError)
78 .done(function(data) {
79 Notification.show(
80 t(appName, 'Successfully applied the following migrations:')
81 + ' '
82 + data.migrations.handled.join(', '),
83 { timeout: 30 });
84 let redirectTimeout = 10;
85 const makeText = timeout => t(appName, 'Redirecting to the orchestra app in {timeout} seconds.', { timeout: redirectTimeout });
86 const row = Notification.show(makeText(redirectTimeout));
87 const second = 1000;
88 const notifier = setInterval(() => {
89 row.text(makeText(--redirectTimeout));
90 }, second);
91 setTimeout(() => {
92 clearInterval(notifier);
93 window.location.reload();
94 }, redirectTimeout * 1000);
95 });
96 },
97 true,
98 true);
99 });
100 };
101 addReadyCallback(handleMigrations);
102
103 $container.on('click', '.pdfletter-download', function(event) {
104 const post = {};
105 fileDownload(
106 'download/test/pdfletter',
107 post,
108 t(appName, 'Unable to download test-letter.')
109 );
110 });
111
112 $container.on('click', '.progress-status.button', function(event) {
113 console.info('Hello World');
114 if (ProgressStatus.poll.active()) {
115 ProgressStatus.poll.stop();
116 return false;
117 }
118 $.post(generateUrl('foregroundjob/progress/create'), { target: 100, current: 0 })
119 .fail(Ajax.handleError)
120 .done(function(data) {
121 console.log('Progress Create', data);
122 const id = data.id;
123 ProgressStatus.poll(
124 id,
125 {
126 update(id, current, target, data) {
127 $('#progress-status-info').html(current + ' of ' + target);
128 console.info(current, target);
129 return parseInt(current) < parseInt(target);
130 },
131 fail(xhr, status, errorThrown) {
132 Ajax.handleError(xhr, status, errorThrown);
133 },
134 });
135 $.post(generateUrl('foregroundjob/progress/test'), { id, target: 100, data: { foo: 'bar' } })
136 .fail(Ajax.handleError)
137 .done(function(data) {});
138 });
139 return false;
140 });
141
142}
