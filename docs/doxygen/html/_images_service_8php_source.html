228 public function getImage(string $joinTable, mixed $ownerId, mixed $imageId = self::IMAGE_ID_ANY):array
229 {
230 $image = null;
231 $fileName = null;
232
233 try {
234 switch ($joinTable) {
235 case self::TMP_STORAGE:
236 $cacheKey = $ownerId;
237 $imageData = $this->fileCache->get($cacheKey);
238
239 $this->logDebug("Requested cache file with key ".$cacheKey);
240
241 $image = new CloudImage();
242 $image->loadFromData($imageData);
243 $fileName = $cacheKey;
244 break;
245 case self::USER_STORAGE:
246 case self::APP_STORAGE:
247 // $ownerId is a directory, $imageId a file in that directory
248 if ($joinTable == self::USER_STORAGE) {
250 $storage = $this->di(UserStorage::class);
251 $directory = $storage->getFolder($ownerId);
252
254 if ($imageId == self::IMAGE_ID_ANY) {
256 foreach ($directory->getDirectoryListing() as $node) {
257 if ($node->getType() == \OCP\Files\Node::TYPE_FILE) {
258 $file = $node;
259 break;
260 }
261 }
262 } else {
263 $file = $directory->get($imageId);
264 }
265 } else {
267 $storage = $this->di(AppStorage::class);
268 $directory = $storage->getFolder($ownerId);
270 if ($imageId == self::IMAGE_ID_ANY) {
271 $file = array_shift($directory->getDirectoryListing());
272 } else {
273 $file = $directory->getFile($imageId);
274 }
275 }
276 if (empty($file)) {
277 $this->logInfo('NO FILE FOR ' . $imageId);
278 break;
279 }
280 $image = new CloudImage();
281 $image->loadFromData($file->getContent());
282 $fileName = $file->getName();
283 break;
284 default: // data-base
285 $imagesRepository = $this->getDatabaseRepository(Entities\Image::class);
286
287 if ($joinTable == self::DATABASE_STORAGE) {
288 // plain data-base id without join table
289 $dbImage = $imagesRepository->find($imageId);
290 if (empty($dbImage)) {
291 $this->logInfo('Unable to load image with id ' . $imageId);
292 }
293 } else {
294 $joinTableClass = $imagesRepository->joinTableClass($joinTable);
295 $this->logDebug("cooked table: ".$joinTableClass);
296
297 $joinTableRepository = $this->getDatabaseRepository($joinTableClass);
298 $findBy = [ 'ownerId' => $ownerId ];
299 if ($imageId > self::IMAGE_ID_PLACEHOLDER) {
300 $findBy['image'] = $imageId;
301 }
302
303 $joinTableEntity = $joinTableRepository->findOneBy($findBy);
304 if ($joinTableEntity == null) {
305 $this->logDebug('NOT FOUND ' . print_r($findBy, true));
306 break;
307 }
308
310 $dbImage = $joinTableEntity->getImage();
311 }
312
313 $fileName = $dbImage->getFileName();
314 if (empty($fileName)) {
315 $fileName = $joinTable . '-' . $ownerId . '-' . $dbImage->getId();
316 }
317
318 // otherwise return image blob as download
319
320 $imageMimeType = $dbImage->getMimeType();
321 $imageData = $dbImage->getFileData()->getData('binary');
322
323 if (empty($imageData)) {
324 $this->logInfo('EMPTY IMAGE DATA FOR FILE ID ' . $dbImage->getId());
325 break;
326 }
327
328 $image = new CloudImage();
329 $image->loadFromData($imageData);
330
331 if ($image->mimeType() !== $imageMimeType) {
332 $this->logError("Mime-types stored / computed: ".$imageMimeType." / ".$image->mimeType());
333 $this->logError('Trying to correct cached mime-type for image id ' . $dbImage->getId() . '.');
334 $dbImage->setMimeType($image->mimeType());
335 $this->flush();
336 }
337 break;
338 }
339
340 } catch (\Throwable $t) {
341 $this->logException($t);
342 return [];
343 }
344
345 return [ $image, $fileName ]; // maybe null
346 }

364 public function getImageIds(string $joinTable, mixed $ownerId):array
365 {
366 $result = [];
367 try {
368 switch ($joinTable) {
369 case self::USER_STORAGE:
370 // $ownerId is a directory, $imageId a file in that directory
371 if (empty($ownerId)) {
372 throw new RuntimeException($this->l->t('Images path must not be empty.'));
373 }
375 $storage = $this->di(UserStorage::class);
376 $directory = $storage->getFolder($ownerId);
378 foreach ($directory->getDirectoryListing() as $node) {
379 if ($node->getType() == \OCP\Files\Node::TYPE_FILE && $node->getMimePart() == 'image') {
380 $result[] = $node->getName();
381 }
382 }
383 break;
384 case self::APP_STORAGE:
385 // $ownerId is a directory, $imageId a file in that directory
387 $storage = $this->di(AppStorage::class);
388 $directory = $storage->getFolder($ownerId);
390 foreach ($directory->getDirectoryListing() as $file) {
391 if (strpos($file->getMimeType(), 'image/') === '0') {
392 $result[] = $file->getName();
393 }
394 }
395 break;
396 default: // data-base
397 $imagesRepository = $this->getDatabaseRepository(Entities\Image::class);
398 $joinTableClass = $imagesRepository->joinTableClass($joinTable);
399 $joinTableRepository = $this->getDatabaseRepository($joinTableClass);
400 $findBy = [ 'ownerId' => $ownerId ];
401 foreach ($joinTableRepository->findBy($findBy) as $joinTableEntity) {
402 $result[] = $joinTableEntity->getImageId();
403 }
404 break;
405 }
406 } catch (\Throwable $t) {
407 $this->logException($t);
408 }
409 return $result;
410 }

431 public function storeImage(
432 CloudImage $image,
433 string $joinTable,
434 mixed $ownerId,
435 string $fileName,
436 int $imageId = self::IMAGE_ID_ANY,
437 ) {
438 switch ($joinTable) {
439 case self::USER_STORAGE:
440 case self::APP_STORAGE:
441 // $ownerId is a directory, $imageId a file in that directory
442 if ($imageId != self::IMAGE_ID_PLACEHOLDER && $imageId != $fileName) {
443 throw new RuntimeException($this->l->t('Image-id "%s" must be identical to the file-name "%s" or %d.', [ $imageId, $fileName, self::IMAGE_ID_PLACEHOLDER ]));
444 }
445 if ($joinTable == self::USER_STORAGE) {
447 $storage = $this->di(UserStorage::class);
448 $directory = $storage->getFolder($ownerId);
449 $existsMethod = 'nodeExists';
450 } else {
452 $storage = $this->di(AppStorage::class);
453 $directory = $storage->getFolder($ownerId);
454 $existsMethod = 'fileExists';
455 }
456
457 if ($imageId === self::IMAGE_ID_PLACEHOLDER && $directory->$existsMethod($fileName)) {
458 $this->logInfo(sprintf('File %s / %s exists, generate versioned file-name.', $imageId, $fileName));
459 $base = pathinfo($fileName, PATHINFO_FILENAME);
460 $ext = pathinfo($fileName, PATHINFO_EXTENSION);
461 $cnt = 0;
462 do {
463 $cnt++;
464 $newFileName = sprintf('%s-%d.%s', $base, $cnt, $ext);
465 } while ($directory->$existsMethod($newFileName));
466 $fileName = $newFileName;
467 }
468 // just create a new file
469 $directory->newFile($fileName, $image->data());
470 return $fileName; // id == file-name
471 default:
472 // data-base
473 $imagesRepository = $this->getDatabaseRepository(Entities\Image::class);
474 $joinTable = Util::dashesToCamelCase($joinTable, true);
475 $joinTableClass = $imagesRepository->joinTableClass($joinTable);
476 $joinTableRepository =$this->getDatabaseRepository($joinTableClass);
477 if ($imageId > self::IMAGE_ID_PLACEHOLDER) {
478 $findBy = [ 'ownerId' => $ownerId, 'imageId' => $imageId, ];
479 $joinTableEntity = $joinTableRepository->findOneBy($findBy);
481 $dbImage = $joinTableEntity->getImage();
482 $dbImage->setFileName($fileName);
483 $dbImage->setMimeType($image->mimeType());
484 $dbImage->getFileData()->setData($image->data(), 'binary');
485 $this->flush();
486 } else {
487 $imagesRepository = $this->getDatabaseRepository(Entities\Image::class);
488 $dbImage = $imagesRepository->persistForEntity($joinTable, $ownerId, $image, $fileName);
489 }
490 break;
491 }
492
493 return $dbImage->getId();
494 }

525 $directory->get($imageId)->delete();
526 }

530 $storage = $this->di(AppStorage::class);

535 $file->delete();
536 }
537 } else {
538 $directory->getFile($imageId)->delete();
539 }
540 break;
541 default: // data-base
542 $imagesRepository = $this->getDatabaseRepository(Entities\Image::class);
543
544 if ($joinTable == self::DATABASE_STORAGE) {
545 $joinTableEntities = [ $imagesRepository->find($imageId) ];
546 } else {
547 $joinTableClass = $imagesRepository->joinTableClass($joinTable);
548 $this->logDebug("cooked table: ".$joinTableClass);
549
550 $joinTableRepository = $this->getDatabaseRepository($joinTableClass);
551
552 $findBy = [ 'ownerId' => $ownerId ];
553 if ($imageId > self::IMAGE_ID_ANY) {
554 $findBy['imageId'] = $imageId;
555 }
556
557 $joinTableEntities = $joinTableRepository->findBy($findBy);
558 }
559
560 if (count($joinTableEntities) < 1) {
561 throw new RuntimeException(
562 $this->l->t('Unable to find image link for ownerId/imageId "%s/%s" in join-table "%s".', [ $ownerId, $imageId, $joinTable ])
563 );
564 }
565 foreach ($joinTableEntities as $joinTableEntity) {
566 $this->remove($joinTableEntity);
567 }
568 $this->flush();
569 break;
