238 public function getImage(string $joinTable, mixed $ownerId, mixed $imageId = self::IMAGE_ID_ANY):array
239 {
240 $image = null;
241 $fileName = null;
242
243 try {
244 switch ($joinTable) {
245 case self::TMP_STORAGE:
246 $cacheKey = $ownerId;
247 $imageData = $this->fileCache->get($cacheKey);
248
249 $this->logDebug("Requested cache file with key ".$cacheKey);
250
251 $image = new CloudImage();
252 $image->loadFromData($imageData);
253 $fileName = $cacheKey;
254 break;
255 case self::USER_STORAGE:
256 case self::APP_STORAGE:
257 // $ownerId is a directory, $imageId a file in that directory
258 if ($joinTable == self::USER_STORAGE) {
260 $storage = $this->di(UserStorage::class);
261 $directory = $storage->getFolder($ownerId);
262
264 if ($imageId == self::IMAGE_ID_ANY) {
266 foreach ($directory->getDirectoryListing() as $node) {
267 if ($node->getType() == \OCP\Files\Node::TYPE_FILE) {
268 $file = $node;
269 break;
270 }
271 }
272 } else {
273 $file = $directory->get($imageId);
274 }
275 } else {
277 $storage = $this->di(AppStorage::class);
278 $directory = $storage->getFolder($ownerId);
280 if ($imageId == self::IMAGE_ID_ANY) {
281 $file = array_shift($directory->getDirectoryListing());
282 } else {
283 $file = $directory->getFile($imageId);
284 }
285 }
286 if (empty($file)) {
287 $this->logInfo('NO FILE FOR ' . $imageId);
288 break;
289 }
290 $image = new CloudImage();
291 $image->loadFromData($file->getContent());
292 $fileName = $file->getName();
293 break;
294 default: // data-base
295 $imagesRepository = $this->getDatabaseRepository(Entities\Image::class);
296
297 if ($joinTable == self::DATABASE_STORAGE) {
298 // plain data-base id without join table
299 $dbImage = $imagesRepository->find($imageId);
300 if (empty($dbImage)) {
301 $this->logInfo('Unable to load image with id ' . $imageId);
302 }
303 } else {
304 $joinTableClass = $imagesRepository->joinTableClass($joinTable);
305 $this->logDebug("cooked table: ".$joinTableClass);
306
307 $joinTableRepository = $this->getDatabaseRepository($joinTableClass);
308 $findBy = [ 'ownerId' => $ownerId ];
309 if ($imageId > self::IMAGE_ID_PLACEHOLDER) {
310 $findBy['image'] = $imageId;
311 }
312
313 $joinTableEntity = $joinTableRepository->findOneBy($findBy);
314 if ($joinTableEntity == null) {
315 $this->logDebug('NOT FOUND ' . print_r($findBy, true));
316 break;
317 }
318
320 $dbImage = $joinTableEntity->getImage();
321 }
322
323 $fileName = $dbImage->getFileName();
324 if (empty($fileName)) {
325 $fileName = $joinTable . '-' . $ownerId . '-' . $dbImage->getId();
326 }
327
328 // otherwise return image blob as download
329
330 $imageMimeType = $dbImage->getMimeType();
331 $imageData = $dbImage->getFileData()->getData('binary');
332
333 if (empty($imageData)) {
334 $this->logInfo('EMPTY IMAGE DATA FOR FILE ID ' . $dbImage->getId());
335 break;
336 }
337
338 $image = new CloudImage();
339 $image->loadFromData($imageData);
340
341 if ($image->mimeType() !== $imageMimeType) {
342 $this->logError("Mime-types stored / computed: ".$imageMimeType." / ".$image->mimeType());
343 $this->logError('Trying to correct cached mime-type for image id ' . $dbImage->getId() . '.');
344 $dbImage->setMimeType($image->mimeType());
345 $this->flush();
346 }
347 break;
348 }
349
350 } catch (\Throwable $t) {
351 $this->logException($t);
352 return [];
353 }
354
355 return [ $image, $fileName ]; // maybe null
356 }

374 public function getImageIds(string $joinTable, mixed $ownerId):array
375 {
376 $result = [];
377 try {
378 switch ($joinTable) {
379 case self::USER_STORAGE:
380 // $ownerId is a directory, $imageId a file in that directory
381 if (empty($ownerId)) {
382 throw new RuntimeException($this->l->t('Images path must not be empty.'));
383 }
385 $storage = $this->di(UserStorage::class);
386 $directory = $storage->getFolder($ownerId);
388 foreach ($directory->getDirectoryListing() as $node) {
389 if ($node->getType() == \OCP\Files\Node::TYPE_FILE && $node->getMimePart() == 'image') {
390 $result[] = $node->getName();
391 }
392 }
393 break;
394 case self::APP_STORAGE:
395 // $ownerId is a directory, $imageId a file in that directory
397 $storage = $this->di(AppStorage::class);
398 $directory = $storage->getFolder($ownerId);
400 foreach ($directory->getDirectoryListing() as $file) {
401 if (strpos($file->getMimeType(), 'image/') === '0') {
402 $result[] = $file->getName();
403 }
404 }
405 break;
406 default: // data-base
407 $imagesRepository = $this->getDatabaseRepository(Entities\Image::class);
408 $joinTableClass = $imagesRepository->joinTableClass($joinTable);
409 $joinTableRepository = $this->getDatabaseRepository($joinTableClass);
410 $findBy = [ 'ownerId' => $ownerId ];
411 foreach ($joinTableRepository->findBy($findBy) as $joinTableEntity) {
412 $result[] = $joinTableEntity->getImageId();
413 }
414 break;
415 }
416 } catch (\Throwable $t) {
417 $this->logException($t);
418 }
419 return $result;

441 public function storeImage(
442 CloudImage $image,
443 string $joinTable,
444 mixed $ownerId,
445 string $fileName,
446 int $imageId = self::IMAGE_ID_ANY,
447 ) {
448 switch ($joinTable) {
449 case self::USER_STORAGE:
450 case self::APP_STORAGE:
451 // $ownerId is a directory, $imageId a file in that directory
452 if ($imageId != self::IMAGE_ID_PLACEHOLDER && $imageId != $fileName) {
453 throw new RuntimeException($this->l->t('Image-id "%s" must be identical to the file-name "%s" or %d.', [ $imageId, $fileName, self::IMAGE_ID_PLACEHOLDER ]));
454 }
455 if ($joinTable == self::USER_STORAGE) {
457 $storage = $this->di(UserStorage::class);
458 $directory = $storage->getFolder($ownerId);
459 $existsMethod = 'nodeExists';
460 } else {
462 $storage = $this->di(AppStorage::class);
463 $directory = $storage->getFolder($ownerId);
464 $existsMethod = 'fileExists';
465 }
466
467 if ($imageId === self::IMAGE_ID_PLACEHOLDER && $directory->$existsMethod($fileName)) {
468 $this->logInfo(sprintf('File %s / %s exists, generate versioned file-name.', $imageId, $fileName));
469 $base = pathinfo($fileName, PATHINFO_FILENAME);
470 $ext = pathinfo($fileName, PATHINFO_EXTENSION);
471 $cnt = 0;
472 do {
473 $cnt++;
474 $newFileName = sprintf('%s-%d.%s', $base, $cnt, $ext);
475 } while ($directory->$existsMethod($newFileName));
476 $fileName = $newFileName;
477 }
478 // just create a new file
479 $directory->newFile($fileName, $image->data());
480 return $fileName; // id == file-name
481 default:
482 // data-base
483 $imagesRepository = $this->getDatabaseRepository(Entities\Image::class);
484 $joinTable = Util::dashesToCamelCase($joinTable, true);
485 $joinTableClass = $imagesRepository->joinTableClass($joinTable);
486 $joinTableRepository =$this->getDatabaseRepository($joinTableClass);
487 if ($imageId > self::IMAGE_ID_PLACEHOLDER) {
488 $findBy = [ 'ownerId' => $ownerId, 'imageId' => $imageId, ];
489 $joinTableEntity = $joinTableRepository->findOneBy($findBy);
491 $dbImage = $joinTableEntity->getImage();
492 $dbImage->setFileName($fileName);
493 $dbImage->setMimeType($image->mimeType());
494 $dbImage->getFileData()->setData($image->data(), 'binary');
495 $this->flush();
496 } else {
497 $imagesRepository = $this->getDatabaseRepository(Entities\Image::class);
498 $dbImage = $imagesRepository->persistForEntity($joinTable, $ownerId, $image, $fileName);
499 }
500 break;
501 }
502
503 return $dbImage->getId();
504 }

551 default: // data-base
552 $imagesRepository = $this->getDatabaseRepository(Entities\Image::class);
553
554 if ($joinTable == self::DATABASE_STORAGE) {
555 $joinTableEntities = [ $imagesRepository->find($imageId) ];
556 } else {
557 $joinTableClass = $imagesRepository->joinTableClass($joinTable);
558 $this->logDebug("cooked table: ".$joinTableClass);
559
560 $joinTableRepository = $this->getDatabaseRepository($joinTableClass);
561
562 $findBy = [ 'ownerId' => $ownerId ];
563 if ($imageId > self::IMAGE_ID_ANY) {
564 $findBy['imageId'] = $imageId;
565 }
566
567 $joinTableEntities = $joinTableRepository->findBy($findBy);
568 }
569
570 if (count($joinTableEntities) < 1) {
571 throw new RuntimeException(
572 $this->l->t('Unable to find image link for ownerId/imageId "%s/%s" in join-table "%s".', [ $ownerId, $imageId, $joinTable ])

576 $this->remove($joinTableEntity);
