77 {
78 $options = $config['handlers'][get_called_class()];
79 $innerSeparator = $options['separator']?:'-';
80 $outerSeparator = $config['separator']?:'.';
81
82 // get the components
83 $slugs = explode($outerSeparator, $slug);
84
85 // prefer a set of components if all are non empty
86 if (is_array($options['preferred'])) {
87 $preferred = array_filter(array_intersect_key($slugs, array_flip($options['preferred'])));
88 // use preferred fields if all are non empty
89 if (count($preferred) == count($options['preferred'])) {
90 $slugs = $preferred;
91 }
92 }
93
94 // remove empty fields
95 $slugs = array_filter($slugs);
96
97 // transliterate them separately, again using $config['separator']
98 $slugs = array_map(function($slugPart) use ($innerSeparator) {
99 // use iconv for transliteration first place
100 $slugPart = $this->transliterate($slugPart);
101
102 // then pass down and replace other "inconvenient" characters
103 return Transliterator::transliterate($slugPart, $innerSeparator);
104 }, $slugs);
105
106 // implode again using the outer separator
107 $slug = implode($outerSeparator, $slugs);
108 }
