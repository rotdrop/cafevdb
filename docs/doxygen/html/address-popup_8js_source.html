24import { appName } from '../app/app-info';
25
26export default {
27 methods: {
28 addressItemUnknownLabel(item) {
29 return t(appName, '{item}: unknown', { item: t(appName, item) });
30 },
31 musicianAddressPopup(option) {
32 if (!option.id === 0) {
33 return this.addressPopup(t(appName, 'selects all musicians'));
34 }
35 const name = option.informalDisplayName || '';
36 const userId = option.userIdSlug ? ` (${option.userIdSlug})` : '';
37 const email = option.email || this.addressItemUnknownLabel('email');
38 const street = option.street || this.addressItemUnknownLabel('street');
39 const streetNumber = option.streetNumber ? ' ' + option.streetNumber : '';
40 const postalCode = option.postalCode && option.postalCode !== '0' ? option.postalCode + ' ' : '';
41 const city = option.city || this.addressItemUnknownLabel('city');
42 const additionalInfo = [email, street + streetNumber, postalCode + city];
43 if (option.countryName) {
44 let country = option.countryName;
45 if (option.country) {
46 country += ` (${option.country})`;
47 }
48 additionalInfo.push(country);
49 }
50 const content = `<h4>${name}${userId}</h4>` + additionalInfo.join('<br/>');
51 return this.addressPopup(content);
52 },
53 contactNameFromContact(option) {
54 let name = option.name.value || option.name;
55 if (typeof name !== 'string') {
56 name = t(appName, '[empty name]');
57 }
58 return name;
59 },
60 contactAddressPopup(option) {
61 const name = this.contactNameFromContact(option);
62 const additionalInfo = [];
63 // const book = option.addressBookName || '';
64 let emails = [];
65 if (option.EMAIL) {
66 for (const email of option.EMAIL) {
67 const emailValue = email.value || email;
68 if (typeof emailValue === 'string') {
69 emails.push(`${emailValue}`);
70 }
71 }
72 }
73 emails = emails.join('<br/>');
74 if (emails) {
75 additionalInfo.push(emails);
76 }
77 let address = [];
78 if (option.ADR && option.ADR.length > 0) {
79 address = (option.ADR[0].value || option.ADR[0]).split(';');
80 }
81 const street = address[2] || this.addressItemUnknownLabel('street');
82 const postalCode = (address[5] + ' ') || '';
83 const city = address[3] || this.addressItemUnknownLabel('city');
84 const country = address[6] || this.addressItemUnknownLabel('country');
85 additionalInfo.splice(additionalInfo.length, 0, street, postalCode + city, country);
86 const content = `<h4>${name}</h4>`
87 + additionalInfo.join('<br/>');
88 return this.addressPopup(content);
89 },
90 addressPopup(content) {
91 return {
92 content,
93 // placement: 'bottom',
94 preventOverflow: false,
95 boundariesElement: 'viewport',
96 html: true,
97 classes: ['vue-tooltip-address-popup'],
98 };
99 },
100 },
101};
