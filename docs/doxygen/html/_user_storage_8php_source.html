224 public function folderWalk(mixed $pathOrFolder, ?callable $callback = null, int $depth = 0):int
225 {
227 if (!($pathOrFolder instanceof Folder)) {
228 $folder = $this->getFolder($pathOrFolder);
229 } else {
230 $folder = $pathOrFolder;
231 }
232
233 if (empty($folder)) {
234 return 0;
235 }
236
237 if (!empty($callback)) {
238 $callback($folder, $depth);
239 }
240 ++$depth;
241
242 $numberOfFiles = 0;
243 $folderContents = $folder->getDirectoryListing();
245 foreach ($folderContents as $node) {
246 if ($node->getType() == FileInfo::TYPE_FILE) {
247 if (!empty($callback)) {
248 $callback($node, $depth);
249 }
250 ++$numberOfFiles;
251 } else {
252 $numberOfFiles += $this->folderWalk($node, $callback, $depth);
253 }
254 }
255 return $numberOfFiles;
256 }

301 public function getFolderArchive($pathOrFolder, int $parentsToStrip = 0, string $format = 'zip'):?string
302 {
304 if (!($pathOrFolder instanceof Folder)) {
305 $folder = $this->getFolder($pathOrFolder);
306 } else {
307 $folder = $pathOrFolder;
308 }
309
310 if (empty($folder)) {
311 return null;
312 }
313
314 // don't try to strip more components than the initial path-depth
315 $parentsToStrip = min($parentsToStrip, count(Util::explode(self::PATH_SEP, $folder->getPath())));
316
317 $dataStream = fopen("php://memory", 'w');
318 $zipStreamOptions = new ArchiveOptions;
319 $zipStreamOptions->setOutputStream($dataStream);
320
321 $zipStream = new ZipStream(opt: $zipStreamOptions);
322
323 $this->archiveFolderRecursively($folder, $parentsToStrip, $zipStream);
324
325 $zipStream->finish();
326 rewind($dataStream);
327 $data = stream_get_contents($dataStream);
328 fclose($dataStream);
329
330 return $data;
331 }

416 public function restore(string $path, ?array $timeInterval = null, bool $overwriteExisting = false):bool
417 {
419 $trashManager = $this->appContainer->get(ITrashManager::class);
420
421 if (empty($trashManager)) {
422 throw new RuntimeException($this->l->t('Unable to restore "%s": TrashManager cannot be loaded', $path));
423 }
424
425 $path = trim($path, self::PATH_SEP);
426
427 $candidate = [ 'time' => 0, 'trashItem' => null ];
428
430 foreach ($trashManager->listTrashRoot($this->user()) as $trashItem) {
431 $originalLocation = trim($trashItem->getOriginalLocation(), self::PATH_SEP);
432 $deletedTime = $trashItem->getDeletedTime();
433
434 if ($originalLocation == $path) {
435 if (empty($timeInterval)
436 || ($deletedTime >= $timeInterval[0] && $deletedTime <= $timeInterval[1])) {
437 if ($candidate['time'] < $deletedTime) {
438 $candidate['time'] = $deletedTime;
439 $candidate['trashItem'] = $trashItem;
440 }
441 }
442 }
443 }
444
445 if (empty($candidate['trashItem'])) {
446 $this->logInfo('Unable to find trash-bin item to undelete');
447 return false;
448 }
449
450 if ($overwriteExisting) {
451 $this->logInfo('Try delete ' . $path);
452 $this->delete($path);
453 }
454
455 $this->logInfo('Try restore ' . $candidate['trashItem']->getOriginalLocation());
456 $trashManager->restoreItem($candidate['trashItem']);
457
458 return true;
459 }

676 public function getDownloadLink($pathOrNode):?string
677 {
678 if (is_string($pathOrNode)) {
679 $node = $this->get($pathOrNode, useCache: true, throw: true);
680 } elseif ($pathOrNode instanceof \OCP\Files\Node) {
681 $node = $pathOrNode;
682 } else {
683 throw new InvalidArgumentException($this->l->t('Argument must be a valid path or already a file-system node.'));
684 }
685 // $this->logDebug('NODE '.$node->getPath().' '.$node->getInternalPath());
686
687 // Internal path is mount-point specific and in particular
688 // relative to the share-root if sharing a folder. getPath()
689 // contains the /USERID/files/ prefix. We use that for the
690 // moment ...
691 $nodePath = substr(strchr($node->getPath(), 'files/'), strlen('files/'));
692
693 if ($node->getType() == FileInfo::TYPE_FILE) {
694 $webDAVRoot = \OCP\Util::linkToRemote('webdav/');
695 return $webDAVRoot.$nodePath;
696 } elseif ($node->getType() == FileInfo::TYPE_FOLDER) {
697 $parent = $node->getParent();
698 $parentPath = substr(strchr($parent->getPath(), 'files/'), strlen('files/'));
699 return \OCP\Util::linkToAbsolute('files', 'ajax/download.php', [
700 'dir' => $parentPath,
701 'files' => $node->getName(),
702 ]);
703 } else {
704 throw new InvalidArgumentException($this->l->t('Unknown file type "%s" for download file "%s".', [ $node->getType(), $node->getName() ]));
705 }
