255 public function getFolderArchive($pathOrFolder, int $parentsToStrip = 0, string $format = 'zip'):?string
256 {
258 if (!($pathOrFolder instanceof Folder)) {
259 $folder = $this->getFolder($pathOrFolder);
260 } else {
261 $folder = $pathOrFolder;
262 }
263
264 if (empty($folder)) {
265 return null;
266 }
267
268 // don't try to strip more components than the initial path-depth
269 $parentsToStrip = min($parentsToStrip, count(Util::explode(self::PATH_SEP, $folder->getPath())));
270
271 $dataStream = fopen("php://memory", 'w');
272 $zipStreamOptions = new ArchiveOptions;
273 $zipStreamOptions->setOutputStream($dataStream);
274
275 $zipStream = new ZipStream(opt: $zipStreamOptions);
276
277 $this->archiveFolderRecursively($folder, $parentsToStrip, $zipStream);
278
279 $zipStream->finish();
280 rewind($dataStream);
281 $data = stream_get_contents($dataStream);
282 fclose($dataStream);
283

331 public function copyTree(mixed $src, mixed $dst, ?string $excludeRegexp = null, int $flags = self::KEEP_DEST):?Folder
332 {
333 $this->logInfo('SRC // DST ' . $src . ' || ' . $dst);
334
336 if (!($src instanceof Folder)) {
337 $sourceFolder = $this->getFolder($src);
338 } else {
339 $sourceFolder = $src;
340 }
341
342 if (empty($sourceFolder)) {
343 $this->logInfo('NO SOURCE FOLDER ' . $src . ' || ' . $dst);
344 return null;
345 }
346
347 $sourceRootPath = $sourceFolder->getPath();
348 $sourceRootLen = strlen($sourceRootPath);
349
350 $userFolderPrefix = $this->userFolder->getPath();
351 $userFolderLen = strlen($userFolderPrefix);
352
353 if ($dst instanceof Folder) {
354 $destFolder = $dst;
355 $destRootPath = $destFolder->getPath();
356 $destUserPath = trim(substr($destRootPath, $userFolderLen), self::PATH_SEP);
357 } else {
358 $destUserPath = $dst;
359 try {
360 $destFolder = $this->userFolder->get($destUserPath);
361 $destRootPath = $destFolder->getPath();
362 } catch (FileNotFoundException $e) {
363 $destFolder = null;
364 }
365 }
366 if ($destFolder
367 && (($flags & self::CLEAN_DEST) || $destFolder->getType() != FileInfo::TYPE_FOLDER)) {
368 $destFolder->delete();
369 $destFolder = null;
370 }
371 if (empty($destFolder)) {
372 $destFolder = $this->ensureFolder($destUserPath);
373 $destRootPath = $destFolder->getPath();
374 }
375
376 $this->folderWalk($sourceFolder, function(
377 Node $sourceNode,
378 int $depth,
379 ) use (
380 $sourceFolder,
381 $sourceRootPath,
382 $sourceRootLen,
383 $destFolder,
384 $destRootPath,
385 $excludeRegexp,
386 $flags,
387 ) {
388 if (!empty($excludeRegexp) && preg_match($excludeRegexp, $sourceNode->getName())) {
389 $this->logInfo('EXCLUDE HIT FOR PATTERN ' . $excludeRegexp);
390 return false;
391 }
392
393 $sourcePath = $sourceNode->getPath();
394 $relativePath = trim(substr($sourcePath, $sourceRootLen), self::PATH_SEP);
395 $destPath = $destRootPath . self::PATH_SEP . $relativePath;
396 switch ($sourceNode->getType()) {
397 case FileInfo::TYPE_FOLDER:
398 try {
399 $destNode = $destFolder->get($relativePath);
400 if ($destNode->getType() != FileInfo::TYPE_FOLDER && ($flags & self::OVWR_DEST)) {
401 $destNode->delete();
402 throw new FileNotFoundException($this->l->t("Deleted non-directory %s", [$destPath]));
403 }
404 $this->logInfo('TARGET FOLDER ALREADY EXISTS: ' . $destPath);
405 } catch (FileNotFoundException $e) {
406 $destNode = $destFolder->newFolder($relativePath);
407 }
408 break;
409 case FileInfo::TYPE_FILE:
410 try {
411 $destNode = $destFolder->get($relativePath);
412 if ($flags & self::OVWR_DEST) {
413 $destNode->delete();
414 throw new FileNotFoundException($this->l->t("Overwrite requested, deleted %s", [$destPath]));
415 } else {
416 $this->logInfo('KEEPING EXISTING FILE: ' . $destPath);
417 }
418 } catch (FileNotFoundException $e) {
419 $sourceNode->copy($destPath);
420 }
421 break;
422 }
423 return true;
424 });
425
426 return null;
427 }

512 public function restore(string $path, ?array $timeInterval = null, bool $overwriteExisting = false):bool
513 {
515 $trashManager = $this->appContainer->get(ITrashManager::class);
516
517 if (empty($trashManager)) {
518 throw new RuntimeException($this->l->t('Unable to restore "%s": TrashManager cannot be loaded', $path));
519 }
520
521 $path = trim($path, self::PATH_SEP);
522
523 $candidate = [ 'time' => 0, 'trashItem' => null ];
524
526 foreach ($trashManager->listTrashRoot($this->user()) as $trashItem) {
527 $originalLocation = trim($trashItem->getOriginalLocation(), self::PATH_SEP);
528 $deletedTime = $trashItem->getDeletedTime();
529
530 if ($originalLocation == $path) {
531 if (empty($timeInterval)
532 || ($deletedTime >= $timeInterval[0] && $deletedTime <= $timeInterval[1])) {
533 if ($candidate['time'] < $deletedTime) {
534 $candidate['time'] = $deletedTime;
535 $candidate['trashItem'] = $trashItem;
536 }
537 }
538 }
539 }
540
541 if (empty($candidate['trashItem'])) {
542 $this->logInfo('Unable to find trash-bin item to undelete');
543 return false;
544 }
545
546 if ($overwriteExisting) {
547 $this->logInfo('Try delete ' . $path);
548 $this->delete($path);
549 }
550
551 $this->logInfo('Try restore ' . $candidate['trashItem']->getOriginalLocation());
552 $trashManager->restoreItem($candidate['trashItem']);
553
554 return true;
555 }
