245 public function getFolderArchive($pathOrFolder, int $parentsToStrip = 0, string $format = 'zip'):?string
246 {
248 if (!($pathOrFolder instanceof Folder)) {
249 $folder = $this->getFolder($pathOrFolder);
250 } else {
251 $folder = $pathOrFolder;
252 }
253
254 if (empty($folder)) {
255 return null;
256 }
257
258 // don't try to strip more components than the initial path-depth
259 $parentsToStrip = min($parentsToStrip, count(Util::explode(self::PATH_SEP, $folder->getPath())));
260
261 $dataStream = fopen("php://memory", 'w');
262 $zipStream = new ZipStream(outputStream: $dataStream);
263
264 $this->archiveFolderRecursively($folder, $parentsToStrip, $zipStream);

267 rewind($dataStream);
268 $data = stream_get_contents($dataStream);
269 fclose($dataStream);

318 public function copyTree(mixed $src, mixed $dst, ?string $excludeRegexp = null, int $flags = self::KEEP_DEST):?Folder
319 {
320 $this->logInfo('SRC // DST ' . $src . ' || ' . $dst);
321
323 if (!($src instanceof Folder)) {
324 $sourceFolder = $this->getFolder($src);
325 } else {
326 $sourceFolder = $src;
327 }
328
329 if (empty($sourceFolder)) {
330 $this->logInfo('NO SOURCE FOLDER ' . $src . ' || ' . $dst);
331 return null;
332 }
333
334 $sourceRootPath = $sourceFolder->getPath();
335 $sourceRootLen = strlen($sourceRootPath);
336
337 $userFolderPrefix = $this->userFolder->getPath();
338 $userFolderLen = strlen($userFolderPrefix);
339
340 if ($dst instanceof Folder) {
341 $destFolder = $dst;
342 $destRootPath = $destFolder->getPath();
343 $destUserPath = trim(substr($destRootPath, $userFolderLen), self::PATH_SEP);
344 } else {
345 $destUserPath = $dst;
346 try {
347 $destFolder = $this->userFolder->get($destUserPath);
348 $destRootPath = $destFolder->getPath();
349 } catch (FileNotFoundException $e) {
350 $destFolder = null;
351 }
352 }
353 if ($destFolder
354 && (($flags & self::CLEAN_DEST) || $destFolder->getType() != FileInfo::TYPE_FOLDER)) {
355 $destFolder->delete();
356 $destFolder = null;
357 }
358 if (empty($destFolder)) {
359 $destFolder = $this->ensureFolder($destUserPath);
360 $destRootPath = $destFolder->getPath();
361 }

363 $this->folderWalk($sourceFolder, function(
364 Node $sourceNode,
365 int $depth,
366 ) use (
367 $sourceFolder,
368 $sourceRootPath,
369 $sourceRootLen,
370 $destFolder,
371 $destRootPath,
372 $excludeRegexp,
373 $flags,
374 ) {
375 if (!empty($excludeRegexp) && preg_match($excludeRegexp, $sourceNode->getName())) {
376 $this->logInfo('EXCLUDE HIT FOR PATTERN ' . $excludeRegexp);
377 return false;
378 }
379
380 $sourcePath = $sourceNode->getPath();
381 $relativePath = trim(substr($sourcePath, $sourceRootLen), self::PATH_SEP);
382 $destPath = $destRootPath . self::PATH_SEP . $relativePath;
383 switch ($sourceNode->getType()) {
384 case FileInfo::TYPE_FOLDER:
385 try {
386 $destNode = $destFolder->get($relativePath);
387 if ($destNode->getType() != FileInfo::TYPE_FOLDER && ($flags & self::OVWR_DEST)) {
388 $destNode->delete();
389 throw new FileNotFoundException($this->l->t("Deleted non-directory %s", [$destPath]));
390 }
391 $this->logInfo('TARGET FOLDER ALREADY EXISTS: ' . $destPath);
392 } catch (FileNotFoundException $e) {
393 $destNode = $destFolder->newFolder($relativePath);
394 }
395 break;
396 case FileInfo::TYPE_FILE:
397 try {
398 $destNode = $destFolder->get($relativePath);
399 if ($flags & self::OVWR_DEST) {
400 $destNode->delete();
401 throw new FileNotFoundException($this->l->t("Overwrite requested, deleted %s", [$destPath]));
402 } else {
403 $this->logInfo('KEEPING EXISTING FILE: ' . $destPath);
404 }
405 } catch (FileNotFoundException $e) {
406 $sourceNode->copy($destPath);
407 }
408 break;
409 }
410 return true;
411 });
412
413 return null;
414 }

499 public function restore(string $path, ?array $timeInterval = null, bool $overwriteExisting = false):bool
500 {
502 $trashManager = $this->appContainer->get(ITrashManager::class);
503
504 if (empty($trashManager)) {
505 throw new RuntimeException($this->l->t('Unable to restore "%s": TrashManager cannot be loaded', $path));
506 }
507
508 $path = trim($path, self::PATH_SEP);
509
510 $candidate = [ 'time' => 0, 'trashItem' => null ];
511
513 foreach ($trashManager->listTrashRoot($this->user()) as $trashItem) {
514 $originalLocation = trim($trashItem->getOriginalLocation(), self::PATH_SEP);
515 $deletedTime = $trashItem->getDeletedTime();
516
517 if ($originalLocation == $path) {
518 if (empty($timeInterval)
519 || ($deletedTime >= $timeInterval[0] && $deletedTime <= $timeInterval[1])) {
520 if ($candidate['time'] < $deletedTime) {
521 $candidate['time'] = $deletedTime;
522 $candidate['trashItem'] = $trashItem;
523 }
524 }
525 }
526 }
527
528 if (empty($candidate['trashItem'])) {
529 $this->logInfo('Unable to find trash-bin item to undelete');
530 return false;
531 }
532
533 if ($overwriteExisting) {
534 $this->logInfo('Try delete ' . $path);
535 $this->delete($path);
536 }
537
538 $this->logInfo('Try restore ' . $candidate['trashItem']->getOriginalLocation());
539 $trashManager->restoreItem($candidate['trashItem']);
540
541 return true;
542 }
