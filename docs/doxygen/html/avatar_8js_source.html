28/**
29 * This plugin inserts the right avatar for the user, depending on, whether a
30 * custom avatar is uploaded - which it uses then - or not, and display a
31 * placeholder with the first letter of the users name instead.
32 * For this it queries the core_avatar_get route, thus this plugin is fit very
33 * tightly for owncloud, and it may not work anywhere else.
34 *
35 * You may use this on any <div></div>
36 * Here I'm using <div class="avatardiv"></div> as an example.
37 *
38 * There are 5 ways to call this:
39 *
40 * 1. $('.avatardiv').avatar('jdoe', 128);
41 * This will make the div to jdoe's fitting avatar, with a size of 128px.
42 *
43 * 2. $('.avatardiv').avatar('jdoe');
44 * This will make the div to jdoe's fitting avatar. If the div already has a
45 * height, it will be used for the avatars size. Otherwise this plugin will
46 * search for 'size' DOM data, to use for avatar size. If neither are available
47 * it will default to 64px.
48 *
49 * 3. $('.avatardiv').avatar();
50 * This will search the DOM for 'user' data, to use as the username. If there
51 * is no username available it will default to a placeholder with the value of
52 * "?". The size will be determined the same way, as the second example.
53 *
54 * 4. $('.avatardiv').avatar('jdoe', 128, true);
55 * This will behave like the first example, except it will also append random
56 * hashes to the custom avatar images, to force image reloading in IE8.
57 *
58 * 5. $('.avatardiv').avatar('jdoe', 128, undefined, true);
59 * This will behave like the first example, but it will hide the avatardiv, if
60 * it will display the default placeholder. undefined is the ie8fix from
61 * example 4 and can be either true, or false/undefined, to be ignored.
62 *
63 * 6. $('.avatardiv').avatar('jdoe', 128, undefined, true, callback);
64 * This will behave like the above example, but it will call the function
65 * defined in callback after the avatar is placed into the DOM.
66 *
67 */
68
69$.fn.avatar = function(user, size, ie8fix, hidedefault, callback, displayname) {
70 const setAvatarForUnknownUser = function(target) {
71 target.imageplaceholder('?')
72 target.css('background-color', '#b9b9b9')
73 }
74
75 if (typeof (user) !== 'undefined') {
76 user = String(user)
77 }
78 if (typeof (displayname) !== 'undefined') {
79 displayname = String(displayname)
80 }
81
82 if (typeof (size) === 'undefined') {
83 if (this.height() > 0) {
84 size = this.height()
85 } else if (this.data('size') > 0) {
86 size = this.data('size')
87 } else {
88 size = 64
89 }
90 }
91
92 this.height(size)
93 this.width(size)
94
95 if (typeof (user) === 'undefined') {
96 if (typeof (this.data('user')) !== 'undefined') {
97 user = this.data('user')
98 } else {
99 setAvatarForUnknownUser(this)
100 return
101 }
102 }
103
104 // sanitize
105 user = String(user).replace(/\//g, '')
106
107 const $div = this
108 let url
109
110 // If this is our own avatar we have to use the version attribute
111 if (user === OC.getCurrentUser().uid) {
112 url = OC.generateUrl(
113 '/avatar/{user}/{size}?v={version}',
114 {
115 user,
116 size: Math.ceil(size * window.devicePixelRatio),
117 version: oc_userconfig.avatar.version,
118 })
119 } else {
120 url = OC.generateUrl(
121 '/avatar/{user}/{size}',
122 {
123 user,
124 size: Math.ceil(size * window.devicePixelRatio),
125 })
126 }
127
128 const img = new Image()
129
130 // If the new image loads successfully set it.
131 img.onload = function() {
132 $div.clearimageplaceholder()
133 $div.append(img)
134
135 if (typeof callback === 'function') {
136 callback()
137 }
138 }
139 // Fallback when avatar loading fails:
140 // Use old placeholder when a displayname attribute is defined,
141 // otherwise show the unknown user placeholder.
142 img.onerror = function() {
143 $div.clearimageplaceholder()
144 if (typeof (displayname) !== 'undefined') {
145 $div.imageplaceholder(user, displayname)
146 } else {
147 setAvatarForUnknownUser($div)
148 }
149
150 if (typeof callback === 'function') {
151 callback()
152 }
153 }
154
155 if (size < 32) {
156 $div.addClass('icon-loading-small')
157 } else {
158 $div.addClass('icon-loading')
159 }
160 img.width = size
161 img.height = size
162 img.src = url
163 img.alt = ''
164}
