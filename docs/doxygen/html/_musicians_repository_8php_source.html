192 public function fetchLastModifiedDate(array $criteria = [], string $indexBy = 'uuid')
193 {
194 $selects = [];
195 if (!empty($criteria)) {
196 $selects[] = 'm.'.$indexBy.' AS '.$indexBy;
197 }
198 $selects[] = "GREATEST(
199 MAX(COALESCE(m.updated, '')),
200 MAX(COALESCE(pp.updated, '')),
201 COALESCE(photo.updated, ''),
202 MAX(COALESCE(mi.updated, '')))
203 AS lastModified";
204
205 $qb = $this->createQueryBuilder('m', 'm.'.$indexBy);
206 $qb->select($selects)
207 ->leftJoin('m.projectParticipation', 'pp')
208 ->leftJoin('m.photo', 'photo')
209 ->leftJoin('m.instruments', 'mi');
210
211 if (!empty($criteria)) {
212 $qb->groupBy('m.id');
213 $andX = $qb->expr()->andX();
214 foreach ($criteria as $key => &$value) {
215 $value = str_replace('*', '%', $value);
216 if (strpos($value, '%') !== false) {
217 $andX->add($qb->expr()->like('m'.'.'.$key, ':'.$key));
218 } else {
219 $andX->add($qb->expr()->eq('m'.'.'.$key, ':'.$key));
220 }
221 }
222 $qb->where($andX);
223 foreach ($criteria as $key => $value) {
224 $type = ($key == 'uuid') ? 'uuid_binary' : null;
225 $qb->setParameter($key, $value, $type);
226 }
227 }
228
