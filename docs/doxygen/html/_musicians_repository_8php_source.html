196 public function fetchLastModifiedDate(array $criteria = [], string $indexBy = 'uuid')
197 {
198 $selects = [];
199 if (!empty($criteria)) {
200 $selects[] = 'm.'.$indexBy.' AS '.$indexBy;
201 }
202 $selects[] = "GREATEST(
203 MAX(COALESCE(m.updated, '')),
204 MAX(COALESCE(pp.updated, '')),
205 MAX(COALESCE(mi.updated, '')))
206 AS lastModified";
207
208 $qb = $this->createQueryBuilder('m', 'm.'.$indexBy);
209 $qb->select($selects)
210 ->leftJoin('m.projectParticipation', 'pp')
211 ->leftJoin('m.instruments', 'mi');
212
213 if (!empty($criteria)) {
214 $qb->groupBy('m.id');
215 $andX = $qb->expr()->andX();
216 foreach ($criteria as $key => &$value) {
217 $value = str_replace('*', '%', $value);
218 if (strpos($value, '%') !== false) {
219 $andX->add($qb->expr()->like('m'.'.'.$key, ':'.$key));
220 } else {
221 $andX->add($qb->expr()->eq('m'.'.'.$key, ':'.$key));
222 }
223 }
224 $qb->where($andX);
225 foreach ($criteria as $key => $value) {
226 $type = ($key == 'uuid') ? 'uuid_binary' : null;
227 $qb->setParameter($key, $value, $type);
228 }
229 }
230
231 // $this->log('CRIT '.print_r($criteria, true));
232 // $this->log('SQL '.$qb->getQuery()->getSql());
233 // $this->log('PARAM '.print_r($qb->getQuery()->getParameters(), true));
234
235 return $qb->getQuery()->getOneOrNullResult();
236 }
