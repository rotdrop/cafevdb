82 public function recordUntranslated(string $phrase, string $locale, string $file, int $line):void
83 {
84 if (array_search($locale, $this->availableLanguages) === false) {
85 return;
86 }
87 $this->entityManager->suspendLogging();
88 $this->setDataBaseRepository(TranslationKey::class);
89 $translationKey = $this->findOneBy([ 'phraseHash' => md5($phrase) ]);
90 if (empty($translationKey)) {
91 $translationKey = TranslationKey::create()->setPhrase($phrase);
92 try {
93 $this->persist($translationKey);
94 $this->flush();
95 } catch (\Throwable $t) {
96 $this->logException($t);
97 }
98 $this->logDebug('Translation key for "'.$phrase.'" was empty, new id '.$translationKey->getId());
99 } else {
100 $this->logDebug('Existing translation key for "'.$phrase.'" has id '.$translationKey->getId());
101 }
102
103 $this->setDataBaseRepository(TranslationLocation::class);
104 $location = $this->findOneBy([
105 'translationKey' => $translationKey,
106 'file' => $file,
107 'line' => $line,
108 ]);
109 if (empty($location)) {
110 $this->logDebug('Empty location for key '.$translationKey->getId());
111 $location = TranslationLocation::create()
112 ->setTranslationKey($translationKey)
113 ->setFile($file)
114 ->setLine($line);
115 try {
116 $this->persist($location);
117 $this->flush();
118 } catch (\Throwable $t) {
119 $this->logException($t);
120 }
121 }
122 $this->setDatabaseRepository(MissingTranslation::class);
123 $missingLocale = $this->findOneBy([
124 'translationKey' => $translationKey,
125 'locale' => $locale,
126 ]);
127 if (empty($missingLocale)) {
128 $missingLocale = MissingTranslation::create()
129 ->setTranslationKey($translationKey)
130 ->setLocale($locale);
131 try {
132 $this->persist($missingLocale);
133 $this->flush();

135 $this->logException($t);

139 $this->flush();

157 public function recordTranslation(string $phrase, string $translatedPhrase, string $locale):bool
158 {
159 if (empty(Util::normalizeSpaces($translatedPhrase))) {
160 throw new Exception('Translation for %s is empty.', $phrase);
161 }
162 $this->setDataBaseRepository(TranslationKey::class);
163 $translationKey = $this->findOneBy([ 'phrase' => $phrase ]);
164 if (empty($translationKey)) {
165 $translationKey = TranslationKey::create()->setPhrase($phrase);
166 try {
167 $this->persist($translationKey);
168 $this->flush();
169 } catch (\Throwable $t) {
170 $this->logException($t);
171 return false;
172 }
173 }
174
175 $this->setDataBaseRepository(Translation::class);
176 $changed = false;
177 $translation = $this->findOneBy([
178 'translationKey' => $translationKey,
179 'locale' => $locale ]);
180 if (empty($translation)) {
181 $translation = Translation::create()
182 ->setTranslationKey($translationKey)
183 ->setTranslation($translatedPhrase)
184 ->setLocale($locale);
185 $changed = true;
186 } elseif ($translation->getTranslation() != $translatedPhrase) {
187 $translation->setTranslation($translatedPhrase);
188 $changed = true;
189 }
190 if ($changed) {
191 try {
192 $this->persist($translation);
193 $this->flush();
194 } catch (\Throwable $t) {
195 $this->logException($t);
196 return false;
197 }
198 }
199 $this->setDatabaseRepository(MissingTranslation::class);
200 $missingLocale = $this->findOneBy([
201 'translationKey' => $translationKey,
202 'locale' => $locale,
203 ]);
204 if (!empty($missingLocale)) {
205 $this->remove($missingLocale, flush: true);
206 }
207 return true;
208 }

245 public function generateCatalogueTemplates():string
246 {
247 $this->setDataBaseRepository(TranslationKey::class);
248 $translationKeys = $this->findAll();
249 $catalogue = [];
250 foreach ($translationKeys as $translationKey) {
251 $entry = [];
252 foreach ($translationKey['locations'] as $location) {
253 $entry[] = "#: {$location['file']}:{$location['line']}";
254 }
255 $entry[] = "#, php-format";
256 $entry[] = 'msgid "'.str_replace('"', '\\"', $translationKey['phrase']).'"';
257 $entry[] = 'msgstr ""';
258 $entry[] = '';
259 $catalogue[] = implode("\n", $entry);
260 }
261 $contents = implode("\n", $catalogue);
262

271 public function eraseTranslationKeys(string $phrase):bool
272 {

275 if (count($translationKeys) == 0) {

278 $this->logWarn('About to erase '.count($translationKeys).' translation-keys.');
