86 public function get(string $parameter):DataResponse

88 $value = null;

91 $value = $this->getAppValue('usergroup');
92 break;

94 $admins = $this->configService->getGroupSubAdmins();
95 $value = [];
97 foreach ($admins as $admin) {
98 $value[] = $admin->getUID();
99 }
100 break;
101 case AdminSettings::WIKI_NAME_SPACE_KEY:
102 $value = $this->getAppValue('wikinamespace');
103 break;
104 case AdminSettings::CLOUD_USER_BACKEND_CONFIG_KEY:
105 $value = $this->di(CloudUserConnectorService::class)->haveCloudUserBackendConfig();
106 break;
107 case FontService::TTF_FONTS_FOLDER_CONFIG:
109 $fontService = $this->di(CloudUserConnectorService::class);

111 break;
112 case AdminSettings::TRUE_TYPE_FONTS:
114 $fontService = $this->di(CloudUserConnectorService::class);
115 $value = $fontService->getFontFolderEntries();
116 break;
117 case FontService::DEFAULT_OFFICE_FONT:
119 $fontService = $this->di(CloudUserConnectorService::class);
120 $value = $fontService->getDefaultFontName();
121 break;
122 }
123 if ($value != null) {
124 return new DataResponse([ 'value' => $value ]);
125 } else {
126 return new DataResponse([ 'key' => $parameter ], Http::STATUS_NOT_FOUND);

176 private function post(string $parameter, mixed $value = null, ?string $operation = null):DataResponse
177 {
178 $wikiNameSpace = $this->getAppValue('wikinamespace');
179 $orchestraUserGroup = $this->getAppValue('usergroup');
180 try {
181 switch ($parameter) {
182 case AdminSettings::ORCHESTRA_USER_GROUP_KEY:
183 $realValue = trim($value);
184 if (!empty($orchestraUserGroup) && !empty($wikiNameSpace)) {
185 $this->revokeWikiAccess($wikiNameSpace, $orchestraUserGroup);
186 }
187 $orchestraUserGroup = $realValue;
188 $this->setAppValue('usergroup', $orchestraUserGroup);
189 $result = [
190 'orchestraUserGroup' => $orchestraUserGroup,
191 ];
192 if (empty($wikiNameSpace)) {
193 $wikiNameSpace = $orchestraUserGroup;
194 $this->setAppValue('wikinamespace', $wikiNameSpace);
195 $result['wikiNameSpace'] = $wikiNameSpace;
196 }
197 $this->grantWikiAccess($wikiNameSpace, $orchestraUserGroup);
198 $result = [
199 'messages' => [
200 'transient' => [
201 $this->l->t('Setting orchestra group to "%s". Please login as group administrator and configure the Camerata DB application.', [$realValue]),
202 ],
203 ],
204 ];
205 return self::dataResponse($result);
206
207 case AdminSettings::ORCHESTRA_USER_GROUP_KEY . 'Admins':
208 if (!is_array($value)) {
209 return self::grumble($this->l->t('Expecting a list of user-ids.'));
210 }
211 $userGroup = $this->group();
212 if (empty($userGroup)) {
213 return self::grumble($this->l->t('Orchestra management group is unset or non-existent'));
214 }
215 $currentAdmins = array_map(fn($user) => $user->getUID(), $this->getGroupSubAdmins());
216 $missing = array_diff($value, $currentAdmins);
217 $remaining = array_intersect($value, $currentAdmins);
218 $excess = array_diff($currentAdmins, $value);
219 $success = [];
220 $failure = [];
221 if (!empty($remaining)) {
222 $success[] = $this->l->t('Already as sub-admin of "%1$s" configured: %2$s.', [ $userGroup->getGID(), implode(', ', $remaining) ]);
223 }
224 foreach ($missing as $userId) {
225 try {
226 $user = $this->user($userId);
227 $this->subAdminManager()->createSubAdmin($user, $userGroup);
228 $success[] = $this->l->t('Added "%1$s" as sub-admin of "%2$s".', [ $userId, $userGroup->getGID(), ]);
229 } catch (Throwable $t) {
230 $this->logException($t);
231 $failure[] = $this->t->t('Failed to add "%1$s" as sub-admin to "%2$s": %3$s', [ $userId, $userGroup->getGID(), $t->getMessage(), ]);
232 }
233 }
234 foreach ($excess as $userId) {
235 try {
236 $user = $this->user($userId);
237 $this->subAdminManager()->deleteSubAdmin($user, $userGroup);
238 $success[] = $this->l->t('Deleted "%1$s" as sub-admin from "%2$s".', [ $userId, $userGroup->getGID(), ]);
239 } catch (Throwable $t) {
240 $this->logException($t);
241 $failure[] = $this->t->t('Failed to delete "%1$s" as sub-admin from "%2$s": %3$s', [ $userId, $userGroup->getGID(), $t->getMessage(), ]);
242 }
243 }
244 $result = [
245 'messages' => [
246 'transient' => $success,
247 'permanent' => $failure,
248 ],
249 ];
250 return self::dataResponse($result);
251
252 case AdminSettings::WIKI_NAME_SPACE_KEY:
253 if (!empty($orchestraUserGroup) && !empty($wikiNameSpace)) {
254 $this->revokeWikiAccess($wikiNameSpace, $orchestraUserGroup);
255 }
256 $realValue = trim($value);
257 $wikiNameSpace = $realValue;
258 $this->setAppValue('wikinamespace', $wikiNameSpace);
259 $result['wikiNameSpace'] = $wikiNameSpace;
260
261 if (!empty($orchestraUserGroup)) {
262 $this->grantWikiAccess($wikiNameSpace, $orchestraUserGroup);
263 }
264
265 $result = [
266 'messages' => [
267 'transient' => [
268 $this->l->t('Setting wiki name-space to "%s".', [$realValue]),
269 ],
270 ],
271 ];

273
274 case AdminSettings::CLOUD_USER_BACKEND_CONFIG_KEY:
275 $delete = $value !== null && $value !== '' && filter_var($value, FILTER_VALIDATE_BOOLEAN, FILTER_NULL_ON_FAILURE) === false;
276 $messages = [];
278 $cloudUserConnector = $this->di(CloudUserConnectorService::class);

280 $cloudUserConnector->setCloudUserSubAdmins(delete: true);

297 '<a class="external settings" href="' . $cloudUserBackendSettings . '" target="' . \md5($cloudUserBackendSettings) . '">' . CloudUserConnectorService::CLOUD_USER_BACKEND . '</a>',
298 CloudUserConnectorService::CLOUD_USER_BACKEND,
299 ]);
300 return self::dataResponse([
301 'messages' => [
302 'transient' => $messages,
303 'permanent' => [ $settingsHint, ],
304 ],
305 ]);
306 case FontService::DEFAULT_OFFICE_FONT_CONFIG:
307 if (empty($value)) {
308 $this->deleteAppValue($parameter);
310 $fontService = $this->di(FontService::class);
311 $defaultFont = $fontService->getDefaultFontName();
312 $value = $defaultFont;

319 return self::dataResponse([

321 'value' => $value,
322 ]);

327 }

329 $this->logException($t);

372 private function fontCache(string $operation):DataResponse
373 {
375 $fontService = $this->di(FontService::class);
376 $fonts = null;
377 try {
378 switch ($operation) {
379 case self::FONT_CACHE_PURGE:
380 $fontService->purgeFontsFolder();
381 $fonts = [];
382 break;
383 case self::FONT_CACHE_RESCAN:
384 $fontService->purgeFontsFolder();
385 // fall through
386 case self::FONT_CACHE_UPDATE:
387 $fonts = $fontService->populateFontsFolder();
388 break;
389 default:
390 return self::grumble($this->l->t('Unknown font cache operation: "%s".', $operation));
391 }
392 } catch (Throwable $t) {
393 $this->logException($t);
394 return self::grumble($this->l->t('Font operation "%1$s" failed with exception "%2$s".', [
395 $operation, $t->getMessage(),
396 ]));
397 }
398 if ($fonts !== null) {
399 return self::dataResponse([
400 'message' => $this->l->t('Font operation "%s" successful.', $operation),
401 'fonts' => $fonts,
402 'default' => $fontService->getDefaultFontName(),
403 ]);
404 }
405 return self::grumble($this->l->t('Unknown internal error during font-cache operation: "%s".', $operation));
406 }
