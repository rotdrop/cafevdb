82 public function get(string $parameter):DataResponse
83 {
84 $value = null;
85 switch ($parameter) {
86 case AdminSettings::USER_AND_GROUP_BACKEND_KEY:
87 $value = $this->getAppValue(ConfigService::USER_AND_GROUP_BACKEND_KEY, AdminSettings::DEFAULT_USER_AND_GROUP_BACKEND);
88 break;
89 case AdminSettings::ORCHESTRA_USER_GROUP_KEY:
90 $value = $this->getAppValue(ConfigService::USER_GROUP_KEY);
91 break;
92 case AdminSettings::ORCHESTRA_USER_GROUP_ADMINS_KEY:
93 $admins = $this->configService->getGroupSubAdmins();
94 $value = [];
96 foreach ($admins as $admin) {
97 $value[] = $admin->getUID();
98 }
99 break;
100 case AdminSettings::WIKI_NAME_SPACE_KEY:
101 $value = $this->getAppValue('wikinamespace');
102 break;
103 case AdminSettings::CLOUD_USER_BACKEND_CONFIG_KEY:
104 $value = $this->di(CloudUserConnectorService::class)->haveCloudUserBackendConfig();
105 break;
106 case FontService::OFFICE_FONTS_FOLDER_CONFIG:
108 $fontService = $this->di(CloudUserConnectorService::class);
109 $value = $fontService->getFontsFolderName();
110 break;
111 case AdminSettings::OFFICE_FONTS:
113 $fontService = $this->di(FontService::class);
114 $value = $fontService->getFontFolderEntries();
115 break;
116 case FontService::DEFAULT_OFFICE_FONT_CONFIG:
118 $fontService = $this->di(FontService::class);
119 $value = $fontService->getDefaultFontName();
120 break;
121 default:
122 return new DataResponse([ 'key' => $parameter ], Http::STATUS_NOT_FOUND);
123 }
124 return new DataResponse([ 'value' => $value ]);

173 private function post(string $parameter, mixed $value = null, ?string $operation = null):DataResponse
174 {
175 $wikiNameSpace = $this->getAppValue('wikinamespace');
176 $orchestraUserGroup = $this->getAppValue(ConfigService::USER_GROUP_KEY);
177 try {
178 switch ($parameter) {
179 case AdminSettings::USER_AND_GROUP_BACKEND_KEY:
180 // @todo something has to be done if this is really set to something
181 // and we want to support that.
182 $this->setAppValue(AdminSettings::USER_AND_GROUP_BACKEND_KEY, $value);
183 $result = [
184 'messages' => [
185 'transient' => [
186 $this->l->t('Setting user and group backend to "%s". Please have a look at the action-menu for additional actions.', [$value]),
187 ],
188 ],
189 ];
190 return self::dataResponse($result);
191 break;
192 case AdminSettings::ORCHESTRA_USER_GROUP_KEY:
193 $realValue = trim($value);
194 if (!empty($orchestraUserGroup) && !empty($wikiNameSpace)) {
195 $this->revokeWikiAccess($wikiNameSpace, $orchestraUserGroup);
196 }
197 $orchestraUserGroup = $realValue;
198 $this->setAppValue(ConfigService::USER_GROUP_KEY, $orchestraUserGroup);
199 $result = [
200 $parameter => $orchestraUserGroup,
201 ];
202 if (empty($wikiNameSpace)) {
203 $wikiNameSpace = $orchestraUserGroup;
204 $this->setAppValue('wikinamespace', $wikiNameSpace);
205 $result['wikiNameSpace'] = $wikiNameSpace;
206 }
207 $this->grantWikiAccess($wikiNameSpace, $orchestraUserGroup);
208 $result = [
209 'messages' => [
210 'transient' => [
211 $this->l->t('Setting orchestra group to "%s". Please login as group administrator and configure the Camerata DB application.', [$realValue]),
212 ],
213 ],
214 ];
215 return self::dataResponse($result);
216
217 case AdminSettings::ORCHESTRA_USER_GROUP_ADMINS_KEY:
218 if (!is_array($value)) {
219 return self::grumble($this->l->t('Expecting a list of user-ids.'));
220 }
221 $userGroup = $this->group();
222 if (empty($userGroup)) {
223 return self::grumble($this->l->t('Orchestra management group is unset or non-existent'));
224 }
225 $currentAdmins = array_map(fn($user) => $user->getUID(), $this->getGroupSubAdmins());
226 $missing = array_diff($value, $currentAdmins);
227 $remaining = array_intersect($value, $currentAdmins);
228 $excess = array_diff($currentAdmins, $value);
229 $success = [];
230 $failure = [];
231 if (!empty($remaining)) {
232 $success[] = $this->l->t('Already as sub-admin of "%1$s" configured: %2$s.', [ $userGroup->getGID(), implode(', ', $remaining) ]);
233 }
234 foreach ($missing as $userId) {
235 try {
236 $user = $this->user($userId);
237 $this->subAdminManager()->createSubAdmin($user, $userGroup);
238 $success[] = $this->l->t('Added "%1$s" as sub-admin of "%2$s".', [ $userId, $userGroup->getGID(), ]);
239 } catch (Throwable $t) {
240 $this->logException($t);
241 $failure[] = $this->t->t('Failed to add "%1$s" as sub-admin to "%2$s": %3$s', [ $userId, $userGroup->getGID(), $t->getMessage(), ]);
242 }
243 }
244 foreach ($excess as $userId) {
245 try {
246 $user = $this->user($userId);
247 $this->subAdminManager()->deleteSubAdmin($user, $userGroup);
248 $success[] = $this->l->t('Deleted "%1$s" as sub-admin from "%2$s".', [ $userId, $userGroup->getGID(), ]);
249 } catch (Throwable $t) {
250 $this->logException($t);
251 $failure[] = $this->t->t('Failed to delete "%1$s" as sub-admin from "%2$s": %3$s', [ $userId, $userGroup->getGID(), $t->getMessage(), ]);
252 }
253 }
254 $result = [
255 'messages' => [
256 'transient' => $success,
257 'permanent' => $failure,
258 ],
259 ];
260 return self::dataResponse($result);
261
262 case AdminSettings::WIKI_NAME_SPACE_KEY:
263 if (!empty($orchestraUserGroup) && !empty($wikiNameSpace)) {
264 $this->revokeWikiAccess($wikiNameSpace, $orchestraUserGroup);
265 }
266 $realValue = trim($value);
267 $wikiNameSpace = $realValue;
268 $this->setAppValue('wikinamespace', $wikiNameSpace);
269 $result['wikiNameSpace'] = $wikiNameSpace;
270
271 if (!empty($orchestraUserGroup)) {
272 $this->grantWikiAccess($wikiNameSpace, $orchestraUserGroup);
273 }
274
275 $result = [
276 'messages' => [
277 'transient' => [
278 $this->l->t('Setting wiki name-space to "%s".', [$realValue]),
279 ],
280 ],
281 ];
282 return self::dataResponse($result);
283
284 case AdminSettings::CLOUD_USER_BACKEND_CONFIG_KEY:
285 $delete = $value !== null && $value !== '' && filter_var($value, FILTER_VALIDATE_BOOLEAN, FILTER_NULL_ON_FAILURE) === false;
286 $messages = [];
288 $cloudUserConnector = $this->di(CloudUserConnectorService::class);
289 if ($delete) {
290 $cloudUserConnector->setCloudUserSubAdmins(delete: true);
291 $responses = $cloudUserConnector->configureCloudUserBackend(erase: true);
292 } else {
293 $cloudUserConnector = $this->di(CloudUserConnectorService::class);
294 $responses = $cloudUserConnector->configureCloudUserBackend(erase: false);
295 $cloudUserConnector->setCloudUserSubAdmins(delete: false);
296 }
297 $messages[] = $this->l->t('"%1$s" controller answered with "%2$s".', [
298 CloudUserConnectorService::CLOUD_USER_BACKEND,
299 implode('", "', $responses)
300 ]);
301
302 // Get front-end link to user-sql config
303 //
304 // https://example.com/nextcloud/index.php/settings/admin/user_sql
305 $cloudUserBackendSettings = $this->urlGenerator()->getBaseUrl() . '/index.php/settings/admin/' . CloudUserConnectorService::CLOUD_USER_BACKEND;
306 $settingsHint = $this->l->t('Please head over to the %1$s settings and check the generated "%2$s"-configuration.', [
307 '<a class="external settings" href="' . $cloudUserBackendSettings . '" target="' . \md5($cloudUserBackendSettings) . '">' . CloudUserConnectorService::CLOUD_USER_BACKEND . '</a>',
308 CloudUserConnectorService::CLOUD_USER_BACKEND,
309 ]);
310 return self::dataResponse([
311 'messages' => [
312 'transient' => $messages,
313 'permanent' => [ $settingsHint, ],
314 ],
315 ]);
316 case FontService::DEFAULT_OFFICE_FONT_CONFIG:
317 if (empty($value)) {
318 $this->deleteAppValue($parameter);
320 $fontService = $this->di(FontService::class);
321 $defaultFont = $fontService->getDefaultFontName();
322 $value = $defaultFont;
323 $transient = [ $this->l->t('Default office font reset to default "%s".', $value), ];
324 } else {
325 $this->setAppValue($parameter, $value);
326 $transient = [ $this->l->t('Default office font set to "%s".', $value), ];
327 }
328
329 return self::dataResponse([
330 'messages' => [ 'transient' => $transient, ],
331 'value' => $value,
332 ]);
333 case self::POST_REQUEST_FONT_CACHE:
334 return $this->fontCache($operation);
335 default:
336 break;
337 }
338 } catch (Throwable $t) {
339 $this->logException($t);
340 return self::grumble($this->exceptionChainData($t));
341 }
342 return self::grumble($this->l->t('Unknown Request: "%s"', $parameter));

382 private function fontCache(string $operation):DataResponse
383 {
385 $fontService = $this->di(FontService::class);
386 $fonts = null;
387 try {
388 switch ($operation) {
389 case self::FONT_CACHE_PURGE:
390 $fontService->purgeFontsFolder();
391 $fonts = [];
392 break;
393 case self::FONT_CACHE_RESCAN:
394 $fontService->purgeFontsFolder();
395 // fall through
396 case self::FONT_CACHE_UPDATE:
397 $fonts = $fontService->populateFontsFolder();
398 break;
399 default:
400 return self::grumble($this->l->t('Unknown font cache operation: "%s".', $operation));
401 }
402 } catch (Throwable $t) {
403 $this->logException($t);
404 return self::grumble($this->l->t('Font operation "%1$s" failed with exception "%2$s".', [
405 $operation, $t->getMessage(),
406 ]));
407 }
408 if ($fonts !== null) {
409 return self::dataResponse([
410 'message' => $this->l->t('Font operation "%s" successful.', $operation),
411 'fonts' => $fonts,
412 'default' => $fontService->getDefaultFontName(),
413 ]);
414 }
415 return self::grumble($this->l->t('Unknown internal error during font-cache operation: "%s".', $operation));
416 }
