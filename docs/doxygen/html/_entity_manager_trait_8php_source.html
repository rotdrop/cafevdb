151 protected function remove(mixed $entity, bool $flush = false, bool $hard = false, bool $soft = false):void
152 {
153 if (filter_var($entity, FILTER_VALIDATE_INT, ['min_range' => 1])) {
154 $entity = [ 'id' => $entity ];
155 }
156 if (is_array($entity)) {
157 $key = $entity;
