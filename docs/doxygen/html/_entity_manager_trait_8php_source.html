156 protected function remove(
157 mixed $entity,
158 bool $flush = false,
159 bool $hard = false,
160 bool $soft = false,
161 bool $useTransaction = false,
162 ):void {
163 if (filter_var($entity, FILTER_VALIDATE_INT, ['min_range' => 1])) {
164 $entity = [ 'id' => $entity ];
165 }
166 if (is_array($entity)) {
167 $key = $entity;
168 // getReference() will not always work, and in particular not with
169 // foreign keys, so just fetch the entity first.
170 //
171 // $entity = $this->entityManager->getReference($this->entityClassName, $key);
172 $entity = $this->entityManager->find($this->entityClassName, $key);
173 }
174 if ($soft && !$hard && method_exists($entity, 'isDeleted') && $entity->isDeleted()) {
175 return;
176 }
177 try {
178 if ($useTransaction && $flush) {
179 $this->entityManager->beginTransaction();
180 }
181 $this->entityManager->remove($entity);
182 if ($hard && (!method_exists($entity, 'isDeleted') || !$entity->isDeleted())) {
183 $this->flush();
184 $this->entityManager->remove($entity);
185 }
186
187 if ($flush) {
188 $this->flush();
189 if ($useTransaction) {
190 $this->entityManager->commit();
191 }
192 }
193 } catch (Throwable $t) {
194 if ($useTransaction) {
195 $this->entityManager->rollback();
196 }
197 throw $t;
198 }
199 }
