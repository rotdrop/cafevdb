69 public static function transposition(string $input, ?callable $callback):array
70 {
71 if (!is_callable($callback)) {
72 return [];
73 }
74 $result = [];
75 $len = strlen($input);
76 for ($idx = 0; $idx < $len - 1; ++$idx) {
77 $victim = $input;
78 $victim[$idx] = $input[$idx+1];
79 $victim[$idx+1] = $input[$idx];
80 if ($callback($victim)) {
81 $result[] = $victim;
82 }
83 }
84 return $result;
85 }

98 public function parseCurrency(string $value, bool $noThrow = false)
99 {
100 $value = Util::normalizeSpaces($value, ' ', true);
101 if (strstr($value, '€') !== false) {
102 $currencyCode = '€';
103 } elseif (strstr($value, '$') !== false) {
104 $currencyCode = '$';
105 } elseif (is_numeric($value)) {
106 // just return e.g. if C-locale is active
107 $currencyCode = '';
108 } elseif ($noThrow) {
109 return false;
110 } else {
111 throw new InvalidArgumentException($this->l->t(
112 "Cannot parse `%s' Only plain number, € and \$ currencies are supported.",
113 (string)$value));
114 }
115
116 switch ($currencyCode) {
117 case '€':
118 // Allow either comma or decimal point as separator
119 if (preg_match('/^(€)?\h*([+-]?)\h*(\d{1,3}(\.\d{3})*|(\d+))(\,(\d{2}))?\h*(€)?$/', $value, $matches)
120 ||
121 preg_match('/^(€)?\h*([+-]?)\h*(\d{1,3}(\,\d{3})*|(\d+))(\.(\d{2}))?\h*(€)?$/', $value, $matches)) {
122 //print_r($matches);
123 // Convert value to number
124 //
125 // $matches[2] optional sign
126 // $matches[3] integral part
127 // $matches[7] fractional part
128 $fraction = isset($matches[7]) ? $matches[7] : '00';
129 $value = (float)($matches[2].str_replace([',', '.', ], '', $matches[3]).'.'.$fraction);
130 } elseif ($noThrow) {
131 return false;
132 } else {
133 throw new InvalidArgumentException($this->l->t('Cannot parse %s-currency string "%s".', [ $currencyCode, (string)$value, ]));
134 }
135 break;
136 case '$':
137 if (preg_match('/^\$? *([+-])? *(\d{1,3}(\,\d{3})*|(\d+))(\.(\d{2}))? *\$?$/', $value, $matches)) {
138 // Convert value to number
139 //
140 // $matches[1] optional sign
141 // $matches[2] integral part
142 // $matches[6] fractional part
143 $fraction = isset($matches[6]) ? $matches[6] : '00';
144 $value = (float)($matches[1].str_replace([ ',', '.', ], '', $matches[2]).'.'.$fraction);
145 } elseif ($noThrow) {
146 return false;
147 } else {
148 throw new InvalidArgumentException($this->l->t('Cannot parse %s-currency string "%s".', [ $currencyCode, (string)$value, ]));
149 }
150 break;
151 }
152 return [
153 'amount' => $value,
154 'currency' => $currencyCode,
155 ];
156 }

169 public function currencyValue(string $value)
170 {
171 $locale = $this->getLocale();
172 $amount = preg_replace('/\s+/u', '', $value);
173 $fmt = new NumberFormatter($locale, NumberFormatter::CURRENCY);
174 $cur = $fmt->getSymbol(\NumberFormatter::CURRENCY_SYMBOL);
175 $amount = str_replace($cur, '', $amount);
176 $cur = $fmt->getSymbol(\NumberFormatter::INTL_CURRENCY_SYMBOL);
177 $amount = str_replace($cur, '', $amount);
178 $fmt = new NumberFormatter($locale, NumberFormatter::DECIMAL);
179 $parsed = $fmt->parse($amount);
180 if ($parsed === false) {
181 $fmt = new NumberFormatter('en_US_POSIX', NumberFormatter::DECIMAL);
182 $parsed = $fmt->parse($amount);
183 }
184 return $parsed !== false ? sprintf('%.02f', $parsed) : $parsed;
185 }

203 public function ensureCamelCase(string $slug, bool $capitalizeFirst = true):string
204 {
205 $slug = Util::normalizeSpaces($slug);
206 $words = explode(' ', $slug);
207 if (count($words) > 1) {
208 foreach ($words as &$word) {
209 $word = preg_replace_callback(
210 '/\b([A-Z][A-Z]+)/',
211 fn($arg) => strtolower($arg[1]),
212 $word);
213 }
214 }
215 foreach ($words as &$word) {
216 $word = Util::camelCaseToDashes($word, ' ');
217 $word = Util::dashesToCamelCase($word, $capitalizeFirst, ' ');
218 }
219 $slug = implode('', $words);
220 return $slug;
221 }

230 public function floatValue(string $value)
231 {
232 $amount = preg_replace('/\s+/u', '', $value);
233 $locales = [ $this->getLocale(), 'en_US_POSIX' ];
234 $parsed = false;
235 foreach ($locales as $locale) {
236 $fmt = new NumberFormatter($locale, \NumberFormatter::DECIMAL);
237
238 $decimalSeparator = $fmt->getSymbol(\NumberFormatter::DECIMAL_SEPARATOR_SYMBOL);
239 $groupingSeparator = $fmt->getSymbol(\NumberFormatter::GROUPING_SEPARATOR_SYMBOL);
240
241 $decPos = strpos($amount, $decimalSeparator);
242 $grpPos = strpos($amount, $groupingSeparator);
243
244 if ($grpPos !== false && $decPos === false) {
245 // unlikely: 1,000, we assume 1,000.00 would be used
246 continue;
247 } elseif ($decPos < $grpPos) {
248 // unlikely: 1.000,00 in en_US
249 continue;
250 }
251
252 $parsed = $fmt->parse($amount);
253 if ($parsed !== false) {
254 $percent = $fmt->getSymbol(\NumberFormatter::PERCENT_SYMBOL);
255 if (preg_match('/'.$percent.'/u', $amount)) {
256 $parsed /= 100.0;
257 }
258 break;
259 }
260 }
261 return $parsed !== false ? (float)$parsed : $parsed;
262 }

273 public function storageValue(string $value, ?string $locale = null)
274 {
275 $factor = [
276 'b' => 1,
277 'kb' => 1000, 'kib' => (1 << 10),
278 'mb' => 1000000, 'mib' => (1 << 20),
279 'gb' => 1000000000, 'gib' => (1 << 30),
280 'tb' => 1000000000000, 'tib' => (1 << 40),
281 'pb' => 1000000000000000, 'pib' => (1 << 50),
282 ];
283 $value = preg_replace('/\s+/u', '', $value);
284 $value = strtolower(
285 str_ireplace(
286 [ 'bytes', 'kilo', 'kibi', 'mega', 'mebi', 'giga', 'gibi', 'tera', 'tibi', 'peta', 'pebi' ],
287 [ 'b', 'k', 'ki', 'm', 'mi', 'g', 'gi', 't', 'ti', 'p', 'pi' ],
288 $value));
289
290 if (preg_match('/([-0-9,.]+)([kmgtp]?i?b?)?$/', $value, $matches)) {
291 $this->logInfo('MATCHES ' . print_r($matches, true));
292 $value = $this->floatValue($matches[1]);
293 if (empty($value)) {
294 return null;
295 }
296 if (!empty($matches[2])) {
297 if (empty($factor[$matches[2]])) {
298 return null;
299 }
300 $value *= $factor[$matches[2]];
301 }
302 return $value;
303 } else {
304 return null;
305 }

318 {
319 $locale = $locale ?? $this->getLocale();
320 $value = Util::normalizeSpaces($value);
321 $fmt = new NumberFormatter($locale, \NumberFormatter::SPELLOUT);
322 $value = implode(
323 ' ',

325 function($value) use ($fmt) {
326 $number = $fmt->parse(strtolower($value));

328 },
329 explode(' ', $value)
330 ));
331 $language = $this->getLanguage($locale);
332 DateInterval::setLocale($language);
333 $value = DateInterval::parseFromLocale($value, $language);
