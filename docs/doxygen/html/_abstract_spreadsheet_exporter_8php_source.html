124 $format = new ExportFormat(strtolower($format));
125
126 $creator = $this->getConfigValue('emailfromname', 'Bilbo Baggins');
127 $email = $this->getConfigValue('emailfromaddress', 'bilbo@nowhere.com');
128
129 $locale = $this->getLocale();
130
131
132 $fontPath = $this->fontService->getFontsFolderName();
133 PhpSpreadsheet\Shared\Font::setTrueTypeFontPath($fontPath);
134 PhpSpreadsheet\Shared\Font::setAutoSizeMethod(PhpSpreadsheet\Shared\Font::AUTOSIZE_METHOD_EXACT);
135
136 $spreadSheet = new PhpSpreadsheet\Spreadsheet();
137 $defaultFont = $this->fontService->getDefaultFontName();
138 $spreadSheet->getDefaultStyle()->getFont()->setName($defaultFont);
139 $spreadSheet->getDefaultStyle()->getFont()->setSize(12);
140
141 $validLocale = PhpSpreadsheet\Settings::setLocale($locale);
142 if (!$validLocale) {
143 $this->logError('Unable to set locale to "'.$locale.'"');
144 }
145
147 $valueBinder = $this->di(PhpSpreadsheetValueBinder::class);
148 PhpSpreadsheet\Cell\Cell::setValueBinder($valueBinder);
149
150 /*
151 *
152 **************************************************************************
153 *
154 * Let the hard work be done by something else ...
155 *
156 */
157
158 $meta = $this->fillSheet($spreadSheet, [
159 'creator' => $creator,
160 'email' => $email,
161 'date' => new DateTimeImmutable,
162 ]);
163
164 /*
165 *
166 **************************************************************************
167 *
168 * Adjust the column height and width computations, PhpSpreadsheet is not
169 * godd in doing that ...
170 *
171 */
172
173 for ($sheetIdx = 0; $sheetIdx < $spreadSheet->getSheetCount(); $sheetIdx++) {
174 $spreadSheet->setActiveSheetIndex($sheetIdx);
175 $sheet = $spreadSheet->getActiveSheet();
176
177 $wrapTextValues = [];
178 $wrapTextRows = [];
179 $highestColumn = $sheet->getHighestColumn();
180 $highestRow = $sheet->getHighestRow();
181
182 // set wrap-text values to empty string in order not to spoil the width
183 // computations.
184 for ($column = 'A'; $column <= $highestColumn; $column++) {
185 $sheet->getColumnDimension($column)->setAutoSize(true);
186 for ($row = 1; $row <= $highestRow; $row++) {
187 $cellAddress = $column . $row;
188 $cell = $sheet->getCell($cellAddress);
189 if ($cell->getStyle()->getAlignment()->getWrapText()) {
190 $wrapTextValues[$cellAddress] = $cell->getValue();
191 $wrapTextRows[] = $row;
192 $cell->setValue('');
193 }
194 }
195 }
196
197 $sheet->calculateColumnWidths();
198
199 // restore cell-values
200 foreach ($wrapTextValues as $cellAddress => $cellValue) {
201 $sheet->getCell($cellAddress)->setValue($cellValue);
202 }
203
204 // disable auto-size and fetch column-width
205 $columnWidth = [];
206 for ($column = 'A'; $column <= $highestColumn; $column++) {
207 $columnDimensions = $sheet->getColumnDimension($column);
208 $columnDimensions->setAutoSize(false);
209 $width = $columnDimensions->getWidth('pt');
210 $width = max($width + 2.0 * self::WIDTH_EXTRA_SPACE, self::MIN_WIDTH);
211 $columnDimensions->setWidth($width, 'pt');
212 $columnWidth[$column] = $columnDimensions->getWidth(); // Excel units
213 }
214
215 // compute the height of the wrap-text rows
216 foreach ($wrapTextRows as $row) {
217 $font = $sheet->getStyle('A' . $row . ':' . $highestColumn . $row)->getFont();
218 $optimalHeight = 0;
219 for ($column = 'A'; $column <= $highestColumn; $column++) {
220 $cell = $sheet->getCell($column . $row);
221 $cellXf = $spreadSheet->getCellXfByIndex($cell->getXfIndex());
222 $font = $cellXf->getFont();
223 $singleLineHeaderWidth = PhpSpreadsheet\Shared\Font::calculateColumnWidth(
224 $font,
225 $cell->getValue(),
226 $cellXf->getAlignment()->getTextRotation(),
227 $font,
228 filterAdjustment: false,
229 indentAdjustment: 0,
230 );
231 $numberOfLines = (int)ceil($singleLineHeaderWidth / $columnWidth[$column]);
232 $fontHeight = PhpSpreadsheet\Shared\Font::getDefaultRowHeightByFont($font);
233 $optimalHeight = max($optimalHeight, $numberOfLines * $fontHeight + $fontHeight / 4);
234 }
235 $sheet->getRowDimension($row)->setRowHeight($optimalHeight);
236 }
237 } // loop over sheets
238
239 /*

241 **************************************************************************
242 *
243 * Set spread-sheet properties after callback to content-filler
244 *
245 */
246
247 $spreadSheet->getProperties()->setCreator($meta['creator'])
248 ->setLastModifiedBy($meta['creator'])
249 ->setTitle('CAFEV-'.$meta['name'])
250 ->setSubject('CAFEV-'.$meta['name'])
251 ->setDescription('Exported Database-Table')
252 ->setKeywords('office 2007 openxml php '.$meta['name'])
253 ->setCategory('Database Table Export');
254

256 *
257 **************************************************************************
258 *
259 * Try to avoid truncated pages
260 *
261 */
262

264 $spreadSheet->setActiveSheetIndex($sheetIdx);
265 $sheet = $spreadSheet->getActiveSheet();
266
267 $pageSetup = $sheet->getPageSetup();
268 $pageSetup->setFitToPage(true);
269 $pageSetup->setOrientation(PhpSpreadsheet\Worksheet\PageSetup::ORIENTATION_LANDSCAPE);
270 $pageSetup->setPaperSize(PhpSpreadsheet\Worksheet\PageSetup::PAPERSIZE_A3);
271 }
