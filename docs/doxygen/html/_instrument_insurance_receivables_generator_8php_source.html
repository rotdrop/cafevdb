99 public function generateReceivables():Collection
100 {
101 $receivableOptions = $this->serviceFeeField->getDataOptions();
102
104 $managementOption = $this->serviceFeeField->getManagementOption();
105 if (empty($managementOption)) {
106 throw new RuntimeException(
107 $this->l->t(
108 'Unable to find management option for participant field "%s".',
109 $this->serviceFeeField->getName()
110 ));
111 }
112 $startingDate = $this->insurancesRepository->startOfInsurances();
113 $managementDate = Util::convertToDateTime($managementOption->getLimit());
114
115 if (!empty($managementDate) && !empty($startingDate)) {
116 if ($managementDate->getTimestamp() < $startingDate->getTimestamp()) {
117 $startingDate = $managementDate;
118 }
119 } elseif (!empty($managementDate)) {
120 $startingDate = $managementDate;
121 } elseif (empty($startingDate)) {
122 $startingDate = new DateTime;
123 }
124 $startingDate = $startingDate->setTimezone($this->timeZone);
125 $managementOption->setLimit($startingDate->getTimestamp());
126
127 $startingYear = $startingDate->format('Y');
128 $endingYear = (new DateTime)->setTimezone($this->timeZone)->format('Y');
129
130 // We (mis-)use year 0 for the initial value, if any
131 $years = array_map(
132 function($value) {
133 return sprintf('%04d', $value);
134 },
135 array_merge([0], range($startingYear, $endingYear)));
136
137 foreach ($years as $year) {
138 if ($year == '0000') {
139 $labelText = $this->l->t($labelTemplate = 'Opening Balance');
140 $tooltipTemplate = $this->toolTipsService['instrument-insurance:opening-balance']??'';
141 $tooltipText = $this->l->t($tooltipTemplate);
142 } else {
143 $labelText = $this->l->t($labelTemplate = 'Insurance Fee %d', $year);
144 $tooltipTemplate = $this->toolTipsService['instrument-insurance:annual-service-fee']??'';
145 $tooltipText = $this->l->t($tooltipTemplate);
146 }
147 $yearReceivables = $receivableOptions->matching(self::criteriaWhere(['data' => (string)$year]));
148 if ($yearReceivables->isEmpty()) {
149 // add a new option
150 $receivable = (new Entities\ProjectParticipantFieldDataOption)
151 ->setField($this->serviceFeeField)
152 ->setKey(Uuid::create())
153 ->setLabel($labelText)
154 ->setToolTip($tooltipText)
155 ->setData($year) // may change in the future
156 ->setLimit(null); // may change in the future
157 $receivableOptions->set($receivable->getKey()->getBytes(), $receivable);
158 } else {
159 // update display things, but keep the essential data untouched
161 $receivable = $yearReceivables->first();
162 $receivable->setLabel($labelText)
163 ->setTooltip($tooltipText);
164 }
165 $this->translate($receivable, 'label', null, sprintf($labelTemplate, $year))
166 ->translate($receivable, 'tooltip', null, $tooltipTemplate);
167 }
168 return $this->serviceFeeField->getSelectableOptions();
169 }

172 protected function updateOne(
173 Entities\ProjectParticipantFieldDataOption $receivable,
174 Entities\ProjectParticipant $participant,
175 string $updateStrategy = self::UPDATE_STRATEGY_EXCEPTION,
176 ):array {
177 // cook-book:
178 // * find list of insurance years
179 // * walk years from start until now
180 // - add missing items if insurance fee != 0
181 // - remove items without payment when insurance fee == 0
182 // - update all existing items with newly computed insurance sum
183
184 $year = $receivable->getData();
185
186 $openingBalance = $year === '0000';
187
188 $removed = false;
189 $added = false;
190 $changed = false;
191 $skipped = false;
192 $notices = [];
193
195 $musician = $participant->getMusician();
197 $project = $participant->getProject();
199 $fileSystemStorage = $this->storageFactory->getProjectParticipantsStorage($participant);
200
201 if (!$openingBalance) {
202 // "now" should in principle just do ...
203 $referenceDate = new DateTime($year.'-06-01');
204
205 // Compute the actual fee
206 $dueInterval = null;
207 $fee = $this->insuranceService->insuranceFee($musician, $referenceDate, $dueInterval);
208
209 // Generate the overview letter as supporting document
210 // @todo: use new OpenDocument stuff
211 $overview = $this->insuranceService->musicianOverview($musician, $referenceDate);
212 $overviewFilename = $this->insuranceService->musicianOverviewFileName($overview);
213 $overviewLetter = $this->insuranceService->musicianOverviewLetter($overview);
214 } else {
215 if (0 == count($this->insuranceService->billableInsurances($musician))) {
216 // bail out early, DO NOT ADD an opening balance
217 return [
218 'added' => 0,
219 'removed' => 0,
220 'changed' => 0,
221 'skipped' => 1,
222 'notices' => [], // no insurance, no message
223 ];
224 }
225 $fee = null;
226 $updateStrategy = self::UPDATE_STRATEGY_SKIP; // set only manually
227 }
228
229 $participantFieldsData = $participant->getParticipantFieldsData();
230 $optionKey = $receivable->getKey();
231 $datum = $participant->getParticipantFieldsDatum($optionKey);
232 if (empty($datum)) {
233 if ($openingBalance || $fee != 0.0) {
234 // add a new option
236 $datum = (new Entities\ProjectParticipantFieldDatum)
237 ->setDataOption($receivable)
238 ->setProjectParticipant($participant)
239 ->setOptionValue($fee);
240
241 if (!$openingBalance) {
242 // store overview letter
243 $supportingDocumentFile = new Entities\EncryptedFile(
244 $overviewFilename, $overviewLetter, 'application/pdf', $musician);
245 $supportingDocument = $fileSystemStorage->addFieldDatumDocument($datum, $supportingDocumentFile, flush: false);
246 $datum->setSupportingDocument($supportingDocument);
247 }
248
249 // @todo Too much connectivity
250 $participantFieldsData->set($optionKey->getBytes(), $datum);
251 $musician->getProjectParticipantFieldsData()->set($optionKey->getBytes(), $datum);
252 $receivable->getFieldData()->set($musician->getId(), $datum);
253 $project->getParticipantFieldsData()->add($datum);
254 $added = true;
255 }
256 } else { // !empty($datum)
257 $optionValue = (float)$datum->getOptionValue();
258 if (!$datum->isDeleted() && $fee != $optionValue) {
259 if ($openingBalance) {
260 $notices[] = $this->l->t('Keeping opening balance of %s.', $this->moneyValue($optionValue));
261 } else {
262 $notices[] = $this->l->t('Data inconsistency for musician %s in year %d: old fee %s, new fee %s.', [
263 $musician->getPublicName(true),
264 $year,
265 $this->moneyValue($optionValue),
266 $this->moneyValue($fee),
267 ]);
268 }
269 switch ($updateStrategy) {
270 case self::UPDATE_STRATEGY_REPLACE:
271 break;
272 case self::UPDATE_STRATEGY_EXCEPTION:
273 throw new Exceptions\EnduserNotificationException(end($notices));
274 break;
275 case self::UPDATE_STRATEGY_SKIP:
276 $skipped = true;
277 break;
278 default:
279 throw new RuntimeException($this->l->t('Unknonw update strategy: "%s".', $updateStrategy));
280 }
281 }
282 if (!$skipped) {
283 if ($fee == 0.0) {
284 // remove current option
285 $this->remove($datum);
286 $this->remove($datum);
287 $participantFieldsData->removeElement($datum);
288 $musician->getProjectParticipantFieldsData()->removeElement($datum);
289 $receivable->getFieldData()->removeElement($datum);
290 $project->getParticipantFieldsData()->removeElement($datum);
291 $removed = true;
292 } else {
294 $supportingDocument = $datum->getSupportingDocument();
295 if (empty($supportingDocument)) {
296 // create overview letter
297 $supportingDocumentFile = new Entities\EncryptedFile(
298 fileName: $overviewFilename,
299 data: $overviewLetter,
300 mimeType: 'application/pdf',
301 owner: $musician
302 );
303 $supportingDocument = $fileSystemStorage->addFieldDatumDocument($datum, $supportingDocumentFile, flush: false);
304 $datum->setSupportingDocument($supportingDocument);
305 } elseif (true || $fee != $datum->getOptionValue()) {
306 // @todo only update letter if fee changes?
307 $supportingDocument
308 ->setName($overviewFilename)
309 ->getFile()
310 ->setFileName($overviewFilename)
311 ->setMimeType('application/pdf')
312 ->setSize(strlen($overviewLetter))
313 ->getFileData()->setData($overviewLetter);
314 }
315 // just update current data to the computed value
316 if ($datum->isDeleted()) {
317 $datum->setDeleted(null);
318 $datum->setOptionValue($fee);
319 $added = true;
320 } elseif ($fee != $datum->getOptionValue()) {
321 $datum->setOptionValue($fee);
322 $changed = true;
323 }
324 }
325 }
326 }
327 return [
328 'added' => (int)$added,
329 'removed' => (int)$removed,
330 'changed' => (int)$changed,
331 'skipped' => (int)$skipped,
332 'notices' => $notices,
333 ];
334 }

337 public function dueDate(?Entities\ProjectParticipantFieldDataOption $receivable = null):?\DateTimeInterface
338 {
339 $timeZone = $this->getDateTimeZone();
340 if ($receivable === null) {
341 $year = (int)(new DateTime)->setTimezone($timeZone)->format('Y');
342 } else {
343 $year = (int)$receivable->getData();
344 if ($year == 0) {
345 return null;
346 }
347 }
348 $dueDate = $this->serviceFeeField->getDueDate()->setTimezone($timeZone);
349 $dueYear = (int)$dueDate->format('Y');
350 return $dueDate->modify('+'.($year - $dueYear).' years');
351 }
