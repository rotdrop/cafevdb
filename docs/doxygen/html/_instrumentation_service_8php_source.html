212 public function tableTabs(?array $participantFields = null, bool $useFinanceTab = false)
213 {
214 $dfltTabs = $this->defaultTableTabs($useFinanceTab);
215
216 if (!is_array($participantFields)) {
217 return $dfltTabs;
218 }
219
220 $extraTabs = array();

222 if (empty($field['Tab'])) {

224 }
225
226 $extraTab = $field['Tab'];
227 foreach ($dfltTabs as $tab) {
228 if ($extraTab === $tab['id'] ||
229 $extraTab === (string)$tab['name']) {
230 $extraTab = false;
231 break;
232 }
233 }
234 if ($extraTab !== false) {
235 $extraTabs[] = [
236 'id' => $extraTab,
237 'name' => $this->l->t($extraTab),
238 'tooltip' => $this->toolTipsService['participant-fields-extra-tab'],

240 }
241 }
