222 {
223 $dfltTabs = $this->defaultTableTabs($useFinanceTab);
224
225 if (!is_array($participantFields)) {
226 return $dfltTabs;
227 }
228
229 $extraTabs = array();
230 foreach ($participantFields as $field) {
231 if (empty($field['Tab'])) {
232 continue;
233 }
234
235 $extraTab = $field['Tab'];
236 foreach ($dfltTabs as $tab) {
237 if ($extraTab === $tab['id'] ||
238 $extraTab === (string)$tab['name']) {
239 $extraTab = false;
240 break;
241 }
242 }
243 if ($extraTab !== false) {
244 $extraTabs[] = [
245 'id' => $extraTab,
246 'name' => $this->l->t($extraTab),
247 'tooltip' => $this->toolTipsService['participant-fields-extra-tab'],
248 ];
249 }
250 }
