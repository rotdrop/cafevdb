86 public function dedicatedBoardMemberContact(string $role):?array
87 {
88 $participant = $this->dedicatedBoardMemberParticipant($role);
89 if (empty($participant)) {
90 return null;
91 }
92 $musician = $participant->getMusician();
93 $data = [
94 'email' => $musician->getEmail(),
95 'name' => $musician->getPublicName(true),
96 'street' => $musician->getStreet(),
97 'streetNumber' => $musician->getStreetNumber(),
98 'streetAndNumber' => $musician->getStreet() . ' ' . $musician->getStreetNumber(),
99 'postalCode' => $musician->getPostalCode(),
100 'city' => $musician->getCity(),
101 'phone' => $musician->getFixedLinePhone(),
102 'mobile' => $musician->getMobilePhone(),
103 ];
104
105 // Override with configured functional address
106 $roleEmail = $this->getConfigValue($role . 'Email', null);
107 if (!empty($roleEmail)) {
108 $data['email'] = $roleEmail;
109 }
110
111 $missingFields = array_keys(array_filter($data, fn($value) => empty($value)));
112
113 if (!empty($missingFields)) {
114 $this->logInfo('Missing fields for ' . $role . ' ' . print_r($missingFields, true) . ' ' . print_r($data, true));
115 }
116
117 // if some field are missing try to fill in from the cloud data
118 if (!empty($missingFields)) {
119
120 $roleUid = $this->getConfigValue($role.'UserId', null);
121 if (!empty($roleUid)) {
122 $user = $this->user($roleUid);
124 $accountManager = $this->di(IAccountManager::class);
126 $account = $accountManager->getAccount($user);
127
128 foreach ($missingFields as $key) {
129 switch ($key) {
130 case 'email':
131 $item = $user->getEMailAddress();
132 break;
133 case 'name':
134 $item = $user->getDisplayName();
135 break;
136 case 'streetAndNumber':
137 $item = $account->getProperty(IAccountManager::PROPERTY_ADDRESS)->getValue();
138 break;
139 case 'phone':
140 $item = $account->getProperty(IAccountManager::PROPERTY_PHONE)->getValue();
141 break;
142 case 'postalCode':
143 case 'city':
144 case 'mobile':
145 $item = null;
146 break;
147 }
148 if (!empty($item)) {
149 $data[$key] = $item;
150 }
151 }
152 }
153 }
154
155 return $data;
156 }

276 public function isDedicatedBoardMember(string $role, ?string $uid = null, bool $allowGroupAccess = false):bool
277 {
278 empty($uid) && $uid = $this->userId();
279 $musicianId = $this->getConfigValue($role.'Id', null);
280 if (empty($musicianId)) {
281 return false;
282 }
283 $roleUid = $this->getConfigValue($role.'UserId', null);
284 if ($this->inGroup($roleUid) && $roleUid === $uid) {
285 return true;
286 }
287 if (!$allowGroupAccess) {
288 return false;
289 }
290 // check for group-membership
291 $groupId = $this->getConfigValue($role.'GroupId', null);
292 return !empty($groupId) && $this->inGroup($uid, $groupId);
293 }
