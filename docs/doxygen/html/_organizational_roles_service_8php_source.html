94 public function dedicatedBoardMemberContact(string $role, int $musicianId = 0):?array
95 {
96 $participant = $this->dedicatedBoardMemberParticipant($role, $musicianId);
97 if (empty($participant)) {
98 return null;
99 }
100 $musician = $participant->getMusician();
101 $data = [
102 'email' => $musician->getEmail(),
103 'name' => $musician->getPublicName(true),
104 'street' => $musician->getStreet(),
105 'streetNumber' => $musician->getStreetNumber(),
106 'streetAndNumber' => $musician->getStreet() . ' ' . $musician->getStreetNumber(),
107 'postalCode' => $musician->getPostalCode(),
108 'city' => $musician->getCity(),
109 'phone' => $musician->getFixedLinePhone(),
110 'mobile' => $musician->getMobilePhone(),
111 ];
112
113 // Override with configured functional address
114 $roleEmail = $this->getConfigValue($role . 'Email', null);
115 if (!empty($roleEmail)) {
116 $data['email'] = $roleEmail;
117 }
118
119 $missingFields = array_keys(array_filter($data, fn($value) => empty($value)));
120
121 if (!empty($missingFields)) {
122 $this->logInfo('Missing fields for ' . $role . ' ' . print_r($missingFields, true) . ' ' . print_r($data, true));
123 }
124
125 // if some field are missing try to fill in from the cloud data
126 if (!empty($missingFields)) {
127
128 $roleUid = $this->getConfigValue($role.'UserId', null);
129 if (!empty($roleUid)) {
130 $user = $this->user($roleUid);
132 $accountManager = $this->di(IAccountManager::class);
134 $account = $accountManager->getAccount($user);
135
136 foreach ($missingFields as $key) {
137 switch ($key) {
138 case 'email':
139 $item = $user->getEMailAddress();
140 break;
141 case 'name':
142 $item = $user->getDisplayName();
143 break;
144 case 'streetAndNumber':
145 $item = $account->getProperty(IAccountManager::PROPERTY_ADDRESS)->getValue();
146 break;
147 case 'phone':
148 $item = $account->getProperty(IAccountManager::PROPERTY_PHONE)->getValue();
149 break;
150 case 'postalCode':
151 case 'city':
152 case 'mobile':
153 $item = null;
154 break;
155 }
156 if (!empty($item)) {
157 $data[$key] = $item;
158 }
159 }
160 }
161 }
162
163 return $data;
164 }
