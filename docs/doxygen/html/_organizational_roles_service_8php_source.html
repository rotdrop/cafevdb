91 public function dedicatedBoardMemberContact(string $role, int $musicianId = 0):?array
92 {
93 $participant = $this->dedicatedBoardMemberParticipant($role, $musicianId);
94 if (empty($participant)) {
95 return null;
96 }
97 $musician = $participant->getMusician();
98 $data = [
99 'email' => $musician->getEmail(),
100 'name' => $musician->getPublicName(true),
101 'street' => $musician->getStreet(),
102 'streetNumber' => $musician->getStreetNumber(),
103 'streetAndNumber' => $musician->getStreet() . ' ' . $musician->getStreetNumber(),
104 'postalCode' => $musician->getPostalCode(),
105 'city' => $musician->getCity(),
106 'phone' => $musician->getFixedLinePhone(),
107 'mobile' => $musician->getMobilePhone(),
108 ];
109
110 // Override with configured functional address
111 $roleEmail = $this->getConfigValue($role . 'Email', null);
112 if (!empty($roleEmail)) {
113 $data['email'] = $roleEmail;
114 }
115
116 $missingFields = array_keys(array_filter($data, fn($value) => empty($value)));
117
118 if (!empty($missingFields)) {
119 $this->logInfo('Missing fields for ' . $role . ' ' . print_r($missingFields, true) . ' ' . print_r($data, true));
120 }
121
122 // if some field are missing try to fill in from the cloud data
123 if (!empty($missingFields)) {
124
125 $roleUid = $this->getConfigValue($role.'UserId', null);
126 if (!empty($roleUid)) {
127 $user = $this->user($roleUid);
129 $accountManager = $this->di(IAccountManager::class);
131 $account = $accountManager->getAccount($user);
132
133 foreach ($missingFields as $key) {
134 switch ($key) {
135 case 'email':
136 $item = $user->getEMailAddress();
137 break;
138 case 'name':
139 $item = $user->getDisplayName();
140 break;
141 case 'streetAndNumber':
142 $item = $account->getProperty(IAccountManager::PROPERTY_ADDRESS)->getValue();
143 break;
144 case 'phone':
145 $item = $account->getProperty(IAccountManager::PROPERTY_PHONE)->getValue();
146 break;
147 case 'postalCode':
148 case 'city':
149 case 'mobile':
150 $item = null;
151 break;
152 }
153 if (!empty($item)) {
154 $data[$key] = $item;
155 }
156 }
157 }
158 }
159
160 return $data;
161 }

295 public function isDedicatedBoardMember(string $role, ?string $uid = null, bool $allowGroupAccess = false):bool
296 {
297 empty($uid) && $uid = $this->userId();
298 $musicianId = $this->getConfigValue($role.'Id', null);
299 if (empty($musicianId)) {
300 return false;
301 }
302 $roleUid = $this->getConfigValue($role.'UserId', null);
303 if ($this->inGroup($roleUid) && $roleUid === $uid) {
304 return true;

306 if (!$allowGroupAccess) {
307 return false;

309 // check for group-membership
