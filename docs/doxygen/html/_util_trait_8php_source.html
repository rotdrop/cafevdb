123 public function floatValue(string $value, ?string $locale = null)
124 {
125 $amount = preg_replace('/\s+/u', '', $value);
126 empty($locale) && $locale = $this->getLocale();
127 $locales = [ $locale, 'en_US_POSIX' ];
128 $parsed = false;
129 foreach ($locales as $locale) {
130 $fmt = new NumberFormatter($locale, \NumberFormatter::DECIMAL);
131
132 $decimalSeparator = $fmt->getSymbol(\NumberFormatter::DECIMAL_SEPARATOR_SYMBOL);
133 $groupingSeparator = $fmt->getSymbol(\NumberFormatter::GROUPING_SEPARATOR_SYMBOL);
134
135 $decPos = strpos($amount, $decimalSeparator);
136 $grpPos = strpos($amount, $groupingSeparator);
137
138 if ($grpPos !== false && $decPos === false) {
139 // unlikely: 1,000, we assume 1,000.00 would be used
140 continue;
141 } elseif ($decPos < $grpPos) {
142 // unlikely: 1.000,00 in en_US
143 continue;
144 }
145
146 $parsed = $fmt->parse($amount);
147 if ($parsed !== false) {
148 $percent = $fmt->getSymbol(\NumberFormatter::PERCENT_SYMBOL);
149 if (preg_match('/'.$percent.'/u', $amount)) {
150 $parsed /= 100.0;
151 }
152 break;
153 }
154 }
155 return $parsed !== false ? (float)$parsed : $parsed;
156 }

172 public function storageValue(?string $value, ?string $locale = null)
173 {
174 if ($value === null || $value === '') {
175 return null;
176 }
177 if ($value === '0') {
178 return (int)0;
179 }
180 $factor = [
181 'b' => 1,
182 'kb' => 1000,
183 'kib' => (1 << 10),
184 'mb' => (1000 * 1000),
185 'mib' => (1 << 20),
186 'gb' => (1000 * 1000 * 1000),
187 'gib' => (1 << 30),
188 'tb' => (1000 * 1000 * 1000 * 1000),
189 'tib' => (1 << 40),
190 'pb' => (1000 * 1000 * 1000 * 1000 * 1000),
191 'pib' => (1 << 50),
192 'eb' => (1000 * 1000 * 1000 * 1000 * 1000 * 1000),
193 'eib' => (1 << 60),
194 'zb' => (1000 * 1000 * 1000 * 1000 * 1000 * 1000 * 1000),
195 'zib' => (1 << 70),
196 'yb' => (1000 * 1000 * 1000 * 1000 * 1000 * 1000 * 1000 * 1000),
197 'yib' => (1 << 80),
198 ];
199 $value = preg_replace('/\s+/u', '', $value);
200 $value = strtolower(
201 str_ireplace(
202 [ 'bytes', 'kilo', 'kibi', 'mega', 'mebi', 'giga', 'gibi', 'tera', 'tibi', 'peta', 'pebi', 'exa', 'exbi', 'zetta', 'zebi', 'yotta', 'yobi', ],
203 [ 'b', 'k', 'ki', 'm', 'mi', 'g', 'gi', 't', 'ti', 'p', 'pi', 'e', 'ei', 'z', 'zi', 'y', 'yi', ],
204 $value));
205
206 if (preg_match('/([-0-9,.]+)([kmgtpezy]?i?b?)?$/', $value, $matches)) {
207 $this->logInfo('MATCHES ' . print_r($matches, true));
208 $value = $this->floatValue($matches[1], $locale);
209 if (empty($value)) {
210 return null;
211 }
212 if (!empty($matches[2])) {
213 if (empty($factor[$matches[2]])) {
214 return null;
215 }
216 $value *= $factor[$matches[2]];
217 if ($value <= PHP_INT_MAX) {
218 $value = (int)$value;
219 }
220 }
221 return $value;
222 } else {

240 protected function formatStorageValue(int $bytes, string $format = 'binary', int $digits = 2, ?string $locale = null):?string
241 {
242 $units = [
243 'decimal' => [
244 '', 'k', 'M', 'G', 'T', 'P', 'E', 'Z', 'Y',
245 ],
246 'longDecimal' => [
247 'bytes', 'kilo', 'mega', 'giga', 'tera', 'peta', 'exa', 'zetta', 'yotta',
248 ],
249 'binary' => [
250 '', 'Ki', 'Mi', 'Gi', 'Ti', 'Pi', 'Ei', 'Zi', 'Yi',
251 ],
252 'longBinary' => [
253 '', 'kibi', 'mebi', 'gibi', 'tebi', 'pebi', 'exbi', 'zebi', 'yobi'
254 ],
255 ];
256 $bytesUnit = [
257 'decimal' => 'B',
258 'longDecimal' => ' ' . $this->l->t('bytes'),
259 'binary' => 'B',
260 'longBinary' => ' ' . $this->l->t('bytes'),
261 ];
262 $radix = [
263 'decimal' => 1000.0,
264 'longDecimal' => 1000.0,
265 'binary' => 1024.0,
266 'longBinary' => 1024.0,
267 ];
268
269 $units = $units[$format] ?? null;
270 $bytesUnit = $bytesUnit[$format] ?? null;
271 $radix = $radix[$format] ?? null;
272
273 if ($units === null || $bytesUnit === null || $radix === null) {
274 // maybe throw InvalidArgumentException
275 return null;
276 }
277
278 empty($locale) && $locale = $this->getLocale();
279
280 $floatVal = $bytes;
281 $exponent = 0;
282 $exponentLimit = count($units) - 1;
283 while ($floatVal > $radix && $exponent < $exponentLimit) {
284 $floatVal /= $radix;
285 ++$exponent;
286 }
287
288 $fmt = new NumberFormatter($locale, \NumberFormatter::DECIMAL);
289 $fmt->setAttribute(NumberFormatter::MIN_FRACTION_DIGITS, 0);
290 $fmt->setAttribute(NumberFormatter::MAX_FRACTION_DIGITS, $digits);
291 $stringVal = $fmt->format($floatVal);
292
293 return $stringVal . ' ' . $units[$exponent] . $bytesUnit;
294 }

326 protected function getCommonPath(array $paths, bool $leadingSlash = true):string
327 {
328 // $this->logInfo('PATHS ' . print_r($paths, true));
329 $lastOffset = (int)$leadingSlash;
330 $common = $leadingSlash ? '/' : '';
331 while (($index = strpos($paths[0], '/', $lastOffset)) !== false) {
332 $dirLen = $index - $lastOffset + 1; // include /
333 $dir = substr($paths[0], $lastOffset, $dirLen);
334 foreach ($paths as $path) {
335 if (substr($path, $lastOffset, $dirLen) != $dir) {
336 return $leadingSlash ? substr($common, 0, -1) : $common;

339 $common .= $dir;
340 $lastOffset = $index + 1;
341 }
342 return $leadingSlash ? substr($common, 0, -1) : $common;
