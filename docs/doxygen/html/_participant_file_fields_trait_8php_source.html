300 private function getFilesAppLink(
301 ?Entities\ProjectParticipantField $field,
302 Entities\Musician $musician,
303 ?Entities\Project $project = null,
304 ?string $subFolder = null,
305 ):?array {
306 $pathChain = [];
307 $project = $project ?? $this->project;
308 if (!empty($field)) {
309 $project = $project ?? $field->getProject();
310 $dataType = $field->getDataType();
311 switch ($dataType) {
312 case FieldType::RECEIVABLES:
313 case FieldType::LIABILITIES:
314 case FieldType::DB_FILE:
315 $pathChain[] = $this->getDocumentsFolderName();
316 if ($dataType != FieldType::DB_FILE) {
317 $pathChain[] = $this->getSupportingDocumentsFolderName();
318 }
319 if ($field->getMultiplicity() != FieldMultiplicity::SIMPLE) {
320 $pathChain[] = $this->participantFieldsService->getFileSystemFieldName($field);
321 }
322 break;
323 case FieldType::CLOUD_FILE:
324 if ($field->getMultiplicity() != FieldMultiplicity::SIMPLE) {
325 $pathChain[] = $this->participantFieldsService->getFileSystemFieldName($field);
326 }
327 break;
328 case FieldType::CLOUD_FOLDER:
329 $pathChain[] = $this->participantFieldsService->getFileSystemFieldName($field);
330 break;
331 default:
332 return null;
333 }
334 }
335 if (!empty($subFolder)) {
336 $pathChain[] = $subFolder;
337 }
338 $participantFolder = $this->projectService->ensureParticipantFolder($project, $musician, dry: true);
339 array_unshift($pathChain, $participantFolder);
340
341 $filesAppLink = null;
342 $filesAppTarget = md5($this->userStorage->getFilesAppLink($participantFolder));
343 while (!empty($pathChain)) {
344 $folderPath = implode(UserStorage::PATH_SEP, $pathChain);
345 try {
346 $filesAppLink = $this->userStorage->getFilesAppLink($folderPath, true);
347 break;
348 } catch (/*\OCP\Files\NotFoundException*/ \Throwable $e) {
349 $this->logException($e, level: \OCP\ILogger::DEBUG);
350 array_pop($pathChain);
351 }
352 }
353 return [ $filesAppLink, $filesAppTarget ];
354 }
