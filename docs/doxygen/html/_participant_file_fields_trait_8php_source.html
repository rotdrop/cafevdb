290 private function getFilesAppLink(
291 ?Entities\ProjectParticipantField $field,
292 Entities\Musician $musician,
293 ?Entities\Project $project = null,
294 ?string $subFolder = null,
295 ):?array {
296 $pathChain = [];
297 $project = $project ?? $this->project;
298 if (!empty($field)) {
299 $project = $project ?? $field->getProject();
300 $dataType = $field->getDataType();
301 switch ($dataType) {
302 case FieldType::RECEIVABLES:
303 case FieldType::LIABILITIES:
304 case FieldType::DB_FILE:
305 $pathChain[] = $this->getDocumentsFolderName();
306 if ($dataType != FieldType::DB_FILE) {
307 $pathChain[] = $this->getSupportingDocumentsFolderName();
308 }
309 if ($field->getMultiplicity() != FieldMultiplicity::SIMPLE) {
310 $pathChain[] = $this->participantFieldsService->getFileSystemFieldName($field);
311 }
312 break;
313 case FieldType::CLOUD_FILE:
314 if ($field->getMultiplicity() != FieldMultiplicity::SIMPLE) {
315 $pathChain[] = $this->participantFieldsService->getFileSystemFieldName($field);
316 }
317 break;
318 case FieldType::CLOUD_FOLDER:
319 $pathChain[] = $this->participantFieldsService->getFileSystemFieldName($field);
320 break;
321 default:
322 return null;
323 }
324 }
325 if (!empty($subFolder)) {
326 $pathChain[] = $subFolder;
327 }
328 $participantFolder = $this->projectService->ensureParticipantFolder($project, $musician, dry: true);
329 array_unshift($pathChain, $participantFolder);
330
331 $filesAppLink = null;
332 $filesAppTarget = md5($this->userStorage->getFilesAppLink($participantFolder));
333 while (!empty($pathChain)) {
334 $folderPath = implode(UserStorage::PATH_SEP, $pathChain);
335 try {
336 $filesAppLink = $this->userStorage->getFilesAppLink($folderPath, true);
337 break;
338 } catch (/*\OCP\Files\NotFoundException*/ \Throwable $e) {
339 $this->logException($e, level: \OCP\ILogger::DEBUG);
340 array_pop($pathChain);
341 }
342 }
343 return [ $filesAppLink, $filesAppTarget ];
344 }
