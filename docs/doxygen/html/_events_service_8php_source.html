317 private function makeEvent(Entities\ProjectEvent $projectEvent):?array
318 {
319 $event = [];
320 $event['projectid'] = $projectEvent->getProject()->getId();
321 $event['uri'] = $projectEvent->getEventUri();
322 $event['uid'] = $projectEvent->getEventUid();
323 $event['calendarid'] = $projectEvent->getCalendarId();
324 $calendarObject = $this->calDavService->getCalendarObject($event['calendarid'], $event['uri']);
325 if (empty($calendarObject)) {
326 $this->logDebug('Orphan project event found: ' . print_r($event, true) . (new Exception())->getTraceAsString());
327 if (false) {
328 // clean up orphaned events
329 try {
330 $this->unregister($event['projectid'], $event['uri']);
331 } catch (\Throwable $t) {
332 $this->logException($t);
333 }
334 }
335 return null;
336 }
337 $vCalendar = VCalendarService::getVCalendar($calendarObject);
338 $vObject = VCalendarService::getVObject($vCalendar);
339 $dtStart = $vObject->DTSTART;
340 $dtEnd = VCalendarService::getDTEnd($vCalendar);
341
342 $start = $dtStart->getDateTime();
343 $end = $dtEnd->getDateTime();
344 $allDay = !$dtStart->hasTime();
345
346 $timeZone = $this->getDateTimezone();
347 if (!$allDay) {
348 if ($dtStart->isFloating()) {
349 $start = $start->setTimezone($timeZone);
350 }
351 if ($dtEnd->isFloating()) {
352 $end = $end->setTimezone($timeZone);
353 }
354 } else {
355 // the following is not overly correct, but make a prettier
356 // display:
357 $start = new DateTimeImmutable($start->format('Y-m-d H:i:s'), $timeZone);
358 $end = new DateTimeImmutable($end->format('Y-m-d H:i:s'), $timeZone);
359 }
360
361 $event['start'] = $start;
362 $event['end'] = $end;
363 $event['allday'] = $allDay;
364
365 // description + summary?
366 $event['summary'] = (string)$vObject->SUMMARY;
367 $event['description'] = (string)$vObject->DESCRIPTION;
368 $event['location'] = (string)$vObject->LOCATION;
369
370 return $event;
371 }

445 private function eventTimes(array$eventObject, ?string $timezone = null, ?string $locale = null):array
446 {
447 if ($timezone === null) {
448 $timezone = $this->getTimezone();
449 }
450 if ($locale === null) {
451 $locale = $this->getLocale();
452 }
453
454 $start = $eventObject['start'];
455 $end = $eventObject['end'];
456 $allDay = $eventObject['allday'];
457
458 $startStamp = $start->getTimestamp();
459
460 /* Event end is inclusive the last second of "end" to generate
461 * non-confusing dates and times for whole-day events.
462 */
463 $endStamp = $end->getTimestamp() + ($allDay ? -1 : 0);
464
465 $startdate = Util::strftime("%x", $startStamp, $timezone, $locale);
466 $starttime = Util::strftime("%H:%M", $startStamp, $timezone, $locale);
467 $enddate = Util::strftime("%x", $endStamp, $timezone, $locale);
468 $endtime = Util::strftime("%H:%M", $endStamp, $timezone, $locale);
469
470 return [
471 'timezone' => $timezone,
472 'locale' => $locale,
473 'allday' => $allDay,
474 'start' => [
475 'stamp' => $startStamp,
476 'date' => $startdate,
477 'time' => $starttime,
478 ],
479 'end' => [
480 'stamp' => $endStamp,
481 'date' => $enddate,
482 'time' => $endtime,
483 ],
484 ];
485 }

826 public function syncCalendarObject(array $objectData, bool $unregister = true):array
827 {
828 $eventURI = $objectData['uri'];
829 $calId = $objectData['calendarid'];
830 $calURI = $objectData['calendaruri'];
831 // $eventData = $objectData['calendardata'];
832 $vCalendar = VCalendarService::getVCalendar($objectData);
833 $categories = VCalendarService::getCategories($vCalendar);
834 $eventUID = VCalendarService::getUid($vCalendar);
835
836 $type = VCalendarService::getVObjectType($vCalendar);
837
838 // As a temporary hack enforce all events to be public as there is
839 // currently no means to share calendars with really full-access. This is
840 // a missing delegation feature in NC.
841 if ($type == 'VEVENT') {
842 $vEvent = VCalendarService::getVObject($vCalendar);
843 if (!empty($vEvent->CLASS) && ($vEvent->CLASS == 'CONFIDENTIAL' || $vEvent->CLASS == 'PRIVATE')) {
844 $this->logInfo('FORCE EVENT ' . $eventURI . ' TO BE PUBLIC ' . $vEvent->CLASS);
845
846 // We first have to fetch the original event, as the data supplied by
847 // the change event carries already the disclosed form of the event.
848 $originalEvent = $this->calDavService->getCalendarObject($calId, $eventURI);
849 $originalVCalendar = VCalendarService::getVCalendar($originalEvent);
850
851 $vEvent = VCalendarService::getVObject($originalVCalendar);
852 $vEvent->CLASS = 'PUBLIC';
853
854 $vEvent->{'LAST-MODIFIED'} = new DateTimeImmutable('now', new DateTimeZone('UTC'));
855 $vEvent->DTSTAMP = new DateTimeImmutable('now', new DateTimeZone('UTC'));
856
857 $this->calDavService->updateCalendarObject($calId, $eventURI, $originalVCalendar);
858 return []; // there will be another event which then is used to update the project links.
859 }
860 }
861
862 // Now fetch all projects and their names ...
863 $projects = $this->projectService->fetchAll();
864
865 $registered = [];
866 $unregistered = [];
867 // Do the sync. The categories stored in the event are
868 // the criterion for this.
869 foreach ($projects as $project) {
870 $prKey = $project->getId();
871 if (in_array($project->getName(), $categories)) {
872 // register or update the event
873 if ($this->register($project, $eventURI, $eventUID, $calId, $calURI, $type)) {
874 $registered[] = $prKey;
875 }
876 } else {
877 // unregister the event
878 if ($this->unregister($prKey, $eventURI)) {
879 $unregistered[] = $prKey;
880 }
881 }
882 }
883 return [ 'registered' => $registered, 'unregistered' => $unregistered ];
884 }
