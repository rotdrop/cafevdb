369 {
370 $vObject = VCalendarService::getVObject($vCalendar);
371 $uid = (string)$vObject->UID;
372 $sequence = (int)(string)($vObject->SEQUENCE ?? 0);
373 $siblings = $this->eventSiblings[$calendarId][$uid][$sequence] ?? null;
374 if ($siblings !== null) {
375 return $siblings;
376 }
377 if (!VCalendarService::isEventRecurring($vObject)) {
378 $siblings = [ 0 => $vObject ];
379 } else {
380 $vEvents = VCalendarService::getAllVObjects($vCalendar);
381 try {
382 // there is also the expand() method on the VCalendar ...
383 $siblings = [];

385 while ($eventIterator->valid()) {
386 $sibling = $eventIterator->getEventObject();

388 $siblings[$recurrenceId] = $sibling;
389 $eventIterator->next();
390 }
391 } catch (NoInstancesException $e) {
392 // This event is recurring, but it doesn't have a single
393 // instance. We are skipping this event from the output

395 $siblings = [];
396 } catch (MaxInstancesExceededException $e) {
397 // hopefully happens never, but ... just live with the sequence we
398 // have.
399 }
400 }
401 $this->eventSiblings[$calendarId][$uid][$sequence] = $siblings;
