122 {
123 if (isset($parameters['postalCode']) && !isset($parameters['postalcode'])) {
124 $parameters['postalcode'] = $parameters['postalCode'];
125 unset($parameters['postalCode']);
126 }
127 if (isset($parameters['country']) && is_array($parameters['country'])) {
128 $countries = $parameters['country'];
129 unset($parameters['country']);
130 $query = http_build_query($parameters, '', '&');
131 foreach ($countries as $country) {
132 $query .= '&'.http_build_query(array('country' => $country));
133 }
134 } else {
135 $query = http_build_query($parameters, '', '&');
136 }
137 if ($type === self::JSON || $type === self::DRY) {
138 $url = self::PROVIDER_URL . '/' .$command . 'JSON' . '?username=' . $this->userName . '&' . $query;
139 if ($type === self::DRY) {
140 return $url;
141 } else {
142 $this->debug($url);
143 }
144 $start = hrtime(true);
145 $ctx = stream_context_create([
146 'http'=> [
147 'timeout' => 240,
148 ]
149 ]);
150 $response = file_get_contents($url, false, $ctx);
151 // $this->logInfo('DURATION ' . (float)(hrtime(true) - $start) / 1e9);
152 if ($response !== false) {
153 $json = json_decode($response, true, 512, JSON_BIGINT_AS_STRING);
154 $this->debug(print_r($json, true));
155 return $json;
156 }
157 }

196 ):array {
197 if (empty($country) && empty($city) && empty($postalCode)) {

200
201 $countryArea =
202 $postalCodeArea =
203 $cityArea = '';
204
205 $oql = '[out:csv("name";false)];
206';

208 $oql .= sprintf('area["ISO3166-1"="%s"]->.country;

225(
226 make x name=_.val;
227 out;
228);
229';

624 private function stampPostalCode(string $postalCode, string $country):void
625 {
626 // Remember we tried ... note that due to typos
627 // (e.g. comma w/o following space) we may actually fail
628 // to update some of those beasts. We simply pretend to
629 // update all, e.g. for the timestamp we give a damn on
630 // the name.
631
632 $expr = $this->expr();
633 $qb = $this->queryBuilder()
634 ->update(GeoPostalCode::class, 'gpc')
635 ->set('gpc.updated', "'".(new DateTimeImmutable)->format('Y-m-d H:i:s.u')."'")
636 ->where(
637 $expr->andX(
638 $expr->eq('gpc.country', ':country'),
639 $expr->eq('gpc.postalCode', ':postalCode')
640 ))
641 ->setParameter('country', $country)
642 ->setParameter('postalCode', $postalCode);
643 $this->debug($qb->getDql());
644 $qb->getQuery()
645 ->execute();
646 $this->flush();
647 $this->debug('Stamped postal code ' . $postalCode . '@' . $country);
648 }

987 public function updateCountriesForLanguage(string $lang, bool $force = false):int
988 {
989 if (!$force && $this->count(['target' => $lang], GeoCountry::class) > 0) {
990 $this->debug('language '.$lang.' already retrieved and update not forced, skipping update.');
991 return 0;
992 }
993
994 // obtain localized info from server
995 $countryInfo = $this->request('countryInfo', array('lang' => $lang));
996 if (!isset($countryInfo[self::GEONAMES_TAG]) ||
997 !is_array($countryInfo[self::GEONAMES_TAG])) {
998 return false; // give up
999 }
1000
1001 $localeCountryNames = $this->localeCountryNames($lang);
1002
1003 // Process each entry in turn
1004 $numRows = 0;
1005 foreach ($countryInfo[self::GEONAMES_TAG] as $country) {
1006 $code = $country['countryCode'];
1007 $continent = $country['continent'];
1008 $name = $country['countryName'];
1009 $continentName = $country['continentName'];
1010 // $languages = $country['languages'];
1011
1012 $localeName = $localeCountryNames[$code];
1013 if ($localeName != $name) {
1014 $this->debug($lang.'_'.$code.': '.$localeName.' / '.$name.' (php/remote)');
1015 }
1016
1017 $this->setDatabaseRepository(GeoCountry::class);
1019 $countryEntity = null;
1020 if (!empty($name)) {
1021 $countryEntity = $this->find(['iso' => $code, 'target' => $lang]);
1022 if (empty($countryEntity)) {
1023 $countryEntity = GeoCountry::create()
1024 ->setIso($code)
1025 ->setTarget($lang);
1026 } else {
1027 $numRows += (int)($countryEntity->getL10nName() != $name);
1028 }
1029 $countryEntity->setL10nName($name);
1030 $this->persist($countryEntity);
1031 }
1032
1033 if (!empty($continentName)) {
1034 // Update continent table
1036 $this->setDatabaseRepository(GeoContinent::class);
1037 $continentEntity = $this->find(['code' => $continent, 'target' => $lang]);

1048 }
