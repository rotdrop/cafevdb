930 public function contacts(string $operation):Response
931 {
933 $contactsService = $this->appContainer->query(ContactsService::class);
934 switch ($operation) {
935 case 'list':
936 // Free-form recipients from Cc: or Bcc:
937 $freeForm = $this->parameterService->getParam('freeFormRecipients', '');
938
939 // Convert the free-form input to an array (possibly)
940 $parser = new Mail_RFC822(null, null, null, false);
941 $recipients = $parser->parseAddressList($freeForm);
942 $parseError = $parser->parseError();
943 if ($parseError !== false) {
944 return self::grumble(
945 $this->l->t(
946 'Unable to parse email-recipients "%s".',
947 vsprintf($parseError['message'], $parseError['data'])
948 ));
949 }
950 $freeForm = [];
951 foreach ($recipients as $emailRecord) {
952 $email = $emailRecord->mailbox.'@'.$emailRecord->host;
953 $name = $emailRecord->personal;
954 $freeForm[$email] = $name;
955 }
956
957 // Fetch all known address-book contacts with email
958 $bookContacts = $contactsService->emailContacts();
959
960 $addressBookEmails = [];
961 foreach ($bookContacts as $entry) {
962 $addressBookEmails[$entry['email']] = $entry['name'];
963 }
964
965 // Convert the free-form input in "book-format", but exclude those
966 // contacts already present in the address-book in order not to list
967 // contacts twice.
968 $formContacts = [];
969 foreach ($freeForm as $email => $name) {
970 if (isset($addressBookEmails[$email]) /* && $addressBookEmails[$email] == $name*/) {
971 // skip free-form if already listed in address-book
972 continue;
973 }
974 $formContacts[] = [
975 'email' => $email,
976 'name' => $name,
977 'addressBook' => $this->l->t('Form Input'),
978 'class' => 'free-form'
979 ];
980 }
981
982 // The total options list is the union of the (remaining) free-form
983 // addresses and the address-book entries
984 $emailOptions = array_merge($formContacts, $bookContacts);
985
986 // Now convert it into a form Navigation::selectOptions()
987 // understands
988 $selectOptions = [];
989 foreach ($emailOptions as $entry) {
990 $email = $entry['email'];
991 if ($entry['name'] == '') {
992 $displayName = $email;
993 } else {
994 $displayName = $entry['name'].' <'.$email.'>';
995 }
996
997 $option = [
998 'value' => $email,
999 'name' => $displayName,
1000 'flags' => isset($freeForm[$email]) ? PageNavigation::SELECTED : 0,
1001 'group' => $entry['addressBook'],
1002 ];
1003 if (isset($entry['class'])) {
1004 $option['groupClass'] = $entry['class'];
1005 }
1006 $selectOptions[] = $option;
1007 }
1008
1009 // $phpMailer = new \OCA\CAFEVDB\CommonPHPMailer(true); could validate addresses here
1010
1011 $html = (new TemplateResponse(
1012 $this->appName,
1013 'emailform/addressbook',
1014 [ 'emailOptions' => $selectOptions ],
1015 'blank'))->render();
1016
1017 return self::dataResponse([ 'contents' => $html ]);
1018
1019 case 'save':
1020 // Get some common post data, rest has to be handled by the
1021 // recipients and the sender class.
1022 $addressBookCandidates = $this->parameterService->getParam('addressBookCandidates', []);
1023
1024 $formContacts = [];
1025 foreach ($addressBookCandidates as $record) {
1026 // This is already pre-parsed. If there is a natural name for the
1027 // person, then it is the thing until the first occurence of '<'.
1028 $text = $record['text']; // use html?
1029 $name = strchr($text, '<', true);
1030 if ($name !== false) {
1031 $name = Util::normalizeSpaces($name);
1032 } else {
1033 $name = '';
1034 }
1035 $email = $record['value'];
1036 $formContacts[] = [
1037 'email' => $email,
1038 'name' => $name,
1039 'display' => htmlspecialchars($name.' <'.$email.'>')
1040 ];
1041 }
1042 $failedContacts = [];
1043 foreach ($formContacts as $contact) {
1044 if ($contactsService->addEmailContact($contact) === false) {
1045 $failedContacts[] = $contact['display'];
1046 }
1047 }

1050 return self::grumble(
1051 $this->l->t(
1052 'The following contacts could not be stored: %s',
1053 implode(', ', $failedContacts)));

1056 return self::response('');

1075 $postMaxSize = \OCP\Util::computerFileSize(ini_get('post_max_size'));

1104 'tmp_name' => $node->getStorage()->getLocalFile($node->getInternalPath()),

1108 ];

1115 $fileRecord['upload_max_file_size'] = $maxUploadFileSize;
1116 $fileRecord['max_human_file_size'] = $maxHumanFileSize;
1117 $files[] = $fileRecord;

1120 case AttachmentOrigin::UPLOAD:
1121 $files = $this->request->files[self::UPLOAD_KEY];

1126 $limit = \OCP\Util::uploadLimit();
1127 if ($contentLength > $limit) {

1132 ]));

1136 return self::grumble(
1137 $this->l->t('No file was uploaded, error message was "%s".', $error['message']));
1138 }
1139 return self::grumble($this->l->t('No file was uploaded. Unknown error'));
1140 }

1142 $files = Util::transposeArray($files);
1143
1144 $totalSize = 0;
1145 foreach ($files as &$file) {
1146
1147 $totalSize += $file['size'];
1148
1149 if ($maxUploadFileSize >= 0 and $totalSize > $maxUploadFileSize) {
1150 return self::grumble([
1151 'message' => $this->l->t('Not enough storage available'),
1152 'upload_max_file_size' => $maxUploadFileSize,
1153 'max_human_file_size' => $maxHumanFileSize,
1154 ]);
1155 }

1157 $file['upload_max_file_size'] = $maxUploadFileSize;
1158 $file['max_human_file_size'] = $maxHumanFileSize;
1159 $file['original_name'] = $file['name']; // clone
1160
1161 $file['str_error'] = Util::fileUploadError($file['error'], $this->l);
1162 if ($file['error'] != UPLOAD_ERR_OK) {
1163 continue;
1164 }
1165

1167 if ($composer->saveAttachment($file) === false) {
1168 $file['error'] = 99;
1169 $file['str_error'] = $this->l->t('Couldn\'t save temporary file for: %s', $file['name']);
1170 continue;
1171 }
